32void ExportTypeStage::start(const QStringList& componentIds,
33 const QMap<QString, QSharedPointer<ComponentData>>& cachedData) {
34 if (m_isRunning.load()) {
35 qWarning() << "ExportTypeStage:" << m_typeName << "already running";
36 return;
37 }
38
39 m_componentIds = componentIds;
40 m_cachedData = cachedData;
41 m_cancelled.store(false);
42 m_isRunning.store(true);
43
44 {
45 QMutexLocker locker(&m_progressMutex);
46 m_progress = ExportTypeProgress();
47 m_progress.typeName = m_typeName;
48 m_progress.totalCount = componentIds.size();
49 m_progress.completedCount = 0;
50 m_progress.successCount = 0;
51 m_progress.failedCount = 0;
52 m_progress.skippedCount = 0;
53 m_progress.inProgressCount = 0;
54 }
55
56 // 清空待处理队列和追踪集合
57 {
58 QMutexLocker locker(&m_workerMutex);
59 m_pendingComponents.clear();
60 m_activeWorkers.clear();
61 }
62
63 qDebug() << "ExportTypeStage:" << m_typeName << "starting for" << componentIds.size() << "components";
64
65 // 如果没有组件需要处理，立即发送完成信号
66 if (componentIds.isEmpty()) {
67 qDebug() << "ExportTypeStage:" << m_typeName << "no components to process, completing immediately";
68 m_isRunning.store(false);
69 emit completed(0, 0, 0);
70 return;
71 }
72
73 // 将所有组件加入待处理队列
74 for (const QString& componentId : componentIds) {
75 initItemProgress(componentId);
76 m_pendingComponents.enqueue(componentId);
77 }
78
79 // 启动初始批次：不超过 maxConcurrent 个 worker
80 int initialBatch = qMin(componentIds.size(), m_threadPool.maxThreadCount());
81 for (int i = 0; i < initialBatch; ++i) {
82 if (m_cancelled.load()) {
83 break;
84 }
85 startNextWorker();
86 }
87
88 qDebug() << "ExportTypeStage:" << m_typeName << "initial batch started, pending:" << m_pendingComponents.size();
89}

91void ExportTypeStage::startNextWorker() {
92 if (m_cancelled.load()) {
93 return;
94 }
95
96 QString componentId;
97 {
98 QMutexLocker locker(&m_workerMutex);
99 if (m_pendingComponents.isEmpty()) {
100 return;
101 }
102 componentId = m_pendingComponents.dequeue();
103 }
104
105 QSharedPointer<ComponentData> data = m_cachedData.value(componentId);
106 QObject* worker = createWorker();
107
108 {
109 QMutexLocker locker(&m_workerMutex);
110 m_activeWorkers.insert(worker); // 使用 QSet 而非 QList，O(1) 插入和删除
111 }
112
113 ExportItemStatus statusSnapshot;
114 ExportTypeProgress progressSnapshot;
115 {
116 QMutexLocker locker(&m_progressMutex);
117 auto it = m_progress.itemStatus.find(componentId);
118 if (it != m_progress.itemStatus.end()) {
119 it->status = ExportItemStatus::Status::InProgress;
120 it->startTime = QDateTime::currentDateTime();
121 statusSnapshot = it.value();
122 }
123 m_progress.inProgressCount++;
124 progressSnapshot = m_progress;
125 }
126
127 qInfo() << "ExportTypeStage::startNextWorker: Emitting itemStatusChanged for" << componentId
128 << "status:" << (int)statusSnapshot.status;
129 emit itemStatusChanged(componentId, statusSnapshot);
130 emit progressChanged(progressSnapshot);
131
132 startWorker(worker, componentId, data);
133}

176bool ExportTypeStage::waitForWorkerThread(QThread*& thread, int timeoutMs) {
177 m_cancelled.store(true);
178
179 if (!thread || !thread->isRunning()) {
180 thread = nullptr;
181 return true;
182 }
183

187 qWarning() << "ExportTypeStage:" << m_typeName << "thread did not finish in" << timeoutMs

219void ExportTypeStage::completeItemProgress(QObject* worker,
220 const QString& componentId,
221 bool success,
222 const QString& error) {
223 // 1. 从活跃worker集合移除
224 bool shouldStartNext = false;
225 bool hasActiveWorkers = false;
226 {
227 QMutexLocker workerLocker(&m_workerMutex);
228 if (worker) {
229 m_activeWorkers.remove(worker);
230 }
231 // 检查是否还有待处理的worker
232 shouldStartNext = !m_pendingComponents.isEmpty();
233 hasActiveWorkers = !m_activeWorkers.isEmpty();
234 }
235
236 ExportItemStatus statusSnapshot;
237 ExportTypeProgress progressSnapshot;
238 bool shouldComplete = false;
239
240 // 2. 更新进度
241 {
242 QMutexLocker locker(&m_progressMutex);
243
244 auto it = m_progress.itemStatus.find(componentId);
245 if (it == m_progress.itemStatus.end()) {
246 qInfo() << "ExportTypeStage::completeItemProgress: componentId not found:" << componentId;
247 return;
248 }
249
250 ExportItemStatus& status = it.value();
251 status.status = success ? ExportItemStatus::Status::Success : ExportItemStatus::Status::Failed;
252 status.endTime = QDateTime::currentDateTime();
253 if (!success) {
254 status.errorMessage = error;
255 }
256 statusSnapshot = status;
257
258 m_progress.completedCount++;
259 if (success) {
260 m_progress.successCount++;
261 } else {
262 m_progress.failedCount++;
263 }
264 m_progress.inProgressCount--;
265 progressSnapshot = m_progress;
266
267 const bool cancelledAndDrained = m_cancelled.load() && !hasActiveWorkers && !shouldStartNext;
268 shouldComplete = m_progress.completedCount >= m_progress.totalCount || cancelledAndDrained;
269 }
270
271 emit itemStatusChanged(componentId, statusSnapshot);
272 emit progressChanged(progressSnapshot);
273
274 if (shouldComplete) {
275 qInfo() << "ExportTypeStage::completeItemProgress: Emitting itemStatusChanged for" << componentId
276 << "status:" << (int)statusSnapshot.status;
277 qDebug() << "ExportTypeStage:" << m_typeName << "completed. Success:" << progressSnapshot.successCount
278 << "Failed:" << progressSnapshot.failedCount << "Skipped:" << progressSnapshot.skippedCount;
279 m_isRunning.store(false);
280 emit completed(progressSnapshot.successCount, progressSnapshot.failedCount, progressSnapshot.skippedCount);
281 return;
282 }
283
284 // 3. 启动下一个待处理的worker（按需创建）
285 if (shouldStartNext && !m_cancelled.load()) {
286 startNextWorker();
287 }
288}
