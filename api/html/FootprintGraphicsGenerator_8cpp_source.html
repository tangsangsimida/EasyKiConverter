12QString FootprintGraphicsGenerator::generatePad(const FootprintPad& pad, double bboxX, double bboxY) const {
13 QString content;
14
15 double x = pxToMmRounded(pad.centerX - bboxX);
16 double y = pxToMmRounded(pad.centerY - bboxY);
17 double width = pxToMmRounded(pad.width);
18 double height = pxToMmRounded(pad.height);
19 double holeRadius = pxToMmRounded(pad.holeRadius);
20
21 QString padNumber = pad.number;
22 if (padNumber.contains("(") && padNumber.contains(")")) {
23 int start = padNumber.indexOf("(");
24 int end = padNumber.indexOf(")");
25 padNumber = padNumber.mid(start + 1, end - start - 1);
26 }
27
28 double rotation = pad.rotation;
29 if (rotation > 180.0) {
30 rotation = rotation - 360.0;
31 }
32
33 bool isCustomShape = (padShapeToKicad(pad.shape) == "custom");
34 QString polygonStr;
35
36 if (isCustomShape) {
37 QStringList pointList = pad.points.split(" ", Qt::SkipEmptyParts);
38
39 if (pointList.isEmpty()) {
40 qWarning() << "Pad" << pad.id << "is a custom polygon, but has no points defined";
41 } else {
42 width = 0.005;
43 height = 0.005;
44
45 rotation = 0.0;
46
47 QString path;
48 for (int i = 0; i < pointList.size(); i += 2) {
49 if (i + 1 < pointList.size()) {
50 bool okX = false;
51 bool okY = false;
52 double px = pointList[i].toDouble(&okX);
53 double py = pointList[i + 1].toDouble(&okY);
54 // 检查转换是否成功
55 if (!okX || !okY) {
56 qWarning() << "Failed to parse point coordinates at index" << i << "for pad" << pad.id;
57 continue;
58 }
59
60 double relX = pxToMmRounded(px - bboxX) - x;
61 double relY = pxToMmRounded(py - bboxY) - y;
62
63 path += QString("(xy %1 %2) ").arg(relX, 0, 'f', 2).arg(relY, 0, 'f', 2);
64 }
65 }
66
67 polygonStr = QString(
68 "\n (primitives\n (gr_poly\n (pts %1)\n (width 0.1)\n (fill "
69 "yes)\n )\n )\n ")
70 .arg(path);
71 }
72 } else {
73 width = qMax(width, 0.01);
74 height = qMax(height, 0.01);
75 }
76
77 QString kicadShape;
78 if (isCustomShape) {
79 kicadShape = "custom";
80 } else {
81 kicadShape = padShapeToKicad(pad.shape);
82 }
83
84 QString kicadType = padTypeToKicad(pad.layerId);
85 QString layers = padLayersToKicad(pad.layerId);
86
87 QString drillStr;
88 if (holeRadius > 0 && pad.holeLength > 0) {
89 double holeLengthMm = pxToMmRounded(pad.holeLength);
90 double maxDistanceHole = qMax(holeRadius * 2, holeLengthMm);
91 double pos0 = height - maxDistanceHole;
92 double pos90 = width - maxDistanceHole;
93 double maxDistance = qMax(pos0, pos90);
94
95 if (qAbs(maxDistance - pos0) < qAbs(maxDistance - pos90)) {
96 drillStr = QString(" (drill oval %1 %2)").arg(holeRadius * 2, 0, 'f', 2).arg(holeLengthMm, 0, 'f', 2);
97 } else {
98 drillStr = QString(" (drill oval %1 %2)").arg(holeLengthMm, 0, 'f', 2).arg(holeRadius * 2, 0, 'f', 2);
99 }
100 } else if (holeRadius > 0) {
101 double drillDiameter = holeRadius * 2;
102
103 drillStr = QString(" (drill %1)").arg(drillDiameter, 0, 'f', 2);
104 }
105
106 content += QString(" (pad %1 %2 %3 (at %4 %5 %6) (size %7 %8) (layers %9)%10%11)\n")
107 .arg(padNumber)
108 .arg(kicadType)
109 .arg(kicadShape)
110 .arg(x, 0, 'f', 2)
111 .arg(y, 0, 'f', 2)
112 .arg(rotation, 0, 'f', 2)
113 .arg(width, 0, 'f', 2)
114 .arg(height, 0, 'f', 2)
115 .arg(layers)
116 .arg(drillStr)
117 .arg(polygonStr);
118
119 return content;
120}

224QString FootprintGraphicsGenerator::generateArc(const FootprintArc& arc, double bboxX, double bboxY) const {
225 QString content;
226
227 QString arcParam = "M " + arc.path;
228
229 GeometryUtils::SvgArcResult svgArc = GeometryUtils::solveSvgArc(arcParam);
230
231 double cx = pxToMmRounded(svgArc.cx - bboxX);
232 double cy = -pxToMmRounded(svgArc.cy - bboxY);
233
234 if (svgArc.rx == svgArc.ry && svgArc.rx > 0) {
235 double startAngle = svgArc.startAngle;
236 double deltaAngle = svgArc.deltaAngle;
237 double step = 180.0;
238
239 if (deltaAngle < 0) {
240 startAngle = startAngle + deltaAngle;
241 deltaAngle = -deltaAngle;
242 }
243
244 while (deltaAngle > 0.1) {
245 if (deltaAngle < step) {
246 step = deltaAngle;
247 }
248
249 svgArc.startAngle = startAngle;
250 svgArc.deltaAngle = step;
251
252 GeometryUtils::SvgArcEndpoints pt = GeometryUtils::calcSvgArc(svgArc);
253
254 double kiEndAngle = svgArc.startAngle;
255 if (step == 180.0)
256 kiEndAngle += 0.1;
257
258 content += QString(" (fp_arc (start %1 %2) (end %3 %4) (angle %5) (layer %6) (width %7))\n")
259 .arg(cx, 0, 'f', 2)
260 .arg(cy, 0, 'f', 2)
261 .arg(pxToMmRounded(pt.x2 - bboxX), 0, 'f', 2)
262 .arg(-pxToMmRounded(pt.y2 - bboxY), 0, 'f', 2)
263 .arg(-kiEndAngle, 0, 'f', 2)
264 .arg(layerIdToKicad(arc.layerId))
265 .arg(pxToMmRounded(arc.strokeWidth), 0, 'f', 2);
266
267 deltaAngle -= step;
268 startAngle += step;
269 }
270 } else {
271 QList<GeometryUtils::SvgPoint> points = GeometryUtils::arcToPath(svgArc, false);
272
273 for (int i = 1; i < points.size(); ++i) {
274 double x1 = pxToMmRounded(points[i - 1].x - bboxX);
275 double y1 = -pxToMmRounded(points[i - 1].y - bboxY);
276 double x2 = pxToMmRounded(points[i].x - bboxX);
277 double y2 = -pxToMmRounded(points[i].y - bboxY);
278
279 content += QString(" (fp_line (start %1 %2) (end %3 %4) (layer %5) (width %6))\n")
280 .arg(x1, 0, 'f', 2)
281 .arg(y1, 0, 'f', 2)
282 .arg(x2, 0, 'f', 2)
283 .arg(y2, 0, 'f', 2)
284 .arg(layerIdToKicad(arc.layerId))
285 .arg(pxToMmRounded(arc.strokeWidth), 0, 'f', 2);
286 }
287 }
288
289 return content;
290}

292QString FootprintGraphicsGenerator::generateText(const FootprintText& text, double bboxX, double bboxY) const {
293 QString content;
294 double x = pxToMmRounded(text.centerX - bboxX);
295 double y = pxToMmRounded(text.centerY - bboxY);
296
297 QString layer = layerIdToKicad(text.layerId);
298
299 if (text.type == "N") {
300 layer = layer.replace(".SilkS", ".Fab");
301 }
302
303 QString displayStr = text.isDisplayed ? "" : " hide";
304
305 bool isNonASCII = false;
306 for (int i = 0; i < text.text.length(); ++i) {
307 if (text.text[i].unicode() > 127) {
308 isNonASCII = true;
309 break;
310 }
311 }
312 if (isNonASCII && !text.textPath.isEmpty()) {
313 qWarning() << "Warning: Converting non-ASCII text to polygon:" << text.text;
314
315 QStringList paths = text.textPath.split("M", Qt::SkipEmptyParts);
316 for (const QString& pathStr : paths) {
317 if (pathStr.trimmed().isEmpty())
318 continue;
319
320 QStringList tokens = pathStr.split(QRegularExpression("[\\s,]+"), Qt::SkipEmptyParts);
321 QList<QPointF> points;
322 for (int i = 0; i + 1 < tokens.size(); i += 2) {
323 double px = tokens[i].toDouble();
324 double py = tokens[i + 1].toDouble();
325
326 double relX = pxToMmRounded(px - bboxX);
327 double relY = pxToMmRounded(py - bboxY);
328 points.append(QPointF(relX, relY));
329 }
330
331 if (points.size() >= 2) {
332 for (int i = 1; i < points.size(); ++i) {
333 content += QString(" (fp_line (start %1 %2) (end %3 %4) (layer %5) (width %6))\n")
334 .arg(points[i - 1].x(), 0, 'f', 2)
335 .arg(points[i - 1].y(), 0, 'f', 2)
336 .arg(points[i].x(), 0, 'f', 2)
337 .arg(points[i].y(), 0, 'f', 2)
338 .arg(layer)
339 .arg(pxToMmRounded(text.strokeWidth) * 0.8, 0, 'f', 2);
340 }
341 }
342 }
343 } else {
344 bool isBottomLayer = layer.startsWith("B");
345 QString mirrorStr = isBottomLayer ? " mirror" : "";
346
347 content += QString(" (fp_text user %1 (at %2 %3 %4) (layer %5)%6\n")
348 .arg(text.text)
349 .arg(x, 0, 'f', 2)
350 .arg(y, 0, 'f', 2)
351 .arg(double(text.rotation), 0, 'f', 2)
352 .arg(layer)
353 .arg(displayStr);
354
355 double fontSize = pxToMmRounded(text.fontSize);
356 double thickness = pxToMmRounded(text.strokeWidth);
357 fontSize = qMax(fontSize, 1.0);
358 thickness = qMax(thickness, 0.01);
359 content += QString(" (effects (font (size %1 %2) (thickness %3)) (justify left%4))\n")
360 .arg(fontSize, 0, 'f', 2)
361 .arg(fontSize, 0, 'f', 2)
362 .arg(thickness, 0, 'f', 2)
363 .arg(mirrorStr);
364 content += " )\n";
365 }
366
367 return content;
368}

370QString FootprintGraphicsGenerator::generateSolidRegion(const FootprintSolidRegion& region,
371 double bboxX,
372 double bboxY) const {
373 QString content;
374
375 bool isCourtYard = (region.layerId == 99);
376 QString layer;
377
378 if (isCourtYard) {
379 layer = "F.CrtYd";
380 } else if (region.layerId == 100) {
381 layer = "F.Fab";
382 } else if (region.layerId == 101) {
383 layer = "F.SilkS";
384 } else {
385 layer = layerIdToKicad(region.layerId);
386 }
387
388 QStringList tokens = region.path.split(QRegularExpression("[\\s,]+"), Qt::SkipEmptyParts);
389 QList<QPointF> points;
390
391 for (int i = 0; i < tokens.size();) {
392 QString token = tokens[i].toUpper();
393
394 if (token == "M") {
395 if (i + 2 < tokens.size()) {
396 double x = tokens[i + 1].toDouble();
397 double y = tokens[i + 2].toDouble();
398 points.append(QPointF(pxToMmRounded(x - bboxX), pxToMmRounded(y - bboxY)));
399 i += 3;
400 } else {
401 i++;
402 }
403 } else if (token == "L") {
404 if (i + 2 < tokens.size()) {
405 double x = tokens[i + 1].toDouble();
406 double y = tokens[i + 2].toDouble();
407 points.append(QPointF(pxToMmRounded(x - bboxX), pxToMmRounded(y - bboxY)));
408 i += 3;
409 } else {
410 i++;
411 }
412 } else if (token == "Z") {
413 if (!points.isEmpty()) {
414 points.append(points.first());
415 }
416 i++;
417 } else {
418 i++;
419 }
420 }
421
422 if (points.size() >= 2) {
423 if (isCourtYard || region.isKeepOut) {
424 for (int i = 1; i < points.size(); ++i) {
425 content += QString(" (fp_line (start %1 %2) (end %3 %4) (layer %5) (width 0.05))\n")
426 .arg(points[i - 1].x(), 0, 'f', 2)
427 .arg(points[i - 1].y(), 0, 'f', 2)
428 .arg(points[i].x(), 0, 'f', 2)
429 .arg(points[i].y(), 0, 'f', 2)
430 .arg(layer);
431 }
432 } else {
433 content += " (fp_poly\n";
434 content += " (pts";
435 for (const QPointF& pt : points) {
436 content += QString(" (xy %1 %2)").arg(pt.x(), 0, 'f', 2).arg(pt.y(), 0, 'f', 2);
437 }
438 content += ")\n";
439 content += QString(" (layer %1)\n").arg(layer);
440 content += " (width 0.1)\n";
441 if (region.fillStyle == "solid") {
442 content += " (fill solid)\n";
443 }
444 content += " )\n";
445 }
446 }
447
448 return content;
449}

481QString FootprintGraphicsGenerator::generateModel3D(const Model3DData& model3D,
482 double bboxX,
483 double bboxY,
484 const QString& model3DPath,
485 const QString& fpType) const {
486 QString content;
487
488 QString finalPath = model3DPath.isEmpty() ? model3D.name() : model3DPath;
489
490 double z = pxToMmRounded(model3D.translation().z);
491
492 if (fpType == "smd") {
493 z = -z;
494
495 } else {
496 z = 0.0;
497 }
498
499 double rotX = (360.0 - model3D.rotation().x);
500 // 使用 fmod 替代 while 循环，避免死循环风险
501 rotX = fmod(rotX, 360.0);
502 if (rotX < 0.0) {
503 rotX += 360.0;
504 }
505
506 double rotY = (360.0 - model3D.rotation().y);
507 // 使用 fmod 替代 while 循环，避免死循环风险
508 rotY = fmod(rotY, 360.0);
509 if (rotY < 0.0) {
510 rotY += 360.0;
511 }
512
513 double rotZ = (360.0 - model3D.rotation().z);
514 // 使用 fmod 替代 while 循环，避免死循环风险
515 rotZ = fmod(rotZ, 360.0);
516 if (rotZ < 0.0) {
517 rotZ += 360.0;
518 }
519
520 // 计算 3D 模型的 offset
521
522 // 检测是否为异常的小值（来自特殊单位的 c_origin）
523
524 // 如果 model3D.translation() 的值远小于 bbox（< 100），说明已经转换为毫米单位
525
526 double finalOffsetX, finalOffsetY;
527
528 if (qAbs(model3D.translation().x) < 100 && qAbs(model3D.translation().y) < 100) {
529 // 异常值：translation 已经是毫米单位
530
531 // 对于异常封装，translation 可能是相对于焊盘中心的偏移，而不是相对于 bbox 的
532
533 // 需要计算焊盘中心相对于 bbox 左上角的偏移
534
535 double padsCenterX = bboxX + 3999.5; // 使用固定的焊盘中心 (4000, 3000) 作为基准
536
537 double padsCenterY = bboxY + 2999.5;
538
539 // 将焊盘中心转换为毫米单位
540
541 double padsCenterX_mm = pxToMmRounded(padsCenterX - bboxX);
542
543 double padsCenterY_mm = pxToMmRounded(padsCenterY - bboxY);
544
545 // 最终偏移 = translation - 焊盘中心偏移
546
547 finalOffsetX = model3D.translation().x - padsCenterX_mm;
548
549 finalOffsetY = -(model3D.translation().y - padsCenterY_mm); // Y 轴取反
550
551 } else {
552 // 正常值：translation 和 bbox 都是像素单位，使用标准的 pxToMm 转换逻辑
553
554 double offsetX = model3D.translation().x - bboxX;
555
556 double offsetY = model3D.translation().y - bboxY;
557
558 finalOffsetX = pxToMmRounded(offsetX);
559
560 finalOffsetY = -pxToMmRounded(offsetY); // Y 轴取反（正常情况）
561 }
562
563 content += QString(" (model \"%1\"\n").arg(finalPath);
564
565 content += QString(" (offset (xyz %1 %2 %3))\n")
566
567 .arg(finalOffsetX, 0, 'f', 3)
568
569 .arg(finalOffsetY, 0, 'f', 3)
570
571 .arg(z, 0, 'f', 3);
572
573 content += " (scale (xyz 1 1 1))\n";
574
575 content += QString(" (rotate (xyz %1 %2 %3))\n").arg(rotX, 0, 'f', 0).arg(rotY, 0, 'f', 0).arg(rotZ, 0, 'f', 0);
576
577 content += " )\n";
578
579 return content;
580}

608QString FootprintGraphicsGenerator::padTypeToKicad(int layerId) const {
609 if (layerId == 1) {
610 return "smd";
