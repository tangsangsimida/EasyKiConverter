41QJsonObject _loadMetadataFromPath(const QString& metaPath) {
42 QFile metaFile(metaPath);
43 if (!metaFile.open(QIODevice::ReadOnly)) {
44 return QJsonObject();
45 }
46 QByteArray data = metaFile.readAll();
47 metaFile.close();
48 QJsonParseError error;
49 QJsonDocument doc = QJsonDocument::fromJson(data, &error);
50 if (error.error != QJsonParseError::NoError || !doc.isObject()) {
51 return QJsonObject();
52 }
53 return doc.object();
54}

56bool _saveMetadataToPath(const QString& metaPath, const QJsonObject& metadata) {
57 QFile metaFile(metaPath);
58 if (!metaFile.open(QIODevice::WriteOnly)) {
59 return false;
60 }
61 QByteArray data = QJsonDocument(metadata).toJson(QJsonDocument::Compact);
62 qint64 written = metaFile.write(data);
63 metaFile.close();
64 return written > 0;

71 QDir rootDir(m_cacheRoot);
72 if (!rootDir.exists()) {

75
76 int repairedComponents = 0;
77 int removedComponents = 0;
78
79 const QStringList componentDirs = rootDir.entryList(QDir::Dirs | QDir::NoDotAndDotDot, QDir::Name);
80 for (const QString& entry : componentDirs) {
81 if (entry == "model3d") {

131bool CacheHealthManager::repairComponentCache(const QString& lcscId) {
132 const QString dirPath = componentCacheDir(lcscId);
133 QDir componentDir(dirPath);
134 if (!componentDir.exists()) {
135 return false;
136 }
137
138 const QString metaPath = metadataPath(lcscId);
139 if (!QFileInfo::exists(metaPath)) {
140 componentDir.removeRecursively();
141 qWarning().noquote() << "Removed invalid cache dir without metadata:" << dirPath;
142 return false;
143 }
144
145 QJsonObject metadata = loadMetadata(lcscId);
146 if (metadata.isEmpty()) {
147 componentDir.removeRecursively();
148 qWarning().noquote() << "Removed invalid cache dir with unreadable metadata:" << dirPath;
149 return false;
150 }
151
152 bool metadataChanged = false;
153 if (metadata.value("lcscId").toString().isEmpty()) {
154 metadata["lcscId"] = lcscId;
155 metadataChanged = true;
156 }
157 if (!metadata.contains("cachedAt")) {
158 metadata["cachedAt"] = QDateTime::currentDateTime().toString(Qt::ISODate);
159 metadataChanged = true;
160 }
161
162 const QString cadPath = componentDir.filePath("cad_data.json");
163 bool hasValidCad = false;
164 if (QFileInfo::exists(cadPath)) {
165 QFile cadFile(cadPath);
166 if (cadFile.open(QIODevice::ReadOnly)) {
167 const QByteArray cadData = cadFile.readAll();
168 cadFile.close();
169
170 QJsonParseError parseError;
171 const QJsonDocument cadDoc = QJsonDocument::fromJson(cadData, &parseError);
172 hasValidCad = !cadData.isEmpty() && parseError.error == QJsonParseError::NoError && cadDoc.isObject();
173 }
174
175 if (!hasValidCad) {
176 QFile::remove(cadPath);
177 qWarning().noquote() << "Removed broken cad_data.json from cache:" << cadPath;
178 }
179 }
180
181 for (int index = 0; index < 3; ++index) {
182 const QString previewPath = previewImagePath(lcscId, index);
183 const QFileInfo previewInfo(previewPath);
184 if (previewInfo.exists() && previewInfo.size() <= 0) {
185 QFile::remove(previewPath);
186 qWarning().noquote() << "Removed empty preview cache file:" << previewPath;
187 }
188 }
189
190 const QString legacyDatasheetPath = datasheetPath(lcscId);
191 if (QFileInfo::exists(legacyDatasheetPath)) {
192 QFile legacyDatasheet(legacyDatasheetPath);
193 if (legacyDatasheet.open(QIODevice::ReadOnly)) {
194 const QByteArray legacyData = legacyDatasheet.read(8);
195 legacyDatasheet.close();
196
197 const QString format = legacyData.startsWith("%PDF-") ? QStringLiteral("pdf") : QStringLiteral("html");
198 const QString targetPath = resolveDatasheetPath(lcscId, format, true);
199 if (QFile::exists(targetPath)) {
200 QFile::remove(targetPath);
201 }
202 if (legacyDatasheet.rename(legacyDatasheetPath, targetPath)) {
203 metadata["datasheetFormat"] = format;
204 metadataChanged = true;
205 qInfo().noquote() << "Migrated legacy datasheet cache file:" << targetPath;
206 } else {
207 QFile::remove(legacyDatasheetPath);
208 qWarning().noquote() << "Removed unreadable legacy datasheet cache file:" << legacyDatasheetPath;
209 }
210 } else {
211 QFile::remove(legacyDatasheetPath);
212 qWarning().noquote() << "Removed unreadable legacy datasheet cache file:" << legacyDatasheetPath;
213 }
214 }
215
216 const QString datasheetFormat = metadata.value("datasheetFormat").toString();
217 const QString resolvedDatasheetPath = resolveDatasheetPath(lcscId, datasheetFormat, false);
218 if (QFileInfo::exists(resolvedDatasheetPath)) {
219 const QFileInfo datasheetInfo(resolvedDatasheetPath);
220 if (datasheetInfo.size() <= 0) {
221 QFile::remove(resolvedDatasheetPath);
222 qWarning().noquote() << "Removed empty datasheet cache file:" << resolvedDatasheetPath;
223 } else if (datasheetFormat.isEmpty()) {
224 metadata["datasheetFormat"] =
225 resolvedDatasheetPath.endsWith(".html") ? QStringLiteral("html") : QStringLiteral("pdf");
226 metadataChanged = true;
227 }
228 }
229
230 const bool hasBasicIdentity =
231 !metadata.value("lcscId").toString().isEmpty() || !metadata.value("name").toString().isEmpty();
232 const bool hasPreviewUrls = !metadata.value("previewImages").toArray().isEmpty();
233 const bool hasDatasheetUrl = !metadata.value("datasheet").toString().isEmpty();
234 const bool hasModel3DUuid = !metadata.value("model3duuid").toString().isEmpty();
235
236 if (!hasValidCad && !hasBasicIdentity && !hasPreviewUrls && !hasDatasheetUrl && !hasModel3DUuid) {
237 componentDir.removeRecursively();
238 qWarning().noquote() << "Removed unusable cache dir after self-heal:" << dirPath;
239 return false;
240 }
241
242 if (metadataChanged) {
243 saveMetadata(lcscId, metadata);
244 }
245
246 return true;
247}

249void CacheHealthManager::repairModel3DCache() {
250 const QString modelCacheDir = m_cacheRoot + "/model3d";
251 QDir dir(modelCacheDir);
252 if (!dir.exists()) {
253 return;
254 }
255
256 const QSet<QString> validSuffixes = {QStringLiteral("step"), QStringLiteral("wrl")};
257 const QFileInfoList files = dir.entryInfoList(QDir::Files | QDir::NoDotAndDotDot, QDir::Name);
258 for (const QFileInfo& fileInfo : files) {
259 const QString suffix = fileInfo.suffix().toLower();
260 const bool removable =
261 fileInfo.size() <= 0 || !validSuffixes.contains(suffix) || fileInfo.fileName().startsWith(".tmp");
262 if (removable) {
263 QFile::remove(fileInfo.absoluteFilePath());
264 qWarning().noquote() << "Removed broken 3D cache file:" << fileInfo.absoluteFilePath();
265 }
266 }
267}
