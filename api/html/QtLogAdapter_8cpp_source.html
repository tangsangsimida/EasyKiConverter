53void QtLogAdapter::qtMessageHandler(QtMsgType type, const QMessageLogContext& context, const QString& msg) {
54 // === 递归调用检测 ===
55 // 检查是否已经在处理日志（防止死锁）
56 if (s_inHandler.loadRelaxed()) {
57 // 递归调用检测到，使用原始处理器或直接输出到 stderr
58 if (s_originalHandler) {
59 s_originalHandler(type, context, msg);
60 } else {
61 // 后备方案：直接输出到 stderr
62 fprintf(stderr, "[RECURSIVE] %s\n", msg.toUtf8().constData());
63 }
64 return;
65 }
66
67 // 设置递归保护标志
68 s_inHandler.storeRelaxed(1);
69
70 // 使用 RAII 确保标志在退出时被清除
71 struct RecursionGuardReset {
72 ~RecursionGuardReset() {
73 s_inHandler.storeRelaxed(0);
74 }
75 } guardReset;
76
77 // 将 Qt 日志类型映射到我们的日志级别
78 LogLevel level;
79 switch (type) {
80 case QtDebugMsg:
81 level = LogLevel::Debug;
82 break;
83 case QtInfoMsg:
84 level = LogLevel::Info;
85 break;
86 case QtWarningMsg:
87 level = LogLevel::Warn;
88 break;
89 case QtCriticalMsg:
90 level = LogLevel::Error;
91 break;
92 case QtFatalMsg:
93 level = LogLevel::Fatal;
94 break;
95 default:
96 level = LogLevel::Info;
97 break;
98 }
99
100 // 安全地获取 Logger 实例并检查是否应该记录
101 Logger* logger = Logger::instance();
102 if (!logger) {
103 return;
104 }
105
106 if (!logger->shouldLog(level, s_defaultModule)) {
107 return;
108 }
109
110 // 构建日志消息（安全地处理字符串）
111 QString message;
112
113 // 避免格式化字符串攻击：不对用户消息进行额外格式化
114 // 仅在消息中添加分类信息
115 if (context.category && strlen(context.category) > 0) {
116 QByteArray category = context.category;
117 if (category != "default") {
118 message = QString("[%1] %2").arg(QString::fromUtf8(category), msg);
119 } else {
120 message = msg;
121 }
122 } else {
123 message = msg;
124 }
125
126 // 记录日志（使用安全的文件名和函数名）
127 const char* safeFile = context.file ? context.file : "";
128 const char* safeFunction = context.function ? context.function : "";
129 int safeLine = context.line >= 0 ? context.line : 0;
130
131 logger->log(level, s_defaultModule, message, safeFile, safeFunction, safeLine);
132
133 // 如果需要保留原始行为
134 if (s_preserveOriginal && s_originalHandler) {
135 // 临时清除递归标志以允许原始处理器工作
136 s_inHandler.storeRelaxed(0);
137 s_originalHandler(type, context, msg);
138 s_inHandler.storeRelaxed(1);
139 }
140
141 // Qt 的 qFatal 默认会终止程序，这里保持这个行为
142 if (type == QtFatalMsg) {
143 logger->flush();
144 // 不调用 abort()，让程序正常退出
145 // 递归保护会在函数退出时自动清除
146 }
147}
