9bool PathSecurity::isValidPathComponent(const QString& name) {
10 if (name.isEmpty())
11 return false;

22bool PathSecurity::isSafePath(const QString& fullPath, const QString& baseDir) {
23 if (fullPath.isEmpty() || baseDir.isEmpty()) {
24 qWarning() << "isSafePath: received empty path";
25 return false;
26 }
27
28 // 1. 获取绝对路径并规范化
29 QFileInfo baseInfo(baseDir);
30 QFileInfo targetInfo(fullPath);
31
32 // 优先使用 canonicalFilePath (要求路径存在)，它能解析符号链接和短文件名，这是最准确的比较方式
33 // 如果路径不存在，回退到 cleanPath(absoluteFilePath)
34 QString absBase = baseInfo.exists() ? baseInfo.canonicalFilePath() : QDir::cleanPath(baseInfo.absoluteFilePath());
35 QString absTarget =
36 targetInfo.exists() ? targetInfo.canonicalFilePath() : QDir::cleanPath(targetInfo.absoluteFilePath());
37
38 // 2. 统一分隔符为 '/' (Qt 默认)
39 absBase = QDir::fromNativeSeparators(absBase);
40 absTarget = QDir::fromNativeSeparators(absTarget);
41
42 // 3. 确保基准路径以 '/' 结尾，用于目录前缀匹配
43 if (!absBase.endsWith('/')) {
44 absBase += '/';
45 }
46
47 // 4. 特殊情况处理：允许目标路径等于基准路径
48 // 移除 target 可能存在的末尾斜杠
49 QString targetNoSlash = absTarget;
50 while (targetNoSlash.endsWith('/'))
51 targetNoSlash.chop(1);
52
53 // 如果 target + '/' == base，说明它们代表同一个目录
54 if (targetNoSlash + '/' == absBase) {
55 return true;
56 }
57 // 或者直接比较（忽略大小写，适应 Windows）
58 if (targetNoSlash.compare(absBase.left(absBase.length() - 1), Qt::CaseInsensitive) == 0) {
59 return true;
60 }
61
62 // 5. 前缀比较：检查 target 是否以 base 开头
63 bool isSafe = absTarget.startsWith(absBase, Qt::CaseInsensitive);
64
65 if (!isSafe) {
66 qWarning() << "isSafePath: path is outside base directory";
67 qWarning() << " Target (norm):" << absTarget;
68 qWarning() << " Base (norm):" << absBase;
69 qWarning() << " Raw Target:" << fullPath;
70 qWarning() << " Raw Base: " << baseDir;
71 }
72
73 return isSafe;
74}

76QString PathSecurity::sanitizeFilename(const QString& name) {
77 QString safeName = name;
78
79 // 替换非法字符: \ / : * ? " < > |
80 // 使用十六进制转义确保安全: \x5C 是反斜杠, \x22 是双引号
81 static const QRegularExpression illegalChars(QStringLiteral("[\\x5C/:*?\\x22<>|]"));
82 safeName.replace(illegalChars, "_");
83
84 // 替换控制字符
85 static const QRegularExpression controlChars(R"([\x00-\x1f])");
86 safeName.replace(controlChars, "");
87
88 // 替换 Unicode 零宽字符 (可能导致显示欺骗)
89 static const QRegularExpression zeroWidth(R"([\u200B-\u200D\uFEFF])");
90 safeName.replace(zeroWidth, "");
91
92 safeName = safeName.trimmed();
93
94 if (safeName.isEmpty()) {
95 safeName = "unnamed_component";
96 }
97
98 static const QStringList reservedNames = {"CON", "PRN", "AUX", "NUL", "COM1", "COM2", "COM3", "COM4",
99 "COM5", "COM6", "COM7", "COM8", "COM9", "LPT1", "LPT2", "LPT3",
100 "LPT4", "LPT5", "LPT6", "LPT7", "LPT8", "LPT9"};
101
102 if (reservedNames.contains(safeName, Qt::CaseInsensitive)) {
103 safeName += "_safe";
104 }
105
106 return safeName;
107}

109bool PathSecurity::safeRemoveRecursively(const QString& path, int maxFileCount) {
110 QDir dir(path);
111 if (!dir.exists())
112 return true;
113
114 int count = countFilesRecursively(path, maxFileCount + 1);
115 if (count > maxFileCount) {
116 qCritical() << "Security Alert: Recursive delete aborted. File count" << count << "exceeds limit of"
117 << maxFileCount << "in path:" << path;
118 return false;
119 }
120
121 qDebug() << "Safe delete: Removing directory with" << count << "items:" << path;
122 return dir.removeRecursively();
123}

125int PathSecurity::countFilesRecursively(const QString& path, int limit, int currentCount) {
126 if (currentCount >= limit)
127 return currentCount;
128
129 QDir dir(path);
130 QFileInfoList list = dir.entryInfoList(QDir::AllEntries | QDir::NoDotAndDotDot | QDir::Hidden);
131
132 for (const QFileInfo& info : list) {
133 currentCount++;
134 if (currentCount >= limit)
135 return limit;
136
137 if (info.isDir()) {
138 if (!info.isSymLink()) {
139 currentCount = countFilesRecursively(info.absoluteFilePath(), limit, currentCount);
140 } else {
141 // 注意：我们跳过符号链接指向的内容，不计入文件总数，也不深入检查
142 // removeRecursively 会删除符号链接本身，但不会删除目标
143 // 这是预期的安全行为
144 currentCount++;
145 }
146 if (currentCount >= limit)
147 return limit;
148 } else {
149 currentCount++;
150 }
151 }
152 return currentCount;
153}
