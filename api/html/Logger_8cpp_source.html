72void Logger::log(LogLevel level,
73 LogModule module,
74 const QString& message,
75 const char* file,
76 const char* function,
77 int line) {
78 // 快速检查是否应该记录（无锁快速路径）
79 if (!shouldLog(level, module)) {
80 return;
81 }
82
83 // 构建日志记录
84 LogRecord record;
85 record.level = level;
86 record.module = module;
87 record.message = message;
88 record.timestamp = QDateTime::currentMSecsSinceEpoch();
89 record.threadId = QThread::currentThreadId();
90
91 if (file) {
92 record.fileName = QString::fromUtf8(file);
93 }
94 if (function) {
95 record.functionName = QString::fromUtf8(function);
96 }
97 record.line = line;
98
99 // 获取线程名称
100 if (s_threadNames.hasLocalData()) {
101 record.threadName = s_threadNames.localData();
102 }
103
104 // 发射信号
105 emit logRecord(record);
106
107 // 分发到各 Appender
108 QMutexLocker locker(&m_appendersMutex);
109 for (auto& appender : m_appenders) {
110 QString formatted;
111 if (appender->formatter()) {
112 formatted = appender->formatter()->format(record);
113 } else {
114 formatted = record.message;
115 }
116 appender->append(record, formatted);
117 }
118}

120bool Logger::shouldLog(LogLevel level, LogModule module) const {
121 // 快速路径：先检查全局级别（使用原子变量，无锁）
122 int globalLvl = m_globalLevel.loadRelaxed();
123 if (static_cast<int>(level) < globalLvl) {
124 // 如果全局级别已经过滤掉了，直接返回
125 // 但还需要检查模块级别（可能模块级别更低）
126
127 // 检查模块级别（需要读锁）
128 QReadLocker locker(&m_moduleLevelsLock);
129 if (m_moduleLevels.isEmpty()) {
130 return false; // 没有模块级别设置，全局过滤生效
131 }
132
133 auto it = m_moduleLevels.find(module);
134 if (it == m_moduleLevels.end()) {
135 return false; // 该模块没有特定设置，全局过滤生效
136 }
137
138 // 模块有特定设置，使用模块级别
139 return static_cast<int>(level) >= static_cast<int>(it.value());
140 }
141
142 // 全局级别检查通过
143 // 但还需要检查模块是否有更严格的级别设置
144 QReadLocker locker(&m_moduleLevelsLock);
145 auto it = m_moduleLevels.find(module);
146 if (it != m_moduleLevels.end()) {
147 return static_cast<int>(level) >= static_cast<int>(it.value());
148 }
149
150 return true; // 模块没有特定设置，使用全局结果
151}

185void LogUtils::replaceFirstPlaceholder(QString& result, const QString& replacement) {
186 // 安全检查：避免对包含格式化字符的输入进行处理
187 // 仅替换第一个 {} 占位符，不处理其他格式化字符
188 int pos = result.indexOf(QStringLiteral("{}"));
189 if (pos >= 0) {
190 // 对替换内容进行安全处理，转义可能的格式化字符
191 QString safeReplacement = replacement;
192 // 不对内容进行额外的格式化处理，直接替换
193 result.replace(pos, 2, safeReplacement);
194 }
195}
