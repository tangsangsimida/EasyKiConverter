57void ComponentDataCollector::handleCadDataFetched(const QJsonObject& data) {
58 qDebug() << "CAD data fetched for:" << m_componentId;
59
60 if (m_isCancelled) {
61 handleError("Data collection cancelled");
62 return;
63 }
64
65 // 检查响应结
66 QJsonObject resultData;
67 if (data.contains("result")) {
68 resultData = data["result"].toObject();
69 } else {
70 resultData = data;
71 }
72
73 if (resultData.isEmpty()) {
74 handleError("Empty CAD data");
75 return;
76 }
77
78 // 设置元件基本信息
79 m_componentData.setLcscId(m_componentId);
80
81 if (resultData.contains("title")) {
82 m_componentData.setName(resultData["title"].toString());
83 }
84 if (resultData.contains("package")) {
85 m_componentData.setPackage(resultData["package"].toString());
86 }
87 if (resultData.contains("manufacturer")) {
88 m_componentData.setManufacturer(resultData["manufacturer"].toString());
89 }
90 if (resultData.contains("datasheet")) {
91 m_componentData.setDatasheet(resultData["datasheet"].toString());
92 }
93
94 // 导入符号数据
95 QSharedPointer<SymbolData> symbolData = m_importer->importSymbolData(resultData);
96 if (symbolData) {
97 m_componentData.setSymbolData(symbolData);
98 qDebug() << "Symbol imported - Name:" << symbolData->info().name;
99 } else {
100 qWarning() << "Failed to import symbol data";
101 }
102
103 // 导入封装数据
104 QSharedPointer<FootprintData> footprintData = m_importer->importFootprintData(resultData);
105 if (footprintData) {
106 m_componentData.setFootprintData(footprintData);
107 qDebug() << "Footprint imported - Name:" << footprintData->info().name;
108 } else {
109 qWarning() << "Failed to import footprint data";
110 }
111
112 // 检查是否需要获3D 模型
113 if (m_export3DModel && footprintData && footprintData->model3D().uuid().isEmpty()) {
114 // 从 CAD 数据中提取 3D 模型 UUID
115 if (resultData.contains("head")) {
116 QJsonObject head = resultData["head"].toObject();
117 if (head.contains("uuid_3d")) {
118 QString uuid = head["uuid_3d"].toString();
119 if (!uuid.isEmpty()) {
120 qDebug() << "Fetching 3D model with UUID:" << uuid;
121
122 // 更新 Model3DData 的 UUID
123 QSharedPointer<Model3DData> model3DData(new Model3DData());
124 model3DData->setUuid(uuid);
125 m_componentData.setModel3DData(model3DData);
126
127 // 进入获取 OBJ 数据状
128 setState(FetchingObjData);
129 m_api->fetch3DModelObj(uuid);
130 return;
131 }
132 }
133 }
134 }
135
136 // 不需3D 模型或没有找UUID，直接完
137 complete();
138}

140void ComponentDataCollector::handleModel3DFetched(const QString& uuid, const QByteArray& data) {
141 qDebug() << "3D model data fetched for UUID:" << uuid << "Size:" << data.size();
142
143 if (m_isCancelled) {
144 handleError("Data collection cancelled");
145 return;
146 }
147
148 // 根据当前状态决定如何处
149 if (m_state == FetchingObjData) {
150 // 保存 OBJ 数据
151 if (m_componentData.model3DData()) {
152 m_componentData.model3DData()->setRawObj(QString::fromUtf8(data));
153 qDebug() << "OBJ data saved for:" << uuid;
154 }
155
156 // 检查是否需要获STEP 数据
157 if (m_export3DModel) {
158 setState(FetchingStepData);
159 m_api->fetch3DModelStep(uuid);
160 } else {
161 complete();
162 }
163 } else if (m_state == FetchingStepData) {
164 // 保存 STEP 数据
165 if (m_componentData.model3DData()) {
166 m_componentData.model3DData()->setStep(data);
167 qDebug() << "STEP data saved for:" << uuid;
168 }
169
170 complete();
171 } else {
172 qDebug() << "Unexpected 3D model data in state:" << m_state;
173 }
174}
