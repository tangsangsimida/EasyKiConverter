44QSharedPointer<ComponentData> ExportWorkerHelpers::loadDiskCachedComponentData(const QString& componentId) {
45 ComponentCacheService* cache = ComponentCacheService::instance();
46 QSharedPointer<ComponentData> cachedData = cache->loadComponentData(componentId);
47 if (!cachedData) {
48 return nullptr;
49 }
50
51 const QByteArray cadJsonData = cache->loadCadDataJson(componentId);
52 if (!cadJsonData.isEmpty()) {
53 QJsonParseError parseError;
54 const QJsonDocument cadDoc = QJsonDocument::fromJson(cadJsonData, &parseError);
55 if (parseError.error == QJsonParseError::NoError && cadDoc.isObject()) {
56 const QJsonObject cadDataObject = cadDoc.object();
57 EasyedaSymbolImporter symbolImporter;
58 EasyedaFootprintImporter footprintImporter;
59 cachedData->setSymbolData(symbolImporter.importSymbolData(cadDataObject));
60 cachedData->setFootprintData(footprintImporter.importFootprintData(cadDataObject));
61 }
62 // 保存原始 CAD JSON 数据用于 debug 导出
63 cachedData->setCadJsonRaw(cadJsonData);
64 }
65
66 // 加载 3D 模型 OBJ 原始数据
67 if (cachedData->model3DData() && !cachedData->model3DData()->uuid().isEmpty()) {
68 const QByteArray model3DObjData = cache->loadModel3D(cachedData->model3DData()->uuid(), QStringLiteral("obj"));
69 if (!model3DObjData.isEmpty()) {
70 cachedData->setModel3DObjRaw(model3DObjData);
71 }
72 }
73
74 const QByteArray datasheetData = cache->loadDatasheet(componentId);
75 if (!datasheetData.isEmpty()) {
76 cachedData->setDatasheetData(datasheetData);
77 }
78
79 return cachedData;
80}

145void ExportWorkerHelpers::recomputeTypeProgressCounts(ExportTypeProgress& progress) {
146 progress.completedCount = 0;
147 progress.successCount = 0;
148 progress.failedCount = 0;
149 progress.skippedCount = 0;
150 progress.inProgressCount = 0;
151
152 for (auto it = progress.itemStatus.cbegin(); it != progress.itemStatus.cend(); ++it) {
153 switch (it.value().status) {
154 case ExportItemStatus::Status::Pending:
155 break;
156 case ExportItemStatus::Status::InProgress:
157 progress.inProgressCount++;
158 break;
159 case ExportItemStatus::Status::Success:
160 progress.completedCount++;
161 progress.successCount++;
162 break;
163 case ExportItemStatus::Status::Failed:
164 progress.completedCount++;
165 progress.failedCount++;
166 break;
167 case ExportItemStatus::Status::Skipped:
168 progress.completedCount++;
169 progress.skippedCount++;
170 break;
171 }
172 }
173}
