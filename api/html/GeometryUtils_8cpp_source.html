25QPointF GeometryUtils::getArcCenter(double startX,
26 double startY,
27 double endX,
28 double endY,
29 int rotationDirection,
30 double radius) {
31 double arcDistance = distance(startX, startY, endX, endY);
32
33 // 检查距离是否有
34 if (arcDistance < std::numeric_limits<double>::epsilon()) {
35 qWarning() << "Arc distance is too small, cannot calculate center";
36 return QPointF(startX, startY);
37 }
38
39 double mX = (startX + endX) / 2.0;
40 double mY = (startY + endY) / 2.0;
41
42 double u = (endX - startX) / arcDistance;
43 double v = (endY - startY) / arcDistance;
44
45 double hSquared = radius * radius - (arcDistance * arcDistance) / 4.0;
46
47 // 检h² 是否为负数（半径太小
48 if (hSquared < 0) {
49 qWarning() << "Radius is too small for the given arc distance";
50 return QPointF(mX, mY);
51 }
52
53 double h = std::sqrt(hSquared);
54
55 double centerX = mX - rotationDirection * h * v;
56 double centerY = mY + rotationDirection * h * u;
57
58 return QPointF(centerX, centerY);
59}

135void GeometryUtils::computeArc(double startX,
136 double startY,
137 double radiusX,
138 double radiusY,
139 double angle,
140 bool largeArcFlag,
141 bool sweepFlag,
142 double endX,
143 double endY,
144 double& centerX,
145 double& centerY,
146 double& angleExtent) {
147 // Compute the half distance between the current and the final point
148 double dx2 = (startX - endX) / 2.0;
149 double dy2 = (startY - endY) / 2.0;
150
151 // Convert angle from degrees to radians
152 angle = degreesToRadians(std::fmod(angle, 360.0));
153 double cosAngle = std::cos(angle);
154 double sinAngle = std::sin(angle);
155
156 // Step 1: Compute (x1, y1)
157 double x1 = cosAngle * dx2 + sinAngle * dy2;
158 double y1 = -sinAngle * dx2 + cosAngle * dy2;
159
160 // Ensure radii are large enough
161 radiusX = std::abs(radiusX);
162 radiusY = std::abs(radiusY);
163 double PradiusX = radiusX * radiusX;
164 double PradiusY = radiusY * radiusY;
165 double Px1 = x1 * x1;
166 double Py1 = y1 * y1;
167
168 // Check that radii are large enough
169 double radiiCheck = 0.0;
170 if (PradiusX != 0.0 && PradiusY != 0.0) {
171 radiiCheck = Px1 / PradiusX + Py1 / PradiusY;
172 }
173
174 if (radiiCheck > 1.0) {
175 radiusX = std::sqrt(radiiCheck) * radiusX;
176 radiusY = std::sqrt(radiiCheck) * radiusY;
177 PradiusX = radiusX * radiusX;
178 PradiusY = radiusY * radiusY;
179 }
180
181 // Step 2: Compute (cx1, cy1)
182 int sign = (largeArcFlag == sweepFlag) ? -1 : 1;
183 double sq = 0.0;
184 if (PradiusX * Py1 + PradiusY * Px1 > 0.0) {
185 sq = (PradiusX * PradiusY - PradiusX * Py1 - PradiusY * Px1) / (PradiusX * Py1 + PradiusY * Px1);
186 }
187 sq = std::max(sq, 0.0);
188 double coef = sign * std::sqrt(sq);
189 double cx1 = coef * ((radiusX * y1) / radiusY);
190 double cy1 = (radiusX != 0.0) ? coef * -((radiusY * x1) / radiusX) : 0.0;
191
192 // Step 3: Compute (cx, cy) from (cx1, cy1)
193 double sx2 = (startX + endX) / 2.0;
194 double sy2 = (startY + endY) / 2.0;
195 centerX = sx2 + (cosAngle * cx1 - sinAngle * cy1);
196 centerY = sy2 + (sinAngle * cx1 + cosAngle * cy1);
197
198 // Step 4: Compute the angle_extent (dangle)
199 double ux = (radiusX != 0.0) ? (x1 - cx1) / radiusX : 0.0;
200 double uy = (radiusY != 0.0) ? (y1 - cy1) / radiusY : 0.0;
201 double vx = (radiusX != 0.0) ? (-x1 - cx1) / radiusX : 0.0;
202 double vy = (radiusY != 0.0) ? (-y1 - cy1) / radiusY : 0.0;
203
204 // Compute the angle extent
205 double n = std::sqrt((ux * ux + uy * uy) * (vx * vx + vy * vy));
206 double p = ux * vx + uy * vy;
207 sign = ((ux * vy - uy * vx) < 0.0) ? -1 : 1;
208
209 if (n != 0.0) {
210 double ratio = p / n;
211 if (std::abs(ratio) < 1.0) {
212 angleExtent = radiansToDegrees(sign * std::acos(ratio));
213 } else {
214 angleExtent = 360.0 + 359.0;
215 }
216 } else {
217 angleExtent = 360.0 + 359.0;
218 }
219
220 if (!sweepFlag && angleExtent > 0.0) {
221 angleExtent -= 360.0;
222 } else if (sweepFlag && angleExtent < 0.0) {
223 angleExtent += 360.0;
224 }
225
226 int angleExtentSign = (angleExtent < 0.0) ? 1 : -1;
227 angleExtent = (std::abs(angleExtent) / 360.0) * 360.0 * angleExtentSign;
228}

234GeometryUtils::SvgArcResult GeometryUtils::solveSvgArc(const QString& param) {
235 SvgArcResult res;
236
237 // 解析 SVG 弧参数字符串
238 // 格式: "M x1 y1 A rx ry phi fa fs x2 y2"
239 QStringList args = param.split(QRegularExpression("\\s+"), Qt::SkipEmptyParts);
240
241 if (args.size() < 10) {
242 qWarning() << "Invalid SVG arc parameter:" << param;
243 return res;
244 }
245
246 if (args[0] != "M" && args[0] != "m") {
247 qWarning() << "SVG arc must start with M command:" << param;
248 return res;
249 }
250
251 if (args[3] != "A" && args[3] != "a") {
252 qWarning() << "Expected A command in SVG arc:" << param;
253 return res;
254 }
255
256 double x1 = args[1].toDouble();
257 double y1 = args[2].toDouble();
258 double rx = args[4].toDouble();
259 double ry = args[5].toDouble();
260 double phiAngle = args[6].toDouble();
261 int fa = args[7].toInt(); // large-arc-flag
262 int fs = args[8].toInt(); // sweep-flag
263 double x2 = args[9].toDouble();
264 double y2 = (args.size() > 10) ? args[10].toDouble() : 0.0;
265
266 res.rx = rx;
267 res.ry = ry;
268 res.startPt.x = x1;
269 res.startPt.y = y1;
270 res.endPt.x = x2;
271 res.endPt.y = y2;
272 res.xRotate = phiAngle;
273
274 // 如果半径，退化为直线
275 if (rx == 0 || ry == 0) {
276 res.cx = (x1 + x2) / 2.0;
277 res.cy = (y1 + y2) / 2.0;
278 res.deltaAngle = 360.0;
279 return res;
280 }
281
282 // 转换旋转角度为弧
283 double phi = degreesToRadians(phiAngle);
284 double cos_phi = std::cos(phi);
285 double sin_phi = std::sin(phi);
286
287 // 计算中点
288 double dx = (x1 - x2) / 2.0;
289 double dy = (y1 - y2) / 2.0;
290
291 // 变换到未旋转的坐标系
292 double x1_ = cos_phi * dx + sin_phi * dy;
293 double y1_ = -sin_phi * dx + cos_phi * dy;
294
295 // 确保半径足够
296 double rx_sq = rx * rx;
297 double ry_sq = ry * ry;
298 double x1_sq = x1_ * x1_;
299 double y1_sq = y1_ * y1_;
300
301 double lambda = x1_sq / rx_sq + y1_sq / ry_sq;
302 if (lambda > 1.0) {
303 double sqrtLambda = std::sqrt(lambda);
304 rx *= sqrtLambda;
305 ry *= sqrtLambda;
306 rx_sq = rx * rx;
307 ry_sq = ry * ry;
308 }
309
310 // 计算圆心
311 double temp =
312 std::sqrt(std::max(0.0, (rx_sq * ry_sq - rx_sq * y1_sq - ry_sq * x1_sq) / (rx_sq * y1_sq + ry_sq * x1_sq)));
313
314 int sign = (fa == fs) ? -1 : 1;
315 double cx_ = sign * temp * rx * y1_ / ry;
316 double cy_ = -sign * temp * ry * x1_ / rx;
317
318 // 变换回旋转的坐标
319 res.cx = cos_phi * cx_ - sin_phi * cy_ + (x1 + x2) / 2.0;
320 res.cy = sin_phi * cx_ + cos_phi * cy_ + (y1 + y2) / 2.0;
321
322 // 计算起始角度和角度增
323 double angle1 = getAngle(1.0, 0.0, (x1_ - cx_) / rx, (y1_ - cy_) / ry);
324 double angle2 = getAngle((x1_ - cx_) / rx, (y1_ - cy_) / ry, (-x1_ - cx_) / rx, (-y1_ - cy_) / ry);
325
326 res.startAngle = radiansToDegrees(angle1);
327 res.deltaAngle = radiansToDegrees(angle2);
328
329 // 规范化起始角
330 while (res.startAngle < 0)
331 res.startAngle += 360.0;
332 while (res.startAngle >= 360.0)
333 res.startAngle -= 360.0;
334 res.startAngle = std::abs(res.startAngle);
335
336 // 根据 sweep-flag 调整角度增量
337 if (fs != 0) {
338 if (res.deltaAngle < 0)
339 res.deltaAngle += 360.0;
340 } else {
341 if (res.deltaAngle > 0)
342 res.deltaAngle -= 360.0;
343 }
344
345 return res;
346}

348double GeometryUtils::getAngle(double x1, double y1, double x2, double y2) {
349 // angle = arccos( (U dot V) / (|U|*|V|) )
350 double factor = (x1 * y2 - y1 * x2 > 0) ? 1.0 : -1.0;
351 double dot = x1 * x2 + y1 * y2;
352 double mag1 = std::sqrt(x1 * x1 + y1 * y1);
353 double mag2 = std::sqrt(x2 * x2 + y2 * y2);
354
355 if (mag1 == 0 || mag2 == 0)
356 return 0.0;
357
358 double ratio = dot / (mag1 * mag2);
359 ratio = std::clamp(ratio, -1.0, 1.0); // 防止数值误
360
361 return factor * std::acos(ratio);
362}

364GeometryUtils::SvgArcEndpoints GeometryUtils::calcSvgArc(const SvgArcResult& arc) {
365 SvgArcEndpoints res;
366
367 double phi = degreesToRadians(arc.xRotate);
368 double theta = -degreesToRadians(arc.startAngle);
369 double dTheta = -degreesToRadians(arc.deltaAngle);
370
371 // 计算起点
372 res.x1 = std::cos(phi) * std::cos(theta) * arc.rx - std::sin(phi) * std::sin(theta) * arc.rx + arc.cx;
373 res.y1 = std::sin(phi) * std::cos(theta) * arc.ry - std::cos(phi) * std::sin(theta) * arc.ry + arc.cy;
374
375 // 计算终点
376 theta += dTheta;
377 res.x2 = std::cos(phi) * std::cos(theta) * arc.rx - std::sin(phi) * std::sin(theta) * arc.rx + arc.cx;
378 res.y2 = std::sin(phi) * std::cos(theta) * arc.ry - std::cos(phi) * std::sin(theta) * arc.ry + arc.cy;
379
380 return res;
381}

383QList<GeometryUtils::SvgPoint> GeometryUtils::arcToPath(const SvgArcResult& arc, bool includeStart) {
384 QList<SvgPoint> res;
385
386 const int splitCount = 32; // 与 lckiconverter 保持一致
387 double step = 360.0 / splitCount;
388
389 double startAngle = arc.startAngle;
390 double deltaAngle = arc.deltaAngle;
391
392 if (deltaAngle < 0)
393 step = -step;
394
395 while (std::abs(deltaAngle) > 0.1) {
396 if (std::abs(deltaAngle) < std::abs(step)) {
397 step = deltaAngle;
398 }
399
400 SvgArcResult tempArc = arc;
401 tempArc.startAngle = startAngle;
402 tempArc.deltaAngle = step;
403
404 SvgArcEndpoints pt = calcSvgArc(tempArc);
405
406 if (res.isEmpty() && includeStart) {
407 res.append(SvgPoint(pt.x1, pt.y1));
408 }
409 res.append(SvgPoint(pt.x2, pt.y2));
410
411 deltaAngle -= step;
412 startAngle += step;
413 }
414
415 return res;
416}

631QList<GeometryUtils::SvgPoint> GeometryUtils::bezierToPolyline(double startX,
632 double startY,
633 double cp1X,
634 double cp1Y,
635 double cp2X,
636 double cp2Y,
637 double endX,
638 double endY,
639 int segments) {
640 QList<SvgPoint> res;
641
642 // 三次贝塞尔曲线公
643 // B(t) = (1-t)^3 * P0 + 3*(1-t)^2*t * P1 + 3*(1-t)*t^2 * P2 + t^3 * P3
644
645 for (int i = 0; i <= segments; ++i) {
646 double t = static_cast<double>(i) / segments;
647 double t2 = t * t;
648 double t3 = t2 * t;
649 double mt = 1.0 - t;
650 double mt2 = mt * mt;
651 double mt3 = mt2 * mt;
652
653 double x = mt3 * startX + 3.0 * mt2 * t * cp1X + 3.0 * mt * t2 * cp2X + t3 * endX;
654 double y = mt3 * startY + 3.0 * mt2 * t * cp1Y + 3.0 * mt * t2 * cp2Y + t3 * endY;
655
656 res.append(SvgPoint(x, y));
657 }
658
659 return res;
660}
