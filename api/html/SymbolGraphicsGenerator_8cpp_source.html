79QString SymbolGraphicsGenerator::generatePin(const SymbolPin& pin, const SymbolBBox& bbox) const {
80 QString content;
81
82 // 使用边界框偏移量计算相对坐标
83 double x = pxToMm(pin.settings.posX - bbox.x);
84 double y = -pxToMm(pin.settings.posY - bbox.y); // Y 轴翻转
85
86 // 计算引脚长度（Python版本的做法：直接从路径字符串中提h'后面的数字）
87 QString path = pin.pinPath.path;
88 double length = 0;
89
90 // 查找 'h' 命令并提取其后的数值作为引脚长度
91 int hIndex = path.indexOf('h');
92 if (hIndex >= 0) {
93 QString lengthStr = path.mid(hIndex + 1);
94 // 提取数值部分（可能包含其他命令）
95 QStringList parts = lengthStr.split(QRegularExpression("[^0-9.-]"), Qt::SkipEmptyParts);
96 if (!parts.isEmpty()) {
97 bool ok = false;
98 double tempLength = parts[0].toDouble(&ok);
99 if (ok) {
100 length = tempLength;
101 }
102 }
103 }
104
105 // 转换为毫米单位
106 length = pxToMm(length);
107
108 // 调试：输出引脚信息
109 qDebug() << "Pin Debug - Name:" << pin.name.text << "Number:" << pin.settings.spicePinNumber;
110 qDebug() << " Original posX:" << pin.settings.posX << "bbox.x:" << bbox.x;
111 qDebug() << " Calculated x:" << x << "y:" << y;
112 qDebug() << " Path:" << path << "Extracted length (px):" << length / 0.0254;
113 qDebug() << " Length (mm):" << length << "Rotation:" << pin.settings.rotation;
114
115 // 确保引脚长度为正数（KiCad 引脚长度必须是正数）
116 length = std::abs(length);
117
118 // 确保引脚长度不为 0
119 if (length < 0.01) {
120 length = 2.54; // 默认引脚长度（100mil）
121 }
122
123 // 直接使用原始引脚类型，不进行自动推断
124 PinType pinType = pin.settings.type;
125 QString kicadPinType = pinTypeToKicad(pinType);
126
127 // 动态计算引脚样式（根据 dot 和 clock 的显示状态）
128 PinStyle pinStyle = PinStyle::Line;
129 if (pin.dot.isDisplayed && pin.clock.isDisplayed) {
130 pinStyle = PinStyle::InvertedClock;
131 } else if (pin.dot.isDisplayed) {
132 pinStyle = PinStyle::Inverted;
133 } else if (pin.clock.isDisplayed) {
134 pinStyle = PinStyle::Clock;
135 }
136 QString kicadPinStyle = pinStyleToKicad(pinStyle);
137
138 // 处理引脚名称和编
139 QString pinName = pin.name.text;
140 pinName.replace(" ", "");
141 QString pinNumber = pin.settings.spicePinNumber;
142 pinNumber.replace(" ", "");
143
144 // 如果引脚名称为空，使用引脚编号
145 if (pinName.isEmpty()) {
146 pinName = pinNumber;
147 }
148
149 double orientation = pin.settings.rotation;
150
151 // Python 版本使用 (180 + orientation) % 360 计算方向
152 // 注意：orientation 为 0, 90, 180, 270 的整数
153 double kicadOrientation = (180.0 + orientation);
154 // 使用 fmod 替代 while 循环，避免死循环风险
155 kicadOrientation = fmod(kicadOrientation, 360.0);
156 if (kicadOrientation < 0.0) {
157 kicadOrientation += 360.0;
158 }
159
160 // 规范化为 KiCad 要求的标准角度：0, 90, 180, 270
161 // 找到最接近的标准角度
162 double standardAngles[] = {0.0, 90.0, 180.0, 270.0};
163 double closestAngle = 0.0;
164 double minDiff = 360.0;
165 for (double angle : standardAngles) {
166 double diff = qAbs(kicadOrientation - angle);
167 if (diff < minDiff) {
168 minDiff = diff;
169 closestAngle = angle;
170 }
171 }
172 kicadOrientation = closestAngle;
173
174 content += QString(" (pin %1 %2\n").arg(kicadPinType, kicadPinStyle);
175 content += QString(" (at %1 %2 %3)\n").arg(x, 0, 'f', 2).arg(y, 0, 'f', 2).arg(kicadOrientation, 0, 'f', 0);
176 content += QString(" (length %1)\n").arg(length, 0, 'f', 2);
177 content += QString(" (name \"%1\" (effects (font (size 1.27 1.27) (thickness 0) )))\n").arg(pinName);
178 content += QString(" (number \"%1\" (effects (font (size 1.27 1.27) (thickness 0) )))\n").arg(pinNumber);
179 content += " )\n";
180
181 return content;
182}

224QString SymbolGraphicsGenerator::generateArc(const SymbolArc& arc) const {
225 QString content;
226 double strokeWidth = pxToMm(arc.strokeWidth);
227
228 // KiCad V6 使用三点法定义圆弧：start、mid、end
229 if (arc.path.size() >= 3) {
230 // 提取起点、中点、终点
231 const auto path = arc.path; // 避免临时对象detach
232 QPointF startPoint = path.first();
233 QPointF endPoint = path.last();
234
235 // 计算中点（取中间的点）
236 int midIndex = path.size() / 2;
237 QPointF midPoint = path[midIndex];
238
239 // 转换为相对于边界框的坐标，并转换为毫米
240 double startX = pxToMm(startPoint.x() - m_currentBBox.x);
241 double startY = -pxToMm(startPoint.y() - m_currentBBox.y); // Y 轴翻转
242 double midX = pxToMm(midPoint.x() - m_currentBBox.x);
243 double midY = -pxToMm(midPoint.y() - m_currentBBox.y); // Y 轴翻转
244 double endX = pxToMm(endPoint.x() - m_currentBBox.x);
245 double endY = -pxToMm(endPoint.y() - m_currentBBox.y); // Y 轴翻转
246
247 content += " (arc\n";
248 content += QString(" (start %1 %2)\n").arg(startX, 0, 'f', 2).arg(startY, 0, 'f', 2);
249 content += QString(" (mid %1 %2)\n").arg(midX, 0, 'f', 2).arg(midY, 0, 'f', 2);
250 content += QString(" (end %1 %2)\n").arg(endX, 0, 'f', 2).arg(endY, 0, 'f', 2);
251 content += QString(" (stroke (width %1) (type default))\n").arg(strokeWidth, 0, 'f', 3);
252
253 // 根据fillColor设置填充类型（与Python版本一致）
254 if (arc.fillColor) {
255 content += " (fill (type background))\n";
256 } else {
257 content += " (fill (type none))\n";
258 }
259
260 content += " )\n";
261 } else if (arc.path.size() == 2) {
262 // 只有两个点，计算中点
263 const auto path = arc.path; // 避免临时对象detach
264 QPointF startPoint = path.first();
265 QPointF endPoint = path.last();
266 QPointF midPoint = (startPoint + endPoint) / 2;
267
268 // 转换为相对于边界框的坐标，并转换为毫米
269 double startX = pxToMm(startPoint.x() - m_currentBBox.x);
270 double startY = -pxToMm(startPoint.y() - m_currentBBox.y); // Y 轴翻转
271 double midX = pxToMm(midPoint.x() - m_currentBBox.x);
272 double midY = -pxToMm(midPoint.y() - m_currentBBox.y); // Y 轴翻转
273 double endX = pxToMm(endPoint.x() - m_currentBBox.x);
274 double endY = -pxToMm(endPoint.y() - m_currentBBox.y); // Y 轴翻转
275
276 content += " (arc\n";
277 content += QString(" (start %1 %2)\n").arg(startX, 0, 'f', 2).arg(startY, 0, 'f', 2);
278 content += QString(" (mid %1 %2)\n").arg(midX, 0, 'f', 2).arg(midY, 0, 'f', 2);
279 content += QString(" (end %1 %2)\n").arg(endX, 0, 'f', 2).arg(endY, 0, 'f', 2);
280 content += QString(" (stroke (width %1) (type default))\n").arg(strokeWidth, 0, 'f', 3);
281
282 // 根据fillColor设置填充类型
283 if (arc.fillColor) {
284 content += " (fill (type background))\n";
285 } else {
286 content += " (fill (type none))\n";
287 }
288
289 content += " )\n";
290 } else {
291 // 点数不足，跳过或生成一个默认的arc
292 qDebug() << "Warning: Arc has insufficient points (" << arc.path.size() << "), skipping";
293 }
294
295 return content;

298QString SymbolGraphicsGenerator::generateEllipse(const SymbolEllipse& ellipse) const {
299 QString content;
300
301 // V6 使用毫米单位
302 double cx = pxToMm(ellipse.centerX - m_currentBBox.x);
303 double cy = -pxToMm(ellipse.centerY - m_currentBBox.y); // Y 轴翻转
304 double radiusX = pxToMm(ellipse.radiusX);
305 double radiusY = pxToMm(ellipse.radiusY);
306 double strokeWidth = pxToMm(ellipse.strokeWidth);
307
308 // 如果是圆形（radiusX 等于 radiusY），使用 circle 元素
309 if (qAbs(radiusX - radiusY) < 0.01) {
310 content += " (circle\n";
311 content += QString(" (center %1 %2)\n").arg(cx, 0, 'f', 2).arg(cy, 0, 'f', 2);
312 content += QString(" (radius %1)\n").arg(radiusX, 0, 'f', 2);
313 content += QString(" (stroke (width %1) (type default))\n").arg(strokeWidth, 0, 'f', 3);
314 content += " (fill (type none))\n";
315 content += " )\n";
316 } else {
317 // 椭圆：转换为路径
318 // 使用 32 段折线近似椭圆
319 content += " (polyline\n";
320 content += " (pts";
321
322 const int segments = 32;
323 for (int i = 0; i <= segments; ++i) {
324 double angle = 2.0 * M_PI * i / segments;
325 double x = cx + radiusX * std::cos(angle);
326 double y = cy + radiusY * std::sin(angle);
327 content += QString(" (xy %1 %2)").arg(x, 0, 'f', 2).arg(y, 0, 'f', 2);
328 }
329
330 content += ")\n";
331 content += QString(" (stroke (width %1) (type default))\n").arg(strokeWidth, 0, 'f', 3);
332
333 // 根据 fillColor 属性设置填充类型
334 if (ellipse.fillColor) {
335 content += " (fill (type background))\n";
336 } else {
337 content += " (fill (type none))\n";
338 }
339 content += " )\n";
340 }
341
342 return content;
343}

345QString SymbolGraphicsGenerator::generatePolygon(const SymbolPolygon& polygon) const {
346 QString content;
347 double strokeWidth = pxToMm(polygon.strokeWidth);
348
349 // 解析点数据
350 QStringList points = polygon.points.split(" ");
351 // 过滤掉空字符串
352 points.removeAll("");
353
354 // 至少需要2 个有效的点（4 个坐标值）
355 if (points.size() >= 4) {
356 // KiCad V6 不支持 polygon 元素，使用 polyline 代替
357 content += " (polyline\n";
358 content += " (pts";
359 // 存储第一个点以便在最后重
360 QString firstPoint;
361 QString lastPoint; // 用于检测重复点
362 for (int i = 0; i < points.size(); i += 2) {
363 if (i + 1 < points.size()) {
364 // 转换为相对于边界框的坐标，并转换为毫米
365 double x = pxToMm(points[i].toDouble() - m_currentBBox.x);
366 double y = -pxToMm(points[i + 1].toDouble() - m_currentBBox.y);
367 QString point = QString(" (xy %1 %2)").arg(x, 0, 'f', 2).arg(y, 0, 'f', 2);
368
369 // 避免重复点
370 if (point != lastPoint) {
371 content += point;
372 if (i == 0) {
373 firstPoint = point;
374 }
375 lastPoint = point;
376 }
377 }
378 }
379 // 多边形总是重复第一个点以闭合
380 if (!firstPoint.isEmpty() && firstPoint != lastPoint) {
381 content += firstPoint;
382 }
383 content += ")\n";
384 content += QString(" (stroke (width %1) (type default))\n").arg(strokeWidth, 0, 'f', 3);
385 // 根据 fillColor 属性设置填充类型
386 if (polygon.fillColor) {
387 content += " (fill (type background))\n";
388 } else {
389 content += " (fill (type none))\n";
390 }
391 content += " )\n";
392 }
393
394 return content;
395}

496QString SymbolGraphicsGenerator::generateText(const SymbolText& text) const {
497 QString content;
498
499 // V6 使用毫米单位
500 double x = pxToMm(text.posX - m_currentBBox.x);
501 double y = -pxToMm(text.posY - m_currentBBox.y); // Y 轴翻转
502
503 // 计算字体大小（从pt转换为mm）
504 double fontSize = text.textSize * 0.352778; // 1pt = 0.352778mm
505
506 // 处理粗体和斜
507 QString fontStyle = "";
508 if (text.bold) {
509 fontStyle += "bold ";
510 }
511 if (text.italic == "1" || text.italic == "Italic" || text.italic == "italic") {
512 fontStyle += "italic ";
513 }
514 if (fontStyle.isEmpty()) {
515 fontStyle = "normal";
516 } else {
517 fontStyle = fontStyle.trimmed();
518 }
519
520 // 处理旋转角度
521 double rotation = text.rotation;
522 if (rotation != 0) {
523 rotation = 360 - rotation;
524 }
525
526 // 处理可见
527 QString hide = text.visible ? "" : "hide";
528
529 // 转义文本内容
530 QString textContent = text.text;
531 textContent.replace(" ", "");
532 if (textContent.isEmpty()) {
533 textContent = "~";
534 }
535
536 // 生成文本元素
537 content += " (text\n";
538 content += QString(" \"%1\"\n").arg(textContent);
539 content += QString(" (at %1 %2 %3)\n").arg(x, 0, 'f', 2).arg(y, 0, 'f', 2).arg(rotation, 0, 'f', 0);
540 content += QString(" (effects (font (size %1 %2) (thickness 0.1) %3) %4)\n")
541 .arg(fontSize, 0, 'f', 2)
542 .arg(fontSize, 0, 'f', 2)
543 .arg(fontStyle)
544 .arg(hide);
545 content += " )\n";
546
547 return content;

615SymbolBBox SymbolGraphicsGenerator::calculatePartBBox(const SymbolPart& part) const {
616 SymbolBBox bbox;
617 bbox.x = 0.0;
618 bbox.y = 0.0;
619 bbox.width = 0.0;
620 bbox.height = 0.0;
621
622 // 如果子部分有坐标原点，使用它作为基准
623 if (part.originX != 0.0 || part.originY != 0.0) {
624 bbox.x = part.originX;
625 bbox.y = part.originY;
626 }
627
628 // 计算图形元素的边界
629 double minX = bbox.x;
630 double minY = bbox.y;
631 double maxX = bbox.x;
632 double maxY = bbox.y;
633
634 // 遍历所有图形元素，计算边界
635 auto updateBounds = [&](double x, double y, double w, double h) {
636 minX = qMin(minX, x);
637 minY = qMin(minY, y);
638 maxX = qMax(maxX, x + w);
639 maxY = qMax(maxY, y + h);
640 };
641
642 // 处理矩形
643 for (const auto& rect : part.rectangles) {
644 updateBounds(rect.posX, rect.posY, rect.width, rect.height);
645 }
646
647 // 处理圆
648 for (const auto& circle : part.circles) {
649 double radius = circle.radius;
650 updateBounds(circle.centerX - radius, circle.centerY - radius, radius * 2, radius * 2);
651 }
652
653 // 处理椭圆
654 for (const auto& ellipse : part.ellipses) {
655 updateBounds(ellipse.centerX - ellipse.radiusX,
656 ellipse.centerY - ellipse.radiusY,
657 ellipse.radiusX * 2,
658 ellipse.radiusY * 2);
659 }
660
661 // 处理圆弧（使用路径点）
662 for (const auto& arc : part.arcs) {
663 for (const auto& point : arc.path) {
664 minX = qMin(minX, point.x());
665 minY = qMin(minY, point.y());
666 maxX = qMax(maxX, point.x());
667 maxY = qMax(maxY, point.y());
668 }
669 }
670
671 // 处理引脚
672 for (const auto& pin : part.pins) {
673 minX = qMin(minX, pin.settings.posX);
674 minY = qMin(minY, pin.settings.posY);
675 maxX = qMax(maxX, pin.settings.posX);
676 maxY = qMax(maxY, pin.settings.posY);
677 }
678
679 // 处理文本
680 for (const auto& text : part.texts) {
681 minX = qMin(minX, text.posX);
682 minY = qMin(minY, text.posY);
683 maxX = qMax(maxX, text.posX);
684 maxY = qMax(maxY, text.posY);
685 }
686
687 // 更新边界框
688 bbox.x = minX;
689 bbox.y = minY;
690 bbox.width = maxX - minX;
691 bbox.height = maxY - minY;
692
693 qDebug() << "Part BBox - x:" << bbox.x << "y:" << bbox.y << "width:" << bbox.width << "height:" << bbox.height;
694
695 return bbox;
696}
