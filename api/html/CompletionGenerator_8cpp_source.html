5QString CompletionGenerator::generate(Shell shell) {

18CompletionGenerator::Shell CompletionGenerator::parseShell(const QString& shellStr) {
19 QString lower = shellStr.toLower();
20 if (lower == "zsh") {
21 return Shell::Zsh;
22 } else if (lower == "fish") {
23 return Shell::Fish;
24 }
25 return Shell::Bash;

28QString CompletionGenerator::getInstallInstructions(Shell shell) {
29 switch (shell) {
30 case Shell::Bash:
31 return QString(
32 "安装方法 (Bash):\n"
33 " # 临时启用 (当前会话)\n"
34 " eval \"$(easykiconverter --completion bash)\"\n"
35 "\n"
36 " # 永久启用 (添加到 ~/.bashrc)\n"
37 " easykiconverter --completion bash >> ~/.bash_completion\n"
38 " echo 'source ~/.bash_completion' >> ~/.bashrc\n");
39 case Shell::Zsh:
40 return QString(
41 "安装方法 (Zsh):\n"
42 " # 临时启用 (当前会话)\n"
43 " eval \"$(easykiconverter --completion zsh)\"\n"
44 "\n"
45 " # 永久启用 (添加到 ~/.zshrc)\n"
46 " easykiconverter --completion zsh > ~/.zsh/_easykiconverter\n"
47 " echo 'fpath=(~/.zsh $fpath)' >> ~/.zshrc\n"
48 " echo 'autoload -Uz compinit && compinit' >> ~/.zshrc\n");
49 case Shell::Fish:
50 return QString(
51 "安装方法 (Fish):\n"
52 " # 永久启用\n"
53 " easykiconverter --completion fish > ~/.config/fish/completions/easykiconverter.fish\n");
54 default:
55 return "";
56 }

59QString CompletionGenerator::generateBash() {
60 return QString(
61 "# EasyKiConverter bash completion script\n"
62 "# Generated by easykiconverter --completion bash\n"
63 "\n"
64 "_easykiconverter_complete() {\n"
65 " local cur prev words cword\n"
66 " _init_completion || return\n"
67 "\n"
68 " # 子命令列表\n"
69 " local subcommands=\"convert\"\n"
70 " local convert_subcommands=\"bom component batch\"\n"
71 "\n"
72 " # 所有可用参数\n"
73 " local global_opts=\"--help -h --version -v --debug -d --log-level --log-file --config --language --theme "
74 "--portable --sync-logging --cache-dir --cache-size-mb\"\n"
75 " local convert_opts=\"--input -i --output -o --lib-name --component -c --symbol --footprint --3d-model "
76 "--3d-model-format --preview --datasheet --progress --quiet -q --weak-network --update-mode "
77 "--3d-path-mode --no-overwrite --symbol-description --footprint-description\"\n"
78 "\n"
79 " # 动态补全 LCSC ID (从缓存获取)\n"
80 " _easykiconverter_complete_lcsc_ids() {\n"
81 " if command -v easykiconverter &> /dev/null; then\n"
82 " easykiconverter --complete lcsc-id 2>/dev/null\n"
83 " fi\n"
84 " }\n"
85 "\n"
86 " # 动态补全文件\n"
87 " _easykiconverter_complete_files() {\n"
88 " local ext=\"$1\"\n"
89 " if [[ -n \"$ext\" ]]; then\n"
90 " compgen -f -X '!*.\"$ext\"' -- \"$cur\"\n"
91 " else\n"
92 " compgen -f -- \"$cur\"\n"
93 " fi\n"
94 " }\n"
95 "\n"
96 " # 动态补全目录\n"
97 " _easykiconverter_complete_dirs() {\n"
98 " compgen -d -- \"$cur\"\n"
99 " }\n"
100 "\n"
101 " # 分析当前命令位置\n"
102 " local has_convert=false\n"
103 " local has_subcommand=false\n"
104 " local convert_type=\"\"\n"
105 "\n"
106 " for ((i=1; i<cword; i++)); do\n"
107 " case \"${words[i]}\" in\n"
108 " convert)\n"
109 " has_convert=true\n"
110 " ;;\n"
111 " bom|component|batch)\n"
112 " if $has_convert; then\n"
113 " has_subcommand=true\n"
114 " convert_type=\"${words[i]}\"\n"
115 " fi\n"
116 " ;;\n"
117 " esac\n"
118 " done\n"
119 "\n"
120 " # 补全逻辑\n"
121 " if [[ \"$prev\" == \"easykiconverter\" ]] || [[ \"$prev\" == \"--completion\" ]]; then\n"
122 " # 第一个参数: 子命令或全局选项\n"
123 " if [[ \"$cur\" == -* ]]; then\n"
124 " COMPREPLY=($(compgen -W \"$global_opts\" -- \"$cur\"))\n"
125 " else\n"
126 " COMPREPLY=($(compgen -W \"$subcommands\" -- \"$cur\"))\n"
127 " fi\n"
128 " elif ! $has_convert; then\n"
129 " # 非 convert 子命令后的参数\n"
130 " COMPREPLY=($(compgen -W \"$global_opts\" -- \"$cur\"))\n"
131 " elif ! $has_subcommand; then\n"
132 " # convert 后的子命令\n"
133 " if [[ \"$cur\" == -* ]]; then\n"
134 " COMPREPLY=($(compgen -W \"$convert_opts $global_opts\" -- \"$cur\"))\n"
135 " else\n"
136 " COMPREPLY=($(compgen -W \"$convert_subcommands\" -- \"$cur\"))\n"
137 " fi\n"
138 " else\n"
139 " # 子命令后的参数\n"
140 " case \"$prev\" in\n"
141 " -i|--input)\n"
142 " # 补全输入文件\n"
143 " COMPREPLY=($(compgen -f -X '!*.{xlsx,xls,csv,txt}' -- \"$cur\"))\n"
144 " ;;\n"
145 " -o|--output)\n"
146 " # 补全输出目录\n"
147 " COMPREPLY=($(compgen -d -- \"$cur\"))\n"
148 " ;;\n"
149 " -c|--component)\n"
150 " # 动态补全 LCSC ID\n"
151 " COMPREPLY=($(compgen -W \"$(_easykiconverter_complete_lcsc_ids)\" -- \"$cur\"))\n"
152 " ;;\n"
153 " --log-level)\n"
154 " COMPREPLY=($(compgen -W \"trace debug info warn error fatal\" -- \"$cur\"))\n"
155 " ;;\n"
156 " --language)\n"
157 " COMPREPLY=($(compgen -W \"zh_CN en\" -- \"$cur\"))\n"
158 " ;;\n"
159 " --theme)\n"
160 " COMPREPLY=($(compgen -W \"dark light\" -- \"$cur\"))\n"
161 " ;;\n"
162 " --3d-path-mode)\n"
163 " COMPREPLY=($(compgen -W \"relative absolute\" -- \"$cur\"))\n"
164 " ;;\n"
165 " --3d-model-format)\n"
166 " COMPREPLY=($(compgen -W \"wrl step both\" -- \"$cur\"))\n"
167 " ;;\n"
168 " --symbol|--footprint|--3d-model)\n"
169 " # 这些是 flag 选项，不需要补全值\n"
170 " ;;\n"
171 " *)\n"
172 " if [[ \"$cur\" == -* ]]; then\n"
173 " COMPREPLY=($(compgen -W \"$convert_opts\" -- \"$cur\"))\n"
174 " else\n"
175 " # 根据子命令类型补全\n"
176 " case \"$convert_type\" in\n"
177 " bom|batch)\n"
178 " COMPREPLY=($(compgen -f -X '!*.{xlsx,xls,csv,txt}' -- \"$cur\"))\n"
179 " ;;\n"
180 " component)\n"
181 " COMPREPLY=($(compgen -W \"$(_easykiconverter_complete_lcsc_ids)\" -- \"$cur\"))\n"
182 " ;;\n"
183 " esac\n"
184 " fi\n"
185 " ;;\n"
186 " esac\n"
187 " fi\n"
188 "\n"
189 " return 0\n"
190 "}\n"
191 "\n"
192 "complete -F _easykiconverter_complete easykiconverter\n");
193}

209 " )\n"

211 " convert_commands=(\n"
212 " 'bom:转换 BOM 表'\n"

214 " 'batch:批量转换'\n"
215 " )\n"
216 "\n"
217 " global_opts=(\n"
218 " '(-h --help)'{-h,--help}'[显示帮助信息]'\n"
219 " '(-v --version)'{-v,--version}'[显示版本信息]'\n"
220 " '(-d --debug)'{-d,--debug}'[调试模式]'\n"
221 " '--log-level[日志级别]:level:(trace debug info warn error fatal)'\n"
222 " '--log-file[日志文件路径]:file:_files'\n"
223 " '--config[配置文件路径]:file:_files'\n"
224 " '--language[界面语言]:lang:(zh_CN en)'\n"
225 " '--theme[界面主题]:theme:(dark light)'\n"
226 " '--portable[便携模式]'\n"
227 " '--sync-logging[同步日志输出]'\n"
228 " '--cache-dir[磁盘缓存目录]:dir:_directories'\n"
229 " '--cache-size-mb[磁盘缓存大小限制 MB]:mb:'\n"
230 " '--completion[生成补全脚本]:shell:(bash zsh fish)'\n"
231 " )\n"
232 "\n"
233 " convert_opts=(\n"
234 " '(-i --input)'{-i,--input}'[输入文件路径]:file:_files -g \"*.{xlsx,xls,csv,txt}\"'\n"
235 " '(-o --output)'{-o,--output}'[输出目录路径]:dir:_directories'\n"
236 " '--lib-name[导出库名称]:name:'\n"
237 " '(-c --component)'{-c,--component}'[LCSC 元器件编号]:id:_easykiconverter_lcsc_ids'\n"
238 " '--symbol[导出符号库]'\n"
239 " '--footprint[导出封装库]'\n"
240 " '--3d-model[导出 3D 模型]'\n"
241 " '--3d-model-format[3D 模型格式]:format:(wrl step both)'\n"
242 " '--preview[导出预览图]'\n"
243 " '--datasheet[导出数据手册]'\n"
244 " '--progress[显示进度条]'\n"
245 " '(-q --quiet)'{-q,--quiet}'[安静模式]'\n"
246 " '--weak-network[弱网模式]'\n"
247 " '--update-mode[更新模式]'\n"
248 " '--3d-path-mode[3D 模型路径模式]:mode:(relative absolute)'\n"
249 " '--no-overwrite[不覆盖已存在的文件]'\n"
250 " '--symbol-description[符号库描述]:text:'\n"
251 " '--footprint-description[封装库描述]:text:'\n"
252 " )\n"
253 "\n"
254 " # 动态补全 LCSC ID\n"
255 " _easykiconverter_lcsc_ids() {\n"
256 " local -a ids\n"
257 " ids=(${(f)\"$(easykiconverter --complete lcsc-id 2>/dev/null)\"})\n"
258 " _describe -t lcsc-ids 'LCSC IDs' ids\n"
259 " }\n"
260 "\n"
261 " _arguments -C \\\n"
262 " $global_opts \\\n"
263 " '1:command:->command' \\\n"
264 " '*::arg:->args'\n"
265 "\n"
266 " case $state in\n"
267 " command)\n"
268 " _describe -t commands 'easykiconverter commands' commands\n"
269 " ;;\n"
270 " args)\n"
271 " case ${words[1]} in\n"
272 " convert)\n"
273 " if (( CURRENT == 2 )); then\n"
274 " _describe -t convert-commands 'convert commands' convert_commands\n"
275 " else\n"
276 " case ${words[2]} in\n"
277 " bom|batch)\n"
278 " _arguments \\\n"
279 " $convert_opts \\\n"
280 " '*:file:_files -g \"*.{xlsx,xls,csv,txt}\"'\n"
281 " ;;\n"
282 " component)\n"
283 " _arguments \\\n"
284 " $convert_opts \\\n"
285 " '*:lcsc-id:_easykiconverter_lcsc_ids'\n"
286 " ;;\n"
287 " esac\n"
288 " fi\n"
289 " ;;\n"
290 " *)\n"
291 " _arguments $global_opts\n"
292 " ;;\n"
293 " esac\n"
294 " ;;\n"
295 " esac\n"
296 "}\n"
297 "\n"
298 "_easykiconverter \"$@\"\n");
299}
