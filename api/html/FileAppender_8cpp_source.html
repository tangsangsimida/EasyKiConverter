59bool FileAppender::validateAndNormalizePath(const QString& filePath) {
60 if (filePath.isEmpty()) {
61 return false;
62 }
63
64 // 检查路径中的危险字符和模式
65 QString cleanPath = filePath;
66
67 // 检查路径遍历攻击模式
68 if (cleanPath.contains("..") || cleanPath.contains("~")) {
69 return false;
70 }
71
72 // 检查绝对路径是否指向系统敏感目录
73 QFileInfo fileInfo(cleanPath);
74 QString absolutePath = fileInfo.absoluteFilePath();
75
76 // 规范化路径（解析符号链接等）
77 QDir dir = fileInfo.absoluteDir();
78 m_baseDir = dir.absolutePath();
79
80 // 获取规范的绝对路径
81 QString canonicalDir = dir.canonicalPath();
82 if (!canonicalDir.isEmpty()) {
83 m_baseDir = canonicalDir;
84 }
85
86 // 检查路径是否在允许的目录范围内
87 if (!isPathInAllowedDir(absolutePath)) {
88 return false;
89 }
90
91 // 存储规范化的路径
92 m_filePath = absolutePath;
93 m_canonicalPath = absolutePath;
94
95 return true;
96}

98bool FileAppender::isPathInAllowedDir(const QString& path) const {
99 // 允许的根目录列表
100 QStringList allowedRoots;
101
102 // 应用程序目录
103 if (QCoreApplication::instance()) {
104 allowedRoots << QCoreApplication::applicationDirPath();
105 }
106
107 // 用户数据目录
108 allowedRoots << QStandardPaths::writableLocation(QStandardPaths::AppDataLocation);
109 allowedRoots << QStandardPaths::writableLocation(QStandardPaths::AppLocalDataLocation);
110 allowedRoots << QStandardPaths::writableLocation(QStandardPaths::TempLocation);
111
112 // 用户主目录
113 allowedRoots << QDir::homePath();
114
115 // 当前工作目录
116 allowedRoots << QDir::currentPath();
117
118 // 检查路径是否在允许的根目录下
119 for (const QString& root : allowedRoots) {
120 if (path.startsWith(root)) {
121 return true;
122 }
123 }
124
125 return false;

154void FileAppender::flush() {
155 if (!m_pathValid) {
156 return;
157 }
158
159 if (m_async) {
160 // 请求刷新
161 m_flushRequested.storeRelaxed(1);
162 m_queueCondition.wakeAll();
163
164 // 等待队列清空（带超时保护）
165 int timeout = 5000; // 5秒超时
166 int waited = 0;
167 while (waited < timeout) {
168 QMutexLocker locker(&m_queueMutex);
169 if (m_writeQueue.isEmpty()) {
170 break;
171 }
172 locker.unlock();
173 QThread::msleep(10);
174 waited += 10;
175 }
176 }
177
178 // 刷新文件缓冲
179 QMutexLocker fileLocker(&m_fileMutex);
180 if (m_file.isOpen()) {
181 m_file.flush();
182 }
183}

185void FileAppender::close() {
186 // 停止异步线程
187 if (m_async && m_writerThread) {
188 m_running.storeRelaxed(0);
189 m_queueCondition.wakeAll();
190
191 // 等待线程结束（带超时保护）
192 if (!m_writerThread->wait(3000)) {
193 m_writerThread->terminate();
194 m_writerThread->wait();
195 }
196 delete m_writerThread;
197 m_writerThread = nullptr;
198 }
199
200 // 关闭文件
201 QMutexLocker locker(&m_fileMutex);
202 if (m_file.isOpen()) {
203 m_file.flush();
204 m_file.close();
205 }

226void FileAppender::rollOver() {
227 // 使用递归锁，允许在 writeDirect 中调用
228 QMutexLocker locker(&m_fileMutex);
229
230 if (m_file.isOpen()) {
231 m_file.flush();
232 m_file.close();
233 }
234
235 // 清理旧文件，为新文件腾出空间
236 cleanupOldFiles();
237
238 // 重命名现有文件
239 QString rolledName = generateRolledFileName(1);
240
241 // 确保目标文件不存在
242 if (QFile::exists(rolledName)) {
243 QFile::remove(rolledName);
244 }
245
246 if (!QFile::rename(m_filePath, rolledName)) {
247 // 重命名失败，尝试直接删除并重新创建
248 QFile::remove(m_filePath);
249 }
250
251 // 重新打开新文件
252 m_file.setFileName(m_filePath);
253 if (!m_file.open(QIODevice::WriteOnly | QIODevice::Append | QIODevice::Text)) {
254 m_pathValid = false;
255 m_lastError = "Failed to reopen log file after rollover";
256 }

259void FileAppender::cleanupOldFiles() {
260 // 注意：此函数在 rollOver 中调用，已持有 m_fileMutex
261 QDir dir = QFileInfo(m_filePath).absoluteDir();
262 QString baseName = QFileInfo(m_filePath).baseName();
263 QString suffix = QFileInfo(m_filePath).completeSuffix();
264 QString pattern = baseName + "_*";
265 if (!suffix.isEmpty()) {
266 pattern += "." + suffix;
267 }
268
269 QStringList filters;
270 filters << pattern;
271 QStringList files = dir.entryList(filters, QDir::Files, QDir::Name);
272
273 // 删除超出数量限制的旧文件
274 while (files.size() >= m_maxFiles - 1) {
275 QString toRemove = dir.absoluteFilePath(files.takeFirst());
276 QFile::remove(toRemove);
277 }
278}

296void FileAppender::writerThreadFunc() {
297 while (m_running.loadRelaxed()) {
298 QString message;
299 bool overflowed = false;
300
301 {
302 QMutexLocker locker(&m_queueMutex);
303
304 // 等待新消息或刷新请求
305 while (m_writeQueue.isEmpty() && m_running.loadRelaxed() && !m_flushRequested.loadRelaxed()) {
306 m_queueCondition.wait(&m_queueMutex, 100);
307 }
308
309 if (!m_writeQueue.isEmpty()) {
310 message = m_writeQueue.dequeue();
311 }
312
313 // 检查是否有溢出发生过
314 if (m_queueOverflow.loadRelaxed()) {
315 overflowed = true;
316 m_queueOverflow.storeRelaxed(0);
317 }
318 }
319
320 // 如果发生过溢出，添加警告消息
321 if (overflowed) {
322 writeDirect("[WARNING] Log queue overflow occurred, some messages were dropped");
323 }
324
325 if (!message.isEmpty()) {
326 writeDirect(message);
327 }
328
329 // 处理刷新请求
330 if (m_flushRequested.loadRelaxed()) {
331 QMutexLocker fileLocker(&m_fileMutex);
332 if (m_file.isOpen()) {
333 m_file.flush();
334 }
335 m_flushRequested.storeRelaxed(0);
336 }
337 }
338
339 // 写入剩余消息
340 while (true) {
341 QString message;
342 {
343 QMutexLocker locker(&m_queueMutex);
344 if (m_writeQueue.isEmpty()) {
345 break;
346 }
347 message = m_writeQueue.dequeue();
348 }
349
350 if (!message.isEmpty()) {
351 writeDirect(message);
352 }
353 }
354}

356void FileAppender::writeDirect(const QString& formatted) {
357 QMutexLocker locker(&m_fileMutex);
358
359 if (!m_pathValid) {
360 return;
361 }
362
363 if (!m_file.isOpen()) {
364 openFile();
365 }
366
367 if (m_file.isOpen()) {
368 m_file.write(formatted.toUtf8());
369 m_file.write("\n");
370
371 // 检查是否需要滚动（避免在锁内调用 rollOver，因为它也使用同一个递归锁）
372 if (needsRollover()) {
373 locker.unlock(); // 释放锁
374 rollOver(); // rollOver 会重新获取锁
375 }
376 }
