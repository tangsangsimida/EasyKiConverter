34struct PipelineProgress {
35 int fetchCompleted = 0; // 抓取完成数
36 int processCompleted = 0; // 处理完成数
37 int writeCompleted = 0; // 写入完成数
38 int totalTasks = 0; // 总任务数
39
40 // 计算各阶段进度（0-100）
41 int fetchProgress() const {
42 return totalTasks > 0 ? (fetchCompleted * 100 / totalTasks) : 0;
43 }
44
45 int processProgress() const {
46 return totalTasks > 0 ? (processCompleted * 100 / totalTasks) : 0;
47 }
48
49 int writeProgress() const {
50 return totalTasks > 0 ? (writeCompleted * 100 / totalTasks) : 0;
51 }
52
53 // 计算加权总进度（抓取30%，处理50%，写入20%）
54 int overallProgress() const {
55 return (fetchProgress() * 30 + processProgress() * 50 + writeProgress() * 20) / 100;
56 }
57};
