116void Exporter3DModel::convertToKiCadCoordinates(Model3DData& modelData) {
117 // EasyEDA 使用右手坐标系，KiCad 使用左手坐标
118 // 需要进行坐标转
119
120 // 获取当前的平移和旋转
121 Model3DBase translation = modelData.translation();
122 Model3DBase rotation = modelData.rotation();
123
124 // 旋转 X 轴 180 度
125 rotation.x += 180.0;
126

142QString Exporter3DModel::generateWrlContent(const Model3DData& modelData, const QByteArray& objData) {
143 QString content;
144
145 // 解析 OBJ 数据
146 QJsonObject parsedData = parseObjData(objData);
147 QJsonArray vertices = parsedData["vertices"].toArray();
148 QJsonArray faces = parsedData["faces"].toArray();
149 QJsonObject materials = parsedData["materials"].toObject();
150 QJsonArray shapes = parsedData["shapes"].toArray();
151
152 // qDebug() << "Generating WRL content - vertices:" << vertices.size()
153 // << "faces:" << faces.size()
154 // << "materials:" << materials.size()
155 // << "shapes:" << shapes.size();
156
157 // WRL 文件头部
158 content += "#VRML V2.0 utf8\n";
159 content += "# 3D model generated by EasyKiConverter (https://github.com/tangsangsimida/EasyKiConverter)\n";
160 content += "\n";
161
162 // 如果有形状数据，使用形状数据生成 WRL
163 if (!shapes.isEmpty()) {
164 for (const QJsonValue& shapeValue : shapes) {
165 QJsonObject shape = shapeValue.toObject();
166 QString materialId = shape["materialId"].toString();
167 QJsonArray shapePoints = shape["points"].toArray();
168 QJsonArray coordIndex = shape["coordIndex"].toArray();
169
170 // qDebug() << "Processing shape - materialId:" << materialId
171 // << "points:" << shapePoints.size()
172 // << "coordIndex:" << coordIndex.size();
173
174 // 关键步骤：在倒数第二个位置插入最后一个点的副本（与 Python 版本保持一致）
175 if (shapePoints.size() > 0) {
176 shapePoints.insert(shapePoints.size() - 1, shapePoints.last());
177 // qDebug() << "Inserted duplicate of last point, new size:" << shapePoints.size();
178 }
179
180 // 获取材质信息
181 QJsonObject material = materials[materialId].toObject();
182 QJsonArray diffuseColor = material["diffuseColor"].toArray();
183 QJsonArray specularColor = material["specularColor"].toArray();
184 QString transparency = material["transparency"].toString("0");
185
186 // 生成 Shape
187 content += "Shape {\n";
188 content += " appearance Appearance {\n";
189 content += " material Material {\n";
190 content += QString(" diffuseColor %1 %2 %3\n")
191 .arg(diffuseColor.isEmpty() ? "0.8" : diffuseColor[0].toString())
192 .arg(diffuseColor.isEmpty() ? "0.8" : diffuseColor[1].toString())
193 .arg(diffuseColor.isEmpty() ? "0.8" : diffuseColor[2].toString());
194 content += QString(" specularColor %1 %2 %3\n")
195 .arg(specularColor.isEmpty() ? "0.2" : specularColor[0].toString())
196 .arg(specularColor.isEmpty() ? "0.2" : specularColor[1].toString())
197 .arg(specularColor.isEmpty() ? "0.2" : specularColor[2].toString());
198 content += " ambientIntensity 0.2\n";
199 content += QString(" transparency %1\n").arg(transparency);
200 content += " shininess 0.5\n";
201 content += " }\n";
202 content += " }\n";
203 content += " geometry IndexedFaceSet {\n";
204 content += " ccw TRUE\n";
205 content += " solid FALSE\n";
206 content += " coord DEF co Coordinate {\n";
207 content += " point [\n";
208
209 // 添加顶点数据
210 for (const QJsonValue& pointValue : shapePoints) {
211 // pointValue 现在是字符串格式，直接输
212 content += " " + pointValue.toString() + ",\n";
213 }
214
215 content += " ]\n";
216 content += " }\n";
217 content += " coordIndex [\n";
218
219 // 添加面索引数
220 for (const QJsonValue& indexValue : coordIndex) {
221 // indexValue 是一QJsonArray，包含面索引
222 QJsonArray faceIndices = indexValue.toArray();
223 QString indexStr;
224 for (const QJsonValue& idx : faceIndices) {
225 indexStr += QString::number(idx.toInt()) + " ";
226 }
227 content += " " + indexStr + "\n";
228 }
229
230 content += " ]\n";
231 content += " }\n";
232 content += "}\n";
233 }
234 } else {
235 // qDebug() << "No shapes found, using simple Transform mode";
236 // 如果没有形状数据，使用简单的 Transform 包装
237 // 获取平移和旋
238 Model3DBase translation = modelData.translation();
239 Model3DBase rotation = modelData.rotation();
240
241 content += "Transform {\n";
242 content += " translation ";
243 content += QString("%1 %2 %3\n").arg(translation.x).arg(translation.y).arg(translation.z);
244 content += " rotation ";
245 content += QString("%1 %2 %3 %4\n").arg(rotation.x).arg(rotation.y).arg(rotation.z).arg(1.0);
246 content += " scale 1 1 1\n";
247 content += " children [\n";
248 content += " Shape {\n";
249 content += " appearance Appearance {\n";
250 content += " material Material {\n";
251 content += " diffuseColor 0.8 0.8 0.8\n";
252 content += " specularColor 0.2 0.2 0.2\n";
253 content += " ambientIntensity 0.2\n";
254 content += " shininess 0.2\n";
255 content += " }\n";
256 content += " }\n";
257 content += " geometry IndexedFaceSet {\n";
258 content += " coord Coordinate {\n";
259 content += " point [\n";
260
261 // 添加顶点数据
262 for (const QJsonValue& vertexValue : vertices) {
263 QJsonArray vertex = vertexValue.toArray();
264 if (vertex.size() >= 3) {
265 content += QString(" %1 %2 %3\n")
266 .arg(vertex[0].toDouble())
267 .arg(vertex[1].toDouble())
268 .arg(vertex[2].toDouble());
269 }
270 }
271
272 content += " ]\n";
273 content += " }\n";
274 content += " coordIndex [\n";
275
276 // 添加面索引数
277 for (const QJsonValue& faceValue : faces) {
278 QJsonArray face = faceValue.toArray();
279 QString faceStr;
280 for (const QJsonValue& indexValue : face) {
281 faceStr += QString::number(indexValue.toInt() - 1) + " ";
282 }
283 faceStr += "-1";
284 content += " " + faceStr + "\n";
285 }
286
287 content += " ]\n";
288 content += " }\n";
289 content += " }\n";
290 content += " ]\n";
291 content += "}\n";
292 }
293
294 return content;

297QJsonObject Exporter3DModel::parseObjData(const QByteArray& objData) {
298 QJsonObject result;
299 QJsonArray vertices;
300 QJsonArray faces;
301 QJsonObject materials;
302 QJsonArray shapes;
303
304 QString objString = QString::fromUtf8(objData);
305 QStringList lines = objString.split('\n');
306
307 // qDebug() << "=== OBJ Data Debug ===";
308 // qDebug() << "OBJ data size:" << objData.size();
309 // qDebug() << "OBJ string size:" << objString.size();
310 // qDebug() << "Total lines:" << lines.size();
311 // qDebug() << "First 500 chars:" << objString.left(500);
312 // qDebug() << "First 20 lines:";
313 // for (int i = 0; i < qMin(20, lines.size()); ++i) {
314 // qDebug() << " Line" << i << ":" << lines[i].left(100);
315 // }
316 // qDebug() << "===================";
317
318 // 第一遍：提取材质信息
319 QString currentMaterialId;
320 QJsonObject currentMaterial;
321 bool inMaterial = false;
322
323 for (const QString& line : lines) {
324 QString trimmedLine = line.trimmed();
325 if (trimmedLine.isEmpty() || trimmedLine.startsWith('#')) {
326 continue;
327 }
328
329 QStringList parts = trimmedLine.split(' ', Qt::SkipEmptyParts);
330 if (parts.isEmpty()) {
331 continue;
332 }
333
334 if (parts[0] == "newmtl") {
335 // 保存上一个材
336 if (!currentMaterialId.isEmpty() && !currentMaterial.isEmpty()) {
337 materials[currentMaterialId] = currentMaterial;
338 }
339 // 开始新材质
340 currentMaterialId = parts[1];
341 currentMaterial = QJsonObject();
342 inMaterial = true;
343 } else if (inMaterial) {
344 if (parts[0] == "endmtl") {
345 // 保存材质
346 if (!currentMaterialId.isEmpty() && !currentMaterial.isEmpty()) {
347 materials[currentMaterialId] = currentMaterial;
348 }
349 currentMaterialId.clear();
350 currentMaterial = QJsonObject();
351 inMaterial = false;
352 } else if (parts[0] == "Ka") {
353 QJsonArray ambientColor;
354 for (int i = 1; i < parts.size(); ++i) {
355 ambientColor.append(parts[i]);
356 }
357 currentMaterial["ambientColor"] = ambientColor;
358 } else if (parts[0] == "Kd") {
359 QJsonArray diffuseColor;
360 for (int i = 1; i < parts.size(); ++i) {
361 diffuseColor.append(parts[i]);
362 }
363 currentMaterial["diffuseColor"] = diffuseColor;
364 } else if (parts[0] == "Ks") {
365 QJsonArray specularColor;
366 for (int i = 1; i < parts.size(); ++i) {
367 specularColor.append(parts[i]);
368 }
369 currentMaterial["specularColor"] = specularColor;
370 } else if (parts[0] == "d") {
371 currentMaterial["transparency"] = parts[1];
372 }
373 }
374 }
375
376 // 保存最后一个材
377 if (!currentMaterialId.isEmpty() && !currentMaterial.isEmpty()) {
378 materials[currentMaterialId] = currentMaterial;
379 }
380
381 // qDebug() << "Found" << materials.size() << "materials";
382
383 // 第二遍：提取顶点数据（存储为字符串，用于 WRL 输出
384 QStringList vertexStrings; // 存储顶点坐标字符
385 for (const QString& line : lines) {
386 QString trimmedLine = line.trimmed();
387 if (trimmedLine.isEmpty() || trimmedLine.startsWith('#')) {
388 continue;
389 }
390
391 QStringList parts = trimmedLine.split(' ', Qt::SkipEmptyParts);
392 if (parts.isEmpty()) {
393 continue;
394 }
395
396 if (parts[0] == 'v') {
397 // 顶点数据，转换为毫米（除以 2.54）
398 if (parts.size() >= 4) {
399 double x = parts[1].toDouble() / 2.54;
400 double y = parts[2].toDouble() / 2.54;
401 double z = parts[3].toDouble() / 2.54;
402 // 保留 4 位小数，与 Python 版本保持一致
403 QString vertexStr = QString("%1 %2 %3").arg(x, 0, 'f', 4).arg(y, 0, 'f', 4).arg(z, 0, 'f', 4);
404 vertexStrings.append(vertexStr);
405
406 // 同时保存JSON 数组，用于后续处
407 QJsonArray vertex;
408 vertex.append(x);
409 vertex.append(y);
410 vertex.append(z);
411 vertices.append(vertex);
412 }
413 }
414 }
415
416 // qDebug() << "Found" << vertices.size() << "vertices";
417
418 // 第三遍：按材质分割形状并提取面数
419 QString currentShapeMaterial = "default";
420 QStringList currentShapePoints; // 使用字符串列
421 QJsonArray currentShapeCoordIndex;
422 QMap<int, int> vertexIndexMap; // 映射原始顶点索引到形状中的索
423 int shapeVertexCounter = 0;
424
425 for (const QString& line : lines) {
426 QString trimmedLine = line.trimmed();
427 if (trimmedLine.isEmpty() || trimmedLine.startsWith('#')) {
428 continue;
429 }
430
431 QStringList parts = trimmedLine.split(' ', Qt::SkipEmptyParts);
432 if (parts.isEmpty()) {
433 continue;
434 }
435
436 if (parts[0] == "usemtl") {
437 // 保存上一个形
438 if (!currentShapePoints.isEmpty()) {
439 QJsonObject shape;
440 shape["materialId"] = currentShapeMaterial;
441 shape["points"] = QJsonArray::fromStringList(currentShapePoints);
442 shape["coordIndex"] = currentShapeCoordIndex;
443 shapes.append(shape);
444 }
445 // 开始新形状
446 currentShapeMaterial = parts[1];
447 currentShapePoints.clear();
448 currentShapeCoordIndex = QJsonArray();
449 vertexIndexMap.clear();
450 shapeVertexCounter = 0;
451 } else if (parts[0] == 'f') {
452 // 面数
453 QJsonArray faceIndices;
454 for (int i = 1; i < parts.size(); ++i) {
455 QString vertexIndexStr = parts[i].split('/')[0];
456 int vertexIndex = vertexIndexStr.toInt() - 1; // OBJ 索引1 开
457
458 if (!vertexIndexMap.contains(vertexIndex)) {
459 // 添加顶点到形状（使用字符串格式）
460 if (vertexIndex >= 0 && vertexIndex < vertexStrings.size()) {
461 currentShapePoints.append(vertexStrings[vertexIndex]);
462 } else {
463 // 默认顶点
464 currentShapePoints.append("0 0 0");
465 }
466 vertexIndexMap[vertexIndex] = shapeVertexCounter;
467 faceIndices.append(shapeVertexCounter);
468 shapeVertexCounter++;
469 } else {
470 faceIndices.append(vertexIndexMap[vertexIndex]);
471 }
472 }
473 faceIndices.append(-1); // WRL 格式要求每个面以 -1 结束
474 currentShapeCoordIndex.append(QJsonArray(faceIndices));
475 }
476 }
477
478 // 保存最后一个形
479 if (!currentShapePoints.isEmpty()) {
480 QJsonObject shape;
481 shape["materialId"] = currentShapeMaterial;
482 shape["points"] = QJsonArray::fromStringList(currentShapePoints);
483 shape["coordIndex"] = currentShapeCoordIndex;
484 shapes.append(shape);
485 }
486
487 // qDebug() << "Found" << shapes.size() << "shapes";
488
489 result["vertices"] = vertices;
490 result["faces"] = faces;
491 result["materials"] = materials;
492 result["shapes"] = shapes;
493
494 return result;
495}
