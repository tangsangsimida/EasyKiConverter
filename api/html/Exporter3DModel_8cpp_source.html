103double Exporter3DModel::calculateObjMinZ(const QByteArray& objData) {
104 if (objData.isEmpty()) {
105 return std::numeric_limits<double>::max();
106 }
107
108 double minZ = std::numeric_limits<double>::max();
109 int start = 0;
110 while (start < objData.size()) {
111 int end = objData.indexOf('\n', start);
112 if (end < 0) {
113 end = objData.size();
114 }
115
116 // 检查行是否以 "v " 开头
117 if (end - start >= 2 && objData.at(start) == 'v' && objData.at(start + 1) == ' ') {
118 // 跳过 "v " 前缀，解析三个坐标分量
119 int pos = start + 2;
120 // 跳过前两个分量 (x, y)
121 int component = 0;
122 while (pos < end && component < 2) {
123 if (objData.at(pos) == ' ') {
124 component++;
125 // 跳过连续空格
126 while (pos < end && objData.at(pos) == ' ') {
127 pos++;
128 }
129 } else {
130 pos++;
131 }
132 }
133
134 // 解析 Z 分量（转换为毫米，与 calculateWrlDisplayMinZ 单位一致）
135 if (pos < end) {
136 bool ok = false;
137 double z = QByteArray(objData.constData() + pos, end - pos).toDouble(&ok);
138 if (ok) {
139 minZ = qMin(minZ, z / WRL_UNIT_TO_MM);
140 }
141 }
142 }
143
144 start = end + 1;
145 }
146 if (minZ == std::numeric_limits<double>::max()) {
147 return minZ;
148 }
149
150 return minZ;
151}

153double Exporter3DModel::calculateWrlDisplayMinZ(const QByteArray& wrlData) {
154 if (wrlData.isEmpty()) {
155 return std::numeric_limits<double>::max();
156 }
157
158 static const QRegularExpression numberRegex(QStringLiteral("[-+]?(?:\\d+\\.?\\d*|\\.\\d+)(?:[eE][-+]?\\d+)?"));
159
160 double minZ = std::numeric_limits<double>::max();
161 bool inPointArray = false;
162 int coordinateComponent = 0;
163 const QStringList lines = QString::fromUtf8(wrlData).split('\n');
164 for (const QString& line : lines) {
165 QString segment = line;
166 if (!inPointArray) {
167 const int pointStart = segment.indexOf(QStringLiteral("point ["));
168 if (pointStart < 0) {
169 continue;
170 }
171 inPointArray = true;
172 coordinateComponent = 0;
173 segment = segment.mid(pointStart + QStringLiteral("point [").size());
174 }
175
176 const int pointEnd = segment.indexOf(QLatin1Char(']'));
177 if (pointEnd >= 0) {
178 segment = segment.left(pointEnd);
179 }
180
181 QRegularExpressionMatchIterator matches = numberRegex.globalMatch(segment);
182 while (matches.hasNext()) {
183 const QRegularExpressionMatch match = matches.next();
184 bool ok = false;
185 const double value = match.captured(0).toDouble(&ok);
186 if (!ok) {
187 continue;
188 }
189 if (coordinateComponent % 3 == 2) {
190 minZ = qMin(minZ, value * WRL_UNIT_TO_MM);
191 }
192 coordinateComponent++;
193 }
194
195 if (pointEnd >= 0) {
196 inPointArray = false;
197 }
198 }
199
200 return minZ;
201}

230QString Exporter3DModel::generateWrlContent(const Model3DData& modelData, const QByteArray& objData) {
231 QString content;
232
233 // 解析 OBJ 数据
234 QJsonObject parsedData = parseObjData(objData);
235 QJsonArray vertices = parsedData["vertices"].toArray();
236 QJsonArray faces = parsedData["faces"].toArray();
237 QJsonObject materials = parsedData["materials"].toObject();
238 QJsonArray shapes = parsedData["shapes"].toArray();
239
240 // qDebug() << "Generating WRL content - vertices:" << vertices.size()
241 // << "faces:" << faces.size()
242 // << "materials:" << materials.size()
243 // << "shapes:" << shapes.size();
244
245 // WRL 文件头部
246 content += "#VRML V2.0 utf8\n";
247 content += "# 3D model generated by EasyKiConverter (https://github.com/tangsangsimida/EasyKiConverter)\n";
248 content += "\n";
249
250 // 如果有形状数据，使用形状数据生成 WRL
251 if (!shapes.isEmpty()) {
252 for (const QJsonValue& shapeValue : shapes) {
253 QJsonObject shape = shapeValue.toObject();
254 QString materialId = shape["materialId"].toString();
255 QJsonArray shapePoints = shape["points"].toArray();
256 QJsonArray coordIndex = shape["coordIndex"].toArray();
257
258 // qDebug() << "Processing shape - materialId:" << materialId
259 // << "points:" << shapePoints.size()
260 // << "coordIndex:" << coordIndex.size();
261
262 // 在倒数第二个位置插入最后一个点的副本
263 if (shapePoints.size() > 0) {
264 shapePoints.insert(shapePoints.size() - 1, shapePoints.last());
265 // qDebug() << "Inserted duplicate of last point, new size:" << shapePoints.size();
266 }
267
268 // 获取材质信息
269 QJsonObject material = materials[materialId].toObject();
270 QJsonArray diffuseColor = material["diffuseColor"].toArray();
271 QJsonArray specularColor = material["specularColor"].toArray();
272 QString transparency = material["transparency"].toString("0");
273
274 // 生成 Shape
275 content += "Shape {\n";
276 content += " appearance Appearance {\n";
277 content += " material Material {\n";
278 content += QString(" diffuseColor %1 %2 %3\n")
279 .arg(diffuseColor.isEmpty() ? "0.8" : diffuseColor[0].toString())
280 .arg(diffuseColor.isEmpty() ? "0.8" : diffuseColor[1].toString())
281 .arg(diffuseColor.isEmpty() ? "0.8" : diffuseColor[2].toString());
282 content += QString(" specularColor %1 %2 %3\n")
283 .arg(specularColor.isEmpty() ? "0.2" : specularColor[0].toString())
284 .arg(specularColor.isEmpty() ? "0.2" : specularColor[1].toString())
285 .arg(specularColor.isEmpty() ? "0.2" : specularColor[2].toString());
286 content += " ambientIntensity 0.2\n";
287 content += QString(" transparency %1\n").arg(transparency);
288 content += " shininess 0.5\n";
289 content += " }\n";
290 content += " }\n";
291 content += " geometry IndexedFaceSet {\n";
292 content += " ccw TRUE\n";
293 content += " solid FALSE\n";
294 content += " coord DEF co Coordinate {\n";
295 content += " point [\n";
296
297 for (const QJsonValue& pointValue : shapePoints) {
298 // pointValue 现在是字符串格式，直接输出
299 content += " " + pointValue.toString() + ",\n";
300 }
301
302 content += " ]\n";
303 content += " }\n";
304 content += " coordIndex [\n";
305
306 for (const QJsonValue& indexValue : coordIndex) {
307 // indexValue 是一QJsonArray，包含面索引
308 QJsonArray faceIndices = indexValue.toArray();
309 QString indexStr;
310 for (const QJsonValue& idx : faceIndices) {
311 indexStr += QString::number(idx.toInt()) + " ";
312 }
313 content += " " + indexStr + "\n";
314 }
315
316 content += " ]\n";
317 content += " }\n";
318 content += "}\n";
319 }
320 } else {
321 // qDebug() << "No shapes found, using simple Transform mode";
322 // 如果没有形状数据，使用简单的 Transform 包装
323 // 获取平移和旋
324 Model3DBase translation = modelData.translation();
325 Model3DBase rotation = modelData.rotation();
326
327 content += "Transform {\n";
328 content += " translation ";
329 content += QString("%1 %2 %3\n").arg(translation.x).arg(translation.y).arg(translation.z);
330 content += " rotation ";
331 content += QString("%1 %2 %3 %4\n").arg(rotation.x).arg(rotation.y).arg(rotation.z).arg(1.0);
332 content += " scale 1 1 1\n";
333 content += " children [\n";
334 content += " Shape {\n";
335 content += " appearance Appearance {\n";
336 content += " material Material {\n";
337 content += " diffuseColor 0.8 0.8 0.8\n";
338 content += " specularColor 0.2 0.2 0.2\n";
339 content += " ambientIntensity 0.2\n";
340 content += " shininess 0.2\n";
341 content += " }\n";
342 content += " }\n";
343 content += " geometry IndexedFaceSet {\n";
344 content += " coord Coordinate {\n";
345 content += " point [\n";
346
347 for (const QJsonValue& vertexValue : vertices) {
348 QJsonArray vertex = vertexValue.toArray();
349 if (vertex.size() >= 3) {
350 content += QString(" %1 %2 %3\n")
351 .arg(vertex[0].toDouble())
352 .arg(vertex[1].toDouble())
353 .arg(vertex[2].toDouble());
354 }
355 }
356
357 content += " ]\n";
358 content += " }\n";
359 content += " coordIndex [\n";
360
361 for (const QJsonValue& faceValue : faces) {
362 QJsonArray face = faceValue.toArray();
363 QString faceStr;
364 for (const QJsonValue& indexValue : face) {
365 faceStr += QString::number(indexValue.toInt() - 1) + " ";
366 }
367 faceStr += "-1";
368 content += " " + faceStr + "\n";
369 }
370
371 content += " ]\n";
372 content += " }\n";
373 content += " }\n";
374 content += " ]\n";
375 content += "}\n";
376 }
377
378 return content;

451QJsonObject Exporter3DModel::parseObjData(const QByteArray& objData) {
452 QJsonObject result;
453 QJsonArray vertices;
454 QJsonArray faces;
455 QJsonObject materials;
456 QJsonArray shapes;
457
458 QString objString = QString::fromUtf8(objData);
459 QStringList lines = objString.split('\n');
460
461 // qDebug() << "=== OBJ Data Debug ===";
462 // qDebug() << "OBJ data size:" << objData.size();
463 // qDebug() << "OBJ string size:" << objString.size();
464 // qDebug() << "Total lines:" << lines.size();
465 // qDebug() << "First 500 chars:" << objString.left(500);
466 // qDebug() << "First 20 lines:";
467 // for (int i = 0; i < qMin(20, lines.size()); ++i) {
468 // qDebug() << " Line" << i << ":" << lines[i].left(100);
469 // }
470 // qDebug() << "===================";
471
472 // 第一遍：提取材质信息
473 QString currentMaterialId;
474 QJsonObject currentMaterial;
475 bool inMaterial = false;
476
477 for (const QString& line : lines) {
478 QString trimmedLine = line.trimmed();
479 if (trimmedLine.isEmpty() || trimmedLine.startsWith('#')) {
480 continue;
481 }
482
483 QStringList parts = trimmedLine.split(' ', Qt::SkipEmptyParts);
484 if (parts.isEmpty()) {
485 continue;
486 }
487
488 if (parts[0] == "newmtl") {
489 // 保存上一个材
490 if (!currentMaterialId.isEmpty() && !currentMaterial.isEmpty()) {
491 materials[currentMaterialId] = currentMaterial;
492 }
493 // 开始新材质
494 currentMaterialId = parts[1];
495 currentMaterial = QJsonObject();
496 inMaterial = true;
497 } else if (inMaterial) {
498 if (parts[0] == "endmtl") {
499 // 保存材质
500 if (!currentMaterialId.isEmpty() && !currentMaterial.isEmpty()) {
501 materials[currentMaterialId] = currentMaterial;
502 }
503 currentMaterialId.clear();
504 currentMaterial = QJsonObject();
505 inMaterial = false;
506 } else if (parts[0] == "Ka") {
507 QJsonArray ambientColor;
508 for (int i = 1; i < parts.size(); ++i) {
509 ambientColor.append(parts[i]);
510 }
511 currentMaterial["ambientColor"] = ambientColor;
512 } else if (parts[0] == "Kd") {
513 QJsonArray diffuseColor;
514 for (int i = 1; i < parts.size(); ++i) {
515 diffuseColor.append(parts[i]);
516 }
517 currentMaterial["diffuseColor"] = diffuseColor;
518 } else if (parts[0] == "Ks") {
519 QJsonArray specularColor;
520 for (int i = 1; i < parts.size(); ++i) {
521 specularColor.append(parts[i]);
522 }
523 currentMaterial["specularColor"] = specularColor;
524 } else if (parts[0] == "d") {
525 currentMaterial["transparency"] = parts[1];
526 }
527 }
528 }
529
530 // 保存最后一个材
531 if (!currentMaterialId.isEmpty() && !currentMaterial.isEmpty()) {
532 materials[currentMaterialId] = currentMaterial;
533 }
534
535 // qDebug() << "Found" << materials.size() << "materials";
536
537 // 第二遍：提取顶点数据
538 QVector<Model3DBase> convertedVertices;
539 double minX = std::numeric_limits<double>::max();
540 double maxX = std::numeric_limits<double>::lowest();
541 double minY = std::numeric_limits<double>::max();
542 double maxY = std::numeric_limits<double>::lowest();
543 double minZ = std::numeric_limits<double>::max();
544 for (const QString& line : lines) {
545 QString trimmedLine = line.trimmed();
546 if (trimmedLine.isEmpty() || trimmedLine.startsWith('#')) {
547 continue;
548 }
549
550 QStringList parts = trimmedLine.split(' ', Qt::SkipEmptyParts);
551 if (parts.isEmpty()) {
552 continue;
553 }
554
555 if (parts[0] == 'v') {
556 // 顶点数据，转换为毫米
557 if (parts.size() >= 4) {
558 double x = parts[1].toDouble() / WRL_UNIT_TO_MM;
559 double y = parts[2].toDouble() / WRL_UNIT_TO_MM;
560 double z = parts[3].toDouble() / WRL_UNIT_TO_MM;
561 convertedVertices.append(Model3DBase(x, y, z));
562 minX = qMin(minX, x);
563 maxX = qMax(maxX, x);
564 minY = qMin(minY, y);
565 maxY = qMax(maxY, y);
566 minZ = qMin(minZ, z);
567 }
568 }
569 }
570
571 const bool hasVertices = !convertedVertices.isEmpty();
572 const double normalizeX = hasVertices ? (minX + maxX) / 2.0 : 0.0;
573 const double normalizeY = hasVertices ? (minY + maxY) / 2.0 : 0.0;
574 const double normalizeZ = hasVertices && minZ > 0.0 ? minZ : 0.0;
575
576 QStringList vertexStrings;
577 for (Model3DBase vertexData : convertedVertices) {
578 vertexData.x -= normalizeX;
579 vertexData.y -= normalizeY;
580 vertexData.z -= normalizeZ;
581
582 // 保留 4 位小数，与 Python 版本保持一致
583 QString vertexStr =
584 QString("%1 %2 %3").arg(vertexData.x, 0, 'f', 4).arg(vertexData.y, 0, 'f', 4).arg(vertexData.z, 0, 'f', 4);
585 vertexStrings.append(vertexStr);
586
587 // 同时保存JSON 数组，用于后续处理
588 QJsonArray vertex;
589 vertex.append(vertexData.x);
590 vertex.append(vertexData.y);
591 vertex.append(vertexData.z);
592 vertices.append(vertex);
593 }
594
595 // qDebug() << "Found" << vertices.size() << "vertices";
596
597 // 第三遍：按材质分割形状并提取面数
598 QString currentShapeMaterial = "default";
599 QStringList currentShapePoints; // 使用字符串列
600 QJsonArray currentShapeCoordIndex;
601 QMap<int, int> vertexIndexMap; // 映射原始顶点索引到形状中的索
602 int shapeVertexCounter = 0;
603
604 for (const QString& line : lines) {
605 QString trimmedLine = line.trimmed();
606 if (trimmedLine.isEmpty() || trimmedLine.startsWith('#')) {
607 continue;
608 }
609
610 QStringList parts = trimmedLine.split(' ', Qt::SkipEmptyParts);
611 if (parts.isEmpty()) {
612 continue;
613 }
614
615 if (parts[0] == "usemtl") {
616 // 保存上一个形
617 if (!currentShapePoints.isEmpty()) {
618 QJsonObject shape;
619 shape["materialId"] = currentShapeMaterial;
620 shape["points"] = QJsonArray::fromStringList(currentShapePoints);
621 shape["coordIndex"] = currentShapeCoordIndex;
622 shapes.append(shape);
623 }
624 // 开始新形状
625 currentShapeMaterial = parts[1];
626 currentShapePoints.clear();
627 currentShapeCoordIndex = QJsonArray();
628 vertexIndexMap.clear();
629 shapeVertexCounter = 0;
630 } else if (parts[0] == 'f') {
631 // 面数
632 QJsonArray faceIndices;
633 for (int i = 1; i < parts.size(); ++i) {
634 QString vertexIndexStr = parts[i].split('/')[0];
635 int vertexIndex = vertexIndexStr.toInt() - 1; // OBJ 索引1 开
636
637 if (!vertexIndexMap.contains(vertexIndex)) {
638 // WRL 格式使用字符串格式的顶点
639 if (vertexIndex >= 0 && vertexIndex < vertexStrings.size()) {
640 currentShapePoints.append(vertexStrings[vertexIndex]);
641 } else {
642 // 默认顶点
643 currentShapePoints.append("0 0 0");
644 }
645 vertexIndexMap[vertexIndex] = shapeVertexCounter;
646 faceIndices.append(shapeVertexCounter);
647 shapeVertexCounter++;
648 } else {
649 faceIndices.append(vertexIndexMap[vertexIndex]);
650 }
651 }
652 faceIndices.append(-1); // WRL 格式要求每个面以 -1 结束
653 currentShapeCoordIndex.append(QJsonArray(faceIndices));
654 }
655 }
656
657 // 保存最后一个形
658 if (!currentShapePoints.isEmpty()) {
659 QJsonObject shape;
660 shape["materialId"] = currentShapeMaterial;
661 shape["points"] = QJsonArray::fromStringList(currentShapePoints);
662 shape["coordIndex"] = currentShapeCoordIndex;
663 shapes.append(shape);
664 }
665
666 // qDebug() << "Found" << shapes.size() << "shapes";
667
668 result["vertices"] = vertices;
669 result["faces"] = faces;
670 result["materials"] = materials;
671 result["shapes"] = shapes;
672
673 return result;
674}
