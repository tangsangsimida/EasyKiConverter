98void NetworkUtils::handleResponse() {
99 m_timeoutTimer->stop();
100
101 if (!m_currentReply) {
102 m_isRequesting = false;
103 return;
104 }
105
106 int statusCode = m_currentReply->attribute(QNetworkRequest::HttpStatusCodeAttribute).toInt();
107
108 if (shouldRetry(statusCode) && m_retryCount < m_maxRetries) {
109 retryRequest();
110 return;
111 }
112
113 QByteArray responseData = m_currentReply->readAll();
114 m_currentReply->deleteLater();
115 m_currentReply = nullptr;
116
117 if (statusCode != 200 && statusCode != 0) { // 某些情况下状态码可能为 0 但数据有效（如 local file，虽然这里不常用）
118 QString errorMsg = QString("HTTP request failed with status code: %1").arg(statusCode);
119 emit requestError(errorMsg);
120 m_isRequesting = false;
121 return;
122 }
123
124 // 检查是否是 gzip 压缩的数据
125 if (responseData.size() >= 2 && static_cast<unsigned char>(responseData[0]) == 0x1F &&
126 static_cast<unsigned char>(responseData[1]) == 0x8B) {
127 responseData = decompressGzip(responseData);
128 }
129
130 if (m_expectBinaryData) {
131 emit binaryDataFetched(responseData);
132 m_isRequesting = false;
133 return;
134 }
135
136 QJsonParseError parseError;
137 QJsonDocument jsonDoc = QJsonDocument::fromJson(responseData, &parseError);
138
139 if (parseError.error != QJsonParseError::NoError) {
140 QString errorMsg = QString("Failed to parse JSON response: %1").arg(parseError.errorString());
141 emit requestError(errorMsg);
142 m_isRequesting = false;
143 return;
144 }
145
146 if (!jsonDoc.isObject()) {
147 emit requestError("JSON response is not an object");
148 m_isRequesting = false;
149 return;
150 }
151
152 QJsonObject jsonObject = jsonDoc.object();
153 m_isRequesting = false;
154 emit requestSuccess(jsonObject);
155}

217QByteArray NetworkUtils::decompressGzip(const QByteArray& compressedData) {
218 if (compressedData.size() < 2)
219 return QByteArray();
220
221 z_stream stream;
222 memset(&stream, 0, sizeof(stream));
223
224 if (inflateInit2(&stream, 15 + 16) != Z_OK)
225 return QByteArray();
226
227 stream.next_in = const_cast<Bytef*>(reinterpret_cast<const Bytef*>(compressedData.constData()));
228 stream.avail_in = compressedData.size();
229
230 QByteArray decompressedData;
231 const int chunkSize = 4096;
232 char buffer[chunkSize];
233
234 int ret;
235 do {
236 stream.next_out = reinterpret_cast<Bytef*>(buffer);
237 stream.avail_out = chunkSize;
238 ret = inflate(&stream, Z_NO_FLUSH);
239 if (ret == Z_OK || ret == Z_STREAM_END) {
240 decompressedData.append(buffer, chunkSize - stream.avail_out);
241 } else {
242 inflateEnd(&stream);
243 return QByteArray();
244 }
245 } while (ret != Z_STREAM_END && stream.avail_in > 0);
246
247 inflateEnd(&stream);
248 return (ret == Z_STREAM_END) ? decompressedData : QByteArray();
249}
