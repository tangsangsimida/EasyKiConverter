541 // 捕获队列的共享指针副本，确保在 Service 被销毁时队列依然有效
542 auto fetchProcessQueue = m_fetchProcessQueue;
543
544 for (int i = 0; i < m_processThreadPool->maxThreadCount(); i++) {
545 m_processThreadPool->start(QRunnable::create([this, fetchProcessQueue]() {
546 while (true) {
547 if (m_isCancelled.loadAcquire()) { // Batch-level cancellation check
548 qDebug() << "Process stage worker exiting due to cancellation.";
549 break;
550 }
551
552 // 使用捕获的队列指针进行操作
553 if (!fetchProcessQueue)
554 break;
555
556 QSharedPointer<ComponentExportStatus> status;
557 if (!fetchProcessQueue->pop(status, 1000)) {
558 if (fetchProcessQueue->isClosed())
559 break;
560 continue;
561 }
562
563 ProcessWorker* worker = new ProcessWorker(status, nullptr);
564
565 // 关键修复：将 worker 移动到主线程，确保 deleteLater() 能在主线程事件循环中被执行
566 // 线程池线程通常不运行事件循环，直接 deleteLater 可能导致内存泄漏
567 worker->moveToThread(this->thread());
568
569 // 注册活跃 worker
570 {
571 QMutexLocker locker(&m_workerMutex);
572 m_activeProcessWorkers.insert(worker);
573 }
574
575 // 使用 Lambda 捕获 worker 指针，避免在槽函数中使用 sender()
576 // Context 设为 this，确保 Service 销毁时连接自动断开

578 worker,
579 &ProcessWorker::processCompleted,
580 this,
581 [this, worker](QSharedPointer<ComponentExportStatus> status) {
582 handleProcessCompleted(status, worker);
583 },
584 Qt::QueuedConnection);
585
586 // 让 Worker 在信号处理完成后自动销毁
587 connect(worker, &ProcessWorker::processCompleted, worker, &QObject::deleteLater, Qt::QueuedConnection);

589 worker->run();
590 }
591 }));

657void ExportServicePipeline::checkPipelineCompletion() {
658 // 防止重复触发：如果已经安排了完成处理，直接返回
659 if (m_completionScheduled.loadAcquire()) {
660 qDebug() << "Completion already scheduled, skipping.";
661 return;
662 }
663
664 // 抓取阶段完成检查
665 bool fetchDone = m_pipelineProgress.fetchCompleted >= m_pipelineProgress.totalTasks;
666 if (m_isCancelled.loadAcquire()) {
667 bool fetchQueueEmpty = m_fetchProcessQueue ? m_fetchProcessQueue->isEmpty() : true;
668 if (fetchQueueEmpty)
669 fetchDone = true;
670 }
671 if (fetchDone && m_fetchProcessQueue && !m_fetchProcessQueue->isClosed())
672 m_fetchProcessQueue->close();
673 if (!fetchDone)
674 return;
675

773void ExportServicePipeline::cleanupPipeline() {
774 // 使用原子标志检查，避免锁带来的死锁问题
775 // cleanupPipeline 只在主线程的事件循环中被调用，此时所有 worker 已停止
776 if (!m_isPipelineRunning) {
777 return;
778 }
779 m_isPipelineRunning = false;
780
781 // 关键修复：移除 waitForDone() 调用
782 // 之前这里调用 waitForDone() 会导致死锁，因为 worker 线程可能正试图通过 Qt::QueuedConnection 发送信号给主线程
783 // 而主线程被 waitForDone() 阻塞，无法处理信号
784 //
785 // 现在我们采用以下策略：
786 // 1. 在 cancelExport() 中已经发出了取消信号并 abort 了所有 worker
787 // 2. 队列使用了 QSharedPointer，并被 worker 的 Lambda 捕获
788 // 3. 这里我们只需要 reset 成员变量中的指针，不仅是安全的，而且是必须的
789 // 4. 当最后一个 worker 结束时，队列的引用计数归零，自动销毁
790
791 if (m_fetchProcessQueue) {
792 m_fetchProcessQueue->close();
793 m_fetchProcessQueue.reset();
794 }
795 if (m_processWriteQueue) {
796 m_processWriteQueue->close();
797 m_processWriteQueue.reset();
798 }
799
800 // 清理 worker 追踪集合
801 {
802 QMutexLocker locker(&m_workerMutex);
803 m_activeFetchWorkers.clear();
804 m_activeProcessWorkers.clear();
805 m_activeWriteWorkers.clear();
806 }
807
808 qDebug() << "Pipeline resources cleaned up (queues released, threads detached).";
809}

850 // 使用原始总数量（包括重试），如果未设置则使用当前批次数量
851 statistics.total = (m_originalTotalTasks > 0) ? m_originalTotalTasks : m_pipelineProgress.totalTasks;

853 // 计算成功和失败数量
854 // 需要去重统计，因为重试的元器件可能在 m_completedStatuses 中有多个状态

856 int actualSuccessCount = 0;
857 int actualFailureCount = 0;
858
859 for (const QSharedPointer<ComponentExportStatus>& status : m_completedStatuses) {
860 QString componentId = status->componentId;
861
862 // 只统计每个元器件的最新状态（最后一次导出结果）
863 if (uniqueComponentIds.contains(componentId)) {
864 continue; // 跳过重复的元器件
865 }
866 uniqueComponentIds.insert(componentId);
867
868 // 统计启用的导出选项
869 if (m_options.exportSymbol && status->symbolWritten)
870 statistics.successSymbol++;
871 if (m_options.exportFootprint && status->footprintWritten)
872 statistics.successFootprint++;
873 if (m_options.exportModel3D && status->model3DWritten)
874 statistics.successModel3D++;
875
876 if (status->isCompleteSuccess()) {
877 actualSuccessCount++;
878 } else {
879 actualFailureCount++;
880 statistics.stageFailures[status->getFailedStage()]++;
881 statistics.failureReasons[status->getFailureReason()]++;
882 }
883 }
884
885 statistics.success = actualSuccessCount;
886 statistics.failed = actualFailureCount;
887
888 // 只在非重试模式下计算时间统计（避免重试时覆盖原始导出的时间信息）
889 if (!m_isRetryMode) {
890 // 计算总耗时
891 qint64 currentTimeMs = QDateTime::currentMSecsSinceEpoch();
892 if (m_originalExportStartTimeMs > 0) {
893 statistics.totalDurationMs = currentTimeMs - m_originalExportStartTimeMs;
894 qDebug() << "Total duration calculated:" << statistics.totalDurationMs << "ms"
895 << "(start:" << m_originalExportStartTimeMs << ", end:" << currentTimeMs << ")";
896 } else if (m_exportStartTimeMs > 0) {
897 // 如果原始开始时间未设置（向后兼容），使用当前开始时间

899 qDebug() << "Total duration calculated (using current start):" << statistics.totalDurationMs << "ms"
900 << "(start:" << m_exportStartTimeMs << ", end:" << currentTimeMs << ")";
901 } else {
902 qWarning() << "Neither m_originalExportStartTimeMs nor m_exportStartTimeMs is initialized, setting "
903 "totalDurationMs to 0";
904 statistics.totalDurationMs = 0;
905 }
906
907 // 计算平均耗时（基于所有完成的元器件状态）
908 qint64 totalFetchTime = 0, totalProcessTime = 0, totalWriteTime = 0;
909 int completedCount = uniqueComponentIds.size();
910
911 for (const QSharedPointer<ComponentExportStatus>& status : m_completedStatuses) {
912 totalFetchTime += status->fetchDurationMs;
913 totalProcessTime += status->processDurationMs;
914 totalWriteTime += status->writeDurationMs;
915 for (const auto& diag : status->networkDiagnostics) {
916 statistics.totalNetworkRequests++;
917 statistics.totalRetries += diag.retryCount;
918 statistics.avgNetworkLatencyMs += diag.latencyMs;
919 if (diag.wasRateLimited)
920 statistics.rateLimitHitCount++;
921 statistics.statusCodeDistribution[diag.statusCode]++;
922 }
923 }
924
925 if (statistics.totalNetworkRequests > 0)
926 statistics.avgNetworkLatencyMs /= statistics.totalNetworkRequests;
927 if (completedCount > 0) {
928 statistics.avgFetchTimeMs = totalFetchTime / completedCount;
929 statistics.avgProcessTimeMs = totalProcessTime / completedCount;
930 statistics.avgWriteTimeMs = totalWriteTime / completedCount;
931 }
932 } else {
933 // 重试模式：保留原始统计信息的时间数据，但使用最新的基本统计
934 if (m_originalStatistics.total > 0) {
935 statistics.totalDurationMs = m_originalStatistics.totalDurationMs;
936 statistics.avgFetchTimeMs = m_originalStatistics.avgFetchTimeMs;
937 statistics.avgProcessTimeMs = m_originalStatistics.avgProcessTimeMs;
938 statistics.avgWriteTimeMs = m_originalStatistics.avgWriteTimeMs;
939 statistics.totalNetworkRequests = m_originalStatistics.totalNetworkRequests;
940 statistics.totalRetries = m_originalStatistics.totalRetries;
941 statistics.avgNetworkLatencyMs = m_originalStatistics.avgNetworkLatencyMs;
942 statistics.rateLimitHitCount = m_originalStatistics.rateLimitHitCount;
943 statistics.statusCodeDistribution = m_originalStatistics.statusCodeDistribution;
944 qDebug() << "Retry mode: Preserving original timing statistics";
945 } else {
946 qDebug() << "Retry mode: No original statistics available, using default values";
947 }
948 }
949
950 qDebug() << "Statistics generated - Total:" << statistics.total << "Success:" << statistics.success
951 << "Failed:" << statistics.failed << "Unique components:" << uniqueComponentIds.size()
952 << "Retry mode:" << m_isRetryMode;
953
954 // v3.0.5: 内存峰值统计
955 // 无论是否重试模式，都使用当前记录的最大值
956 statistics.peakMemoryUsage = m_originalStatistics.peakMemoryUsage;
957
958 return statistics;

961bool ExportServicePipeline::safePushToQueue(
962 QSharedPointer<BoundedThreadSafeQueue<QSharedPointer<ComponentExportStatus>>> queue,
963 QSharedPointer<ComponentExportStatus> status) {
964 if (!queue || m_isCancelled.loadAcquire())
965 return false;
966
967 // 统计内存峰值
968 MemorySnapshot snapshot = status->getMemorySnapshot();
969 // 无论是否重试模式，只要当前峰值更高，就更新
970 // 由于 safePushToQueue 是在主线程调用的（通过 handle...Completed），所以是线程安全的
971 if (snapshot.totalSize > m_originalStatistics.peakMemoryUsage) {
972 m_originalStatistics.peakMemoryUsage = snapshot.totalSize;
973 }
974
975 // 快速尝试
976 if (queue->tryPush(status))
977 return true;
978
979 // 指数退避策略
980 constexpr int MAX_RETRIES = 5;
981 constexpr int BASE_DELAY_MS = 10; // 减少初始延迟到 10ms
982 constexpr int MAX_TOTAL_TIMEOUT_MS = 500; // 减少总超时到 500ms
983
984 int retryCount = 0;
985 int totalDelay = 0;
986
987 while (retryCount < MAX_RETRIES) {
988 if (m_isCancelled.loadAcquire())
989 return false;
990
991 // 防止 UI 冻结
992 QCoreApplication::processEvents(QEventLoop::ExcludeUserInputEvents);
993
994 // 尝试非阻塞推送
995 if (queue->tryPush(status)) {
996 if (retryCount > 0) {
997 qDebug() << "Queue backoff successful after" << retryCount << "retries for" << status->componentId;
998 }
999 return true;
1000 }
1001
1002 // 计算延迟
1003 int delay = BASE_DELAY_MS * (1 << retryCount);
1004 delay += QRandomGenerator::global()->bounded(0, 20); // 抖动
1005
1006 if (totalDelay + delay > MAX_TOTAL_TIMEOUT_MS) {
1007 qWarning() << "Queue push timed out after" << totalDelay << "ms for component:" << status->componentId;
1008 return false;
1009 }
1010
1011 // QThread::msleep 是阻塞当前线程（主线程）
1012 // 因为我们在循环中调用了 processEvents，所以界面不会完全死锁
1013 QThread::msleep(delay);
1014 totalDelay += delay;
1015 retryCount++;
1016 }
1017
1018 return false;
1019}

1162bool ExportServicePipeline::waitForCompletion(int timeoutMs) {
1163 if (!m_isPipelineRunning)
1164
1165 return true;
1166
1167 // 检查 QApplication 实例是否存在
1168
1169 if (!QCoreApplication::instance()) {
1170 qWarning() << "QCoreApplication instance not available, skipping graceful wait.";
1171
1172 return false;
1173 }
1174
1175 qDebug() << "Waiting for pipeline completion (timeout:" << timeoutMs << "ms)...";
1176
1177 // 确保已设置取消标志，加速 Worker 退出
1178
1179 if (!m_isCancelled.loadAcquire()) {
1180 m_isCancelled.storeRelease(1);
1181 }
1182
1183 QElapsedTimer timer;
1184
1185 timer.start();
1186
1187 // 缓存主线程指针
1188
1189 QThread* mainThread = QCoreApplication::instance()->thread();
1190
1191 // 定义一个辅助 lambda 用于安全等待线程池
1192
1193 auto safeWaitForPool = [&](QThreadPool* pool, const char* name) -> bool {
1194 if (!pool)
1195 return true;
1196
1197 while (pool->activeThreadCount() > 0) {
1198 if (timer.elapsed() > timeoutMs) {
1199 qWarning() << name << "thread pool wait timed out";
1200
1201 return false;
1202 }
1203
1204 // 尝试非阻塞等待一小段时间
1205
1206 if (pool->waitForDone(50)) {
1207 break;
1208 }
1209
1210 // 关键：仅在主线程处理事件循环，防止死锁并确保线程安全
1211
1212 if (QThread::currentThread() == mainThread) {
1213 QCoreApplication::processEvents(QEventLoop::ExcludeUserInputEvents);
1214 }
1215 }
1216
1217 return true;
1218 };
1219
1220 // 1. 等待 fetch 线程池
1221
1222 if (!safeWaitForPool(m_fetchThreadPool, "Fetch"))
1223 return false;
1224
1225 // 2. 等待 process 线程池
1226
1227 if (!safeWaitForPool(m_processThreadPool, "Process"))
1228 return false;
1229
1230 // 3. 等待 write 线程池
1231
1232 if (!safeWaitForPool(m_writeThreadPool, "Write"))
1233 return false;

1235 qDebug() << "Pipeline wait finished successfully in" << timer.elapsed() << "ms";
