17void LayerMapper::initializeLayerMapping() {
18 // 初始KiCad 图层名称
19 m_kicadLayerNames[F_Cu] = "F.Cu";
20 m_kicadLayerNames[B_Cu] = "B.Cu";
21 for (int i = 1; i <= 30; i++) {
22 m_kicadLayerNames[i] = QString("In%1.Cu").arg(i);
23 }
24 m_kicadLayerNames[F_SilkS] = "F.SilkS";
25 m_kicadLayerNames[B_SilkS] = "B.SilkS";
26 m_kicadLayerNames[F_Mask] = "F.Mask";
27 m_kicadLayerNames[B_Mask] = "B.Mask";
28 m_kicadLayerNames[F_Paste] = "F.Paste";
29 m_kicadLayerNames[B_Paste] = "B.Paste";
30 m_kicadLayerNames[F_Adhes] = "F.Adhes";
31 m_kicadLayerNames[B_Adhes] = "B.Adhes";
32 m_kicadLayerNames[Edge_Cuts] = "Edge.Cuts";
33 m_kicadLayerNames[F_CrtYd] = "F.CrtYd";
34 m_kicadLayerNames[B_CrtYd] = "B.CrtYd";
35 m_kicadLayerNames[F_Fab] = "F.Fab";
36 m_kicadLayerNames[B_Fab] = "B.Fab";
37 m_kicadLayerNames[Dwgs_User] = "Dwgs.User";
38 m_kicadLayerNames[Cmts_User] = "Cmts.User";
39 m_kicadLayerNames[Eco1_User] = "Eco1.User";
40 m_kicadLayerNames[Eco2_User] = "Eco2.User";
41 m_kicadLayerNames[Margin] = "Margin";
42 for (int i = 1; i <= 9; i++) {
43 m_kicadLayerNames[User_1 + i - 1] = QString("User.%1").arg(i);
44 }
45
46 // 初始化嘉立创图层 ID -> KiCad 图层 ID 映射
47 // 一、电气信号层
48 m_layerIdMapping[1] = F_Cu; // TopLayer -> F.Cu
49 m_layerIdMapping[2] = B_Cu; // BottomLayer -> B.Cu
50
51 // 内电Inner1~Inner32 (ID 21-52) -> In1.Cu~In32.Cu
52 for (int i = 0; i < 32; i++) {
53 m_layerIdMapping[21 + i] = In1_Cu + i;
54 }
55
56 // 二、丝印层
57 m_layerIdMapping[3] = F_SilkS; // TopSilkLayer -> F.SilkS
58 m_layerIdMapping[4] = B_SilkS; // BottomSilkLayer -> B.SilkS
59
60 // 三、阻焊层
61 m_layerIdMapping[7] = F_Mask; // TopSolderMaskLayer -> F.Mask
62 m_layerIdMapping[8] = B_Mask; // BottomSolderMaskLayer -> B.Mask
63
64 // 四、助焊层
65 m_layerIdMapping[5] = F_Paste; // TopPasteMaskLayer -> F.Paste
66 m_layerIdMapping[6] = B_Paste; // BottomPasteMaskLayer -> B.Paste
67
68 // 五、机械与结构
69 m_layerIdMapping[10] = Edge_Cuts; // BoardOutLine -> Edge.Cuts
70 m_layerIdMapping[15] = F_Fab; // Mechanical -> F.Fab (顶层装配)
71 m_layerIdMapping[99] = F_CrtYd; // ComponentShapeLayer -> F.CrtYd (元件占位)
72 m_layerIdMapping[100] = F_Fab; // LeadShapeLayer -> F.Fab (引脚形状)
73 m_layerIdMapping[101] = F_SilkS; // ComponentPolarityLayer -> F.SilkS (极性标
74
75 // 六、装配与文档
76 m_layerIdMapping[13] = F_Fab; // TopAssembly -> F.Fab
77 m_layerIdMapping[14] = B_Fab; // BottomAssembly -> B.Fab
78 m_layerIdMapping[12] = Dwgs_User; // Document -> Dwgs.User
79
80 // 七、特殊功能层
81 m_layerIdMapping[9] = Dwgs_User; // Ratlines -> Dwgs.User (飞线)
82 m_layerIdMapping[11] = F_Cu; // Multi-Layer -> F.Cu (通孔焊盘，实际在所有层)
83 m_layerIdMapping[19] = Dwgs_User; // 3DModel -> Dwgs.User (3D 模型引用)
84 m_layerIdMapping[102] = Dwgs_User; // Hole -> Dwgs.User (非金属化
85 m_layerIdMapping[103] = Cmts_User; // DRCError -> Cmts.User (DRC 错误)
86
87 // 初始化嘉立创图层名称 -> KiCad 图层 ID 映射
88 // 内层索引计算：In.(n-14).Cu
89 // 例如：Inner1 -> In1.Cu, Inner2 -> In2.Cu...
90 m_layerNameMapping["TopLayer"] = F_Cu;
91 m_layerNameMapping["BottomLayer"] = B_Cu;
92 for (int i = 1; i <= 32; i++) {
93 // 使用清晰的公式：In1_Cu + (i - 1)
94 // 与 LCKiConverter 的 In.(n-14).Cu 更直观
95 m_layerNameMapping[QString("Inner%1").arg(i)] = In1_Cu + (i - 1);
96 }
97 m_layerNameMapping["TopSilkLayer"] = F_SilkS;
98 m_layerNameMapping["BottomSilkLayer"] = B_SilkS;
99 m_layerNameMapping["TopSolderMaskLayer"] = F_Mask;
100 m_layerNameMapping["BottomSolderMaskLayer"] = B_Mask;
101 m_layerNameMapping["TopPasteMaskLayer"] = F_Paste;
102 m_layerNameMapping["BottomPasteMaskLayer"] = B_Paste;
103 m_layerNameMapping["BoardOutLine"] = Edge_Cuts;
104 m_layerNameMapping["Mechanical"] = F_Fab;
105 m_layerNameMapping["ComponentShapeLayer"] = F_CrtYd;
106 m_layerNameMapping["LeadShapeLayer"] = F_Fab;
107 m_layerNameMapping["ComponentPolarityLayer"] = F_SilkS;
108 m_layerNameMapping["TopAssembly"] = F_Fab;
109 m_layerNameMapping["BottomAssembly"] = B_Fab;
110 m_layerNameMapping["Document"] = Dwgs_User;
111 m_layerNameMapping["Ratlines"] = Dwgs_User;
112 m_layerNameMapping["Multi-Layer"] = F_Cu;
113 m_layerNameMapping["3DModel"] = Dwgs_User;
114 m_layerNameMapping["Hole"] = Dwgs_User;
115 m_layerNameMapping["DRCError"] = Cmts_User;
116
117 qDebug() << "LayerMapper initialized with" << m_layerIdMapping.size() << "layer mappings";
118}

174bool LayerMapper::isMechanicalLayer(int kicadLayerId) {
175 // 机械层：Edge.Cuts (44), F.CrtYd (45), B.CrtYd (46), F.Fab (47), B.Fab (48)
176 // Dwgs.User (49), Cmts.User (50), Eco1.User (51), Eco2.User (52), Margin (53)
177 return (kicadLayerId >= Edge_Cuts && kicadLayerId <= Margin) || (kicadLayerId >= User_1 && kicadLayerId <= User_9);
