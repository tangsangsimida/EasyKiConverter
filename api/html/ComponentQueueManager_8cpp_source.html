70void ComponentQueueManager::checkAndProcessNext() {
71 if (!m_isRunning) {
72 m_checkTimer->stop();
73 return;
74 }
75
76 int activeCount = m_pendingComponents.size();
77
78 // 如果所有槽位都满了，启用定时器作为安全网（防止某些请求完成但信号未触发的情况）
79 // 否则禁用定时器（正常情况下 requestCompleted 会直接触发处理）
80 if (activeCount >= m_maxConcurrentRequests && !m_requestQueue.isEmpty()) {
81 if (!m_checkTimer->isActive()) {
82 m_checkTimer->start();
83 }
84 } else {
85 m_checkTimer->stop();
86 }
87
88 while (activeCount < m_maxConcurrentRequests && !m_requestQueue.isEmpty()) {
89 QString componentId = m_requestQueue.takeFirst();
90
91 if (m_pendingComponents.contains(componentId)) {
92 qDebug() << "Component" << componentId << "already pending, skipping";
93 continue;
94 }
95
96 m_pendingComponents.insert(componentId);
97 activeCount++;
98
99 qDebug() << "ComponentQueueManager: Request ready for:" << componentId << "(Active:" << activeCount << "/"
100 << m_maxConcurrentRequests << "Remaining:" << m_requestQueue.size() << ")";
101
102 emit requestReady(componentId);
103 }
104
105 if (m_requestQueue.isEmpty() && activeCount == 0) {
106 qDebug() << "ComponentQueueManager: All components processed. Queue empty.";
107 m_checkTimer->stop();
108 emit queueEmpty();
109 stop();
110 }
111}
