62bool isDebugMode(const EasyKiConverter::CommandLineParser& parser) {
63 // 优先检查命令行参数
64 if (parser.isDebugMode()) {
65 return true;
66 }
67
68 // 检查环境变量（向后兼容）
69 if (qEnvironmentVariableIsSet("EASYKICONVERTER_DEBUG_MODE")) {
70 QString debugValue = qEnvironmentVariable("EASYKICONVERTER_DEBUG_MODE", "false").toLower();
71 return (debugValue == "true" || debugValue == "1" || debugValue == "yes");
72 }
73 return false;
74}

86void applyCacheConfiguration(const EasyKiConverter::CommandLineParser& parser) {
87 auto* configService = EasyKiConverter::ConfigService::instance();
88 auto* cacheService = EasyKiConverter::ComponentCacheService::instance();
89

99
100 cacheService->setCacheDir(cacheDir);

119#ifdef _WIN32

203 auto fileAppender =
204

207 fileAppender->setFormatter(QSharedPointer<PatternFormatter>::create(PatternFormatter::simplePattern()));

210
211 LOG_INFO(LogModule::Core,

213 logLevelStr.isEmpty() ? "default" : logLevelStr,

215
216 } else {
217 LOG_INFO(LogModule::Core,
218 "日志系统已初始化 - 级别: {} (仅控制台输出)",

222 // 安装 Qt 日志适配器（将 qDebug/qWarning/qCritical 重定向到新系统）

250 }

252 return appDir.chopped(4); // 去掉 /bin
253 }

320bool applyTaskbarIcon(bool darkMode, QWindow* window = nullptr, const char* context = "应用程序") {
321 const QStringList iconPaths = themedTaskbarIconPaths(darkMode);
322
323 for (const QString& iconPath : iconPaths) {
324 const bool canTryIconTheme = !iconPath.contains('/');
325 if (!canTryIconTheme && !QFile::exists(iconPath)) {
326 continue;
327 }
328
329 const QIcon icon = canTryIconTheme ? QIcon::fromTheme(iconPath) : QIcon(iconPath);
330 if (icon.isNull()) {
331 continue;
332 }
333
334 QGuiApplication::setWindowIcon(icon);
335 if (auto* application = qobject_cast<QApplication*>(QCoreApplication::instance())) {
336 application->setWindowIcon(icon);
337 }
338 if (window) {
339 window->setIcon(icon);
340 }
341
342 qDebug() << context << "任务栏图标已设置:" << iconPath << "darkMode=" << darkMode
343 << "availableSizes=" << icon.availableSizes();
344 return true;
345 }
346
347 qWarning() << context << "任务栏图标设置失败，darkMode=" << darkMode << "尝试路径数:" << iconPaths.size();
348 return false;
349}

353int main(int argc, char* argv[]) {
354 QSurfaceFormat surfaceFormat = QSurfaceFormat::defaultFormat();
355 surfaceFormat.setAlphaBufferSize(8);
356 QSurfaceFormat::setDefaultFormat(surfaceFormat);
357 QQuickWindow::setDefaultAlphaBuffer(true);
358
359#ifdef _WIN32
360 // Windows 透明无边框窗口在 Vulkan 后端下可能出现圆角透明区域被合成为黑色。
361 // Direct3D 11 是 Qt Quick 在 Windows 上更稳定的默认路径。
362 QQuickWindow::setGraphicsApi(QSGRendererInterface::Direct3D11);
363 qInfo() << "已将 Windows 渲染后端设置为 Direct3D 11，并启用 alpha buffer";
364#else
365 // 非 Windows 平台保留 Vulkan 优先策略。
366 QQuickWindow::setGraphicsApi(QSGRendererInterface::Vulkan);
367 qInfo() << "已将渲染后端优先设置为 Vulkan";
368#endif
369
370 // 设置 stdout 为无缓冲模式，确保日志颜色正常显示
371 setvbuf(stdout, nullptr, _IONBF, 0);
372 setvbuf(stderr, nullptr, _IONBF, 0);
373
374 // 在 QApplication 构造函数之前检查命令行参数
375 bool showHelp = false;
376 bool showVersion = false;
377 bool cliModeRequested = false;
378 for (int i = 1; i < argc; ++i) {
379 QString arg = QString::fromLocal8Bit(argv[i]);
380 // 支持 Unix 风格（-h, --help）和 Windows 风格（/h, /help, /?, ?）的参数
381 if (arg == "--help" || arg == "-h" || arg == "/h" || arg == "/help" || arg == "/?" || arg == "?") {
382 showHelp = true;
383 } else if (arg == "--version" || arg == "-v" || arg == "/v" || arg == "/version") {
384 showVersion = true;
385 }
386 }
387
388 // 创建命令行参数解析器（在 QApplication 之前，以支持纯 CLI 模式）
389 EasyKiConverter::CommandLineParser cmdParser(argc, argv);
390
391 // 早期解析：检查是否为纯 CLI 模式（不需要 GUI）
392 bool parsed = cmdParser.parse();
393 bool isCompletionMode = parsed && cmdParser.isCompletionRequested();
394 bool isCompleteMode = parsed && cmdParser.isCompleteRequested();
395 bool isCliMode = parsed && cmdParser.isCliMode();
396 bool isConvertCommand = parsed && cmdParser.hasConvertCommand();
397
398 // 纯 CLI 模式：--completion, --complete, --help, --version, convert 子命令
399 // 这些模式不需要 QApplication，只需要 QCoreApplication
400 cliModeRequested = isCompletionMode || isCompleteMode || showHelp || showVersion || isCliMode || isConvertCommand;
401
402 if (cliModeRequested) {
403 // 纯 CLI 模式：使用 QCoreApplication
404 QCoreApplication app(argc, argv);
405 app.setApplicationName("EasyKiConverter");
406 app.setApplicationVersion(EasyKiConverter::APP_VERSION);
407 app.setOrganizationName("EasyKiConverter");
408 app.setOrganizationDomain("easykiconverter.com");
409
410 if (showHelp) {
411 QString helpText = cmdParser.helpText();
412 QTextStream consoleOut(stdout);
413 consoleOut << helpText;
414 return 0;
415 }
416
417 if (showVersion) {
418 QString versionText = "EasyKiConverter " + app.applicationVersion() + "\n";
419 QTextStream consoleOut(stdout);
420 consoleOut << versionText;
421 return 0;
422 }
423
424 if (!parsed) {
425 QTextStream err(stderr);
426 err << QCoreApplication::translate("main", "错误: 无效的命令行参数") << "\n";
427 err << cmdParser.helpText();
428 return 1;
429 }
430
431 // 尽早初始化语言管理器，使后续验证错误消息能正确翻译
432 {
433 auto* langManager = EasyKiConverter::LanguageManager::instance();
434 QString cliLang = cmdParser.language();
435 if (!cliLang.isEmpty() && (cliLang == "zh_CN" || cliLang == "en")) {
436 langManager->setLanguage(cliLang, /*force=*/true);
437 }
438 }
439
440 if (!cmdParser.validate()) {
441 QTextStream err(stderr);
442 err << QCoreApplication::translate("main", "错误: 参数值无效") << "\n";
443 err << cmdParser.validationError() << "\n\n";
444 err << cmdParser.helpText();
445 return 1;
446 }
447
448 const QString configFilePath = resolveConfigFilePath(cmdParser);
449 EasyKiConverter::TempFileManager::recoverIncompleteTransactions();
450 EasyKiConverter::ConfigService::instance()->loadConfig(configFilePath);
451 applyCacheConfiguration(cmdParser);
452
453 // 处理补全请求
454 if (isCompletionMode) {
455 QString shell = cmdParser.completionShell();
456 auto shellType = EasyKiConverter::CompletionGenerator::parseShell(shell);
457 QTextStream out(stdout);
458 out << EasyKiConverter::CompletionGenerator::generate(shellType);
459 return 0;
460 }
461
462 // 处理动态补全请求
463 if (isCompleteMode) {
464 QString completeType = cmdParser.completeType();
465 QTextStream out(stdout);
466
467 if (completeType == "lcsc-id") {
468 // 输出缓存中的 LCSC ID 列表
469 auto* cache = EasyKiConverter::ComponentCacheService::instance();
470 if (cache) {
471 QStringList ids = cache->getCachedComponentIds();
472 for (const QString& id : ids) {
473 out << id << "\n";
474 }
475 }
476 }
477
478 return 0;
479 }
480
481 // CLI 转换模式
482 if (isCliMode) {
483#ifdef _WIN32
484 // 尝试附加到父进程的控制台
485 if (!AttachConsole(ATTACH_PARENT_PROCESS)) {
486 AllocConsole();
487 SetConsoleTitleA("EasyKiConverter - CLI");
488 }
489 (void)reopenConsoleStream(stdout, "CONOUT$", "w", "stdout");
490 (void)reopenConsoleStream(stderr, "CONOUT$", "w", "stderr");
491#endif
492 // 初始化日志系统（CLI 模式也需要日志）
493 bool debugMode = isDebugMode(cmdParser);
494 setupLogging(debugMode, cmdParser.logLevel(), cmdParser.logFile(), cmdParser.isSyncLogging());
495
496 // 创建并执行 CLI 转换器
497 EasyKiConverter::CliConverter cliConverter(cmdParser);
498 bool success = cliConverter.execute();
499
500 if (!success) {
501 QTextStream err(stderr);
502 err << QCoreApplication::translate("main", "错误: ") << cliConverter.errorMessage() << "\n";
503 }
504
505 // 清理日志
506 EasyKiConverter::QtLogAdapter::uninstall();
507 auto* logger = EasyKiConverter::Logger::instance();
508 if (logger) {
509 logger->flush();
510 logger->close();
511 }
512
513 return success ? 0 : 1;
514 }
515
516 // convert 命令存在但无效子命令（如 easykiconverter convert abc）
517 if (isConvertCommand && !isCliMode) {
518 QTextStream err(stderr);
519 err << QCoreApplication::translate("main", "错误: 无效的 convert 子命令") << "\n";
520 err << QCoreApplication::translate("main", "有效的子命令: bom, component, batch") << "\n\n";
521 err << cmdParser.cliHelpText();
522 return 1;
523 }
524
525 // 到达此处的唯一可能是 isConvertCommand 但 isCliMode=false（无效子命令）
526 // 该路径已在上面返回 1，此处不会执行到此
527 // 防御性断言：确保所有 CLI 模式请求都在上面正确处理
528 Q_ASSERT_X(false, "main", "Unhandled CLI mode path");
529 return 1;
530 }
531
532 // GUI 模式：使用完整的 QApplication
533 // 设置 QML 样式为 Basic，以消除原生样式自定义警告
534 QQuickStyle::setStyle("Basic");
535 // High DPI 适配：使用 PassThrough 策略避免分数缩放下的模糊和舍入误差
536 QGuiApplication::setHighDpiScaleFactorRoundingPolicy(Qt::HighDpiScaleFactorRoundingPolicy::PassThrough);
537
538 QApplication app(argc, argv);
539
540 // 自动检测并记录实际生效的渲染后端
541 QTimer::singleShot(0, []() {
542 QQuickWindow tempWindow;
543 const char* apiName = "Unknown";

547 break;

550 break;
551 case QSGRendererInterface::OpenVG:
552 apiName = "OpenVG";
553 break;
554 case QSGRendererInterface::OpenGL:
555 apiName = "OpenGL";
556 break;
557 case QSGRendererInterface::Direct3D11:
558 apiName = "Direct3D 11";
559 break;
560 case QSGRendererInterface::Vulkan:
561 apiName = "Vulkan";
562 break;
563 case QSGRendererInterface::Metal:
564 apiName = "Metal";
565 break;
566 case QSGRendererInterface::Null:
567 apiName = "Null (headless)";
568 break;
569 case QSGRendererInterface::Direct3D12:
570 apiName = "Direct3D 12 (DX12)";
571 break;
572 default:
573 apiName = "Other/Unknown";
574 break;
575 }
576 qInfo() << "当前实际运行的渲染 API:" << apiName;
577 });
578
579 // 设置应用程序信息
580 app.setApplicationName("EasyKiConverter");
581 app.setApplicationVersion(EasyKiConverter::APP_VERSION);
582 app.setOrganizationName("EasyKiConverter");
583 app.setOrganizationDomain("easykiconverter.com");
584
585 // 注意：所有 CLI 模式处理（--help, --version, --completion, --complete, convert 子命令）
586 // 已在 QCoreApplication 阶段处理完毕。如果到达此处，说明是纯 GUI 启动。
587 // 命令行参数已在 CLI 路径解析并验证，无需重复解析。
588
589 // 检查调试模式（命令行参数优先，环境变量向后兼容）
590 bool debugMode = isDebugMode(cmdParser);
591
592 // 初始化日志系统（在 QApplication 创建后立即初始化）
593 setupLogging(debugMode, cmdParser.logLevel(), cmdParser.logFile(), cmdParser.isSyncLogging());
594 EasyKiConverter::TempFileManager::recoverIncompleteTransactions();
595
596 // 尝试设置应用程序图标（使用 QGuiApplication::setWindowIcon 确保可靠性）
597 QString appImagePath = qgetenv("APPIMAGE");
598 QString appDir = resolveRuntimeAppDir(appImagePath);
599
600 qDebug() << "应用目录:" << appDir;
601
602 // 处理命令行参数 - 配置文件路径
603 QString configFilePath = resolveConfigFilePath(cmdParser);
604 if (!cmdParser.configFile().isEmpty()) {
