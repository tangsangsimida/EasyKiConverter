40bool isDebugMode(const EasyKiConverter::CommandLineParser& parser) {
41 // 优先检查命令行参数
42 if (parser.isDebugMode()) {
43 return true;
44 }
45
46 // 检查环境变量（向后兼容）
47 if (qEnvironmentVariableIsSet("EASYKICONVERTER_DEBUG_MODE")) {
48 QString debugValue = qEnvironmentVariable("EASYKICONVERTER_DEBUG_MODE", "false").toLower();
49 return (debugValue == "true" || debugValue == "1" || debugValue == "yes");
50 }
51 return false;
52}
