196NetworkResult NetworkClient::executeRequest(const QUrl& url,
197 const QByteArray& body,
198 ResourceType resourceType,
199 const RetryPolicy& policy) {
200 // 同步路径：在创建 request 前做 URL 校验，避免白名单拒绝后死锁
201 // （QWaitCondition::wait() 不处理事件循环，queued 完成信号无法到达）
202 {
203 QString errorMsg;
204 if (!UrlUtils::isAllowedUrl(url, resourceType, &errorMsg)) {
205 qWarning() << "NetworkClient: URL rejected by whitelist (sync):" << errorMsg;
206 NetworkResult result;
207 populateDiagnostic(result.diagnostic, url, resourceType);
208 result.error = errorMsg;
209 result.success = false;
210 result.diagnostic.errorType = NetworkErrorType::Other;
211 result.diagnostic.errorMessage = errorMsg;
212 return result;
213 }
214 }
215
216 auto* request = enqueueAsyncRequest(url, body, resourceType, policy);
217 if (!request) {
218 NetworkResult result;
219 populateDiagnostic(result.diagnostic, url, resourceType);
220 result.error = QStringLiteral("Failed to create async request");
221 result.success = false;
222 result.diagnostic.errorType = NetworkErrorType::Other;
223 result.diagnostic.errorMessage = result.error;
224 return result;
225 }
226
227 auto context = std::make_shared<BlockingRequestContext>();
228 QObject::connect(
229 request,
230 &AsyncNetworkRequest::finished,
231 request,
232 [context](const NetworkResult& result) { context->complete(result); },
233 Qt::DirectConnection);
234
235 const NetworkResult result = context->wait();
236 QMetaObject::invokeMethod(request, &QObject::deleteLater, Qt::QueuedConnection);
237 return result;
238}

240AsyncNetworkRequest* NetworkClient::enqueueAsyncRequest(const QUrl& url,
241 const QByteArray& body,
242 ResourceType resourceType,
243 const RetryPolicy& policy) {
244 if (!m_asyncNetworkManager) {
245 return nullptr;
246 }
247
248 // URL 白名单校验：拒绝非预期的 scheme 或 host
249 {
250 QString errorMsg;
251 if (!UrlUtils::isAllowedUrl(url, resourceType, &errorMsg)) {
252 qWarning() << "NetworkClient: URL rejected by whitelist:" << errorMsg;
253 // 返回已完成的失败 request，避免调用方需处理 nullptr
254 NetworkResult failResult;
255 failResult.success = false;
256 failResult.wasCancelled = false;
257 failResult.error = errorMsg;
258 failResult.diagnostic.url = url.toString();
259 failResult.diagnostic.host = url.host();
260 failResult.diagnostic.resourceType = resourceType;
261 failResult.diagnostic.profileName = RequestProfiles::fromType(resourceType).name;
262 failResult.diagnostic.errorType = NetworkErrorType::Other;
263 failResult.diagnostic.errorMessage = errorMsg;
264 return AsyncNetworkRequest::createFinished(failResult);
265 }
266 }
267
268 auto* request = new AsyncNetworkRequest(url, m_asyncNetworkManager, resourceType, policy, body, nullptr);
269 request->moveToThread(&m_networkThread);
270
271 const RequestProfile profile = RequestProfiles::fromType(resourceType);
272 {
273 QMutexLocker locker(&m_asyncQueueMutex);
274 PendingAsyncRequest pending;
275 pending.request = request;
276 pending.resourceType = resourceType;
277 pending.priority = profile.priority;
278 pending.sequence = m_asyncSequence++;
279 pending.enqueuedAtMs = QDateTime::currentMSecsSinceEpoch();
280 m_pendingAsyncRequests.append(pending);
281 m_liveAsyncRequests.append(request);
282 updateStatsForEnqueuedRequest(resourceType);
283 }
284
285 QObject::connect(
286 request,
287 &AsyncNetworkRequest::finished,
288 this,
289 [this, request, resourceType](const NetworkResult& result) {
290 updateStatsForCompletedRequest(resourceType, result);
291 {
292 QMutexLocker locker(&m_asyncQueueMutex);
293 m_liveAsyncRequests.removeAll(request);
294 }
295 onAsyncRequestFinished(resourceType);
296 },
297 Qt::QueuedConnection);
298
299 scheduleAsyncPump();
300 return request;
301}

318void NetworkClient::pumpAsyncQueue() {
319 while (true) {
320 QPointer<AsyncNetworkRequest> requestToStart;
321 ResourceType resourceType = ResourceType::Unknown;
322
323 {
324 QMutexLocker locker(&m_asyncQueueMutex);
325 int bestIndex = -1;
326 int bestPriority = std::numeric_limits<int>::max();
327 quint64 bestSequence = std::numeric_limits<quint64>::max();
328
329 for (int i = m_pendingAsyncRequests.size() - 1; i >= 0; --i) {
330 const PendingAsyncRequest& pending = m_pendingAsyncRequests.at(i);
331 if (!pending.request || pending.request->isFinished()) {
332 m_pendingAsyncRequests.removeAt(i);
333 }
334 }
335
336 for (int i = 0; i < m_pendingAsyncRequests.size(); ++i) {
337 const PendingAsyncRequest& pending = m_pendingAsyncRequests.at(i);
338 if (!pending.request || pending.request->isFinished()) {
339 continue;
340 }
341
342 const RequestProfile profile = RequestProfiles::fromType(pending.resourceType);
343 const int activeCount = m_activeAsyncRequestsByType.value(static_cast<int>(pending.resourceType), 0);
344 if (activeCount >= profile.maxConcurrent) {
345 continue;
346 }
347
348 if (pending.priority < bestPriority ||
349 (pending.priority == bestPriority && pending.sequence < bestSequence)) {
350 bestIndex = i;
351 bestPriority = pending.priority;
352 bestSequence = pending.sequence;
353 }
354 }
355
356 if (bestIndex < 0) {
357 m_asyncPumpScheduled = false;
358 break;
359 }
360
361 PendingAsyncRequest pending = m_pendingAsyncRequests.takeAt(bestIndex);
362 requestToStart = pending.request;
363 resourceType = pending.resourceType;
364 m_activeAsyncRequestsByType[static_cast<int>(resourceType)] =
365 m_activeAsyncRequestsByType.value(static_cast<int>(resourceType), 0) + 1;
366 const qint64 queueDelayMs = qMax<qint64>(0, QDateTime::currentMSecsSinceEpoch() - pending.enqueuedAtMs);
367 updateStatsForDequeuedRequest(resourceType, queueDelayMs);
368 updateStatsForStartedRequest(resourceType);
369 }
370
371 if (!requestToStart) {
372 onAsyncRequestFinished(resourceType);
373 continue;
374 }
375
376 QMetaObject::invokeMethod(
377 requestToStart,
378 [requestToStart]() {
379 if (requestToStart) {
380 requestToStart->start();
381 }
382 },
383 Qt::QueuedConnection);
384 }
385}

388 QList<QPointer<AsyncNetworkRequest>> requestsToCancel;
389 {
390 QMutexLocker locker(&m_asyncQueueMutex);
391 requestsToCancel = m_liveAsyncRequests;
392 m_pendingAsyncRequests.clear();
393 m_asyncPumpScheduled = false;
394 }
395
396 for (const QPointer<AsyncNetworkRequest>& request : std::as_const(requestsToCancel)) {
397 if (!request || request->isFinished()) {
398 continue;
399 }
400
401 if (request->thread() == QThread::currentThread()) {
402 request->cancel();
403 } else if (QThread::currentThread() == &m_networkThread) {
404 request->cancel();
405 } else {
406 const Qt::ConnectionType connectionType =

408 QMetaObject::invokeMethod(request, &AsyncNetworkRequest::cancel, connectionType);
409 }

474void NetworkClient::updateStatsForCompletedRequest(ResourceType resourceType, const NetworkResult& result) {
475 QMutexLocker locker(&m_asyncQueueMutex);
476 const int key = static_cast<int>(resourceType);
477 MutableResourceStats& stats = m_resourceStats[key];
478 const RequestProfile profile = RequestProfiles::fromType(resourceType);
479 stats.snapshot.resourceType = resourceType;
480 stats.snapshot.profileName = profile.name;
481 stats.snapshot.maxConcurrent = profile.maxConcurrent;
482 ++stats.snapshot.completedRequests;
483 stats.snapshot.retryAttempts += static_cast<quint64>(qMax(0, result.retryCount));
484 stats.snapshot.lastLatencyMs = result.elapsedMs;
485 stats.totalLatencyMs += result.elapsedMs;
486 stats.snapshot.averageLatencyMs =
487 stats.snapshot.completedRequests > 0
488 ? (stats.totalLatencyMs / static_cast<qint64>(stats.snapshot.completedRequests))
489 : 0;
490 stats.snapshot.maxLatencyMs = qMax(stats.snapshot.maxLatencyMs, result.elapsedMs);
491
492 if (result.wasCancelled || result.diagnostic.wasCanceled) {
493 ++stats.snapshot.cancelledRequests;
494 } else if (result.success) {
495 ++stats.snapshot.succeededRequests;
496 } else {
497 ++stats.snapshot.failedRequests;
498 }
499
500 if (result.diagnostic.errorType == NetworkErrorType::Timeout) {
501 ++stats.snapshot.timeoutRequests;
502 }
503 if (result.diagnostic.wasRateLimited || result.statusCode == 429) {
504 ++stats.snapshot.rateLimitedRequests;
505 }
506 refreshDynamicStatsLocked(resourceType);
507}

509void NetworkClient::refreshDynamicStatsLocked(ResourceType resourceType) {
510 const int key = static_cast<int>(resourceType);
511 MutableResourceStats& stats = m_resourceStats[key];
512 int queuedCount = 0;
513 for (const PendingAsyncRequest& pending : std::as_const(m_pendingAsyncRequests)) {
514 if (pending.request && !pending.request->isFinished() && pending.resourceType == resourceType) {
515 ++queuedCount;
516 }
517 }
518
519 stats.snapshot.queuedRequests = queuedCount;
520 stats.snapshot.activeRequests = m_activeAsyncRequestsByType.value(key, 0);
521 stats.snapshot.peakQueuedRequests = qMax(stats.snapshot.peakQueuedRequests, queuedCount);
522 stats.snapshot.peakActiveRequests = qMax(stats.snapshot.peakActiveRequests, stats.snapshot.activeRequests);
523}

525int NetworkClient::calculateRetryDelay(int retryCount, const RetryPolicy& policy) {
526 if (retryCount >= static_cast<int>(policy.delays.size())) {
527 return policy.delays.back();
528 }
529 int baseDelay = policy.delays[retryCount];
530
531 int jitterRange = static_cast<int>(baseDelay * policy.jitterFactor);
532 int jitter = QRandomGenerator::global()->bounded(-jitterRange, jitterRange + 1);
533 return baseDelay + jitter;
534}

536bool NetworkClient::shouldRetry(int statusCode,
537 QNetworkReply::NetworkError error,
538 int retryCount,
539 const RetryPolicy& policy) {
540 if (retryCount >= policy.maxRetries) {
541 return false;
542 }
543
544 if (policy.retryableStatusCodes.contains(statusCode)) {
545 return true;
546 }
547
548 switch (error) {
549 case QNetworkReply::TimeoutError:
550 case QNetworkReply::TemporaryNetworkFailureError:
551 case QNetworkReply::NetworkSessionFailedError:
552 case QNetworkReply::UnknownNetworkError:
553 case QNetworkReply::RemoteHostClosedError:
554 case QNetworkReply::HostNotFoundError:
555 case QNetworkReply::ConnectionRefusedError:
556 case QNetworkReply::ProxyTimeoutError:
557 return true;
558 default:
559 return false;
560 }
561}
