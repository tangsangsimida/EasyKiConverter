487QList<QStringList> EasyedaSymbolImporter::parsePinDataString(const QString& pinData) const {
488 QList<QStringList> result;
489 QStringList segments;
490 int start = 0;
491 int pos = 0;
492
493 while (pos < pinData.length()) {
494 if (pinData[pos] == '^' && pos + 1 < pinData.length() && pinData[pos + 1] == '^') {
495 segments.append(pinData.mid(start, pos - start));
496 pos += 2;
497 start = pos;
498 continue;
499 }
500
501 pos++;
502 }
503
504 if (start < pinData.length()) {
505 segments.append(pinData.mid(start));
506 }
507
508 for (const QString& segment : segments) {
509 QStringList subSegments = segment.split("~", Qt::KeepEmptyParts);
510 result.append(subSegments);
511 }
512
513 return result;
514}
