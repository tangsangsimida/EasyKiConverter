39QString extractComponentCode(const QJsonObject& product) {
40 static const QStringList directKeys = {QStringLiteral("component_code"),
41 QStringLiteral("productCode"),
42 QStringLiteral("product_code"),
43 QStringLiteral("productNo"),
44 QStringLiteral("product_no"),
45 QStringLiteral("number"),
46 QStringLiteral("code"),
47 QStringLiteral("lcscPart"),
48 QStringLiteral("lcscPartNumber")};
49
50 for (const QString& key : directKeys) {
51 const QString value = product.value(key).toString().trimmed();
52 if (!value.isEmpty()) {
53 return value.toUpper();
54 }
55 }
56
57 const QJsonObject deviceInfo = product.value(QStringLiteral("device_info")).toObject();
58 if (!deviceInfo.isEmpty()) {
59 for (const QString& key : directKeys) {
60 const QString value = deviceInfo.value(key).toString().trimmed();
61 if (!value.isEmpty()) {
62 return value.toUpper();
63 }
64 }
65
66 const QJsonObject attributes = deviceInfo.value(QStringLiteral("attributes")).toObject();
67 static const QStringList attributeKeys = {QStringLiteral("LCSC Part"),
68 QStringLiteral("LCSC Part #"),
69 QStringLiteral("LCSC Part Number"),
70 QStringLiteral("Part Number")};
71 for (const QString& key : attributeKeys) {
72 const QString value = attributes.value(key).toString().trimmed();
73 if (!value.isEmpty()) {
74 return value.toUpper();
75 }
76 }
77 }
78
79 return QString();
80}

82QJsonObject selectBestProductForComponent(const QString& componentId, const QJsonArray& productList) {
83 const QString normalizedId = componentId.trimmed().toUpper();
84 for (const QJsonValue& value : productList) {
85 const QJsonObject product = value.toObject();
86 if (extractComponentCode(product) == normalizedId) {
87 return product;
88 }
89 }
90
91 // 不再 fallback 到第一个产品，如果没有精确匹配则返回空对象
92 // 原因：之前的 fallback 逻辑会导致显示错误元器件的预览图，造成用户混淆
93 return QJsonObject();
94}

203bool LcscImageService::tryLoadCachedPreviewImages(const QString& componentId) {
204 ComponentCacheService* cache = ComponentCacheService::instance();
205 if (!cache) {
206 return false;
207 }
208
209 int cachedCount = 0;
210 for (int i = 0; i < MAX_IMAGES_PER_COMPONENT; ++i) {
211 QString path = cache->previewImagePath(componentId, i);
212 if (QFileInfo::exists(path)) {
213 cachedCount++;
214 }
215 }
216
217 if (cachedCount == 0) {
218 qDebug() << "LcscImageService: No cached preview images for" << componentId;
219 return false;
220 }
221
222 qDebug() << "LcscImageService: Found" << cachedCount << "cached preview images for" << componentId
223 << ", scheduling async cache load...";
224
225 QTimer::singleShot(0, this, [this, componentId, cache]() { loadCachedPreviewImagesAsync(componentId, cache); });

229void LcscImageService::loadCachedPreviewImagesAsync(const QString& componentId, ComponentCacheService* cache) {
230 if (m_isCancelled) {
231 return;
232 }
233
234 QList<QPair<int, QString>> pathsToLoad;
235 for (int i = 0; i < MAX_IMAGES_PER_COMPONENT; ++i) {
236 QString path = cache->previewImagePath(componentId, i);
237 if (QFileInfo::exists(path)) {
238 pathsToLoad.append({i, path});
239 }
240 }
241
242 if (pathsToLoad.isEmpty()) {
243 qDebug() << "LcscImageService: No cached images found for" << componentId;
244 m_requestedComponents.remove(componentId);
245 checkDownloadCompletion(componentId);
246 return;
247 }
248
249 m_expectedCounts[componentId] = pathsToLoad.size();
250 m_downloadCounts[componentId] = 0;
251
252 auto componentImageData = QSharedPointer<QMap<int, QByteArray>>::create();
253 auto loadedCount = QSharedPointer<int>::create(0);
254 const int totalImages = pathsToLoad.size();
255
256 for (const auto& [imageIndex, path] : pathsToLoad) {
257 QFuture<QByteArray> future = QtConcurrent::run(m_cacheThreadPool, [path]() {
258 QFile file(path);
259 if (file.open(QIODevice::ReadOnly)) {
260 return file.readAll();
261 }
262 return QByteArray();
263 });
264
265 auto* watcher = new QFutureWatcher<QByteArray>(this);
266 connect(watcher,
267 &QFutureWatcher<QByteArray>::finished,
268 this,
269 [this, watcher, componentId, imageIndex, componentImageData, loadedCount, totalImages]() {
270 QByteArray imageData = watcher->result();
271 if (!imageData.isEmpty()) {
272 (*componentImageData)[imageIndex] = imageData;
273 }
274
275 ++(*loadedCount);
276 if (*loadedCount >= totalImages) {
277 m_downloadCounts[componentId] = totalImages;
278 checkDownloadCompletion(componentId);
279 }
280
281 m_pendingImageWatchers.removeOne(watcher);
282 watcher->deleteLater();
283 });
284
285 m_pendingImageWatchers.append(watcher);
286 watcher->setFuture(future);
287 }

306void LcscImageService::performApiSearch(const QString& componentId) {
307 if (m_isCancelled) {
308 return;
309 }
310
311 QByteArray postData;
312 QUrlQuery query;
313 query.addQueryItem("keyword", componentId);
314 query.addQueryItem("currPage", "1");
315 query.addQueryItem("pageSize", "50");
316 query.addQueryItem("needAggs", "true");
317 postData = query.toString(QUrl::FullyEncoded).toUtf8();
318
319 RequestProfile profile = RequestProfiles::productSearch();
320 profile.connectTimeoutMs = 30000;
321 profile.readTimeoutMs = 30000;
322 profile.maxRetries = MAX_RETRY_COUNT;
323
324 AsyncNetworkRequest* request =
325 NetworkClient::instance().postAsync(QUrl("https://pro.lceda.cn/api/v2/eda/product/search"),
326 postData,
327 ResourceType::ProductSearch,
328 RetryPolicy::fromProfile(profile, isWeakNetworkEnabled()));
329 trackAsyncRequest(request);
330
331 QObject::connect(
332 request, &AsyncNetworkRequest::finished, this, [this, request, componentId](const NetworkResult& result) {
333 untrackAsyncRequest(request);
334
335 if (m_isCancelled) {
336 request->deleteLater();
337 return;
338 }
339
340 // 检查组件是否已被取消
341 if (!m_requestedComponents.contains(componentId)) {
342 request->deleteLater();
343 return;
344 }
345
346 if (!result.success) {
347 qWarning() << "LcscImageService: API search failed for component:" << componentId
348 << "error:" << result.error;
349 request->deleteLater();
350 m_requestedComponents.remove(componentId);
351 // 不再触发fallback，直接报告无预览图
352 emit error(componentId, "No preview image available from API");
353 return;
354 }
355
356 const QJsonDocument doc = QJsonDocument::fromJson(result.data);
357 if (doc.isNull()) {
358 qWarning() << "LcscImageService: Failed to parse JSON response for component:" << componentId;
359 request->deleteLater();
360 m_requestedComponents.remove(componentId);
361 // 不再触发fallback，直接报告无预览图
362 emit error(componentId, "No preview image available from API");
363 return;
364 }
365
366 const QJsonObject root = doc.object();
367
368 if (root.contains("result")) {
369 const QJsonObject resultObject = root["result"].toObject();
370 if (resultObject.contains("productList")) {
371 const QJsonArray productList = resultObject["productList"].toArray();
372 if (!productList.isEmpty()) {
373 const QJsonObject product = selectBestProductForComponent(componentId, productList);
374 const QString matchedComponentCode = extractComponentCode(product);
375
376 // 如果没有找到匹配的元器件，直接报告错误，不再使用 fallback
377 if (matchedComponentCode.isEmpty()) {
378 qWarning() << "LcscImageService: No matching product found for component:" << componentId;
379 request->deleteLater();
380 m_requestedComponents.remove(componentId);
381 emit error(componentId, "No preview image available from API");
382 return;
383 }
384
385 QStringList imageUrls = extractPreviewImageUrlsFromProduct(product);
386
387 QString manufacturerPart;
388 QString datasheetUrl;
389 if (product.contains("device_info")) {
390 const QJsonObject deviceInfo = product["device_info"].toObject();
391 if (deviceInfo.contains("attributes")) {
392 const QJsonObject attributes = deviceInfo["attributes"].toObject();
393 manufacturerPart = attributes.value("Manufacturer Part").toString();
394 datasheetUrl = attributes.value("Datasheet").toString();
395 }
396 }
397
398 while (imageUrls.size() > MAX_IMAGES_PER_COMPONENT) {
399 imageUrls.removeLast();
400 }
401
402 emit lcscDataReady(componentId, manufacturerPart, datasheetUrl, imageUrls);
403
404 if (!imageUrls.isEmpty()) {
405 startPreviewImageDownloads(componentId, imageUrls);
406 } else {
407 emit error(componentId, "No images available");
408 }
409
410 request->deleteLater();
411 return;
412 }
413 }
414 }
415
416 request->deleteLater();
417 m_requestedComponents.remove(componentId);
418 // 不再触发fallback，直接报告无预览图
419 emit error(componentId, "No preview image available from API");
420 });
421}

423void LcscImageService::performDownload(const QString& componentId, const QString& imageUrl, int imageIndex) {
424 if (m_isCancelled) {
425 return;
426 }
427
428 const uint64_t gen = ComponentCacheService::instance()->currentGeneration();
429 AsyncNetworkRequest* request = NetworkClient::instance().getAsync(
430 QUrl(imageUrl),
431 ResourceType::PreviewImage,
432 RetryPolicy::fromProfile(RequestProfiles::previewImage(), isWeakNetworkEnabled()));
433 trackAsyncRequest(request);
434 connect(request,
435 &AsyncNetworkRequest::finished,
436 this,
437 [this, request, componentId, imageIndex, gen](const NetworkResult& result) {
438 untrackAsyncRequest(request);
439
440 if (m_isCancelled) {
441 request->deleteLater();
442 return;
443 }
444
445 // 检查组件是否已被取消
446 if (!m_requestedComponents.contains(componentId)) {
447 request->deleteLater();
448 return;
449 }
450
451 if (!result.success) {
452 m_downloadCounts[componentId]++;
453 checkDownloadCompletion(componentId);
454 request->deleteLater();
455 return;
456 }
457
458 const QByteArray imageData = result.data;
459 if (!imageData.isEmpty() && !QString::fromUtf8(imageData).contains("<!DOCTYPE", Qt::CaseInsensitive)) {
460 ComponentCacheService::instance()->savePreviewImage(componentId, imageData, imageIndex, gen);
461 m_downloadCounts[componentId]++;

463 checkDownloadCompletion(componentId);
464 } else {
465 qWarning() << "LcscImageService: Downloaded invalid data for" << componentId << "index"
466 << imageIndex << "- may be blocked (403), falling back to API";
467 m_downloadCounts[componentId]++;
468 checkDownloadCompletion(componentId);
469 emit error(componentId, "Image blocked (403), please retry");
470 }
471
472 request->deleteLater();
473 });
