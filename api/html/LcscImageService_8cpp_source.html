83void LcscImageService::performApiSearch(const QString& componentId, int retryCount) {
84 qDebug() << "LcscImageService: Starting API search for component:" << componentId << "retry:" << retryCount
85 << "active_requests:" << m_activeRequests;
86
87 // 使用官方 API 搜索产品
88 QString apiUrl = "https://pro.lceda.cn/api/v2/eda/product/search";
89
90 // 构建表单数据（与 Python 版本一致）
91 QByteArray postData;
92 QUrlQuery query;
93 query.addQueryItem("keyword", componentId);
94 query.addQueryItem("currPage", "1");
95 query.addQueryItem("pageSize", "50"); // 从 1 改为 50，以获取完整信息
96 query.addQueryItem("needAggs", "true");
97 postData = query.toString(QUrl::FullyEncoded).toUtf8();
98
99 QNetworkRequest request{QUrl(apiUrl)};
100 request.setHeader(QNetworkRequest::ContentTypeHeader, "application/x-www-form-urlencoded; charset=UTF-8");
101 request.setHeader(QNetworkRequest::UserAgentHeader,
102 "Mozilla/5.0 (Windows NT 10.0; Win64; x64) AppleWebKit/537.36 (KHTML, like Gecko) "
103 "Chrome/145.0.0.0 Safari/537.36");
104 request.setRawHeader("Accept", "application/json, text/javascript, */*; q=0.01");
105 request.setRawHeader("X-Requested-With", "XMLHttpRequest");
106 request.setTransferTimeout(30000); // 从 15 秒增加到 30 秒
107
108 QNetworkReply* reply = m_networkManager->post(request, postData);
109 connect(reply, &QNetworkReply::finished, this, [this, reply, componentId, retryCount]() {
110 handleApiResponse(reply, componentId, retryCount);
111 });

114void LcscImageService::handleApiResponse(QNetworkReply* reply, const QString& componentId, int retryCount) {
115 reply->deleteLater();
116
117 if (reply->error() != QNetworkReply::NoError) {
118 if (retryCount < MAX_RETRY_COUNT) { // 从 2 改为 MAX_RETRY_COUNT (3)
119 QTimer::singleShot(1000 * (retryCount + 1), this, [this, componentId, retryCount]() {
120 performApiSearch(componentId, retryCount + 1);
121 });
122 } else {
123 // API 失败，尝试回退到爬虫方式
124 performFallback(componentId);
125 }
126 return;
127 }
128
129 QJsonDocument doc = QJsonDocument::fromJson(reply->readAll());
130 QJsonObject root = doc.object();
131
132 // 解析 API 响应
133 if (root.contains("result")) {
134 QJsonObject result = root["result"].toObject();
135
136 if (result.contains("productList")) {
137 QJsonArray productList = result["productList"].toArray();
138
139 if (!productList.isEmpty()) {
140 QJsonObject product = productList[0].toObject();
141
142 // 提取制造商部件号
143 QString manufacturerPart;
144 if (product.contains("device_info")) {
145 QJsonObject deviceInfo = product["device_info"].toObject();
146 if (deviceInfo.contains("attributes")) {
147 QJsonObject attributes = deviceInfo["attributes"].toObject();
148 if (attributes.contains("Manufacturer Part")) {
149 manufacturerPart = attributes["Manufacturer Part"].toString();
150 m_manufacturerParts[componentId] = manufacturerPart;
151 qDebug() << "Extracted manufacturer part from LCSC API for" << componentId << ":"
152 << manufacturerPart;
153 }
154 }
155 }
156
157 // 提取图片列表
158 QStringList imageUrls;
159 if (product.contains("image")) {
160 QString imagesStr = product["image"].toString();
161 if (!imagesStr.isEmpty()) {
162 // EasyEDA 使用 <$> 分隔多张图片
163 imageUrls = imagesStr.split("<$>", Qt::SkipEmptyParts);
164 }
165 }
166
167 qDebug() << "Extracted" << imageUrls.size() << "image URLs for component:" << componentId;
168 for (int i = 0; i < imageUrls.size(); ++i) {
169 qDebug() << " Image" << i << "URL:" << imageUrls[i];
170 }
171
172 // 限制最多 3 张图片
173 while (imageUrls.size() > MAX_IMAGES_PER_COMPONENT) {
174 imageUrls.removeLast();
175 }
176
177 qDebug() << "After limiting to" << MAX_IMAGES_PER_COMPONENT
178 << "images, final count:" << imageUrls.size();
179
180 // 提取数据手册 URL（按照 Python 代码的方式）
181 QString datasheetUrl;
182 if (product.contains("device_info")) {
183 QJsonObject deviceInfo = product["device_info"].toObject();
184 if (deviceInfo.contains("attributes")) {
185 QJsonObject attributes = deviceInfo["attributes"].toObject();
186 if (attributes.contains("Datasheet")) {
187 datasheetUrl = attributes["Datasheet"].toString();
188 qDebug() << "Extracted datasheet URL from LCSC API for" << componentId << ":"
189 << datasheetUrl;
190 }
191 }
192 }
193
194 // 发送 LCSC 数据就绪信号（包含制造商部件号、数据手册 URL 和预览图 URL 列表）
195 emit lcscDataReady(componentId, manufacturerPart, datasheetUrl, imageUrls);
196 qDebug() << "Emitted lcscDataReady for" << componentId << "with manufacturer part:" << manufacturerPart
197 << "images:" << imageUrls.size()
198 << "and datasheet:" << (datasheetUrl.isEmpty() ? "none" : datasheetUrl);
199
200 // 下载图片
201 if (!imageUrls.isEmpty()) {
202 m_pendingImages[componentId] = imageUrls;
203 m_downloadedImages[componentId].clear();
204 m_downloadCounts[componentId] = 0;
205
206 // 预先创建指定数量的空元素，确保索引能够正确对应
207 // 这样当图片按乱序下载时，能够填充到正确的索引位置
208 m_downloadedImages[componentId].resize(imageUrls.size());
209
210 qDebug() << "Starting download of" << imageUrls.size() << "images for component:" << componentId;
211 qDebug() << " Pre-allocated container size:" << m_downloadedImages[componentId].size();
212
213 // 开始下载所有图片
214 for (int i = 0; i < imageUrls.size(); ++i) {
215 qDebug() << " Scheduling image download" << i << "from:" << imageUrls[i];
216 performDownload(componentId, imageUrls[i], i, 0);
217 }
218 }
219
220 // 下载数据手册
221 if (!datasheetUrl.isEmpty()) {
222 m_pendingDatasheets[componentId] = datasheetUrl;
223 m_datasheetDownloadStatus[componentId] = 0; // pending
224 qDebug() << "Starting datasheet download for component:" << componentId << "from:" << datasheetUrl;
225 performDatasheetDownload(componentId, datasheetUrl, 0);
226 }
227
228 return;
229 }
230 }
231 }
232
233 // API 没有找到图片，尝试回退
234 performFallback(componentId);
235}

296void LcscImageService::handleDownloadResponse(QNetworkReply* reply,
297 const QString& componentId,
298 const QString& imageUrl,
299 int imageIndex,
300 int retryCount) {
301 reply->deleteLater();
302
303 if (reply->error() != QNetworkReply::NoError) {
304 if (retryCount < MAX_RETRY_COUNT) { // 从 2 改为 MAX_RETRY_COUNT (3)
305 qDebug() << "Image download failed for component:" << componentId << "index:" << imageIndex
306 << "error:" << reply->errorString() << "retry:" << retryCount;
307 performDownload(componentId, imageUrl, imageIndex, retryCount + 1);
308 } else {
309 // 下载失败，记录但继续处理其他图片
310 qDebug() << "Image download failed after all retries for" << componentId << "index:" << imageIndex
311 << "error:" << reply->errorString();
312
313 // 增加下载计数，确保 checkDownloadCompletion 能正确判断是否所有图片都已尝试下载
314 m_downloadCounts[componentId]++;
315
316 // 检查下载完成状态
317 checkDownloadCompletion(componentId);
318 }
319 return;
320 }
321
322 // 直接保存到内存中，不写文件
323 QByteArray imageData = reply->readAll();
324 qDebug() << "Image download response for component:" << componentId << "index:" << imageIndex
325 << "size:" << imageData.size() << "bytes" << "URL:" << imageUrl.left(80);
326
327 if (!imageData.isEmpty()) {
328 // 使用索引直接赋值，确保图片数据被保存到正确的索引位置
329 if (imageIndex >= 0 && imageIndex < m_downloadedImages[componentId].size()) {
330 m_downloadedImages[componentId][imageIndex] = imageData;
331 qDebug() << " Saved image data at index:" << imageIndex;
332 } else {
333 // 如果索引超出范围（异常情况），使用 append
334 m_downloadedImages[componentId].append(imageData);
335 qDebug() << " Appended image data (index out of range):" << imageIndex;
336 }
337 m_downloadCounts[componentId]++;
338
339 // 发送信号，传递图片数据（内存）
340 emit imageReady(componentId, imageData, imageIndex);
341
342 // 检查是否所有图片都下载完成
343 checkDownloadCompletion(componentId);
344 } else {
345 qDebug() << "Failed to read image data for" << componentId << "index:" << imageIndex;
346 checkDownloadCompletion(componentId);
347 }

350void LcscImageService::checkDownloadCompletion(const QString& componentId) {
351 if (!m_pendingImages.contains(componentId)) {
352 return;
353 }
354
355 int expectedCount = m_pendingImages[componentId].size();
356 int downloadedCount = m_downloadCounts.value(componentId, 0);
357
358 qDebug() << "checkDownloadCompletion for component:" << componentId << "expected:" << expectedCount
359 << "downloaded:" << downloadedCount;
360
361 // 统计已尝试下载的图片数量（包括成功的和失败的）
362 int attemptedCount = 0;
363 for (int i = 0; i < m_downloadedImages[componentId].size(); ++i) {
364 if (!m_downloadedImages[componentId][i].isEmpty()) {
365 attemptedCount++;
366 }
367 }
368
369 // 只要下载的图片数量达到预期，就认为完成
370 if (downloadedCount >= expectedCount) {
371 // 所有图片下载完成
372 qDebug() << "All images downloaded successfully, emitting allImagesReady";
373 emitAllImagesReady(componentId);
374
375 m_activeRequests--;
376 processQueue();
377 } else if (attemptedCount == m_downloadedImages[componentId].size()) {
378 // 所有位置都已填充（即使有些是空数据），也认为完成
379 qDebug() << "All image slots filled (some may be empty), emitting allImagesReady";
380 emitAllImagesReady(componentId);
381
382 m_activeRequests--;
383 processQueue();
384 }
