51void LcscImageService::handleSearchResponse(QNetworkReply* reply, const QString& componentId, int retryCount) {
52 reply->deleteLater();
53
54 if (reply->error() != QNetworkReply::NoError) {
55 if (retryCount < 2) {
56 QTimer::singleShot(
57 1000, this, [this, componentId, retryCount]() { performSearch(componentId, retryCount + 1); });
58 } else {
59 performFallback(componentId);
60 }
61 return;
62 }
63
64 QJsonDocument doc = QJsonDocument::fromJson(reply->readAll());
65 QJsonObject root = doc.object();
66 QString imageUrl;
67 bool found = false;
68
69 if (root.contains("result")) {
70 QJsonArray resultArray = root["result"].toArray();
71 for (const auto& resVal : resultArray) {
72 QJsonObject resObj = resVal.toObject();
73 if (resObj.contains("products")) {
74 QJsonArray products = resObj["products"].toArray();
75 if (!products.isEmpty()) {
76 imageUrl = products[0].toObject()["selfBreviaryImageUrl"].toString();
77 if (!imageUrl.isEmpty()) {
78 found = true;
79 break;
80 }
81 }
82 }
83 }
84 }

86 if (found) {
