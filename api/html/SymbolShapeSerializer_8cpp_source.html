92QJsonObject SymbolShapeSerializer::toJson(const SymbolArc& arc) {
93 QJsonObject json;
94 QJsonArray pathArray;
95 for (const QPointF& point : arc.path) {
96 QJsonObject pointObj;
97 pointObj["x"] = point.x();
98 pointObj["y"] = point.y();
99 pathArray.append(pointObj);
100 }
101 json["path"] = pathArray;
102 json["helper_dots"] = arc.helperDots;
103 json["stroke_color"] = arc.strokeColor;
104 json["stroke_width"] = arc.strokeWidth;
105 json["stroke_style"] = arc.strokeStyle;
106 json["fill_color"] = arc.fillColor;
107 json["id"] = arc.id;
108 json["is_locked"] = arc.isLocked;
109 return json;
110}
111
112bool SymbolShapeSerializer::fromJson(SymbolArc& arc, const QJsonObject& json) {
113 if (json.contains("path") && json["path"].isArray()) {
114 QJsonArray pathArray = json["path"].toArray();
115 arc.path.clear();
116 for (const QJsonValue& value : pathArray) {
117 if (value.isObject()) {
118 QJsonObject pointObj = value.toObject();
119 QPointF point(pointObj["x"].toDouble(0.0), pointObj["y"].toDouble(0.0));
120 arc.path.append(point);
121 }
122 }
123 }
124 arc.helperDots = json["helper_dots"].toString();
125 arc.strokeColor = json["stroke_color"].toString();
126 arc.strokeWidth = json["stroke_width"].toDouble(0.0);
127 arc.strokeStyle = json["stroke_style"].toString();
128 arc.fillColor = json["fill_color"].toBool(false);
129 arc.id = json["id"].toString();
130 arc.isLocked = json["is_locked"].toBool(false);
131 return true;
132}
