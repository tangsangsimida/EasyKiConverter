52void WriteWorker::run() {
53 // 检查是否已被取消 (v3.0.5+)
54 if (m_isAborted.loadRelaxed()) {
55 m_status->writeSuccess = false;
56 m_status->writeMessage = "Export cancelled";
57 m_status->isCancelled = true;
58 m_status->addDebugLog(QString("WriteWorker cancelled for component: %1").arg(m_status->componentId));
59 emit writeCompleted(m_status);
60 return;
61 }
62
63 QElapsedTimer writeTimer;
64 writeTimer.start();
65
66 m_status->addDebugLog(QString("WriteWorker started for component: %1").arg(m_status->componentId));
67
68 // 初始化所有写入状态
69
70 m_status->symbolWritten = false;
71 m_status->footprintWritten = false;
72 m_status->model3DWritten = false;
73
74 if (!m_status->fetchSuccess || !m_status->processSuccess) {
75 m_status->writeDurationMs = 0;
76 m_status->writeSuccess = false;
77 m_status->writeMessage = "Skipped writing due to previous stage failure";
78 m_status->addDebugLog("Skipping write stage because fetch or process failed.");
79 emit writeCompleted(m_status);
80 return;
81 }
82
83 if ((m_exportSymbol && (!m_status->symbolData || m_status->symbolData->info().name.isEmpty())) &&
84 (m_exportFootprint && (!m_status->footprintData || m_status->footprintData->info().name.isEmpty()))) {
85 m_status->writeDurationMs = writeTimer.elapsed();
86 m_status->writeSuccess = false;
87 m_status->writeMessage = "No valid symbol or footprint data to write";
88 m_status->addDebugLog("ERROR: Symbol and Footprint data are empty or invalid.");
89 emit writeCompleted(m_status);
90 return;
91 }
92
93 if (!createOutputDirectory(m_outputPath)) {
94 m_status->writeDurationMs = writeTimer.elapsed();
95 m_status->writeSuccess = false;
96 m_status->writeMessage = "Failed to create output directory";
97 m_status->addDebugLog(
98 QString("ERROR: Failed to create output directory, Duration: %1ms").arg(m_status->writeDurationMs));
99 emit writeCompleted(m_status);
100 return;
101 }
102
103 // 串行写入文件，每个函数独立更新自己的状态
104 if (m_exportSymbol && m_status->symbolData) {
105 writeSymbolFile(*m_status);
106 if (m_isAborted.loadRelaxed())
107 goto cleanup;
108 }
109
110 if (m_exportFootprint && m_status->footprintData) {
111 writeFootprintFile(*m_status);
112 if (m_isAborted.loadRelaxed())
113 goto cleanup;
114 }
115
116 if (m_exportModel3D && m_status->model3DData && !m_status->model3DData->rawObj().isEmpty()) {
117 write3DModelFile(*m_status);
118 }
119
120cleanup:
121 if (m_isAborted.loadRelaxed()) {
122 m_status->writeDurationMs = writeTimer.elapsed();
123 m_status->writeSuccess = false;
124 m_status->writeMessage = "Export cancelled";
125 m_status->isCancelled = true;
126 // 状态已在开头初始化为 false，无需再次设置
127 m_status->addDebugLog(QString("WriteWorker cancelled for component: %1").arg(m_status->componentId));
128 emit writeCompleted(m_status);
129 return;
130 }
131
132 // 根据用户请求的导出项和实际完成情况，决定最终的成功状态
133 // 注意：3D模型导出失败不影响符号和封装的导出成功状态
134 // 符号和封装是捆绑的核心输出，3D模型是可选附加输出
135 bool coreSuccess = true;
136 bool model3DFailed = false;
137 if (m_exportSymbol && !m_status->symbolWritten) {
138 coreSuccess = false;
139 }
140 if (m_exportFootprint && !m_status->footprintWritten) {
141 coreSuccess = false;
142 }
143 if (m_exportModel3D && !m_status->model3DWritten) {
144 model3DFailed = true; // 仅记录，不影响核心成功状态
145 m_status->addDebugLog("WARNING: 3D model export failed, but symbol/footprint export is not affected.");
146 }
147
148 m_status->writeSuccess = coreSuccess;
149 if (coreSuccess && model3DFailed) {
150 m_status->writeMessage = "Write completed (3D model export failed, symbol/footprint OK)";
151 } else if (coreSuccess) {
152 m_status->writeMessage = "Write completed successfully";
153 } else {
154 m_status->writeMessage = "Failed to write one or more core files (symbol/footprint)";
155 }
156
157 if (m_debugMode) {
158 exportDebugData(*m_status);
159 }
160
161 m_status->writeDurationMs = writeTimer.elapsed();
162 m_status->addDebugLog(QString("WriteWorker completed for component: %1, Success: %2, Duration: %3ms")
163 .arg(m_status->componentId)
164 .arg(m_status->writeSuccess)
165 .arg(m_status->writeDurationMs));
166
167 emit writeCompleted(m_status);
168}

170bool WriteWorker::writeSymbolFile(ComponentExportStatus& status) {
171 if (!status.symbolData) {
172 return true; // 没有数据可写，不应视为失败
173 }
174
175 // 1. 检查临时目录是否存在
176 if (!QDir(m_tempDir).exists()) {
177 status.addDebugLog(QString("ERROR: Temp directory does not exist: %1").arg(m_tempDir));
178 return false;
179 }
180 // 2. 检查路径安全性
181 if (!PathSecurity::isSafePath(m_tempDir, m_outputPath)) {
182 status.addDebugLog(QString("SECURITY ERROR: Temp dir outside output path: %1").arg(m_tempDir));
183 return false;
184 }
185
186 // 使用线程ID和当前时间生成唯一的临时文件名，避免并发冲突
187 qint64 threadId = reinterpret_cast<qint64>(QThread::currentThreadId());
188 qint64 timestamp = QDateTime::currentMSecsSinceEpoch();
189 // 使用统一的临时文件夹
190 QString tempFilePath =
191 QString("%1/%2_%3_%4.kicad_sym.tmp").arg(m_tempDir, status.componentId).arg(threadId).arg(timestamp);
192 QString finalFilePath = QString("%1/%2.kicad_sym").arg(m_tempDir, status.componentId);
193
194 // 先写入临时文件
195 if (!m_symbolExporter.exportSymbol(*status.symbolData, tempFilePath)) {
196 status.addDebugLog(QString("ERROR: Failed to write symbol file: %1").arg(tempFilePath));
197 QFile::remove(tempFilePath); // 清理临时文件
198 return false; // 导出器报告失败
199 }
200
201 // 验证临时文件是否真的被写入
202 if (!QFile::exists(tempFilePath)) {
203 status.addDebugLog(QString("ERROR: Symbol file not found after export: %1").arg(tempFilePath));
204 return false; // 文件未找到，视为失败
205 }
206
207 // 使用互斥锁保护文件删除和重命名操作，防止并发竞态条件
208 {
209 QMutexLocker locker(&s_fileWriteMutex);
210
211 // 删除旧文件（如果存在）
212 if (QFile::exists(finalFilePath)) {
213 if (!QFile::remove(finalFilePath)) {
214 status.addDebugLog(QString("WARNING: Failed to remove old symbol file: %1").arg(finalFilePath));
215 // 继续尝试重命名，可能操作系统会处理
216 }
217 }
218
219 // 原子性地重命名临时文件到最终文件
220 if (!QFile::rename(tempFilePath, finalFilePath)) {
221 status.addDebugLog(
222 QString("ERROR: Failed to rename symbol file: %1 -> %2").arg(tempFilePath, finalFilePath));
223 QFile::remove(tempFilePath); // 清理临时文件
224 return false;
225 }
226 }
227
228 // 验证最终文件是否存在
229 if (QFile::exists(finalFilePath)) {
230 status.addDebugLog(QString("Symbol file written atomically: %1").arg(finalFilePath));
231 status.symbolWritten = true;
232 return true;
233 } else {
234 status.addDebugLog(QString("ERROR: Final symbol file not found after rename: %1").arg(finalFilePath));
235 return false;
236 }

239bool WriteWorker::writeFootprintFile(ComponentExportStatus& status) {
240 if (!status.footprintData) {
241 return true; // 没有数据可写，不应视为失败
242 }
243
244 // 1. 检查临时目录是否存在
245 if (!QDir(m_tempDir).exists()) {
246 status.addDebugLog(QString("ERROR: Temp directory does not exist: %1").arg(m_tempDir));
247 return false;
248 }
249 // 2. 检查路径安全性
250 if (!PathSecurity::isSafePath(m_tempDir, m_outputPath)) {
251 status.addDebugLog(QString("SECURITY ERROR: Temp dir outside output path: %1").arg(m_tempDir));
252 return false;
253 }
254
255 QString footprintLibPath = QString("%1/%2.pretty").arg(m_outputPath, m_libName);
256 if (!createOutputDirectory(footprintLibPath)) {
257 status.addDebugLog(QString("ERROR: Failed to create footprint library directory: %1").arg(footprintLibPath));
258 return false;
259 }
260
261 QString modelsDirPath = QString("%1/%2.3dmodels").arg(m_outputPath, m_libName);
262 if (m_exportModel3D) {
263 // 确保3D模型目录存在，但它的创建失败不直接导致封装写入失败
264 if (!createOutputDirectory(modelsDirPath)) {
265 status.addDebugLog(QString("WARNING: Failed to create 3D models directory: %1").arg(modelsDirPath));
266 }
267 }
268
269 QString footprintName = PathSecurity::sanitizeFilename(status.footprintData->info().name);
270 QString filePath = QString("%1/%2.kicad_mod").arg(footprintLibPath, footprintName);
271
272 // 使用线程ID和当前时间生成唯一的临时文件名，避免并发冲突
273 qint64 threadId = reinterpret_cast<qint64>(QThread::currentThreadId());
274 qint64 timestamp = QDateTime::currentMSecsSinceEpoch();
275 // 使用统一的临时文件夹
276 QString tempFilePath =
277 QString("%1/%2_%3_%4.kicad_mod.tmp").arg(m_tempDir, footprintName).arg(threadId).arg(timestamp);
278
279 QString model3DWrlPath;
280 QString model3DStepPath;
281 if (m_exportModel3D && status.model3DData && !status.model3DData->uuid().isEmpty()) {
282 // 始终使用封装名作为 3D 模型的文件名和引用路径，以确保与 write3DModelFile 写入磁盘的命名一致
283 model3DWrlPath = QString("../%1.3dmodels/%2.wrl").arg(m_libName, footprintName);
284 if (!status.model3DStepRaw.isEmpty()) {
285 model3DStepPath = QString("../%1.3dmodels/%2.step").arg(m_libName, footprintName);
286 }
287 }
288
289 // 先写入临时文件
290 bool exportSuccess = false;
291 if (!model3DStepPath.isEmpty()) {
292 // 有 WRL 和 STEP 两个模型，调用双参数版本
293 exportSuccess =
294 m_footprintExporter.exportFootprint(*status.footprintData, tempFilePath, model3DWrlPath, model3DStepPath);
295 } else if (!model3DWrlPath.isEmpty()) {
296 // 只有 WRL 模型，调用单参数版本
297 exportSuccess = m_footprintExporter.exportFootprint(*status.footprintData, tempFilePath, model3DWrlPath);
298 } else {
299 // 没有 3D 模型，调用无参数版本
300 exportSuccess = m_footprintExporter.exportFootprint(*status.footprintData, tempFilePath);
301 }
302
303 if (!exportSuccess) {
304 status.addDebugLog(QString("ERROR: Failed to write footprint file (exporter): %1").arg(tempFilePath));
305 QFile::remove(tempFilePath); // 清理临时文件
306 return false; // 导出器报告失败
307 }
308
309 // 验证临时文件是否真的被写入
310 if (!QFile::exists(tempFilePath)) {
311 status.addDebugLog(QString("ERROR: Footprint file not found after export: %1").arg(tempFilePath));
312 return false; // 文件未找到，视为失败
313 }
314
315 // 使用互斥锁保护文件删除和重命名操作，防止并发竞态条件
316 {
317 QMutexLocker locker(&s_fileWriteMutex);
318
319 // 删除旧文件（如果存在）
320 if (QFile::exists(filePath)) {
321 if (!QFile::remove(filePath)) {
322 status.addDebugLog(QString("WARNING: Failed to remove old footprint file: %1").arg(filePath));
323 // 继续尝试重命名
324 }
325 }
326
327 // 原子性地重命名临时文件到最终文件
328 if (!QFile::rename(tempFilePath, filePath)) {
329 status.addDebugLog(QString("ERROR: Failed to rename footprint file: %1 -> %2").arg(tempFilePath, filePath));
330 QFile::remove(tempFilePath); // 清理临时文件
331 return false;
332 }
333 }
334
335 // 验证最终文件是否存在
336 if (QFile::exists(filePath)) {
337 status.addDebugLog(QString("Footprint file written atomically: %1").arg(filePath));
338 status.footprintWritten = true;
339 return true;
340 } else {
341 status.addDebugLog(QString("ERROR: Final footprint file not found after rename: %1").arg(filePath));
342 return false;
343 }

346bool WriteWorker::write3DModelFile(ComponentExportStatus& status) {
347 // 注意：model3DObjRaw 可能在 Process 阶段被 clearIntermediateData 清理
348 // 所以必须检查 model3DData 对象中的 rawObj 字符串
349 if (!status.model3DData || status.model3DData->rawObj().isEmpty()) {
350 return true; // 没有 WRL 数据可写，不应视为失败 (如果只请求 STEP，则应在ProcessWorker中处理)
351 }
352
353 // 1. 检查临时目录是否存在
354 if (!QDir(m_tempDir).exists()) {
355 status.addDebugLog(QString("ERROR: Temp directory does not exist: %1").arg(m_tempDir));
356 return false;
357 }
358 // 2. 检查路径安全性（仅对临时目录路径检查）
359 if (!PathSecurity::isSafePath(m_tempDir, m_outputPath)) {
360 status.addDebugLog(
361 QString("SECURITY ERROR: Temp dir %1 is outside output path %2").arg(m_tempDir, m_outputPath));
362 return false;
363 }
364
365 QString modelsDirPath = QString("%1/%2.3dmodels").arg(m_outputPath, m_libName);
366 if (!createOutputDirectory(modelsDirPath)) {
367 status.addDebugLog(QString("ERROR: Failed to create 3D models directory: %1").arg(modelsDirPath));
368 return false;
369 }
370
371 QString footprintName = status.footprintData ? status.footprintData->info().name : status.componentId;
372 footprintName = PathSecurity::sanitizeFilename(footprintName);
373
374 bool wrlSuccess = false;
375 bool stepSuccess = false;
376
377 // 导出 WRL 文件（使用 Write-Temp-Move 模式）
378 QString wrlFilePath = QString("%1/%2.wrl").arg(modelsDirPath, footprintName);
379
380 // 使用线程ID和当前时间生成唯一的临时文件名，避免并发冲突
381 qint64 threadId = reinterpret_cast<qint64>(QThread::currentThreadId());
382 qint64 timestamp = QDateTime::currentMSecsSinceEpoch();
383 // 使用统一的临时文件夹
384 QString wrlTempFilePath = QString("%1/%2_%3_%4.wrl.tmp").arg(m_tempDir, footprintName).arg(threadId).arg(timestamp);
385
386 wrlSuccess = m_model3DExporter.exportToWrl(*status.model3DData, wrlTempFilePath);
387
388 if (wrlSuccess && QFile::exists(wrlTempFilePath)) {
389 // 使用互斥锁保护文件删除和重命名操作
390 QMutexLocker locker(&s_fileWriteMutex);
391
392 // 删除旧文件（如果存在）
393 if (QFile::exists(wrlFilePath) && !QFile::remove(wrlFilePath)) {
394 status.addDebugLog(QString("WARNING: Failed to remove old WRL file: %1").arg(wrlFilePath));
395 }
396 // 原子性地重命名
397 if (QFile::rename(wrlTempFilePath, wrlFilePath)) {
398 status.addDebugLog(QString("3D model WRL file written atomically: %1").arg(wrlFilePath));
399 } else {
400 status.addDebugLog(QString("ERROR: Failed to rename WRL file: %1 -> %2").arg(wrlTempFilePath, wrlFilePath));
401 QFile::remove(wrlTempFilePath);
402 wrlSuccess = false;
403 }
404 } else if (wrlSuccess) {
405 status.addDebugLog(QString("ERROR: WRL file not found after export: %1").arg(wrlTempFilePath));
406 wrlSuccess = false;
407 } else {
408 status.addDebugLog(QString("ERROR: Failed to write WRL file: %1").arg(wrlTempFilePath));
409 QFile::remove(wrlTempFilePath);
410 }
411
412 // 导出 STEP 文件（如果有，使用 Write-Temp-Move 模式）
413 if (!status.model3DStepRaw.isEmpty()) {
414 QString stepFilePath = QString("%1/%2.step").arg(modelsDirPath, footprintName);
415
416 // 使用线程ID和当前时间生成唯一的临时文件名，避免并发冲突
417 qint64 threadId = reinterpret_cast<qint64>(QThread::currentThreadId());
418 qint64 timestamp = QDateTime::currentMSecsSinceEpoch();
419 // 使用统一的临时文件夹
420 QString stepTempFilePath =
421 QString("%1/%2_%3_%4.step.tmp").arg(m_tempDir, footprintName).arg(threadId).arg(timestamp);
422
423 QFile stepFile(stepTempFilePath);
424 if (stepFile.open(QIODevice::WriteOnly)) {
425 stepFile.write(status.model3DStepRaw);
426 stepFile.close();
427
428 if (QFile::exists(stepTempFilePath)) {
429 // 使用互斥锁保护文件删除和重命名操作
430 QMutexLocker locker(&s_fileWriteMutex);
431
432 // 删除旧文件（如果存在）
433 if (QFile::exists(stepFilePath) && !QFile::remove(stepFilePath)) {
434 status.addDebugLog(QString("WARNING: Failed to remove old STEP file: %1").arg(stepFilePath));
435 }
436 // 原子性地重命名
437 if (QFile::rename(stepTempFilePath, stepFilePath)) {
438 status.addDebugLog(QString("3D model STEP file written atomically: %1").arg(stepFilePath));
439 stepSuccess = true;
440 } else {
441 status.addDebugLog(
442 QString("ERROR: Failed to rename STEP file: %1 -> %2").arg(stepTempFilePath, stepFilePath));
443 QFile::remove(stepTempFilePath);
444 }
445 }
446 } else {
447 status.addDebugLog(QString("ERROR: Failed to write STEP file: %1").arg(stepTempFilePath));
448 // STEP 文件失败不影响整体 3D 模型成功
449 }
450
451 // 内存优化：写入完成后立即清理 STEP 数据
452 status.clearStepData();
453 }
454
455 // 只有在 WRL 文件成功导出且存在时才设置 model3DWritten 标志
456 status.model3DWritten = wrlSuccess;
457
458 // 整个3D模型导出成功，取决于 WRL 是否成功。STEP 失败不影响此结果。
459 return wrlSuccess;
460}

505 }

507
508 if (!status.advJsonRaw.isEmpty()) {

512 advFile.write(status.advJsonRaw);
513 advFile.close();
514 qDebug() << "Debug: adv_raw.json written";

516 }
517
518 if (!status.model3DObjRaw.isEmpty()) {
519 QString objFilePath = QString("%1/model3d_raw.obj").arg(componentDebugDir);
520 QFile objFile(objFilePath);

522 objFile.write(status.model3DObjRaw);
523 objFile.close();
524 status.addDebugLog("Debug: model3d_raw.obj written");
525 }
526 }
527
528 if (!status.model3DStepRaw.isEmpty()) {
529 QString stepFilePath = QString("%1/model3d_raw.step").arg(componentDebugDir);
530 QFile stepFile(stepFilePath);
531 if (stepFile.open(QIODevice::WriteOnly)) {
532 stepFile.write(status.model3DStepRaw);
533 stepFile.close();
534 status.addDebugLog("Debug: model3d_raw.step written");
535 }
536 }
537
538 QJsonObject debugInfo;
539 debugInfo["componentId"] = status.componentId;
540 debugInfo["fetchSuccess"] = status.fetchSuccess;
541 debugInfo["fetchMessage"] = status.fetchMessage;
542 debugInfo["processSuccess"] = status.processSuccess;
543 debugInfo["processMessage"] = status.processMessage;
544 debugInfo["writeSuccess"] = status.writeSuccess;
545 debugInfo["writeMessage"] = status.writeMessage;
546
547 if (!status.debugLog.isEmpty()) {
548 QJsonArray logArray;
549 for (const QString& log : status.debugLog) {
550 logArray.append(log);
551 }
552 debugInfo["debugLog"] = logArray;
553 }
554
555 if (status.symbolData) {
556 QJsonObject symbolInfo = SymbolDataSerializer::toJson(status.symbolData->info());
557 symbolInfo["pinCount"] = status.symbolData->pins().size();
558 symbolInfo["rectangleCount"] = status.symbolData->rectangles().size();
559 symbolInfo["circleCount"] = status.symbolData->circles().size();
560 symbolInfo["arcCount"] = status.symbolData->arcs().size();
561 symbolInfo["polylineCount"] = status.symbolData->polylines().size();
562 symbolInfo["polygonCount"] = status.symbolData->polygons().size();
563 symbolInfo["pathCount"] = status.symbolData->paths().size();
564 symbolInfo["ellipseCount"] = status.symbolData->ellipses().size();
565
566 QJsonObject bbox;
