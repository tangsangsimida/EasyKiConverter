55void WriteWorker::run() {
56 if (!m_status) {
57 qWarning() << "WriteWorker: null export status, skip run";
58 return;
59 }
60
61 // 检查是否已被取消 (v3.0.5+)
62 if (m_isAborted.loadRelaxed()) {
63 m_status->writeSuccess = false;
64 m_status->writeMessage = "Export cancelled";
65 m_status->isCancelled = true;
66 m_status->addDebugLog(QString("WriteWorker cancelled for component: %1").arg(m_status->componentId));
67 emit writeCompleted(m_status);
68 return;
69 }
70
71 QElapsedTimer writeTimer;
72 writeTimer.start();
73
74 m_status->addDebugLog(QString("WriteWorker started for component: %1").arg(m_status->componentId));
75
76 // 初始化所有写入状态
77 m_status->symbolWritten = false;
78 m_status->footprintWritten = false;
79 m_status->model3DWritten = false;
80 m_status->previewImageWritten = false;
81 m_status->datasheetWritten = false;
82
83 if (!m_status->fetchSuccess || !m_status->processSuccess) {
84 m_status->writeDurationMs = 0;
85 m_status->writeSuccess = false;
86 m_status->writeMessage = "Skipped writing due to previous stage failure";
87 m_status->addDebugLog("Skipping write stage because fetch or process failed.");
88 emit writeCompleted(m_status);
89 return;
90 }
91
92 if ((m_exportSymbol && (!m_status->symbolData || m_status->symbolData->info().name.isEmpty())) &&
93 (m_exportFootprint && (!m_status->footprintData || m_status->footprintData->info().name.isEmpty()))) {
94 m_status->writeDurationMs = writeTimer.elapsed();
95 m_status->writeSuccess = false;
96 m_status->writeMessage = "No valid symbol or footprint data to write";
97 m_status->addDebugLog("ERROR: Symbol and Footprint data are empty or invalid.");
98 emit writeCompleted(m_status);
99 return;
100 }
101
102 if (!createOutputDirectory(m_outputPath)) {
103 m_status->writeDurationMs = writeTimer.elapsed();
104 m_status->writeSuccess = false;
105 m_status->writeMessage = "Failed to create output directory";
106 m_status->addDebugLog(
107 QString("ERROR: Failed to create output directory, Duration: %1ms").arg(m_status->writeDurationMs));
108 emit writeCompleted(m_status);
109 return;
110 }
111
112 // 并行写入文件
113 // 使用 QtConcurrent::run() + QFutureWatcher 实现并行执行
114 // 每项写完后立即发出 itemWriteCompleted 信号，实现实时进度更新
115 // 注意：QtConcurrent 不支持工作线程抛出异常，所有 lambda 必须捕获异常
116 QList<QFutureWatcher<bool>*> watchers;
117
118 // 符号写入任务
119 if (m_exportSymbol && m_status->symbolData) {
120 QFuture<bool> future = QtConcurrent::run([this]() {
121 try {
122 QMutexLocker locker(&m_symbolMutex);
123 return writeSymbolFile(*m_status);
124 } catch (const std::bad_alloc& e) {
125 qCritical() << "Memory allocation failed in symbol write:" << e.what();
126 return false;
127 } catch (const std::exception& e) {
128 qCritical() << "Exception in symbol write:" << e.what();
129 return false;
130 }
131 });
132 QFutureWatcher<bool>* watcher = new QFutureWatcher<bool>();
133 connect(watcher, &QFutureWatcher<bool>::finished, this, [this, watcher]() {
134 bool result = watcher->result();
135 m_status->symbolWritten = result;
136 emit itemWriteCompleted(m_status->componentId, static_cast<int>(ExportItemType::Symbol), result);
137 // deleteLater removed - watchers are cleaned up by qDeleteAll below
138 });
139 watcher->setFuture(future);
140 watchers.append(watcher);
141 }
142
143 // 封装写入任务
144 if (m_exportFootprint && m_status->footprintData) {
145 QFuture<bool> future = QtConcurrent::run([this]() {
146 try {
147 QMutexLocker locker(&m_footprintMutex);
148 return writeFootprintFile(*m_status);
149 } catch (const std::bad_alloc& e) {
150 qCritical() << "Memory allocation failed in footprint write:" << e.what();
151 return false;
152 } catch (const std::exception& e) {
153 qCritical() << "Exception in footprint write:" << e.what();
154 return false;
155 }
156 });
157 QFutureWatcher<bool>* watcher = new QFutureWatcher<bool>();
158 connect(watcher, &QFutureWatcher<bool>::finished, this, [this, watcher]() {
159 bool result = watcher->result();
160 m_status->footprintWritten = result;
161 emit itemWriteCompleted(m_status->componentId, static_cast<int>(ExportItemType::Footprint), result);
162 // deleteLater removed - watchers are cleaned up by qDeleteAll below
163 });
164 watcher->setFuture(future);
165 watchers.append(watcher);
166 }
167
168 // 3D模型写入任务
169 if (m_exportModel3D && m_status->model3DData &&
170 (!m_status->model3DData->rawObj().isEmpty() || !m_status->cachedModel3DWrlPath.isEmpty())) {
171 QFuture<bool> future = QtConcurrent::run([this]() {
172 try {
173 QMutexLocker locker(&m_model3DMutex);
174 return write3DModelFile(*m_status);
175 } catch (const std::bad_alloc& e) {
176 qCritical() << "Memory allocation failed in 3D model write:" << e.what();
177 return false;
178 } catch (const std::exception& e) {
179 qCritical() << "Exception in 3D model write:" << e.what();
180 return false;
181 }
182 });
183 QFutureWatcher<bool>* watcher = new QFutureWatcher<bool>();
184 connect(watcher, &QFutureWatcher<bool>::finished, this, [this, watcher]() {
185 bool result = watcher->result();
186 m_status->model3DWritten = result;
187 emit itemWriteCompleted(m_status->componentId, static_cast<int>(ExportItemType::Model3D), result);
188 // deleteLater removed - watchers are cleaned up by qDeleteAll below
189 });
190 watcher->setFuture(future);
191 watchers.append(watcher);
192 }
193
194 // 预览图写入任务
195 if (m_exportPreviewImages && !m_status->previewImageDataList.isEmpty()) {
196 QFuture<bool> future = QtConcurrent::run([this]() {
197 try {
198 return writePreviewImageFile(*m_status);
199 } catch (const std::bad_alloc& e) {
200 qCritical() << "Memory allocation failed in preview image write:" << e.what();
201 return false;
202 } catch (const std::exception& e) {
203 qCritical() << "Exception in preview image write:" << e.what();
204 return false;
205 }
206 });
207 QFutureWatcher<bool>* watcher = new QFutureWatcher<bool>();
208 connect(watcher, &QFutureWatcher<bool>::finished, this, [this, watcher]() {
209 bool result = watcher->result();
210 m_status->previewImageWritten = result;
211 emit itemWriteCompleted(m_status->componentId, static_cast<int>(ExportItemType::PreviewImage), result);
212 // deleteLater removed - watchers are cleaned up by qDeleteAll below
213 });
214 watcher->setFuture(future);
215 watchers.append(watcher);
216 }
217
218 // 手册写入任务
219 if (m_exportDatasheet && !m_status->datasheetData.isEmpty()) {
220 QFuture<bool> future = QtConcurrent::run([this]() {
221 try {
222 return writeDatasheetFile(*m_status);
223 } catch (const std::bad_alloc& e) {
224 qCritical() << "Memory allocation failed in datasheet write:" << e.what();
225 return false;
226 } catch (const std::exception& e) {
227 qCritical() << "Exception in datasheet write:" << e.what();
228 return false;
229 }
230 });
231 QFutureWatcher<bool>* watcher = new QFutureWatcher<bool>();
232 connect(watcher, &QFutureWatcher<bool>::finished, this, [this, watcher]() {
233 bool result = watcher->result();
234 m_status->datasheetWritten = result;
235 emit itemWriteCompleted(m_status->componentId, static_cast<int>(ExportItemType::Datasheet), result);
236 // deleteLater removed - watchers are cleaned up by qDeleteAll below
237 });
238 watcher->setFuture(future);
239 watchers.append(watcher);
240 }
241
242 // 等待所有写入任务完成
243 for (QFutureWatcher<bool>* watcher : watchers) {
244 watcher->waitForFinished();
245 }
246
247 // 清理 watchers
248 qDeleteAll(watchers);
249 watchers.clear();
250
251 // 检查是否被取消
252 if (m_isAborted.loadRelaxed()) {
253 m_status->writeDurationMs = writeTimer.elapsed();
254 m_status->writeSuccess = false;
255 m_status->writeMessage = "Export cancelled";
256 m_status->isCancelled = true;
257 m_status->addDebugLog(QString("WriteWorker cancelled for component: %1").arg(m_status->componentId));
258 emit writeCompleted(m_status);
259 return;
260 }
261
262 // 根据用户请求的导出项和实际完成情况，决定最终的成功状态
263 bool symbolSuccess = (!m_exportSymbol) || m_status->symbolWritten;
264 bool footprintSuccess = (!m_exportFootprint) || m_status->footprintWritten;
265 bool model3DSuccess = (!m_exportModel3D) || m_status->model3DWritten;
266 bool previewImageSuccess = (!m_exportPreviewImages) || m_status->previewImageWritten;
267 bool datasheetSuccess = (!m_exportDatasheet) || m_status->datasheetWritten;
268
269 bool overallSuccess =
270 symbolSuccess && footprintSuccess && model3DSuccess && previewImageSuccess && datasheetSuccess;
271
272 QStringList successItems;
273 QStringList failedItems;
274 if (m_exportSymbol) {
275 symbolSuccess ? successItems.append("S") : failedItems.append("S");
276 }
277 if (m_exportFootprint) {
278 footprintSuccess ? successItems.append("F") : failedItems.append("F");
279 }
280 if (m_exportModel3D) {
281 model3DSuccess ? successItems.append("3D") : failedItems.append("3D");
282 }
283 if (m_exportPreviewImages) {
284 previewImageSuccess ? successItems.append("P") : failedItems.append("P");
285 }
286 if (m_exportDatasheet) {
287 datasheetSuccess ? successItems.append("D") : failedItems.append("D");
288 }
289
290 m_status->writeSuccess = overallSuccess;
291 if (overallSuccess) {
292 m_status->writeMessage = "Write completed successfully";
293 } else {
294 m_status->writeMessage = QString("Export completed with failures: %1").arg(failedItems.join(", "));
295 }
296
297 m_status->addDebugLog(QString("WriteWorker completed - Symbol:%1 Footprint:%2 3D:%3 Preview:%4 Datasheet:%5")
298 .arg(symbolSuccess ? "OK" : "FAIL")
299 .arg(footprintSuccess ? "OK" : "FAIL")
300 .arg(model3DSuccess ? "OK" : "FAIL")
301 .arg(previewImageSuccess ? "OK" : "FAIL")
302 .arg(datasheetSuccess ? "OK" : "FAIL"));
303
304 if (m_debugMode) {
305 exportDebugData(*m_status);
306 }
307
308 m_status->writeDurationMs = writeTimer.elapsed();
309 m_status->addDebugLog(QString("WriteWorker completed for component: %1, Success: %2, Duration: %3ms")
310 .arg(m_status->componentId)
311 .arg(m_status->writeSuccess)
312 .arg(m_status->writeDurationMs));
313
314 emit writeCompleted(m_status);

353 if (!QDir(m_tempDir).exists()) {

413bool WriteWorker::write3DModelFile(ComponentExportStatus& status) {
414 // 如果既没有原始数据也没有缓存路径，则跳过
415 const bool hasWrlData =
416 status.model3DData && (!status.model3DData->rawObj().isEmpty() || !status.cachedModel3DWrlPath.isEmpty());
417 const bool hasStepData = !status.model3DStepRaw.isEmpty() || !status.cachedModel3DStepPath.isEmpty();
418 if (!hasWrlData && !hasStepData) {
419 return true;
420 }
421
422 if (!QDir(m_tempDir).exists()) {
423 status.addDebugLog(QString("ERROR: Temp directory does not exist: %1").arg(m_tempDir));
424 return false;
425 }
426 if (!PathSecurity::isSafePath(m_tempDir, m_outputPath)) {
427 status.addDebugLog(
428 QString("SECURITY ERROR: Temp dir %1 is outside output path %2").arg(m_tempDir, m_outputPath));
429 return false;
430 }
431
432 QString modelsDirPath = QString("%1/%2.3dmodels").arg(m_outputPath, m_libName);
433 if (!createOutputDirectory(modelsDirPath)) {
434 status.addDebugLog(QString("ERROR: Failed to create 3D models directory: %1").arg(modelsDirPath));
435 return false;
436 }
437
438 QString footprintName = status.footprintData ? status.footprintData->info().name : status.componentId;
439 footprintName = PathSecurity::sanitizeFilename(footprintName);
440
441 bool wrlSuccess = false;
442
443 // WRL 是 OBJ 派生文件；如果有 OBJ 原始数据，始终重新生成，避免旧 WRL 缓存携带过期坐标归一化。
444 if (status.model3DData && !status.model3DData->rawObj().isEmpty()) {
445 QString wrlFilePath = QString("%1/%2.wrl").arg(modelsDirPath, footprintName);
446 wrlSuccess = AtomicFileWriter::writeAtomically(
447 m_tempDir, wrlFilePath, ".wrl.tmp", [this, &status](const QString& tempPath) -> bool {
448 return m_model3DExporter.exportToWrl(*status.model3DData, tempPath);
449 });
450 if (wrlSuccess) {
451 status.addDebugLog(QString("3D model WRL file written atomically: %1").arg(wrlFilePath));
452 } else {
453 status.addDebugLog(QString("ERROR: Failed to write WRL file"));
454 }
455 } else if (hasWrlData && !status.cachedModel3DWrlPath.isEmpty()) {
456 QString wrlFilePath = QString("%1/%2.wrl").arg(modelsDirPath, footprintName);
457 wrlSuccess = AtomicFileWriter::copyAtomically(status.cachedModel3DWrlPath, wrlFilePath, m_tempDir);
458 if (wrlSuccess) {
459 status.addDebugLog(QString("3D model WRL file copied from cache: %1").arg(status.cachedModel3DWrlPath));
460 } else {
461 status.addDebugLog(QString("ERROR: Failed to copy WRL file from cache"));
462 }
463 }
464
465 // STEP 文件保持服务器原始坐标系，避免破坏它与 WRL 模型原本一致的装配关系。
466 bool stepSuccess = false;
467 if (!status.cachedModel3DStepPath.isEmpty()) {
468 QString stepFilePath = QString("%1/%2.step").arg(modelsDirPath, footprintName);
469 stepSuccess = AtomicFileWriter::copyAtomically(status.cachedModel3DStepPath, stepFilePath, m_tempDir);
470 if (stepSuccess) {
471 status.addDebugLog(QString("3D model STEP file copied from cache: %1").arg(status.cachedModel3DStepPath));
472 } else {
473 status.addDebugLog(QString("WARNING: Failed to copy STEP file from cache: %1").arg(stepFilePath));
474 }
475 } else if (!status.model3DStepRaw.isEmpty()) {
476 QString stepFilePath = QString("%1/%2.step").arg(modelsDirPath, footprintName);
477 stepSuccess = AtomicFileWriter::writeAtomically(
478 m_tempDir, stepFilePath, ".step.tmp", [&status](const QString& tempPath) -> bool {
479 QFile file(tempPath);
480 if (!file.open(QIODevice::WriteOnly)) {
481 return false;
482 }
483 qint64 written = file.write(status.model3DStepRaw);
484 file.close();
485 return written == status.model3DStepRaw.size();
486 });
487
488 if (stepSuccess) {
489 status.addDebugLog(QString("3D model STEP file written"));
490 } else {
491 status.addDebugLog(QString("WARNING: Failed to write STEP file: %1").arg(stepFilePath));
492 }
493 status.clearStepData();
494 }
495
496 status.model3DWritten = wrlSuccess || stepSuccess;
497 return status.model3DWritten;
498}

570 stepFile.write(status.model3DStepRaw);

575
576 QJsonObject debugInfo;
577 debugInfo["componentId"] = status.componentId;
578 debugInfo["fetchSuccess"] = status.fetchSuccess;
579 debugInfo["fetchMessage"] = status.fetchMessage;
580 debugInfo["processSuccess"] = status.processSuccess;
581 debugInfo["processMessage"] = status.processMessage;
582 debugInfo["writeSuccess"] = status.writeSuccess;
583 debugInfo["writeMessage"] = status.writeMessage;
584
585 if (!status.debugLog.isEmpty()) {
586 QJsonArray logArray;
587 for (const QString& log : status.debugLog) {
588 logArray.append(log);
589 }
590 debugInfo["debugLog"] = logArray;
591 }
592
593 if (status.symbolData) {
594 QJsonObject symbolInfo = SymbolDataSerializer::toJson(status.symbolData->info());
595 symbolInfo["pinCount"] = status.symbolData->pins().size();
596 symbolInfo["rectangleCount"] = status.symbolData->rectangles().size();
597 symbolInfo["circleCount"] = status.symbolData->circles().size();
598 symbolInfo["arcCount"] = status.symbolData->arcs().size();
599 symbolInfo["polylineCount"] = status.symbolData->polylines().size();
600 symbolInfo["polygonCount"] = status.symbolData->polygons().size();
601 symbolInfo["pathCount"] = status.symbolData->paths().size();
602 symbolInfo["ellipseCount"] = status.symbolData->ellipses().size();

759bool WriteWorker::writePreviewImageFile(ComponentExportStatus& status) {
760 QDir outputDir(m_outputPath);
761 QString imagesDir = outputDir.filePath("images");
762 if (!createOutputDirectory(imagesDir)) {
763 status.addDebugLog(QString("ERROR: Failed to create images directory: %1").arg(imagesDir));
764 return false;
765 }
766
767 QString safeName = status.componentId;
768 safeName.replace(INVALID_FILENAME_CHARS, "_");
769
770 bool allSuccess = true;
771 int exportedCount = 0;
772
773 for (int i = 0; i < status.previewImageDataList.size(); ++i) {
774 if (status.previewImageDataList[i].isEmpty()) {
775 continue;
776 }
777
778 QString filename = QString("%1_%2.jpg").arg(safeName).arg(i);
779 QString filePath = QDir(imagesDir).filePath(filename);
780
781 QFile file(filePath);
782 if (file.open(QIODevice::WriteOnly)) {
783 file.write(status.previewImageDataList[i]);
784 file.close();
785 exportedCount++;
786 } else {
787 allSuccess = false;
788 status.addDebugLog(QString("ERROR: Failed to write preview image: %1").arg(filePath));
789 }
790 }
791
792 if (exportedCount > 0) {
793 status.addDebugLog(QString("Preview images written: %1 files to %2").arg(exportedCount).arg(imagesDir));
794 }
795 status.previewImageWritten = allSuccess;
796 return allSuccess;
797}
