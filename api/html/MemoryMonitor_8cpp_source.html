32qint64 MemoryMonitor::getCurrentUsage() {
33#if defined(Q_OS_LINUX)
34 QFile file("/proc/self/status");
35 if (!file.open(QIODevice::ReadOnly)) {
36 qWarning() << "Failed to open /proc/self/status:" << file.errorString();
37 return 0;
38 }
39 QByteArray content = file.readAll();
40 file.close();
41
42 QList<QByteArray> lines = content.split('\n');
43 for (const QByteArray& line : lines) {
44 if (line.startsWith("VmRSS:")) {
45 QByteArray value = line.mid(6).trimmed();
46 if (value.endsWith("kB") || value.endsWith("kb")) {
47 value = value.left(value.length() - 2);
48 } else if (value.endsWith("k")) {
49 value = value.left(value.length() - 1);
50 }
51 value = value.trimmed();
52 bool ok = false;
53 qint64 rssKb = value.toLongLong(&ok);
54 if (ok) {
55 return rssKb * 1024;
56 }
57 }
58 }
59 return 0;
60
61#elif defined(Q_OS_WIN) || defined(_WIN32)
62 PROCESS_MEMORY_COUNTERS pmc;
63 if (GetProcessMemoryInfo(GetCurrentProcess(), &pmc, sizeof(pmc))) {
64 return static_cast<qint64>(pmc.WorkingSetSize);
65 }
66 return 0;
67
68#elif defined(Q_OS_MAC)
69 struct task_basic_info info;
70 mach_msg_type_number_t size = TASK_BASIC_INFO_COUNT;
71 if (task_info(mach_task_self(), TASK_BASIC_INFO, reinterpret_cast<task_info_t>(&info), &size) == KERN_SUCCESS) {
72 return static_cast<qint64>(info.resident_size);
73 }
74 return 0;
75
76#else
77 return 0;
78#endif
79}
