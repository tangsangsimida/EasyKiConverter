281QByteArray FetchWorker::httpGet(const QString& url, int timeoutMs, QSharedPointer<ComponentExportStatus> status) {
282 QByteArray result;
283 int retryCount = 0;
284 QElapsedTimer requestTimer;
285
286 while (retryCount <= MAX_HTTP_RETRIES) {
287 // 检查是否已被中断 (原子检查，不加锁)
288 if (m_isAborted.loadRelaxed()) {
289 return QByteArray();
290 }
291
292 requestTimer.restart();
293
294 if (retryCount > 0) {
295 int delay = calculateRetryDelay(retryCount - 1);
296 qDebug() << "Retrying request to" << url << "in" << delay << "ms (Retry" << retryCount << "/"
297 << MAX_HTTP_RETRIES << ")";
298
299 // 使用分段休眠并发检查中断标志，而不是长时间阻塞休眠
300 const int step = 100;
301 for (int i = 0; i < delay; i += step) {
302 if (m_isAborted.loadRelaxed()) {
303 return QByteArray();
304 }
305 QThread::msleep(qMin(step, delay - i));
306 }
307 }
308 QNetworkRequest request{QUrl(url)};
309 request.setRawHeader("User-Agent", "EasyKiConverter/1.0");
310 request.setRawHeader("Accept", "application/json, text/javascript, */*; q=0.01");
311
312 QNetworkReply* reply = m_ownNetworkManager->get(request);
313
314 // 跟踪当前 reply 以便中断
315 {
316 QMutexLocker locker(&m_replyMutex);
317 if (m_isAborted.loadRelaxed()) {
318 reply->abort();
319 reply->deleteLater();
320 return QByteArray();
321 }
322 m_currentReply = reply;
323 }
324
325 QEventLoop loop;
326 QTimer timeoutTimer;
327 timeoutTimer.setSingleShot(true);
328
329 connect(reply, &QNetworkReply::finished, &loop, &QEventLoop::quit);
330 connect(&timeoutTimer, &QTimer::timeout, [&]() {
331 if (reply->isRunning()) {
332 reply->abort();
333 }
334 loop.quit();
335 });
336
337 timeoutTimer.start(timeoutMs);
338 loop.exec();
339
340 // 清除 reply 跟踪
341 {
342 QMutexLocker locker(&m_replyMutex);
343 m_currentReply = nullptr;
344 }
345
346 bool success = false;
347 if (reply->error() == QNetworkReply::NoError) {
348 int statusCode = reply->attribute(QNetworkRequest::HttpStatusCodeAttribute).toInt();
349 if (statusCode == 200) {
350 result = reply->readAll();
351 success = true;
352
353 // 记录网络诊断信息
354 if (status) {
355 ComponentExportStatus::NetworkDiagnostics diag;
356 diag.url = url;
357 diag.statusCode = statusCode;
358 diag.errorString = "";
359 diag.retryCount = retryCount;
360 diag.latencyMs = requestTimer.elapsed();
361 diag.wasRateLimited = false;
362 status->networkDiagnostics.append(diag);
363 }
364 } else if (statusCode == 429) {
365 qWarning() << "Rate limit detected (HTTP 429) for URL:" << url;
366 // 触发速率限制退避
367 QMutexLocker locker(&s_rateLimitMutex);
368 s_lastRateLimitTime = QDateTime::currentDateTime();
369 s_backoffMs = qMin(s_backoffMs * 2, 8000); // 指数退避，最大8秒
370 if (s_backoffMs == 0)
371 s_backoffMs = 1000; // 初始退避1秒
372 qWarning() << "Rate limit backoff increased to" << s_backoffMs
373 << "ms, active requests:" << s_activeRequests.loadRelaxed();
374
375 // 记录网络诊断信息
376 if (status) {
377 ComponentExportStatus::NetworkDiagnostics diag;
378 diag.url = url;
379 diag.statusCode = statusCode;
380 diag.errorString = "Rate Limited";
381 diag.retryCount = retryCount;
382 diag.latencyMs = requestTimer.elapsed();
383 diag.wasRateLimited = true;
384 status->networkDiagnostics.append(diag);
385 }
386 // Will retry
387 } else if (statusCode >= 500) {
388 qWarning() << "HTTP error" << statusCode << "for URL:" << url;
389 // Will retry
390 } else {
391 qWarning() << "HTTP error" << statusCode << "for URL:" << url << "(No retry for this code)";
392 retryCount = MAX_HTTP_RETRIES + 1; // Don't retry for other 4xx errors
393 }
394 } else if (reply->error() == QNetworkReply::OperationCanceledError) {
395 // 请求被取消（超时），不再重试
396 qWarning() << "Request timeout (cancelled) for URL:" << url << "after" << timeoutMs << "ms";
397
398 // 记录网络诊断信息
399 if (status) {
400 ComponentExportStatus::NetworkDiagnostics diag;
401 diag.url = url;
402 diag.statusCode = 0;
403 diag.errorString = "Timeout";
404 diag.retryCount = retryCount;
405 diag.latencyMs = requestTimer.elapsed();
406 diag.wasRateLimited = false;
407 status->networkDiagnostics.append(diag);
408 }
409
410 // 超时后进行正常重试（弱网环境下超时是最常见的错误类型）
411 // Will retry
412 } else {
413 qWarning() << "Network error:" << reply->errorString() << "URL:" << url;
414
415 // 记录网络诊断信息
416 if (status) {
417 ComponentExportStatus::NetworkDiagnostics diag;
418 diag.url = url;
419 diag.statusCode = reply->attribute(QNetworkRequest::HttpStatusCodeAttribute).toInt();
420 diag.errorString = reply->errorString();
421 diag.retryCount = retryCount;
422 diag.latencyMs = requestTimer.elapsed();
423 diag.wasRateLimited = false;
424 status->networkDiagnostics.append(diag);
425 }
426
427 // Will retry for other network errors (except timeout/cancelled)
428 }
429
430 reply->deleteLater();
431
432 if (success) {
433 if (!result.isEmpty() && result.size() >= 2 && (unsigned char)result[0] == 0x1f &&
434 (unsigned char)result[1] == 0x8b) {
435 result = decompressGzip(result);
436 }
437 return result;
438 }
439
440 retryCount++;
441 }
442
443 return QByteArray();
444}

446QByteArray FetchWorker::decompressGzip(const QByteArray& compressedData) {
447 if (compressedData.size() < 18) {
448 qWarning() << "Invalid gzip data size";
449 return QByteArray();
450 }
451
452 z_stream stream;
453 memset(&stream, 0, sizeof(stream));
454
455 if (inflateInit2(&stream, 15 + 16) != Z_OK) {
456 qWarning() << "Failed to initialize zlib";
457 return QByteArray();
458 }
459
460 stream.next_in = (Bytef*)compressedData.data();
461 stream.avail_in = compressedData.size();
462 stream.next_out = nullptr;
463 stream.avail_out = 0;
464
465 QByteArray decompressed;
466 const int bufferSize = 8192;
467 char buffer[bufferSize];
468
469 int ret;
470 do {
471 stream.avail_out = bufferSize;
472 stream.next_out = (Bytef*)buffer;
473
474 ret = inflate(&stream, Z_NO_FLUSH);
475
476 if (ret == Z_OK || ret == Z_STREAM_END) {
477 decompressed.append(buffer, bufferSize - stream.avail_out);
478 }
479 } while (ret == Z_OK);
480
481 inflateEnd(&stream);
482
483 if (ret != Z_STREAM_END) {
484 qWarning() << "Failed to decompress gzip data, error:" << ret;
485 return QByteArray();
486 }
487
488 return decompressed;
489}

491QByteArray FetchWorker::decompressZip(const QByteArray& zipData) {
492 if (zipData.size() < 30) {
493 qWarning() << "Invalid ZIP data size";
494 return QByteArray();
495 }
496
497 QByteArray result;
498 int pos = 0;
499
500 while (pos < zipData.size() - 30) {
501 if (zipData[pos] != 0x50 || zipData[pos + 1] != 0x03 || zipData[pos + 2] != 0x04 || zipData[pos + 3] != 0x4b) {
502 pos++;
503 continue;
504 }
505
506 std::uint16_t compressedSize = ((std::uint8_t)zipData[pos + 18] | ((std::uint8_t)zipData[pos + 19] << 8));
507 std::uint16_t uncompressedSize = ((std::uint8_t)zipData[pos + 22] | ((std::uint8_t)zipData[pos + 23] << 8));
508 std::uint16_t fileNameLen = ((std::uint8_t)zipData[pos + 26] | ((std::uint8_t)zipData[pos + 27] << 8));
509 std::uint16_t extraFieldLen = ((std::uint8_t)zipData[pos + 28] | ((std::uint8_t)zipData[pos + 29] << 8));
510 int dataStart = pos + 30 + fileNameLen + extraFieldLen;
511 std::uint16_t compressionMethod = ((std::uint8_t)zipData[pos + 8] | ((std::uint8_t)zipData[pos + 9] << 8));
512 if (compressionMethod == 8) {
513 QByteArray compressedData = zipData.mid(dataStart, compressedSize);
514 QByteArray decompressedData = decompressGzip(compressedData);
515 result.append(decompressedData);
516 } else if (compressionMethod == 0) {
517 QByteArray uncompressedData = zipData.mid(dataStart, uncompressedSize);
518 result.append(uncompressedData);
519 } else {
520 qWarning() << "Unsupported ZIP compression method:" << compressionMethod;
521 }
522
523 pos = dataStart + compressedSize;
524 }
525
526 return result;
527}

681int FetchWorker::calculateRetryDelay(int retryCount) {
682 // 使用递增延迟表，超出范围则使用最大值
683 int baseDelay;
684 if (retryCount >= 0 && retryCount < static_cast<int>(sizeof(RETRY_DELAYS_MS) / sizeof(RETRY_DELAYS_MS[0]))) {
685 baseDelay = RETRY_DELAYS_MS[retryCount];
686 } else {
687 baseDelay = RETRY_DELAYS_MS[sizeof(RETRY_DELAYS_MS) / sizeof(RETRY_DELAYS_MS[0]) - 1];
688 }
689
690 // 添加 ±20% 随机抖动，避免惊群效应
691 int jitter = static_cast<int>(baseDelay * 0.2);
692 int randomOffset = QRandomGenerator::global()->bounded(-jitter, jitter + 1);
693 return qMax(100, baseDelay + randomOffset); // 确保最小延迟100ms
694}
