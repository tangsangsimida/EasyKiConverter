30 QMutexLocker locker(&m_mutex);

258QByteArray FetchWorker::httpGet(const QString& url,
259 int timeoutMs,
260 QSharedPointer<ComponentExportStatus> status,
261 ResourceType resourceType) {
262 if (m_isAborted.loadRelaxed()) {
263 return QByteArray();
264 }
265
266 RequestProfile profile = RequestProfiles::fromType(resourceType);
267 profile.connectTimeoutMs = timeoutMs;
268 profile.readTimeoutMs = qMax(profile.readTimeoutMs, timeoutMs);
269 bool weakNetworkSupport = false;
270 if (!m_networkClient) {
271 weakNetworkSupport = ConfigService::instance()->getWeakNetworkSupport();
272 }
273 RetryPolicy policy = RetryPolicy::fromProfile(profile, weakNetworkSupport);
274
275 auto finishResult = [this, url, status](const NetworkResult& result) -> QByteArray {
276 if (result.wasCancelled && m_isAborted.loadRelaxed()) {
277 return QByteArray();
278 }
279
280 if (status) {
281 ComponentExportStatus::NetworkDiagnostics diag;
282 diag.url = result.diagnostic.url.isEmpty() ? url : result.diagnostic.url;
283 diag.statusCode = result.statusCode;
284 diag.errorString = result.success ? QString() : result.error;
285 diag.retryCount = result.retryCount;
286 diag.latencyMs = result.elapsedMs;
287 diag.wasRateLimited = result.diagnostic.wasRateLimited;
288 status->networkDiagnostics.append(diag);
289 }
290
291 if (!result.success) {
292 if (result.statusCode == 429) {
293 QMutexLocker locker(&s_rateLimitMutex);
294 s_lastRateLimitTime = QDateTime::currentDateTime();
295 s_backoffMs = qMin(s_backoffMs == 0 ? 1000 : s_backoffMs * 2, 8000);
296 }
297 return QByteArray();
298 }
299
300 // Note: AsyncNetworkRequest already handles gzip decompression internally
301 // based on Content-Encoding header or magic bytes. result.data is already decompressed.
302 return result.data;
303 };
304
305 if (m_networkClient) {
306 return finishResult(m_networkClient->get(QUrl(url), resourceType, policy));
307 }
308
309 AsyncNetworkRequest* request = NetworkClient::instance().getAsync(QUrl(url), resourceType, policy);
310 if (!request) {
311 if (status) {
312 ComponentExportStatus::NetworkDiagnostics diag;
313 diag.url = url;
314 diag.errorString = QStringLiteral("Failed to create network request");
315 status->networkDiagnostics.append(diag);
316 }
317 return QByteArray();
318 }
319
320 // 白名单拒绝等场景：request 已完成，直接返回结果，避免阻塞等待
321 if (request->isFinished()) {
322 const NetworkResult result = request->result();
323 delete request;
324 return finishResult(result);
325 }
326
327 {
328 QMutexLocker locker(&m_replyMutex);
329 m_currentRequest = request;
330 }
331
332 auto context = std::make_shared<BlockingRequestContext>();
333 QObject::connect(
334 request,
335 &AsyncNetworkRequest::finished,
336 request,
337 [context](const NetworkResult& result) { context->complete(result); },
338 Qt::DirectConnection);
339
340 const NetworkResult result = context->wait();
341
342 {
343 QMutexLocker locker(&m_replyMutex);
344 if (m_currentRequest == request) {
345 m_currentRequest = nullptr;
346 }
347 }
348 QMetaObject::invokeMethod(request, &QObject::deleteLater, Qt::QueuedConnection);
349
350 return finishResult(result);
351}

353QByteArray FetchWorker::decompressZip(const QByteArray& zipData) {
354 if (zipData.size() < 30) {
355 qWarning() << "Invalid ZIP data size";
356 return QByteArray();
357 }
358
359 QByteArray result;
360 int pos = 0;
361
362 while (pos < zipData.size() - 30) {
363 if (zipData[pos] != 0x50 || zipData[pos + 1] != 0x03 || zipData[pos + 2] != 0x04 || zipData[pos + 3] != 0x4b) {
364 pos++;
365 continue;
366 }
367
368 std::uint16_t compressedSize = ((std::uint8_t)zipData[pos + 18] | ((std::uint8_t)zipData[pos + 19] << 8));
369 std::uint16_t uncompressedSize = ((std::uint8_t)zipData[pos + 22] | ((std::uint8_t)zipData[pos + 23] << 8));
370 std::uint16_t fileNameLen = ((std::uint8_t)zipData[pos + 26] | ((std::uint8_t)zipData[pos + 27] << 8));
371 std::uint16_t extraFieldLen = ((std::uint8_t)zipData[pos + 28] | ((std::uint8_t)zipData[pos + 29] << 8));
372 int dataStart = pos + 30 + fileNameLen + extraFieldLen;
373 std::uint16_t compressionMethod = ((std::uint8_t)zipData[pos + 8] | ((std::uint8_t)zipData[pos + 9] << 8));
374 if (compressionMethod == 8) {
375 QByteArray compressedData = zipData.mid(dataStart, compressedSize);
376 GzipUtils::DecompressResult decompResult = GzipUtils::decompress(compressedData);
377 if (decompResult.success) {
378 result.append(decompResult.data);
379 } else {
380 qWarning() << "Failed to decompress ZIP entry at pos" << pos;
381 }
382 } else if (compressionMethod == 0) {
383 QByteArray uncompressedData = zipData.mid(dataStart, uncompressedSize);
384 result.append(uncompressedData);
385 } else {
386 qWarning() << "Unsupported ZIP compression method:" << compressionMethod;
387 }
388
389 pos = dataStart + compressedSize;
390 }
391
392 return result;
393}

395bool FetchWorker::fetch3DModelData(QSharedPointer<ComponentExportStatus> status) {
396 QString uuid;
397
398 QJsonParseError parseError;
399 QJsonDocument doc = QJsonDocument::fromJson(status->cadDataRaw, &parseError);
400 if (parseError.error != QJsonParseError::NoError) {
401 QString msg = "Failed to parse CAD data for 3D model UUID";
402 status->addDebugLog(msg);
403 qWarning() << msg;
404 return false;
405 }
406
407 QJsonObject rootObj = doc.object();
408 QJsonObject obj;
409 if (rootObj.contains("result") && rootObj["result"].isObject()) {
410 obj = rootObj["result"].toObject();
411 } else {
412 obj = rootObj;
413 }
414
415 if (obj.contains("packageDetail") && obj["packageDetail"].isObject()) {
416 QJsonObject packageDetail = obj["packageDetail"].toObject();
417 if (packageDetail.contains("dataStr")) {
418 QJsonObject dataObj;
419 if (packageDetail["dataStr"].isObject()) {
420 dataObj = packageDetail["dataStr"].toObject();
421 } else if (packageDetail["dataStr"].isString()) {
422 QString dataStrRaw = packageDetail["dataStr"].toString();
423 if (!dataStrRaw.isEmpty()) {
424 QJsonDocument dataDoc = QJsonDocument::fromJson(dataStrRaw.toUtf8());
425 dataObj = dataDoc.object();
426 }
427 }
428
429 if (!dataObj.isEmpty() && dataObj.contains("shape") && dataObj["shape"].isArray()) {
430 QJsonArray shapes = dataObj["shape"].toArray();
431 for (const QJsonValue& shapeVal : shapes) {
432 QString shapeStr = shapeVal.toString();
433 if (!shapeStr.startsWith("SVGNODE~")) {
434 continue;
435 }
436 int tildeIndex = shapeStr.indexOf('~');
437 if (tildeIndex == -1) {
438 continue;
439 }
440 QString jsonStr = shapeStr.mid(tildeIndex + 1);
441 QJsonDocument nodeDoc = QJsonDocument::fromJson(jsonStr.toUtf8());
442 QJsonObject nodeObj = nodeDoc.object();
443 if (!nodeObj.contains("attrs") || !nodeObj["attrs"].isObject()) {
444 continue;
445 }
446 QJsonObject attrs = nodeObj["attrs"].toObject();
447 if (attrs.contains("c_etype") && attrs["c_etype"].toString() == "outline3D" &&
448 attrs.contains("uuid") && !attrs["uuid"].toString().isEmpty()) {
449 uuid = attrs["uuid"].toString();
450 qDebug() << "Found 3D model UUID from outline3D SVGNODE:" << uuid;
451 break;
452 }
453 }
454 }
455 }
456 }
457
458 if (uuid.isEmpty()) {
459 QString msg = "No 3D model UUID found in CAD data";
460 status->addDebugLog(msg);
461 qDebug() << msg;
462 return false;
463 }
464
465 status->addDebugLog(QString("Fetching 3D model for UUID: %1").arg(uuid));
466
467 QString objUrl = QString("https://modules.easyeda.com/3dmodel/%1").arg(uuid);
468 status->addDebugLog(QString("Downloading 3D model from: %1").arg(objUrl));
469 QByteArray objData = httpGet(objUrl, MODEL_3D_TIMEOUT_MS, status, ResourceType::Model3DObj);
470 if (objData.isEmpty()) {
471 QString msg = QString("ERROR: Failed to download 3D model from: %1").arg(objUrl);
472 status->addDebugLog(msg);
473 qWarning() << msg;
474 return false;
475 }
476
477 QByteArray actualObjData = objData;
478 if (objData.size() >= 2 && objData[0] == 0x50 && objData[1] == 0x4B) {
479 status->addDebugLog("3D model data is ZIP compressed, decompressing...");
480 actualObjData = decompressZip(objData);
481 }
482
483 if (actualObjData.size() >= 2 && static_cast<unsigned char>(actualObjData[0]) == 0x1f &&
484 static_cast<unsigned char>(actualObjData[1]) == 0x8b) {
485 status->addDebugLog("3D model data is gzip compressed, decompressing...");
486 GzipUtils::DecompressResult decompResult = GzipUtils::decompress(actualObjData);
487 if (decompResult.success) {
488 actualObjData = decompResult.data;
489 } else {
490 status->addDebugLog("ERROR: Gzip decompression failed for 3D model data");
491 }
492 }
493
494 if (actualObjData.isEmpty()) {
495 QString msg = "ERROR: Failed to decompress 3D model data";
496 status->addDebugLog(msg);
497 qWarning() << msg;
498 return false;
499 }
500
501 status->model3DObjRaw = actualObjData;
502 status->addDebugLog(QString("3D model (OBJ) data fetched successfully for: %1").arg(status->componentId));
503
504 QString stepUrl = QString("https://modules.easyeda.com/qAxj6KHrDKw4blvCG8QJPs7Y/%1").arg(uuid);
505 status->addDebugLog(QString("Downloading STEP model from: %1").arg(stepUrl));
506 QByteArray stepData = httpGet(stepUrl, MODEL_3D_TIMEOUT_MS, status, ResourceType::Model3DStep);
507 if (!stepData.isEmpty() && stepData.size() > 100) {
508 status->model3DStepRaw = stepData;
509 status->addDebugLog(QString("3D model (STEP) data fetched successfully for: %1, Size: %2 bytes")
510 .arg(status->componentId)
511 .arg(stepData.size()));
512 } else {
513 status->addDebugLog(
514 QString("WARNING: Failed to download STEP model or STEP data too small for: %1").arg(status->componentId));
515 }
516
517 return true;
518}

531int FetchWorker::calculateRetryDelay(int retryCount) {
532 int baseDelay;
533 if (retryCount >= 0 && retryCount < static_cast<int>(sizeof(RETRY_DELAYS_MS) / sizeof(RETRY_DELAYS_MS[0]))) {
534 baseDelay = RETRY_DELAYS_MS[retryCount];
535 } else {
536 baseDelay = RETRY_DELAYS_MS[sizeof(RETRY_DELAYS_MS) / sizeof(RETRY_DELAYS_MS[0]) - 1];
537 }
538
539 int jitter = static_cast<int>(baseDelay * 0.2);
540 int randomOffset = QRandomGenerator::global()->bounded(-jitter, jitter + 1);
541 return qMax(100, baseDelay + randomOffset);
542}
