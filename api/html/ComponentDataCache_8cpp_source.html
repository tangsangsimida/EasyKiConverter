82void ComponentDataCache::flushToDisk(const QString& componentId) {
83 QMutexLocker locker(&m_cacheMutex);
84
85 if (m_cacheDir.isEmpty()) {
86 qWarning() << "ComponentDataCache: Cache dir not set";
87 return;
88 }
89
90 QStringList idsToFlush;
91 if (componentId.isEmpty()) {
92 idsToFlush = m_cache.keys();
93 } else if (m_cache.contains(componentId)) {
94 idsToFlush.append(componentId);
95 } else {
96 return;
97 }
98
99 for (const QString& id : idsToFlush) {
100 const auto& data = m_cache.value(id);
101 if (!data)
102 continue;
103
104 // 构建元器件数据文件路径
105 QString dataFilePath = m_cacheDir + QStringLiteral("/") + id + QStringLiteral(".json");
106
107 QJsonObject json;
108 json[QStringLiteral("componentId")] = id;
109
110 // 序列化符号数据
111 if (data->symbolData()) {
112 QJsonObject symbolJson = SymbolDataSerializer::toJson(*data->symbolData());
113 json[QStringLiteral("symbolData")] = symbolJson;
114 }
115
116 // 序列化封装数据
117 if (data->footprintData()) {
118 QJsonObject footprintJson = FootprintDataSerializer::toJson(*data->footprintData());
119 json[QStringLiteral("footprintData")] = footprintJson;
120 }
121
122 // 写入JSON文件
123 QFile file(dataFilePath);
124 if (file.open(QIODevice::WriteOnly)) {
125 file.write(QJsonDocument(json).toJson());
126 file.close();
127 } else {
128 qWarning() << "ComponentDataCache: Failed to write cache file" << dataFilePath;
129 }
130 }
131}

133bool ComponentDataCache::loadFromDisk(const QString& componentId) {
134 if (m_cacheDir.isEmpty()) {
135 return false;
136 }
137
138 QString dataFilePath = m_cacheDir + QStringLiteral("/") + componentId + QStringLiteral(".json");
139 QFile file(dataFilePath);
140
141 if (!file.exists()) {
142 return false;
143 }
144
145 if (!file.open(QIODevice::ReadOnly)) {
146 return false;
147 }
148
149 QByteArray jsonData = file.readAll();
150 file.close();
151
152 QJsonParseError error;
153 QJsonDocument doc = QJsonDocument::fromJson(jsonData, &error);
154 if (error.error != QJsonParseError::NoError) {
155 qWarning() << "ComponentDataCache: Failed to parse cache file" << error.errorString();
156 return false;
157 }
158
159 // 反序列化为ComponentData
160 QJsonObject json = doc.object();
161 auto data = QSharedPointer<ComponentData>::create();
162
163 if (json.contains(QStringLiteral("symbolData"))) {
164 QJsonObject symbolJson = json[QStringLiteral("symbolData")].toObject();
165 SymbolData symbolData;
166 if (SymbolDataSerializer::fromJson(symbolData, symbolJson)) {
167 data->setSymbolData(QSharedPointer<SymbolData>::create(symbolData));
168 }
169 }
170
171 if (json.contains(QStringLiteral("footprintData"))) {
172 QJsonObject footprintJson = json[QStringLiteral("footprintData")].toObject();
173 FootprintData footprintData;
174 if (FootprintDataSerializer::fromJson(footprintData, footprintJson)) {
175 data->setFootprintData(QSharedPointer<FootprintData>::create(footprintData));
176 }
177 }
178
179 put(componentId, data);
180 return true;
181}

183void ComponentDataCache::clearDiskCache(const QString& componentId) {
184 if (m_cacheDir.isEmpty()) {
185 return;
186 }
187
188 if (componentId.isEmpty()) {
189 // 清除整个缓存目录
190 QDir cacheDir(m_cacheDir);
191 cacheDir.removeRecursively();
192 setCacheDir(m_cacheDir); // 重建目录结构
193 } else {
194 // 清除指定元器件的缓存文件
195 QString dataFilePath = m_cacheDir + QStringLiteral("/") + componentId + QStringLiteral(".json");
196 QFile::remove(dataFilePath);
197
198 // 清除关联的文件
199 QStringList subDirs = {QStringLiteral("symbols"),
200 QStringLiteral("footprints"),
201 QStringLiteral("3dmodels"),
202 QStringLiteral("datasheets"),
203 QStringLiteral("previews")};
204
205 for (const QString& subDir : subDirs) {
206 QString dirPath = m_cacheDir + QStringLiteral("/") + subDir;
207 QDir subDirHandle(dirPath);
208 QStringList filters = {componentId + QStringLiteral("*")};
209 QStringList files = subDirHandle.entryList(filters);
210 for (const QString& file : files) {
211 subDirHandle.remove(file);
212 }
213 }
214 }
215}
