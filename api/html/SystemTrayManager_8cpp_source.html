120void SystemTrayManager::showExportNotification(int successCount,
121 int failureCount,
122 int symbolCount,
123 int footprintCount,
124 int model3DCount,
125 const QString& duration) {
126 if (!m_trayIcon) {
127 qDebug() << "System tray icon not initialized";
128 return;
129 }
130
131 if (!QSystemTrayIcon::supportsMessages()) {
132 qDebug() << "System tray does not support messages on this platform";
133 return;
134 }
135
136 QString title;
137 QString message;
138 QSystemTrayIcon::MessageIcon iconType = QSystemTrayIcon::Information;
139
140 if (failureCount > 0) {
141 title = tr("导出完成");
142
143 double successRate = 0.0;
144 if (successCount + failureCount > 0) {
145 successRate = (static_cast<double>(successCount) * 100.0) / (successCount + failureCount);
146 }
147
148 if (successCount == 0) {
149 iconType = QSystemTrayIcon::Critical;
150 message = tr("导出失败：%1 个元器件全部失败").arg(successCount + failureCount);
151 } else if (successRate < 50.0) {
152 iconType = QSystemTrayIcon::Warning;
153 message = tr("成功 %1 个，失败 %2 个").arg(successCount).arg(failureCount);
154 } else {
155 message = tr("成功 %1 个，失败 %2 个").arg(successCount).arg(failureCount);
156 }
157
158 if (successCount > 0) {
159 message += tr("\n输出：符号 %1 · 封装 %2 · 3D %3").arg(symbolCount).arg(footprintCount).arg(model3DCount);
160 }
161 } else {
162 iconType = QSystemTrayIcon::Information;
163 title = tr("导出完成");
164
165 if (successCount == 1) {
166 message = tr("成功导出 1 个元器件");
167 } else {
168 message = tr("成功导出 %1 个元器件").arg(successCount);
169 }
170
171 QStringList stats;
172 stats << tr("输出：符号 %1 · 封装 %2 · 3D %3").arg(symbolCount).arg(footprintCount).arg(model3DCount);
173
174 if (!duration.isEmpty()) {
175 stats << tr("耗时：%1").arg(duration);
176 }
177
178 message += "\n" + stats.join("\n");
179 }
180
181 qDebug() << "Attempting to show notification:" << title << message;
182 m_trayIcon->showMessage(title, message, iconType, 12000);
183 qDebug() << "Notification shown successfully";
184}
