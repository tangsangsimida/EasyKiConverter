22class BlockingRequestContext {
23public:
24 void complete(const NetworkResult& result) {
25 QMutexLocker locker(&m_mutex);
26 if (m_finished) {
27 return;
28 }
29 m_result = result;
30 m_finished = true;
31 m_condition.wakeAll();
32 }
33
34 NetworkResult wait() {
35 QMutexLocker locker(&m_mutex);
36 while (!m_finished) {
37 m_condition.wait(&m_mutex);
38 }
39 return m_result;
40 }
41
42private:
43 QMutex m_mutex;
44 QWaitCondition m_condition;
45 NetworkResult m_result;

60void NetworkWorker::run() {
61 qDebug() << "NetworkWorker started for:" << m_componentId << "- TaskType:" << static_cast<int>(m_taskType);
62
63 bool success = false;
64
65 switch (m_taskType) {
66 case TaskType::FetchComponentInfo:
67 success = fetchComponentInfo();
68 break;
69 case TaskType::FetchCadData:
70 success = fetchCadData();
71 break;
72 case TaskType::Fetch3DModelObj:
73 success = fetch3DModelObj();
74 break;
75 case TaskType::Fetch3DModelMtl:
76 success = fetch3DModelMtl();
77 break;
78 default:
79 qWarning() << "Unknown task type:" << static_cast<int>(m_taskType);
80 emit fetchError(m_componentId, "Unknown task type");
81 return;
82 }
83
84 if (!success) {
85 qWarning() << "Network request failed for:" << m_componentId;
86 }
87}

167bool NetworkWorker::executeRequest(const QUrl& url,
168 ResourceType resourceType,
169 int timeoutMs,
170 int maxRetries,
171 QByteArray& outData,
172 QString& errorMsg) {
173 RequestProfile profile = RequestProfiles::fromType(resourceType);
174 profile.connectTimeoutMs = timeoutMs;
175 profile.readTimeoutMs = qMax(profile.readTimeoutMs, timeoutMs);
176 profile.maxRetries = maxRetries;
177 RetryPolicy policy = RetryPolicy::fromProfile(profile, ConfigService::instance()->getWeakNetworkSupport());
178
179 AsyncNetworkRequest* request = NetworkClient::instance().getAsync(url, resourceType, policy);
180 if (!request) {
181 errorMsg = QStringLiteral("Failed to create network request");
182 return false;
183 }
184
185 // 白名单拒绝等场景：request 已完成，直接返回结果，避免阻塞等待
186 if (request->isFinished()) {
187 const NetworkResult doneResult = request->result();
188 delete request;
189 if (doneResult.wasCancelled) {
190 errorMsg = "Request cancelled";
191 return false;
192 }
193 if (!doneResult.success) {
194 errorMsg =
195 doneResult.error.isEmpty()
196 ? QString("Network error for %1 after %2 retries").arg(url.toString()).arg(doneResult.retryCount)
197 : doneResult.error;
198 qWarning() << "NetworkWorker:" << errorMsg << "for:" << m_componentId;
199 return false;
200 }
201 outData = doneResult.data;
202 return true;
203 }
204
205 {
206 QMutexLocker locker(&m_mutex);
207 m_currentRequest = request;
208 }
209
210 auto context = std::make_shared<BlockingRequestContext>();
211 QObject::connect(
212 request,
213 &AsyncNetworkRequest::finished,
214 request,
215 [context](const NetworkResult& result) { context->complete(result); },
216 Qt::DirectConnection);
217
218 const NetworkResult result = context->wait();
219
220 {
221 QMutexLocker locker(&m_mutex);
222 if (m_currentRequest == request) {
223 m_currentRequest = nullptr;
224 }
225 }
226 QMetaObject::invokeMethod(request, &QObject::deleteLater, Qt::QueuedConnection);
227
228 if (result.wasCancelled) {
229 errorMsg = "Request cancelled";
230 return false;
231 }
232
233 if (!result.success) {
234 errorMsg = result.error.isEmpty()
235 ? QString("Network error for %1 after %2 retries").arg(url.toString()).arg(result.retryCount)
236 : result.error;
237 qWarning() << "NetworkWorker:" << errorMsg << "for:" << m_componentId;
238 return false;
239 }
240
241 outData = result.data;
242 return true;
243}
