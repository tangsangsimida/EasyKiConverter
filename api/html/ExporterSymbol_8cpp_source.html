43bool ExporterSymbol::exportSymbolLibrary(const QList<SymbolData>& symbols,
44 const QString& libName,
45 const QString& filePath,
46 bool appendMode,
47 bool updateMode,
48 const QString& libraryDescription) {
49 qDebug() << "=== Export Symbol Library ===";
50 qDebug() << "Library name:" << libName;
51 qDebug() << "Output path:" << filePath;
52 qDebug() << "Symbol count:" << symbols.count();
53 qDebug() << "KiCad version: V6";
54 qDebug() << "Append mode:" << appendMode;
55 qDebug() << "Update mode:" << updateMode;
56 qDebug() << "Library description:" << libraryDescription;
57
58 // 重置检测到的版本，避免影响后续调用
59 m_detectedVersion.clear();
60
61 QFile file(filePath);
62 bool fileExists = file.exists();
63
64 // 如果文件不存在，直接创建新库
65 if (!fileExists) {
66 qDebug() << "Creating new symbol library...";
67 if (!file.open(QIODevice::WriteOnly | QIODevice::Text)) {
68 qWarning() << "Failed to open file for writing:" << filePath;
69 return false;
70 }
71
72 QTextStream out(&file);
73 out.setEncoding(QStringConverter::Utf8);
74
75 // 生成头部
76 out << generateHeader(libName);
77
78 // 生成所有符号
79 int index = 0;
80 for (const SymbolData& symbol : symbols) {
81 qDebug() << "Exporting symbol" << (++index) << "of" << symbols.count() << ":" << symbol.info().name;
82 out << generateSymbolContent(symbol, libName);
83 }
84
85 // 生成尾部
86 out << ")\n"; // 闭合 kicad_symbol_lib
87
88 file.close();
89 qDebug() << "Symbol library created successfully:" << filePath;
90 return true;
91 }
92
93 // 文件存在，需要处理追加或更新
94 qDebug() << "Existing library found, reading content...";
95
96 // 读取现有库内容
97 QString existingContent;
98 if (file.open(QIODevice::ReadOnly | QIODevice::Text)) {
99 existingContent = QTextStream(&file).readAll();
100 file.close();
101 qDebug() << "Read" << existingContent.length() << "bytes from existing library";
102 } else {
103 qWarning() << "Failed to open existing library for reading:" << filePath;
104 return false;
105 }
106
107 // 检测现有文件的版本
108 QRegularExpression versionRegex(R"(^\s*\‍(version\s+(\d+)\))", QRegularExpression::MultilineOption);
109 QRegularExpressionMatch versionMatch = versionRegex.match(existingContent);
110 if (versionMatch.hasMatch()) {
111 m_detectedVersion = versionMatch.captured(1);
112 qDebug() << "Detected existing library version:" << m_detectedVersion;
113 } else {
114 m_detectedVersion = "20211014";
115 qDebug() << "Could not detect version, using default:" << m_detectedVersion;
116 }
117
118 // 提取现有符号
119 QMap<QString, QString> existingSymbols; // 符号-> 符号内容
120 QSet<QString> subSymbolNames; // 属于分体式符号的子符号名称
121
122 // 使用栈来正确追踪符号的嵌套关系
123 QStringList lines = existingContent.split('\n');
124 int braceCount = 0;
125
126 // 栈元素: (symbolName, startLine, braceCountAfterStart)
127 struct SymbolInfo {
128 QString name;
129 int startLine;
130 int braceAfterStart;
131 };
132
133 QList<SymbolInfo> symbolStack;
134
135 for (int i = 0; i < lines.size(); ++i) {
136 QString line = lines[i];
137
138 // 逐字符计算括号变化
139 for (int j = 0; j < line.length(); ++j) {
140 if (line[j] == '(') {
141 braceCount++;
142 } else if (line[j] == ')') {
143 braceCount--;
144 }
145 }
146
147 // 检查是否有符号开始
148 int symbolPosInLine = line.indexOf("(symbol \"");
149 if (symbolPosInLine >= 0) {
150 int nameStart = line.indexOf("\"", symbolPosInLine) + 1;
151 int nameEnd = line.indexOf("\"", nameStart);
152 if (nameEnd > nameStart) {
153 QString symbolName = line.mid(nameStart, nameEnd - nameStart);
154 SymbolInfo info;
155 info.name = symbolName;
156 info.startLine = i;
157 info.braceAfterStart = braceCount;
158 symbolStack.append(info);
159 qDebug() << "Found symbol start:" << symbolName << "at line" << i << "with braceCount" << braceCount;
160 }
161 }
162
163 // 检查是否有符号结束 - 当 braceCount 回到符号开始后的值时
164 // 注意：可能有多个符号同时结束
165 while (!symbolStack.isEmpty() && braceCount < symbolStack.last().braceAfterStart) {
166 SymbolInfo info = symbolStack.last();
167 symbolStack.removeLast();
168
169 // 检查是否是子符号 - 如果栈不为空，说明有父符号
170 if (!symbolStack.isEmpty()) {
171 // 这是一个子符号，不应该被单独提取
172 subSymbolNames.insert(info.name);
173 qDebug() << "Found sub-symbol (skipping):" << info.name << "inside parent" << symbolStack.last().name;
174 } else {
175 // 这是一个顶层符号，提取它
176 QString symbolContent;
177 for (int k = info.startLine; k <= i; ++k) {
178 symbolContent += lines[k] + "\n";
179 }
180 existingSymbols[info.name] = symbolContent;
181 qDebug() << "Extracted top-level symbol:" << info.name << "from lines" << info.startLine << "-" << i;
182 }
183 }
184 }
185
186 qDebug() << "Existing symbols count:" << existingSymbols.count();
187 qDebug() << "Sub-symbol names:" << subSymbolNames;
188
189 // 确定要导出的符号
190 QList<SymbolData> symbolsToExport;
191 int overwriteCount = 0;
192 int appendCount = 0;
193 int skipCount = 0;
194
195 for (const SymbolData& symbol : symbols) {
196 QString symbolName = symbol.info().name;
197
198 if (existingSymbols.contains(symbolName)) {
199 if (appendMode && !updateMode) {
200 // 追加模式（非更新）：跳过已存在的符号
201 qDebug() << "Symbol already exists, skipping (append mode):" << symbolName;
202 skipCount++;
203 } else {
204 // 更新模式或覆盖模式：替换已存在的符号
205 qDebug() << "Symbol already exists, overwriting (update mode):" << symbolName;
206 overwriteCount++;
207 symbolsToExport.append(symbol);
208 }
209 } else {
210 // 新符号
211 qDebug() << "New symbol, adding:" << symbolName;
212 appendCount++;
213 symbolsToExport.append(symbol);
214 }
215 }
216
217 qDebug() << "Symbols to export:" << symbolsToExport.count() << "(Overwrite:" << overwriteCount
218 << ", Append:" << appendCount << ", Skip:" << skipCount << ")";
219
220 // 如果没有符号需要导出，直接返回
221 if (symbolsToExport.isEmpty()) {
222 qDebug() << "No symbols to export, skipping";
223 return true;
224 }
225
226 // 打开文件进行写入
227 if (!file.open(QIODevice::WriteOnly | QIODevice::Text)) {
228 qWarning() << "Failed to open file for writing:" << filePath;
229 return false;
230 }
231
232 QTextStream out(&file);
233 out.setEncoding(QStringConverter::Utf8);
234
235 // 生成头部
236 out << generateHeader(libName);
237
238 // 生成所有符号（包括未覆盖的现有符号和新导出的符号）
239 int index = 0;
240
241 // 收集要被覆盖的符号名
242 QSet<QString> overwrittenSymbolNames;
243 for (const SymbolData& symbol : symbolsToExport) {
244 overwrittenSymbolNames.insert(symbol.info().name);
245 }
246
247 // 收集要被删除的子符号名称（属于被覆盖的父符号的子符号）
248 // 改进的逻辑：准确识别分体式符号的子符号
249 QSet<QString> subSymbolsToDelete;
250
251 // 首先分析现有符号，确定哪些是分体式符号（有多个子符号）
252 QMap<QString, QStringList> parentToSubSymbols; // 父符号名 -> 子符号列
253 for (const QString& subSymbolName : subSymbolNames) {
254 // 从子符号名称中提取父符号
255 // 子符号格式：{parentName}_{unitNumber}_1
256 int lastUnderscore = subSymbolName.lastIndexOf('_');
257 if (lastUnderscore > 0) {
258 int secondLastUnderscore = subSymbolName.lastIndexOf('_', lastUnderscore - 1);
259 if (secondLastUnderscore > 0) {
260 QString parentName = subSymbolName.left(secondLastUnderscore);
261 parentToSubSymbols[parentName].append(subSymbolName);
262 }
263 }
264 }
265
266 // 对于每个被覆盖的父符号，收集其所有子符号
267 for (const QString& parentSymbolName : overwrittenSymbolNames) {
268 if (parentToSubSymbols.contains(parentSymbolName)) {
269 // 这是一个分体式符号，删除所有子符号
270 for (const QString& subSymbolName : parentToSubSymbols[parentSymbolName]) {
271 subSymbolsToDelete.insert(subSymbolName);
272 qDebug() << "Marking sub-symbol for deletion (multipart):" << subSymbolName
273 << "(parent:" << parentSymbolName << ")";
274 }
275 } else {
276 // 这是一个单体符号，查找其子符号
277 // KiCad 使用 _1_1 格式，旧版本 exporter 使用 _0_1 格式
278 QString expectedSubSymbolName_v1 = parentSymbolName + "_0_1";
279 QString expectedSubSymbolName_v2 = parentSymbolName + "_1_1";
280 if (subSymbolNames.contains(expectedSubSymbolName_v1)) {
281 subSymbolsToDelete.insert(expectedSubSymbolName_v1);
282 qDebug() << "Marking sub-symbol for deletion (single part v1):" << expectedSubSymbolName_v1
283 << "(parent:" << parentSymbolName << ")";
284 } else if (subSymbolNames.contains(expectedSubSymbolName_v2)) {
285 subSymbolsToDelete.insert(expectedSubSymbolName_v2);
286 qDebug() << "Marking sub-symbol for deletion (single part v2):" << expectedSubSymbolName_v2
287 << "(parent:" << parentSymbolName << ")";
288 }
289 }
290 }
291
292 // 先导出未覆盖的现有符号（需要过滤掉被覆盖的子符号和以被覆盖符号名开头的顶层符号）
293 for (const QString& symbolName : existingSymbols.keys()) {
294 bool isOverwritten = overwrittenSymbolNames.contains(symbolName);
295
296 // 检查是否是以被覆盖符号名开头的顶层符号（可能是之前错误导出的子符号
297 bool isOrphanedSubSymbol = false;
298 if (!isOverwritten) {
299 for (const QString& parentSymbolName : overwrittenSymbolNames) {
300 if (symbolName.startsWith(parentSymbolName + "_")) {
301 isOrphanedSubSymbol = true;
302 qDebug() << "Skipping orphaned sub-symbol (top-level):" << symbolName
303 << "(parent:" << parentSymbolName << ")";
304 break;
305 }
306 }
307 }
308
309 if (!isOverwritten && !isOrphanedSubSymbol) {
310 // 导出未覆盖的符号，但需要过滤掉子符号
311 QString symbolContent = existingSymbols[symbolName];
312 QStringList lines = symbolContent.split('\n');
313 QString filteredContent;
314 bool skipNextSymbol = false;
315 int braceCount = 0;
316
317 for (const QString& line : lines) {
318 QString trimmedLine = line.trimmed();
319
320 // 检查是否是子符号定义的开始
321 if (trimmedLine.startsWith("(symbol \"")) {
322 int nameStart = trimmedLine.indexOf("\"") + 1;
323 int nameEnd = trimmedLine.indexOf("\"", nameStart);
324 if (nameEnd > nameStart) {
325 QString subSymbolName = trimmedLine.mid(nameStart, nameEnd - nameStart);
326 if (subSymbolsToDelete.contains(subSymbolName)) {
327 skipNextSymbol = true;
328 braceCount = 1;
329 qDebug() << "Skipping deleted sub-symbol:" << subSymbolName;
330 continue;
331 }
332 }
333 }
334
335 if (skipNextSymbol) {
336 // 计算括号数量
337 for (int j = 0; j < line.length(); ++j) {
338 if (line[j] == '(')
339 braceCount++;
340 else if (line[j] == ')') {
341 braceCount--;
342 if (braceCount == 0) {
343 skipNextSymbol = false;
344 break;
345 }
346 }
347 }
348 continue;
349 }
350
351 filteredContent += line + "\n";
352 }
353
354 out << filteredContent;
355 qDebug() << "Keeping existing symbol:" << symbolName;
356 }
357 }
358
359 // 再导出新符号和被覆盖的符号
360 for (const SymbolData& symbol : symbolsToExport) {
361 qDebug() << "Exporting symbol" << (++index) << "of" << symbolsToExport.count() << ":" << symbol.info().name;
362 out << generateSymbolContent(symbol, libName);
363 }
364
365 // 生成尾部
366 out << ")\n"; // 闭合 kicad_symbol_lib
367
368 file.close();
369 qDebug() << "Symbol library exported successfully:" << filePath;
370 return true;
371}

385QString ExporterSymbol::generateSymbolContent(const SymbolData& symbolData, const QString& libName) const {
386 QString content;
387
388 // V6 格式 - 主符号定义（包含属性）
389 // 在更新模式下，保持符号名称不变；在新符号中，替换空格为下划线
390 QString cleanSymbolName = symbolData.info().name;
391 // 注意：KiCad 6.x 允许符号名称中包含空格，所以不需要替换
392 // 保持原始名称以确保更新模式下能正确匹配和替换
393 content += QString(" (symbol \"%1\"\n").arg(cleanSymbolName);
394 content += " (in_bom yes)\n";
395 content += " (on_board yes)\n";
396
397 // 设置当前边界框，用于图形元素的相对坐标计算
398 SymbolBBox originalBBox = symbolData.bbox();
399 m_graphicsGenerator.setCurrentBBox(originalBBox);
400 qDebug() << "BBox - x:" << originalBBox.x << "y:" << originalBBox.y;
401
402 // 找到最左上角的引脚（x最小且y最小），将其位置作为新的坐标原点
403 double originX = originalBBox.x;
404 double originY = originalBBox.y;
405 QList<SymbolPin> pins = symbolData.pins();
406 if (!pins.isEmpty()) {
407 bool firstPin = true;
408 for (const SymbolPin& pin : pins) {
409 // 最左上角：x最小（最左），y最小（KiCad中y小=上方）
410 if (firstPin) {
411 originX = pin.settings.posX;
412 originY = pin.settings.posY;
413 firstPin = false;
414 } else {
415 // x更小，或者x相同但y更小（更上方）
416 if (pin.settings.posX < originX || (pin.settings.posX == originX && pin.settings.posY < originY)) {
417 originX = pin.settings.posX;
418 originY = pin.settings.posY;
419 }
420 }
421 }
422 qDebug() << "Top-left pin at:" << originX << originY;
423 }
424
425 // 修改边界框原点为最近引脚位置，使所有图形元素相对于该点定位
426 SymbolBBox currentBBox = originalBBox;
427 currentBBox.x = originX;
428 currentBBox.y = originY;
429 m_graphicsGenerator.setCurrentBBox(currentBBox);
430 qDebug() << "Adjusted BBox to nearest pin origin - x:" << currentBBox.x << "y:" << currentBBox.y;
431
432 // 计算所有图形元素的实际边界，用于文本左右居中对齐
433 double minX = std::numeric_limits<double>::max();
434 double maxX = -std::numeric_limits<double>::max();
435 auto updateBounds = [&](double x, double w) {
436 minX = qMin(minX, x);
437 maxX = qMax(maxX, x + w);
438 };
439 auto parseAndUpdatePoints = [&](const QString& pointsStr) {
440 QStringList points = pointsStr.split(" ");
441 points.removeAll("");
442 for (int i = 0; i + 1 < points.size(); i += 2) {
443 double px = points[i].toDouble();
444 minX = qMin(minX, px);
445 maxX = qMax(maxX, px);
446 }
447 };
448 for (const auto& rect : symbolData.rectangles()) {
449 updateBounds(rect.posX, rect.width);
450 }
451 for (const auto& circle : symbolData.circles()) {
452 double r = circle.radius;
453 minX = qMin(minX, circle.centerX - r);
454 maxX = qMax(maxX, circle.centerX + r);
455 }
456 for (const auto& ellipse : symbolData.ellipses()) {
457 minX = qMin(minX, ellipse.centerX - ellipse.radiusX);
458 maxX = qMax(maxX, ellipse.centerX + ellipse.radiusX);
459 }
460 for (const auto& arc : symbolData.arcs()) {
461 for (const auto& pt : arc.path) {
462 minX = qMin(minX, pt.x());
463 maxX = qMax(maxX, pt.x());
464 }
465 }
466 for (const auto& polyline : symbolData.polylines()) {
467 parseAndUpdatePoints(polyline.points);
468 }
469 for (const auto& polygon : symbolData.polygons()) {
470 parseAndUpdatePoints(polygon.points);
471 }
472 for (const auto& text : symbolData.texts()) {
473 minX = qMin(minX, text.posX);
474 maxX = qMax(maxX, text.posX);
475 }
476 for (const auto& pin : pins) {
477 minX = qMin(minX, pin.settings.posX);
478 maxX = qMax(maxX, pin.settings.posX);
479 }
480 // 如果没有任何图形元素，使用默认宽度
481 double graphWidth = (minX <= maxX) ? (maxX - minX) : 0.0;
482 double graphCenterOffsetX = 0.0;
483 if (minX <= maxX) {
484 graphCenterOffsetX = m_graphicsGenerator.pxToMm(minX + graphWidth / 2.0 - originX);
485 }
486 qDebug() << "Graph bounds - minX:" << minX << "maxX:" << maxX << "width:" << graphWidth
487 << "centerOffsetX(mm):" << graphCenterOffsetX;
488
489 // 计算 y_high 和 y_low（使用引脚坐标，与 Python 版本保持一致）
490 // 如果没有引脚，使用默认值以确保属性位置正确
491 double yHigh = 2.54; // 默认值：100mil
492 double yLow = -2.54; // 默认值：-100mil
493 if (!pins.isEmpty()) {
494 for (const SymbolPin& pin : pins) {
495 double pinY = -m_graphicsGenerator.pxToMm(pin.settings.posY - originY);
496 yHigh = qMax(yHigh, pinY);
497 yLow = qMin(yLow, pinY);
498 }
499 }
500
501 // 生成属性
502 double fieldOffset = 5.08; // FIELD_OFFSET_START
503 double fontSize = 1.27; // PROPERTY_FONT_SIZE
504
505 // 辅助函数：转义属性值
506 auto escapePropertyValue = [](const QString& value) -> QString {
507 QString escaped = value;
508 escaped.replace("\"", "\\\""); // 转义引号
509 escaped.replace("\n", " "); // 移除换行符
510 escaped.replace("\t", " "); // 移除制表符
511 return escaped.trimmed();
512 };
513
514 // Reference 属性
515 QString refPrefix = symbolData.info().prefix;
516 refPrefix.replace("?", ""); // 移除 "?" 后缀
517 content += QString(" (property\n");
518 content += QString(" \"Reference\"\n");
519 content += QString(" \"%1\"\n").arg(escapePropertyValue(refPrefix));
520 content += " (id 0)\n";
521 content += QString(" (at %1 %2 0)\n").arg(graphCenterOffsetX, 0, 'f', 2).arg(yHigh + fieldOffset, 0, 'f', 2);
522 content += QString(" (effects (font (size %1 %2) (thickness 0) ) )\n")
523 .arg(fontSize, 0, 'f', 2)
524 .arg(fontSize, 0, 'f', 2);
525 content += " )\n";
526
527 // Value 属性
528 content += QString(" (property\n");
529 content += QString(" \"Value\"\n");
530 content += QString(" \"%1\"\n").arg(escapePropertyValue(symbolData.info().name));
531 content += " (id 1)\n";
532 content += QString(" (at %1 %2 0)\n").arg(graphCenterOffsetX, 0, 'f', 2).arg(yLow - fieldOffset, 0, 'f', 2);
533 content += QString(" (effects (font (size %1 %2) (thickness 0) ) )\n")
534 .arg(fontSize, 0, 'f', 2)
535 .arg(fontSize, 0, 'f', 2);
536 content += " )\n";
537
538 // Footprint 属性
539 if (!symbolData.info().package.isEmpty()) {
540 fieldOffset += 2.54; // FIELD_OFFSET_INCREMENT
541 content += QString(" (property\n");
542 content += QString(" \"Footprint\"\n");
543 // KiCad 符号需要库前缀格式
544 QString footprintPath = QString("%1:%2").arg(libName, symbolData.info().package);
545 content += QString(" \"%1\"\n").arg(escapePropertyValue(footprintPath));
546 content += " (id 2)\n";
547 content +=
548 QString(" (at %1 %2 0)\n").arg(graphCenterOffsetX, 0, 'f', 2).arg(yLow - fieldOffset, 0, 'f', 2);
549 content += QString(" (effects (font (size %1 %2) (thickness 0) ) hide)\n")
550 .arg(fontSize, 0, 'f', 2)
551 .arg(fontSize, 0, 'f', 2);
552 content += " )\n";
553 }
554
555 // Datasheet 属性
556 if (!symbolData.info().datasheet.isEmpty()) {
557 fieldOffset += 2.54;
558 content += QString(" (property\n");
559 content += QString(" \"Datasheet\"\n");
560 content += QString(" \"%1\"\n").arg(escapePropertyValue(symbolData.info().datasheet));
561 content += " (id 3)\n";
562 content +=
563 QString(" (at %1 %2 0)\n").arg(graphCenterOffsetX, 0, 'f', 2).arg(yLow - fieldOffset, 0, 'f', 2);
564 content += QString(" (effects (font (size %1 %2) (thickness 0) ) hide)\n")
565 .arg(fontSize, 0, 'f', 2)
566 .arg(fontSize, 0, 'f', 2);
567 content += " )\n";
568 }
569
570 // Manufacturer 属性
571 if (!symbolData.info().manufacturer.isEmpty()) {
572 fieldOffset += 2.54;
573 content += QString(" (property\n");
574 content += QString(" \"Manufacturer\"\n");
575 content += QString(" \"%1\"\n").arg(escapePropertyValue(symbolData.info().manufacturer));
576 content += " (id 4)\n";
577 content +=
578 QString(" (at %1 %2 0)\n").arg(graphCenterOffsetX, 0, 'f', 2).arg(yLow - fieldOffset, 0, 'f', 2);
579 content += QString(" (effects (font (size %1 %2) (thickness 0) ) hide)\n")
580 .arg(fontSize, 0, 'f', 2)
581 .arg(fontSize, 0, 'f', 2);
582 content += " )\n";
583 }
584
585 // LCSC Part 属性
586 if (!symbolData.info().lcscId.isEmpty()) {
587 fieldOffset += 2.54;
588 content += QString(" (property\n");
589 content += QString(" \"LCSC Part\"\n");
590 content += QString(" \"%1\"\n").arg(escapePropertyValue(symbolData.info().lcscId));
591 content += " (id 5)\n";

612 .arg(fontSize, 0, 'f', 2);
613 content += " )\n";
614 }
615
616 // 检查是否为多部分符
617 bool isMultiPart = symbolData.isMultiPart();
618 qDebug() << "=== Symbol Type Check ===";
619 qDebug() << "Symbol name:" << cleanSymbolName;

621 qDebug() << "Is multi-part:" << isMultiPart;
622
623 if (isMultiPart) {
624 qDebug() << "Exporting multi-part symbol with" << symbolData.parts().size() << "parts";
625
626 // 为每个部分生成子符号
627 for (const SymbolPart& part : symbolData.parts()) {
628 qDebug() << "Generating sub-symbol" << part.unitNumber << "with" << part.pins.size() << "pins";

630 }
631 } else {

640 }
641
642 content += " )\n"; // 结束主符号
643
644 return content;
645}
