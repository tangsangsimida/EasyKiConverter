231 for (const QString& subSymbolName : subSymbolNames) {

233 // 子符号格式：{parentName}_{unitNumber}_1
234 int lastUnderscore = subSymbolName.lastIndexOf('_');
235 if (lastUnderscore > 0) {

241 }

246 if (parentToSubSymbols.contains(parentSymbolName)) {
247 // 这是一个分体式符号，删除所有子符号
248 for (const QString& subSymbolName : parentToSubSymbols[parentSymbolName]) {

267

269 bool isOrphanedSubSymbol = false;
270 if (!isOverwritten) {

272 if (symbolName.startsWith(parentSymbolName + "_")) {
273 isOrphanedSubSymbol = true;
274 qDebug() << "Skipping orphaned sub-symbol (top-level):" << symbolName
275 << "(parent:" << parentSymbolName << ")";
276 break;
277 }
278 }
279 }
280
281 if (!isOverwritten && !isOrphanedSubSymbol) {
282 // 导出未覆盖的符号，但需要过滤掉子符号

284 QStringList lines = symbolContent.split('\n');
285 QString filteredContent;
286 bool skipNextSymbol = false;
287 int braceCount = 0;
288
289 for (const QString& line : lines) {
290 QString trimmedLine = line.trimmed();
291

293 if (trimmedLine.startsWith("(symbol \"")) {

295 int nameEnd = trimmedLine.indexOf("\"", nameStart);
296 if (nameEnd > nameStart) {
297 QString subSymbolName = trimmedLine.mid(nameStart, nameEnd - nameStart);
298 if (subSymbolsToDelete.contains(subSymbolName)) {
299 skipNextSymbol = true;
300 braceCount = 1;

303 }
304 }
305 }
306
307 if (skipNextSymbol) {
308 // 计算括号数量
309 for (int j = 0; j < line.length(); ++j) {
310 if (line[j] == '(')
311 braceCount++;
312 else if (line[j] == ')') {
313 braceCount--;
314 if (braceCount == 0) {
315 skipNextSymbol = false;
316 break;
317 }
318 }
319 }
320 continue;
321 }
322
323 filteredContent += line + "\n";
324 }
325
326 out << filteredContent;
327 qDebug() << "Keeping existing symbol:" << symbolName;
328 }
329 }

332 for (const SymbolData& symbol : symbolsToExport) {
333 qDebug() << "Exporting symbol" << (++index) << "of" << symbolsToExport.count() << ":" << symbol.info().name;

353QString ExporterSymbol::generateSymbolContent(const SymbolData& symbolData, const QString& libName) const {
354 QString content;
355
356 // V6 格式 - 主符号定义（包含属性）
357 // 在更新模式下，保持符号名称不变；在新符号中，替换空格为下划线
358 QString cleanSymbolName = symbolData.info().name;
359 // 注意：KiCad 6.x 允许符号名称中包含空格，所以不需要替换
360 // 保持原始名称以确保更新模式下能正确匹配和替换
361 content += QString(" (symbol \"%1\"\n").arg(cleanSymbolName);
362 content += " (in_bom yes)\n";
363 content += " (on_board yes)\n";
364
365 // 设置当前边界框，用于图形元素的相对坐标计算
366 SymbolBBox originalBBox = symbolData.bbox();
367 SymbolBBox currentBBox = originalBBox;
368 m_graphicsGenerator.setCurrentBBox(currentBBox);
369 qDebug() << "BBox - x:" << currentBBox.x << "y:" << currentBBox.y << "width:" << currentBBox.width
370 << "height:" << currentBBox.height;
371 // 计算符号中心点（用于将符号居中显示）
372 double centerX = currentBBox.x + currentBBox.width / 2.0;
373 double centerY = currentBBox.y + currentBBox.height / 2.0;
374 qDebug() << "Symbol center - centerX:" << centerX << "centerY:" << centerY;
375 // 修改边界框，使其指向中心点，这样所有图形元素都会相对于中心点定位
376 currentBBox.x = centerX;
377 currentBBox.y = centerY;
378 m_graphicsGenerator.setCurrentBBox(currentBBox);
379 qDebug() << "Adjusted BBox for centering - x:" << currentBBox.x << "y:" << currentBBox.y;
380 // 计算 y_high 和 y_low（使用引脚坐标，与 Python 版本保持一致）
381 // 如果没有引脚，使用默认值以确保属性位置正
382 double yHigh = 2.54; // 默认值：100mil
383 double yLow = -2.54; // 默认值：-100mil
384 QList<SymbolPin> pins = symbolData.pins();
385 if (!pins.isEmpty()) {
386 qDebug() << "Pin coordinates calculation:";
387 for (const SymbolPin& pin : pins) {
388 double pinY = -m_graphicsGenerator.pxToMm(pin.settings.posY - centerY);
389 qDebug() << " Pin" << pin.settings.spicePinNumber << "- raw Y:" << pin.settings.posY
390 << "converted Y:" << pinY;
391 yHigh = qMax(yHigh, pinY);
392 yLow = qMin(yLow, pinY);
393 }
394 } else {
395 qDebug() << "No pins found, using default yHigh and yLow values";
396 }
397 qDebug() << "Final yHigh:" << yHigh << "yLow:" << yLow;
398
399 // 生成属性
400 double fieldOffset = 5.08; // FIELD_OFFSET_START
401 double fontSize = 1.27; // PROPERTY_FONT_SIZE
402
403 // 辅助函数：转义属性值
404 auto escapePropertyValue = [](const QString& value) -> QString {
405 QString escaped = value;
406 escaped.replace("\"", "\\\""); // 转义引号
407 escaped.replace("\n", " "); // 移除换行符
408 escaped.replace("\t", " "); // 移除制表符
409 return escaped.trimmed();
410 };
411
412 // Reference 属性
413 QString refPrefix = symbolData.info().prefix;
414 refPrefix.replace("?", ""); // 移除 "?" 后缀
415 content += QString(" (property\n");
416 content += QString(" \"Reference\"\n");
417 content += QString(" \"%1\"\n").arg(escapePropertyValue(refPrefix));
418 content += " (id 0)\n";
419 content += QString(" (at 0 %1 0)\n").arg(yHigh + fieldOffset, 0, 'f', 2);
420 content += QString(" (effects (font (size %1 %2) (thickness 0) ) )\n")
421 .arg(fontSize, 0, 'f', 2)
422 .arg(fontSize, 0, 'f', 2);
423 content += " )\n";
424
425 // Value 属性
426 content += QString(" (property\n");
427 content += QString(" \"Value\"\n");
428 content += QString(" \"%1\"\n").arg(escapePropertyValue(symbolData.info().name));
429 content += " (id 1)\n";
430 content += QString(" (at 0 %1 0)\n").arg(yLow - fieldOffset, 0, 'f', 2);
431 content += QString(" (effects (font (size %1 %2) (thickness 0) ) )\n")
432 .arg(fontSize, 0, 'f', 2)
433 .arg(fontSize, 0, 'f', 2);
434 content += " )\n";
435
436 // Footprint 属性
437 if (!symbolData.info().package.isEmpty()) {
438 fieldOffset += 2.54; // FIELD_OFFSET_INCREMENT
439 content += QString(" (property\n");
440 content += QString(" \"Footprint\"\n");
441 // 添加库前缀：libName:package
442 QString footprintPath = QString("%1:%2").arg(libName, symbolData.info().package);
443 content += QString(" \"%1\"\n").arg(escapePropertyValue(footprintPath));
444 content += " (id 2)\n";
445 content += QString(" (at 0 %1 0)\n").arg(yLow - fieldOffset, 0, 'f', 2);
446 content += QString(" (effects (font (size %1 %2) (thickness 0) ) hide)\n")
447 .arg(fontSize, 0, 'f', 2)
448 .arg(fontSize, 0, 'f', 2);
449 content += " )\n";
450 }
451
452 // Datasheet 属性
453 if (!symbolData.info().datasheet.isEmpty()) {
454 fieldOffset += 2.54;
455 content += QString(" (property\n");
456 content += QString(" \"Datasheet\"\n");
457 content += QString(" \"%1\"\n").arg(escapePropertyValue(symbolData.info().datasheet));
458 content += " (id 3)\n";
459 content += QString(" (at 0 %1 0)\n").arg(yLow - fieldOffset, 0, 'f', 2);
460 content += QString(" (effects (font (size %1 %2) (thickness 0) ) hide)\n")
461 .arg(fontSize, 0, 'f', 2)
462 .arg(fontSize, 0, 'f', 2);
463 content += " )\n";
464 }
465
466 // Manufacturer 属性
467 if (!symbolData.info().manufacturer.isEmpty()) {
468 fieldOffset += 2.54;
469 content += QString(" (property\n");
470 content += QString(" \"Manufacturer\"\n");
471 content += QString(" \"%1\"\n").arg(escapePropertyValue(symbolData.info().manufacturer));
472 content += " (id 4)\n";
473 content += QString(" (at 0 %1 0)\n").arg(yLow - fieldOffset, 0, 'f', 2);
474 content += QString(" (effects (font (size %1 %2) (thickness 0) ) hide)\n")
475 .arg(fontSize, 0, 'f', 2)
476 .arg(fontSize, 0, 'f', 2);
477 content += " )\n";
478 }
479
480 // LCSC Part 属性
481 if (!symbolData.info().lcscId.isEmpty()) {
482 fieldOffset += 2.54;
483 content += QString(" (property\n");
484 content += QString(" \"LCSC Part\"\n");
485 content += QString(" \"%1\"\n").arg(escapePropertyValue(symbolData.info().lcscId));
486 content += " (id 5)\n";
487 content += QString(" (at 0 %1 0)\n").arg(yLow - fieldOffset, 0, 'f', 2);
488 content += QString(" (effects (font (size %1 %2) (thickness 0) ) hide)\n")
489 .arg(fontSize, 0, 'f', 2)
490 .arg(fontSize, 0, 'f', 2);
491 content += " )\n";
492 }
493
494 // 检查是否为多部分符
495 bool isMultiPart = symbolData.isMultiPart();
496 qDebug() << "=== Symbol Type Check ===";
497 qDebug() << "Symbol name:" << cleanSymbolName;
498 qDebug() << "Parts count:" << symbolData.parts().size();
499 qDebug() << "Is multi-part:" << isMultiPart;
500
501 if (isMultiPart) {
502 qDebug() << "Exporting multi-part symbol with" << symbolData.parts().size() << "parts";
503
504 // 为每个部分生成子符号
505 for (const SymbolPart& part : symbolData.parts()) {
506 qDebug() << "Generating sub-symbol" << part.unitNumber << "with" << part.pins.size() << "pins";
507 content += generateSubSymbol(symbolData, part, cleanSymbolName, libName, centerX, centerY);
508 }
509 } else {
510 // 单部分符号：直接在主符号中包含图形元素，不使用子符号
511 qDebug() << "Exporting single-part symbol";
512
513 // 生成图形元素
514 content += m_graphicsGenerator.generateDrawings(symbolData);
515
516 // 生成引脚
517 content += m_graphicsGenerator.generatePins(symbolData.pins(), m_graphicsGenerator.currentBBox());
518 }
519
520 content += " )\n"; // 结束主符号
521
522 qDebug() << "=== Generated Symbol Content ===";
523 qDebug() << "Content length:" << content.length();
524 qDebug() << "Content contains sub-symbol 0:" << content.contains("MCIMX6Y2CVM08AB_0_1");
525 qDebug() << "Content contains sub-symbol 1:" << content.contains("MCIMX6Y2CVM08AB_1_1");
526 qDebug() << "Content preview (first 1000 chars):" << content.left(1000);
527 qDebug() << "Content preview (last 1000 chars):" << content.right(1000);
528
529 return content;
530}

554QString ExporterSymbol::generateSubSymbol(const SymbolData& symbolData,
555 const SymbolPart& part,
556 const QString& symbolName,
557 const QString& libName,
558 double centerX,
559 double centerY) const {
560 QString content;
561
562 // 多部分符号：使用 _{unitNumber}_1 作为子符号名称
563 // 注意：Unit 编号必须从1 开始，而不是从 0 开始
564 content += QString(" (symbol \"%1_%2_1\"\n").arg(symbolName).arg(part.unitNumber + 1);
565
566 // 计算子部分的边界框和中心点
567 SymbolBBox partBBox = m_graphicsGenerator.calculatePartBBox(part);
568 double partCenterX = partBBox.x + partBBox.width / 2.0;
569 double partCenterY = partBBox.y + partBBox.height / 2.0;
570
571 qDebug() << "Sub-symbol" << part.unitNumber << "- partCenterX:" << partCenterX << "partCenterY:" << partCenterY;
572
573 // 临时保存当前的边界框
574 SymbolBBox originalBBox = m_graphicsGenerator.currentBBox();
575
576 // 设置边界框为子部分的中心点，以便图形元素使用正确的偏移
577 SymbolBBox subBBox;
578 subBBox.x = partCenterX;
579 subBBox.y = partCenterY;
580 subBBox.width = partBBox.width;
581 subBBox.height = partBBox.height;
582 m_graphicsGenerator.setCurrentBBox(subBBox);
583
584 qDebug() << "Setting m_currentBBox for sub-symbol - x:" << subBBox.x << "y:" << subBBox.y;
585
586 // 创建以子部分中心为基准的边界框（用于引脚）
