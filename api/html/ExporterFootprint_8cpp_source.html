65bool ExporterFootprint::exportFootprintLibrary(const QList<FootprintData>& footprints,
66 const QString& libName,
67 const QString& filePath,
68 bool preferWrl,
69 bool exportStep,
70 const QString& libraryDescription,
71 const QString& libraryKeywords,
72 bool useAbsolutePaths,
73 const QString& model3DBaseDir) {
74 qDebug() << "=== Export Footprint Library ===";
75 qDebug() << "Library name:" << libName;
76 qDebug() << "Output path:" << filePath;
77 qDebug() << "Footprint count:" << footprints.count();
78 qDebug() << "Library description:" << libraryDescription;
79 qDebug() << "Library keywords:" << libraryKeywords;
80
81 QDir libDir(filePath);
82
83 if (!libDir.exists()) {
84 qDebug() << "Creating footprint library directory:" << filePath;
85 if (!libDir.mkpath(".")) {
86 qWarning() << "Failed to create footprint library directory:" << filePath;
87 return false;
88 }
89 }
90
91 QSet<QString> existingFootprintNames;
92 QStringList existingFiles = libDir.entryList(QStringList("*.kicad_mod"), QDir::Files);
93 for (const QString& fileName : existingFiles) {
94 QString footprintName = fileName;
95 footprintName.remove(".kicad_mod");
96 existingFootprintNames.insert(footprintName);
97 qDebug() << "Found existing footprint:" << footprintName;
98 }
99
100 qDebug() << "Existing footprints count:" << existingFootprintNames.count();
101
102 const QString safeLibName = PathSecurity::sanitizeFilename(libName);
103 const QString resolvedBaseDir = model3DBaseDir.isEmpty() ? QFileInfo(filePath).absolutePath() : model3DBaseDir;
104
105 int exportedCount = 0;
106 int overwrittenCount = 0;
107 for (const FootprintData& footprint : footprints) {
108 QString footprintName = footprint.info().name;
109 QString fileName = footprintName + ".kicad_mod";
110 QString fullPath = libDir.filePath(fileName);
111
112 bool exists = existingFootprintNames.contains(footprintName);
113 if (exists) {
114 qDebug() << "Overwriting existing footprint:" << footprintName;
115 overwrittenCount++;
116 } else {
117 qDebug() << "Exporting new footprint:" << footprintName;
118 exportedCount++;
119 }
120
121 QString content;
122 if (!footprint.model3D().name().isEmpty() || !footprint.model3D().uuid().isEmpty()) {
123 QString modelName = footprint.model3D().name();
124 if (modelName.isEmpty()) {
125 modelName = footprint.info().name;
126 }
127 modelName = PathSecurity::sanitizeFilename(modelName);
128
129 const bool hasModel3D = !footprint.model3D().name().isEmpty();
130 const bool useWrl = preferWrl && hasModel3D;
131 const bool useStep = exportStep && hasModel3D;
132
133 if (useWrl && useStep) {
134 QString wrlPath =
135 buildModel3DPath(safeLibName, modelName, QStringLiteral("wrl"), useAbsolutePaths, resolvedBaseDir);
136 QString stepPath =
137 buildModel3DPath(safeLibName, modelName, QStringLiteral("step"), useAbsolutePaths, resolvedBaseDir);
138 content = generateFootprintContent(footprint, wrlPath, stepPath, libraryDescription, libraryKeywords);
139 } else if (useWrl) {
140 QString wrlPath =
141 buildModel3DPath(safeLibName, modelName, QStringLiteral("wrl"), useAbsolutePaths, resolvedBaseDir);
142 content = generateFootprintContent(footprint, wrlPath, libraryDescription, libraryKeywords);
143 } else if (useStep) {
144 QString stepPath =
145 buildModel3DPath(safeLibName, modelName, QStringLiteral("step"), useAbsolutePaths, resolvedBaseDir);
146 content = generateFootprintContent(footprint, stepPath, libraryDescription, libraryKeywords);
147 } else {
148 content = generateFootprintContent(footprint, QString(), libraryDescription, libraryKeywords);
149 }
150 } else {
151 content = generateFootprintContent(footprint, QString(), libraryDescription, libraryKeywords);
152 }
153
154 QFile file(fullPath);
155 if (!file.open(QIODevice::WriteOnly | QIODevice::Text)) {
156 qWarning() << "Failed to open file for writing:" << fullPath;
157 continue;
158 }
159
160 QTextStream out(&file);
161 out.setEncoding(QStringConverter::Utf8);
162 out << content;
163 file.close();
164
165 qDebug() << " Exported to:" << fullPath;
166 }
167
168 qDebug() << "Export complete - New:" << exportedCount << "Overwritten:" << overwrittenCount;
169 qDebug() << "Footprint library exported to:" << filePath;
170
171 return true;
172}

205void ExporterFootprint::generateFootprintBaseContent(const FootprintData& footprintData,
206 QString& content,
207 double& outBboxX,
208 double& outBboxY,
209 const QString& libraryDescription,
210 const QString& libraryKeywords) const {
211 content += QString("(footprint easykiconverter:%1\n").arg(footprintData.info().name);
212 content += " (version 20221018)\n";
213
214 // 导出封装描述 (descr) - 单个元器件描述；库描述写入 fp-lib-table
215 const QString footprintDescription = footprintData.info().description.trimmed();
216 if (!footprintDescription.isEmpty()) {
217 content += QString(" (descr \"%1\")\n").arg(footprintDescription);
218 }
219
220 // 导出封装标签 (tags) - 用户输入
221 if (!libraryKeywords.isEmpty()) {
222 content += QString(" (tags \"%1\")\n").arg(libraryKeywords);
223 }
224
225 bool isThroughHole = false;
226 for (const FootprintPad& pad : footprintData.pads()) {
227 if (pad.holeRadius > 0) {
228 isThroughHole = true;
229 break;
230 }
231 }
232
233 if (isThroughHole) {
234 content += " (attr through_hole)\n";
235 } else {
236 content += " (attr smd)\n";
237 }
238
239 double yLow = 0;
240 double yHigh = 0;
241 if (!footprintData.pads().isEmpty()) {
242 yLow = footprintData.pads().first().centerY;
243 yHigh = footprintData.pads().first().centerY;
244 for (const FootprintPad& pad : footprintData.pads()) {
245 if (pad.centerY < yLow)
246 yLow = pad.centerY;
247 if (pad.centerY > yHigh)
248 yHigh = pad.centerY;
249 }
250 }
251
252 double bboxX = footprintData.bbox().x;
253 double bboxY = footprintData.bbox().y;
254
255 double centerX = bboxX + footprintData.bbox().width / 2.0;
256 double centerY = bboxY + footprintData.bbox().height / 2.0;
257 bboxX = centerX;
258 bboxY = centerY;
259
260 content += QString(" (fp_text reference REF** (at 0 %1) (layer F.SilkS)\n")
261 .arg(m_graphicsGenerator.pxToMm(yLow - bboxY - 4));
262 content += " (effects (font (size 1 1) (thickness 0.15)))\n";
263 content += " )\n";
264
265 content += QString(" (fp_text value %1 (at 0 %2) (layer F.Fab)\n")
266 .arg(footprintData.info().name)
267 .arg(m_graphicsGenerator.pxToMm(yHigh - bboxY + 4));
268 content += " (effects (font (size 1 1) (thickness 0.15)))\n";
269 content += " )\n";
270
271 content += " (fp_text user %R (at 0 0) (layer F.Fab)\n";
272 content += " (effects (font (size 1 1) (thickness 0.15)))\n";
273 content += " )\n";
274
275 for (const FootprintTrack& track : footprintData.tracks()) {
276 content += m_graphicsGenerator.generateTrack(track, bboxX, bboxY);
277 }
278 for (const FootprintRectangle& rect : footprintData.rectangles()) {
279 content += m_graphicsGenerator.generateRectangle(rect, bboxX, bboxY);
280 }
281
282 for (const FootprintPad& pad : footprintData.pads()) {
283 content += m_graphicsGenerator.generatePad(pad, bboxX, bboxY);
284 }
285
286 for (const FootprintHole& hole : footprintData.holes()) {
287 content += m_graphicsGenerator.generateHole(hole, bboxX, bboxY);
288 }
289
290 for (const FootprintCircle& circle : footprintData.circles()) {
291 content += m_graphicsGenerator.generateCircle(circle, bboxX, bboxY);
292 }
293
294 for (const FootprintArc& arc : footprintData.arcs()) {
295 content += m_graphicsGenerator.generateArc(arc, bboxX, bboxY);
296 }
297
298 for (const FootprintText& text : footprintData.texts()) {
299 content += m_graphicsGenerator.generateText(text, bboxX, bboxY);
300 }
301
302 bool hasCourtYard = false;
303 for (const FootprintSolidRegion& region : footprintData.solidRegions()) {
304 QString regionContent = m_graphicsGenerator.generateSolidRegion(region, bboxX, bboxY);
305 content += regionContent;
306 if (region.layerId == 99) {
307 hasCourtYard = true;
308 }
309 }
310
311 if (!hasCourtYard && footprintData.bbox().width > 0 && footprintData.bbox().height > 0) {
312 content += m_graphicsGenerator.generateCourtyardFromBBox(footprintData.bbox(), bboxX, bboxY);
313 qWarning() << "Warning: No courtyard found, generated from BBox";
314 }
315
316 outBboxX = bboxX;
317 outBboxY = bboxY;
318}
