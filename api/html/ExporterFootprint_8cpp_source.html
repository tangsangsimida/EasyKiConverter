62bool ExporterFootprint::exportFootprintLibrary(const QList<FootprintData>& footprints,
63 const QString& libName,
64 const QString& filePath) {
65 qDebug() << "=== Export Footprint Library ===";
66 qDebug() << "Library name:" << libName;
67 qDebug() << "Output path:" << filePath;
68 qDebug() << "Footprint count:" << footprints.count();
69
70 QDir libDir(filePath);
71
72 if (!libDir.exists()) {
73 qDebug() << "Creating footprint library directory:" << filePath;
74 if (!libDir.mkpath(".")) {
75 qWarning() << "Failed to create footprint library directory:" << filePath;
76 return false;
77 }
78 }
79
80 QSet<QString> existingFootprintNames;
81 QStringList existingFiles = libDir.entryList(QStringList("*.kicad_mod"), QDir::Files);
82 for (const QString& fileName : existingFiles) {
83 QString footprintName = fileName;
84 footprintName.remove(".kicad_mod");
85 existingFootprintNames.insert(footprintName);
86 qDebug() << "Found existing footprint:" << footprintName;
87 }
88
89 qDebug() << "Existing footprints count:" << existingFootprintNames.count();
90
91 int exportedCount = 0;
92 int overwrittenCount = 0;
93 for (const FootprintData& footprint : footprints) {
94 QString footprintName = footprint.info().name;
95 QString fileName = footprintName + ".kicad_mod";
96 QString fullPath = libDir.filePath(fileName);
97
98 bool exists = existingFootprintNames.contains(footprintName);
99 if (exists) {
100 qDebug() << "Overwriting existing footprint:" << footprintName;
101 overwrittenCount++;
102 } else {
103 qDebug() << "Exporting new footprint:" << footprintName;
104 exportedCount++;
105 }
106
107 QString content = generateFootprintContent(footprint);
108
109 QFile file(fullPath);
110 if (!file.open(QIODevice::WriteOnly | QIODevice::Text)) {
111 qWarning() << "Failed to open file for writing:" << fullPath;
112 continue;
113 }
114
115 QTextStream out(&file);
116 out.setEncoding(QStringConverter::Utf8);
117 out << content;
118 file.close();
119
120 qDebug() << " Exported to:" << fullPath;
121 }
122
123 qDebug() << "Export complete - New:" << exportedCount << "Overwritten:" << overwrittenCount;
124 qDebug() << "Footprint library exported to:" << filePath;
125
126 return true;
127}

138QString ExporterFootprint::generateFootprintContent(const FootprintData& footprintData,
139 const QString& model3DPath) const {
140 QString content;
141
142 content += QString("(footprint easykiconverter:%1\n").arg(footprintData.info().name);
143 content += " (version 20221018)\n";
144
145 bool isThroughHole = false;
146 for (const FootprintPad& pad : footprintData.pads()) {
147 if (pad.holeRadius > 0) {
148 isThroughHole = true;
149 break;
150 }
151 }
152
153 if (isThroughHole) {
154 content += " (attr through_hole)\n";
155 } else {
156 content += " (attr smd)\n";
157 }
158
159 double yLow = 0;
160 double yHigh = 0;
161 if (!footprintData.pads().isEmpty()) {
162 yLow = footprintData.pads().first().centerY;
163 yHigh = footprintData.pads().first().centerY;
164 for (const FootprintPad& pad : footprintData.pads()) {
165 if (pad.centerY < yLow)
166 yLow = pad.centerY;
167 if (pad.centerY > yHigh)
168 yHigh = pad.centerY;
169 }
170 }
171
172 double bboxX = footprintData.bbox().x;
173 double bboxY = footprintData.bbox().y;
174
175 content += QString(" (fp_text reference REF** (at 0 %1) (layer F.SilkS)\n")
176 .arg(m_graphicsGenerator.pxToMm(yLow - bboxY - 4));
177 content += " (effects (font (size 1 1) (thickness 0.15)))\n";
178 content += " )\n";
179
180 content += QString(" (fp_text value %1 (at 0 %2) (layer F.Fab)\n")
181 .arg(footprintData.info().name)
182 .arg(m_graphicsGenerator.pxToMm(yHigh - bboxY + 4));
183 content += " (effects (font (size 1 1) (thickness 0.15)))\n";
184 content += " )\n";
185
186 content += " (fp_text user %R (at 0 0) (layer F.Fab)\n";
187 content += " (effects (font (size 1 1) (thickness 0.15)))\n";
188 content += " )\n";
189
190 for (const FootprintTrack& track : footprintData.tracks()) {
191 content += m_graphicsGenerator.generateTrack(track, bboxX, bboxY);
192 }
193 for (const FootprintRectangle& rect : footprintData.rectangles()) {
194 content += m_graphicsGenerator.generateRectangle(rect, bboxX, bboxY);
195 }
196
197 for (const FootprintPad& pad : footprintData.pads()) {
198 content += m_graphicsGenerator.generatePad(pad, bboxX, bboxY);
199 }
200
201 for (const FootprintHole& hole : footprintData.holes()) {
202 content += m_graphicsGenerator.generateHole(hole, bboxX, bboxY);
203 }
204
205 for (const FootprintCircle& circle : footprintData.circles()) {
206 content += m_graphicsGenerator.generateCircle(circle, bboxX, bboxY);
207 }
208
209 for (const FootprintArc& arc : footprintData.arcs()) {
210 content += m_graphicsGenerator.generateArc(arc, bboxX, bboxY);
211 }
212
213 for (const FootprintText& text : footprintData.texts()) {
214 content += m_graphicsGenerator.generateText(text, bboxX, bboxY);
215 }
216
217 bool hasCourtYard = false;
218 for (const FootprintSolidRegion& region : footprintData.solidRegions()) {
219 QString regionContent = m_graphicsGenerator.generateSolidRegion(region, bboxX, bboxY);
220 content += regionContent;
221 if (region.layerId == 99) {
222 hasCourtYard = true;
223 }
224 }
225
226 if (!hasCourtYard && footprintData.bbox().width > 0 && footprintData.bbox().height > 0) {
227 content += m_graphicsGenerator.generateCourtyardFromBBox(footprintData.bbox(), bboxX, bboxY);
228 qWarning() << "Warning: No courtyard found, generated from BBox";
229 }
230
231 if (!footprintData.model3D().name().isEmpty() || !model3DPath.isEmpty()) {
232 content += m_graphicsGenerator.generateModel3D(
233 footprintData.model3D(), bboxX, bboxY, model3DPath, footprintData.info().type);
234 }
235
236 content += ")\n";
237
238 return content;
239}
