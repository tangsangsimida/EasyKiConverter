84bool ComponentCacheService::migrateCacheDirectory(const QString& oldCacheDir, const QString& newCacheDir) const {
85 if (oldCacheDir.isEmpty() || newCacheDir.isEmpty() || oldCacheDir == newCacheDir) {
86 return true;
87 }
88
89 QDir source(oldCacheDir);
90 if (!source.exists()) {
91 return true;
92 }
93
94 QDir target;
95 if (!target.exists(newCacheDir) && !target.mkpath(newCacheDir)) {
96 LOG_WARN(LogModule::Core, "Failed to create cache migration target directory: {}", newCacheDir);
97 return false;
98 }
99
100 const bool moved = moveDirectoryContents(oldCacheDir, newCacheDir);
101 if (moved) {
102 source.rmdir(oldCacheDir);
103 LOG_DEBUG(LogModule::Core, "Migrated cache directory from {} to {}", oldCacheDir, newCacheDir);
104 } else {
105 LOG_WARN(LogModule::Core,
106 "Cache directory migration completed with skipped or failed entries: {} -> {}",
107 oldCacheDir,
108 newCacheDir);
109 }
110 return moved;

113bool ComponentCacheService::moveDirectoryContents(const QString& sourceDir, const QString& targetDir) const {
114 QDir source(sourceDir);
115 if (!source.exists()) {
116 return true;
117 }
118
119 QDir target;
120 if (!target.exists(targetDir) && !target.mkpath(targetDir)) {
121 LOG_WARN(LogModule::Core, "Failed to create cache migration directory: {}", targetDir);
122 return false;
123 }
124
125 bool allMoved = true;
126 const QFileInfoList entries = source.entryInfoList(QDir::AllEntries | QDir::NoDotAndDotDot);
127 for (const QFileInfo& entryInfo : entries) {
128 const QString sourcePath = entryInfo.absoluteFilePath();
129 const QString targetPath = QDir(targetDir).filePath(entryInfo.fileName());
130 if (!moveCacheEntry(sourcePath, targetPath)) {
131 allMoved = false;
132 }
133 }

138bool ComponentCacheService::moveCacheEntry(const QString& sourcePath, const QString& targetPath) const {
139 QFileInfo sourceInfo(sourcePath);
140 if (!sourceInfo.exists()) {
141 return true;
142 }
143
144 QFileInfo targetInfo(targetPath);
145 if (targetInfo.exists()) {
146 if (sourceInfo.isDir() && targetInfo.isDir()) {
147 const bool moved = moveDirectoryContents(sourcePath, targetPath);
148 QDir sourceDir(sourcePath);
149 if (sourceDir.entryList(QDir::AllEntries | QDir::NoDotAndDotDot).isEmpty()) {
150 sourceDir.rmdir(sourcePath);
151 }
152 return moved;
153 }
154
155 LOG_WARN(LogModule::Core, "Skipping cache migration entry because target already exists: {}", targetPath);
156 return false;
157 }
158
159 QDir targetParent(targetInfo.absolutePath());
160 if (!targetParent.exists() && !targetParent.mkpath(QStringLiteral("."))) {
161 LOG_WARN(LogModule::Core, "Failed to create cache migration parent directory: {}", targetInfo.absolutePath());
162 return false;
163 }
164
165 if (sourceInfo.isDir()) {
166 QDir dir;
167 if (dir.rename(sourcePath, targetPath)) {
168 return true;
169 }
170
171 if (!moveDirectoryContents(sourcePath, targetPath)) {
172 return false;
173 }
174
175 QDir sourceDir(sourcePath);
176 return sourceDir.removeRecursively();
177 }
178
179 if (QFile::rename(sourcePath, targetPath)) {
180 return true;
181 }
182
183 if (QFile::copy(sourcePath, targetPath)) {
184 return QFile::remove(sourcePath);
185 }
186
187 LOG_WARN(LogModule::Core, "Failed to migrate cache file: {} -> {}", sourcePath, targetPath);
188 return false;
189}

424void ComponentCacheService::saveComponentMetadata(const QString& componentId,
425 const ComponentData& data,
426 uint64_t expectedGeneration) {
427 QJsonObject metadata = buildMetadata(componentId, data);
428 const QJsonObject existingMetadata = loadMetadata(componentId);
429 metadata = mergeMetadata(existingMetadata, metadata);
430
431 QJsonDocument doc(metadata);
432 QString key = makeMemoryKey(componentId, "metadata");
433 QByteArray* newData = new QByteArray(doc.toJson(QJsonDocument::Compact));
434 qint64 sizeAfterUpdate = 0;
435
436 // 代次检查 + L1 写入 + 磁盘写入在同一个互斥区内，彻底关闭竞态窗口
437 {
438 QMutexLocker diskLocker(&m_diskWriteMutex);
439 if (expectedGeneration != 0) {
440 if (m_cacheGeneration.load() != expectedGeneration) {
441 LOG_DEBUG(LogModule::Core, "Discarded stale write for {} (generation mismatch)", componentId);
442 delete newData;
443 return;
444 }
445 if (isTombstoned(componentId)) {
446 LOG_DEBUG(LogModule::Core, "Discarded write for tombstoned component {}", componentId);
447 delete newData;
448 return;
449 }
450 }
451 {
452 QMutexLocker locker(&m_mutex);
453 m_memoryCache.insert(key, newData, newData->size());
454 sizeAfterUpdate = m_memoryCache.totalCost();
455 }
456 saveMetadata(componentId, metadata);
457 }
458 enforceDiskCacheLimit();
459 emit memoryCacheSizeChanged(sizeAfterUpdate);
460 emit cacheSaved(componentId);
461 LOG_DEBUG(LogModule::Core, "Saved component metadata to cache: {}", componentId);
462}

647bool ComponentCacheService::hasSymbolFootprintCache(const QString& lcscId) const {
648 // Fast path: if metadata is in memory cache, cad_data.json exists and was previously valid
649 // This avoids disk I/O for components that were cached in this session
650 QString metadataKey = makeMemoryKey(lcscId, "metadata");
651 {
652 QMutexLocker locker(&m_mutex);
653 if (m_memoryCache.contains(metadataKey)) {
654 // Metadata in memory means component was cached before
655 // Just need to verify cad_data.json still exists (fast stat call)
656 QString cadDataPath = componentCacheDir(lcscId) + "/cad_data.json";
657 bool exists = QFileInfo::exists(cadDataPath);
658 LOG_DEBUG(LogModule::Core,
659 "hasSymbolFootprintCache: memory hit for {}, cad_data.json exists: {}",
660 lcscId,
661 exists);
662 return exists;
663 }
664 }
665
666 // Slow path: memory cache miss - do fast disk check without full JSON parse
667 // The actual JSON validity will be verified when loading the data
668 QString cadDataPath = componentCacheDir(lcscId) + "/cad_data.json";
669 QFileInfo fileInfo(cadDataPath);
670 if (!fileInfo.exists()) {
671 LOG_DEBUG(LogModule::Core, "hasSymbolFootprintCache: no cad_data.json for {}", lcscId);
672 return false;
673 }
674
675 // Quick integrity check: verify file has minimum size and ends with valid JSON terminator
676 // This guards against truncated files from crashes during write, without full JSON parse
677 const qint64 fileSize = fileInfo.size();
678 if (fileSize < 2) { // Minimum valid JSON: "{}"
679 LOG_DEBUG(LogModule::Core, "hasSymbolFootprintCache: cad_data.json too small for {}", lcscId);
680 return false;
681 }
682
683 // Check last non-whitespace character is '}' (JSON object terminator)
684 QFile file(cadDataPath);
685 if (file.open(QIODevice::ReadOnly)) {
686 const qint64 readSize = qMin(fileSize, qint64(16));
687 file.seek(fileSize - readSize); // Read last 16 bytes
688 QByteArray tail = file.read(readSize);
689 file.close();
690 // Find last non-whitespace character
691 for (int i = tail.size() - 1; i >= 0; --i) {
692 char c = tail[i];

1097void ComponentCacheService::removeCache(const QString& lcscId) {
1098 // 锁顺序：先 disk，后 tombstone（与其他方法一致）
1099 QMutexLocker diskLocker(&m_diskWriteMutex);
1100 // 标记为 tombstone，阻止旧回调写回
1101 {
1102 QMutexLocker tombLocker(&m_tombstoneMutex);
1103 m_tombstones.insert(lcscId.toUpper());
1104 }
1105 // 先删除L2磁盘缓存
1106 QString dirPath = componentCacheDir(lcscId);
1107 if (dirPath.isEmpty()) {
1108 return;
1109 }
1110 {
1111 QDir dir(dirPath);
1112 if (dir.exists()) {
1113 dir.removeRecursively();
1114 LOG_DEBUG(LogModule::Core, "Removed disk cache for: {}", lcscId);
1115 }
1116 }
1117
1118 // 再删除L1内存缓存（需要锁）
1119 qint64 sizeAfterUpdate = 0;
1120 {
1121 QMutexLocker locker(&m_mutex);
1122
1123 QString metadataKey = makeMemoryKey(lcscId, "metadata");
1124 QString symbolKey = makeMemoryKey(lcscId, "symbol");
1125 QString footprintKey = makeMemoryKey(lcscId, "footprint");
1126
1127 // take() 会自动从 QCache 的 totalCost() 中扣除被移除项的 cost
1128 delete m_memoryCache.take(metadataKey);
1129 delete m_memoryCache.take(symbolKey);
1130 delete m_memoryCache.take(footprintKey);
1131
1132 sizeAfterUpdate = m_memoryCache.totalCost();
1133 }
1134 // 锁外发送信号
1135 emit memoryCacheSizeChanged(sizeAfterUpdate);
1136}
