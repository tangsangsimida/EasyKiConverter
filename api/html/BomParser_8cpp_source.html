57QStringList BomParser::parseCsv(const QString& filePath) {
58 QStringList componentIds;
59 QFile file(filePath);
60

85QStringList BomParser::parseExcel(const QString& filePath) {
86 QStringList componentIds;
87 QXlsx::Document xlsx(filePath);
88
89 if (!xlsx.load()) {

93

95 xlsx.selectSheet(sheetName);
96 QXlsx::CellRange range = xlsx.dimension();
97
98 for (int row = range.firstRow(); row <= range.lastRow(); ++row) {
99 for (int col = range.firstColumn(); col <= range.lastColumn(); ++col) {
100 if (auto cell = xlsx.cellAt(row, col)) {
101 processCellText(cell->readValue().toString(), componentIds);
102 }
103 }
104 }

110void BomParser::processCellText(const QString& text, QStringList& result) {
111 QString trimmedCell = text.trimmed();
112
113 // 去除可能的引号
