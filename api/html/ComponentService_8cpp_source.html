35int previewImageIndexFromPath(const QString& path) {
36 static const QRegularExpression re(QStringLiteral("preview_(\\d+)\\.jpg$"),
37 QRegularExpression::CaseInsensitiveOption);
38 const QRegularExpressionMatch match = re.match(path);
39 if (!match.hasMatch()) {
40 return -1;
41 }
42
43 bool ok = false;
44 const int index = match.captured(1).toInt(&ok);
45 return ok ? index : -1;
46}

158void ComponentService::fetchComponentDataInternal(const QString& componentId, bool fetch3DModel) {
159 qDebug() << "Fetching component data (internal) for:" << componentId << "Fetch 3D:" << fetch3DModel
160 << "at:" << QDateTime::currentDateTime().toString("hh:mm:ss.zzz");
161
162 // 确保 componentId 格式统一（大写）
163 QString normalizedId = componentId.toUpper();
164
165 // 使用互斥锁保护 m_currentComponentId 的并发访问
166 {
167 QMutexLocker locker(&m_currentIdMutex);
168 m_currentComponentId = normalizedId;
169 }
170
171 ComponentData reusableData;
172 uint64_t gen = 0; // 请求创建时的缓存 generation，用于过滤旧回调
173 bool reuseExistingData = false;
174 bool shouldSkipAsDuplicate = false;
175
176 // 先添加到 m_fetchingComponents（防止重复调度）
177 {
178 QMutexLocker locker(&m_fetchingComponentsMutex);
179 if (m_fetchingComponents.contains(normalizedId)) {
180 const FetchingComponent& existing = m_fetchingComponents[normalizedId];
181 const bool hasReusableData =
182 !existing.data.lcscId().isEmpty() && (existing.hasCadData || existing.data.symbolData() != nullptr ||
183 existing.data.footprintData() != nullptr);
184
185 if (hasReusableData) {
186 reusableData = existing.data;
187 reuseExistingData = true;
188 } else if (m_parallelContext != nullptr || !existing.requestActive) {
189 qDebug() << "ComponentService: Removing stale fetching state for" << normalizedId << "before retry";
190 m_fetchingComponents.remove(normalizedId);
191 } else {
192 shouldSkipAsDuplicate = true;
193 }
194 }
195
196 if (!reuseExistingData && !shouldSkipAsDuplicate && !m_fetchingComponents.contains(normalizedId)) {
197 // 创建占位条目，防止重复调度
198 FetchingComponent& fc = m_fetchingComponents[normalizedId];
199 initializeFetchingComponent(fc, normalizedId, fetch3DModel);
200 // 真正创建新请求时才解除 tombstone，允许缓存写入
201 ComponentCacheService::instance()->clearTombstone(normalizedId);
202 }
203 }
204
205 if (reuseExistingData) {
206 qDebug() << "ComponentService: Reusing existing fetched data for:" << normalizedId;
207 updateComponentCache(normalizedId, reusableData);
208 emit cadDataReady(normalizedId, reusableData);
209
210 if (m_parallelContext != nullptr) {
211 handleParallelDataCollected(normalizedId, reusableData);
212 }
213 return;
214 }
215
216 if (shouldSkipAsDuplicate) {
217 qDebug() << "ComponentService: Already fetching for" << normalizedId << ", skipping duplicate request";
218 return;
219 }
220
221 // 检查是否有符号封装缓存，有则直接从缓存加载（元器件存在）
222 ComponentCacheService* cache = ComponentCacheService::instance();
223 if (m_api) {
224 m_api->setWeakNetworkSupport(ConfigService::instance()->getWeakNetworkSupport());
225 }
226 if (cache->hasSymbolFootprintCache(normalizedId)) {
227 qDebug() << "ComponentService: Symbol/Footprint cache hit for" << normalizedId
228 << ", loading from cache (component exists)";
229 QTimer::singleShot(0, this, [this, normalizedId, fetch3DModel, cache]() {
230 loadComponentDataFromCacheAsync(normalizedId, fetch3DModel, cache);
231 });
232 return;
233 }
234
235 // 符号封装缓存不存在，改为后台线程直接获取并解析 CAD 数据，避免主线程卡顿。
236 // 从 FetchingComponent 读取请求创建时的 generation
237 {
238 QMutexLocker locker(&m_fetchingComponentsMutex);
239 auto it = m_fetchingComponents.find(normalizedId);
240 if (it != m_fetchingComponents.end()) {
241 gen = it->cacheGeneration;
242 }
243 }
244 if (gen == 0) {
245 qWarning() << "fetchComponentDataInternal: No FetchingComponent for" << normalizedId << ", aborting";
246 return;
247 }
248 auto future = QtConcurrent::run([normalizedId]() { return CadDataLoader::fetchAndParseCadData(normalizedId); });
249 auto* watcher = new QFutureWatcher<CadFetchTaskResult>(this);
250 connect(watcher, &QFutureWatcher<CadFetchTaskResult>::finished, this, [this, watcher, gen]() {
251 const CadFetchTaskResult result = watcher->result();
252 watcher->deleteLater();
253
254 {
255 QMutexLocker locker(&m_fetchingComponentsMutex);
256 if (!m_fetchingComponents.contains(result.componentId)) {
257 return;
258 }
259 }
260
261 if (!result.success) {
262 emitFetchErrorAndClearState(result.componentId, result.errorMessage);
263 return;
264 }
265
266 {
267 QMutexLocker locker(&m_fetchingComponentsMutex);
268 auto it = m_fetchingComponents.find(result.componentId);
269 if (it == m_fetchingComponents.end()) {
270 return;
271 }
272 it->data = result.parsed.componentData;
273 it->hasCadData = true;
274 it->requestActive = false;
275 }
276
277 updateComponentCache(result.componentId, result.parsed.componentData);
278 emit cadDataReady(result.componentId, result.parsed.componentData);
279 // 使用异步保存，不阻塞UI
280 ComponentCacheService::instance()->saveComponentMetadataAsync(
281 result.componentId, result.parsed.componentData, gen);
282 ComponentCacheService::instance()->saveCadDataJson(
283 result.componentId, QJsonDocument(result.parsed.resultData).toJson(QJsonDocument::Compact), gen);
284
285 if (m_parallelContext != nullptr) {
286 handleParallelDataCollected(result.componentId, result.parsed.componentData);
287 }
288 });
289 watcher->setFuture(future);
290}

292void ComponentService::loadComponentDataFromCacheAsync(const QString& normalizedId,
293 bool fetch3DModel,
294 ComponentCacheService* cache) {
295 // 从 FetchingComponent 读取请求创建时的 generation
296 uint64_t gen = 0;
297 {
298 QMutexLocker locker(&m_fetchingComponentsMutex);
299 auto it = m_fetchingComponents.find(normalizedId);
300 if (it != m_fetchingComponents.end()) {
301 gen = it->cacheGeneration;
302 }
303 }
304 if (gen == 0) {
305 qWarning() << "loadComponentDataFromCacheAsync: No FetchingComponent for" << normalizedId << ", aborting";
306 return;
307 }
308 // 在后台线程执行缓存加载（I/O 密集型）
309 QFuture<CacheLoadResult> future = QtConcurrent::run([normalizedId, fetch3DModel, cache]() -> CacheLoadResult {
310 CacheLoadResult result;
311 result.componentId = normalizedId;
312 result.success = false;
313
314 QSharedPointer<ComponentData> cachedData = cache->loadComponentData(normalizedId);
315 if (!cachedData) {
316 return result;
317 }
318
319 // 从缓存加载CAD数据并重新导入符号和封装
320 QByteArray cadJsonData = cache->loadCadDataJson(normalizedId);
321 if (!cadJsonData.isEmpty()) {
322 QJsonParseError parseError;
323 QJsonDocument cadDoc = QJsonDocument::fromJson(cadJsonData, &parseError);
324 if (parseError.error == QJsonParseError::NoError && cadDoc.isObject()) {
325 QJsonObject cadDataObj = cadDoc.object();
326 // 在后台线程创建 importer 并解析符号和封装数据
327 EasyedaSymbolImporter symbolImporter;
328 result.symbolData = symbolImporter.importSymbolData(cadDataObj);
329 EasyedaFootprintImporter footprintImporter;
330 result.footprintData = footprintImporter.importFootprintData(cadDataObj);
331 }
332 }
333
334 // 从 metadata.json 恢复 model3DData
335 // 注意：loadComponentData 已经加载了 model3DData（包含 name、translation、rotation）
336 // 这里只在没有 model3DData 或 UUID 缺失时进行补充，不要覆盖已有数据
337 QJsonObject metadata = cache->loadMetadata(normalizedId);
338 if (metadata.contains("model3duuid")) {
339 const QString uuid = metadata.value("model3duuid").toString();
340 if (!uuid.isEmpty()) {
341 if (!cachedData->model3DData()) {
342 auto model3DData = QSharedPointer<Model3DData>::create();
343 model3DData->setUuid(uuid);
344 cachedData->setModel3DData(model3DData);
345 } else if (cachedData->model3DData()->uuid().isEmpty()) {
346 cachedData->model3DData()->setUuid(uuid);
347 }
348 }
349 }
350
351 if (!result.symbolData || !result.footprintData) {
352 qWarning() << "Cache load incomplete for" << normalizedId << "- symbol:" << (result.symbolData != nullptr)
353 << "footprint:" << (result.footprintData != nullptr);
354 result.success = false;
355 return result;
356 }
357
358 result.cachedData = cachedData;
359 result.success = true;
360
361 // 加载预览图数据（返回字节数据，在主线程创建 QImage）
362 result.encodedPreviewImages = QStringList(3);
363 for (int i = 0; i < 3; ++i) {
364 QByteArray imageData = cache->loadPreviewImage(normalizedId, i);
365 if (!imageData.isEmpty()) {
366 result.previewImageData.append({i, imageData});
367 result.encodedPreviewImages[i] = QString::fromLatin1(imageData.toBase64());
368 }
369 }
370
371 // 加载数据手册
372 QByteArray datasheetData = cache->loadDatasheet(normalizedId);
373 if (!datasheetData.isEmpty()) {
374 result.datasheetData = datasheetData;
375 }
376
377 // 预加载 OBJ 数据，避免导出阶段重复磁盘 I/O
378 if (fetch3DModel && cachedData->model3DData() && !cachedData->model3DData()->uuid().isEmpty()) {
379 QByteArray objData = cache->loadModel3D(cachedData->model3DData()->uuid(), QStringLiteral("obj"));
380 if (!objData.isEmpty()) {
381 cachedData->setModel3DObjRaw(objData);
382 }
383 }
384
385 return result;
386 });
387
388 // 等待后台任务完成并在主线程处理结果
389 QFutureWatcher<CacheLoadResult>* watcher = new QFutureWatcher<CacheLoadResult>(this);
390 connect(
391 watcher, &QFutureWatcher<CacheLoadResult>::finished, this, [this, watcher, normalizedId, fetch3DModel, gen]() {
392 CacheLoadResult result = watcher->result();
393 watcher->deleteLater();
394
395 if (!result.success || !result.cachedData) {
396 qWarning() << "ComponentService: Failed to load cache for" << normalizedId
397 << ", falling back to network fetch";
398 // 缓存加载失败时，回退到网络获取
399 uint64_t retryGen = 0;
400 {
401 QMutexLocker locker(&m_fetchingComponentsMutex);
402 auto it = m_fetchingComponents.find(normalizedId);
403 if (it == m_fetchingComponents.end()) {
404 return;
405 }
406 FetchingComponent& fetchingComponent = *it;
407 initializeFetchingComponent(fetchingComponent, normalizedId, fetch3DModel);
408 // retry 视为新请求，使用新的 generation
409 retryGen = fetchingComponent.cacheGeneration;
410 }
411 if (retryGen == 0) {
412 qWarning() << "Cache retry: No FetchingComponent for" << normalizedId << ", aborting";
413 return;
414 }
415 auto retryFuture =
416 QtConcurrent::run([normalizedId]() { return CadDataLoader::fetchAndParseCadData(normalizedId); });
417 auto* retryWatcher = new QFutureWatcher<CadFetchTaskResult>(this);
418 connect(retryWatcher,
419 &QFutureWatcher<CadFetchTaskResult>::finished,
420 this,
421 [this, retryWatcher, retryGen]() {
422 const CadFetchTaskResult result = retryWatcher->result();
423 retryWatcher->deleteLater();
424
425 {
426 QMutexLocker locker(&m_fetchingComponentsMutex);
427 if (!m_fetchingComponents.contains(result.componentId)) {
428 return;
429 }
430 }
431
432 if (!result.success) {
433 emitFetchErrorAndClearState(result.componentId, result.errorMessage);
434 return;
435 }
436
437 {
438 QMutexLocker locker(&m_fetchingComponentsMutex);
439 auto it = m_fetchingComponents.find(result.componentId);
440 if (it == m_fetchingComponents.end()) {
441 return;
442 }
443 it->data = result.parsed.componentData;
444 it->hasCadData = true;
445 it->requestActive = false;
446 }
447
448 updateComponentCache(result.componentId, result.parsed.componentData);
449 emit cadDataReady(result.componentId, result.parsed.componentData);
450 // 使用异步保存，不阻塞UI
451 ComponentCacheService::instance()->saveComponentMetadataAsync(
452 result.componentId, result.parsed.componentData, retryGen);
453 ComponentCacheService::instance()->saveCadDataJson(
454 result.componentId,
455 QJsonDocument(result.parsed.resultData).toJson(QJsonDocument::Compact),
456 retryGen);
457
458 if (m_parallelContext != nullptr) {
459 handleParallelDataCollected(result.componentId, result.parsed.componentData);
460 }
461 });
462 retryWatcher->setFuture(retryFuture);
463 return;
464 }
465
466 // 使用后台线程预解析的符号和封装数据
467 if (result.symbolData) {
468 result.cachedData->setSymbolData(result.symbolData);
469 }
470 if (result.footprintData) {
471 result.cachedData->setFootprintData(result.footprintData);
472 const Model3DData parsedModel3D = result.footprintData->model3D();
473 if (!parsedModel3D.uuid().isEmpty()) {
474 auto model3DData = QSharedPointer<Model3DData>::create();
475 model3DData->setUuid(parsedModel3D.uuid());
476 model3DData->setName(parsedModel3D.name());
477 model3DData->setTranslation(parsedModel3D.translation());
478 model3DData->setRotation(parsedModel3D.rotation());
479 result.cachedData->setModel3DData(model3DData);
480 }
481 }
482
483 // 更新 m_fetchingComponents
484 {
485 QMutexLocker locker(&m_fetchingComponentsMutex);
486 auto it = m_fetchingComponents.find(normalizedId);
487 if (it == m_fetchingComponents.end()) {
488 return;
489 }
490 FetchingComponent& fetchingComponent = *it;
491 fetchingComponent.componentId = normalizedId;
492 fetchingComponent.data = *result.cachedData;
493 fetchingComponent.fetch3DModel = fetch3DModel;
494 fetchingComponent.hasCadData =
495 (result.cachedData->symbolData() != nullptr && result.cachedData->footprintData() != nullptr);
496 fetchingComponent.requestActive = false;
497 }
498
499 // 发送缓存加载的信号
500 qDebug() << "ComponentService: Emitting cadDataReady for" << normalizedId
501 << "with symbolData:" << (result.cachedData->symbolData() != nullptr)
502 << "footprintData:" << (result.cachedData->footprintData() != nullptr);
503
504 // 更新缓存
505 updateComponentCache(normalizedId, *result.cachedData);
506
507 emit cadDataReady(normalizedId, *result.cachedData);
508
509 // 如果在并行模式，处理并行数据收集
510 if (m_parallelContext != nullptr) {
511 handleParallelDataCollected(normalizedId, *result.cachedData);
512 }
513
514 // 从缓存加载预览图数据（批量发送，避免频繁 UI 更新）
515 if (!result.encodedPreviewImages.isEmpty()) {
516 const QStringList encodedImages = result.encodedPreviewImages;
517 QTimer::singleShot(0, this, [this, normalizedId, encodedImages]() {
518 emit previewImagesReady(normalizedId, encodedImages);
519 });
520 }
521
522 // 加载数据手册
523 if (!result.datasheetData.isEmpty()) {
524 emit datasheetReady(normalizedId, result.datasheetData);
525 }
526
527 qDebug() << "ComponentService: Cache loaded successfully for" << normalizedId;
528 });
529 watcher->setFuture(future);
530}

566void ComponentService::handleLcscDataReady(const QString& componentId,
567 const QString& manufacturerPart,
568 const QString& datasheetUrl,
569 const QStringList& imageUrls) {
570 qDebug() << "LCSC data ready for component:" << componentId
571 << "Manufacturer Part:" << (manufacturerPart.isEmpty() ? "none" : manufacturerPart)
572 << "Datasheet:" << (datasheetUrl.isEmpty() ? "none" : datasheetUrl) << "Images:" << imageUrls.size();
573
574 // 准备要更新的数据（在锁外构建，避免长时间持锁）
575 ComponentData updatedData;
576 bool hasValidUpdate = false;
577 uint64_t gen = 0;
578
579 // 加锁保护共享数据的访问
580 {
581 QMutexLocker locker(&m_fetchingComponentsMutex);
582
583 // 更新 m_fetchingComponents 中的数据
584 if (m_fetchingComponents.contains(componentId)) {
585 FetchingComponent& fetchingComponent = m_fetchingComponents[componentId];
586
587 // 保存制造商部件号
588 if (!manufacturerPart.isEmpty()) {
589 fetchingComponent.data.setManufacturerPart(manufacturerPart);
590 qDebug() << "Manufacturer part saved to ComponentData:" << manufacturerPart;
591 }
592
593 // 保存数据手册 URL
594 if (!datasheetUrl.isEmpty()) {
595 fetchingComponent.data.setDatasheet(datasheetUrl);
596
597 // 检测数据手册格式
598 QString format = "pdf";
599 if (datasheetUrl.toLower().contains(".html")) {
600 format = "html";
601 }
602 fetchingComponent.data.setDatasheetFormat(format);
603
604 qDebug() << "Datasheet saved to ComponentData:" << datasheetUrl << "format:" << format;
605 }
606
607 // 保存预览图 URL 列表
608 if (!imageUrls.isEmpty()) {
609 fetchingComponent.data.setPreviewImages(imageUrls);
610 qDebug() << "Preview images saved to ComponentData:" << imageUrls.size() << "images";
611
612 // 预先创建指定数量的空元素，确保索引能够正确对应
613 // 这样当图片按乱序下载时，能够填充到正确的索引位置
614 QList<QByteArray> emptyImageDataList;
615 emptyImageDataList.resize(imageUrls.size());
616 fetchingComponent.data.setPreviewImageData(emptyImageDataList);
617 qDebug() << "Pre-allocated" << imageUrls.size() << "empty image data slots";
618 }
619
620 // 复制数据用于锁外处理
621 updatedData = fetchingComponent.data;
622 gen = fetchingComponent.cacheGeneration;
623 hasValidUpdate = true;
624 } else {
625 qWarning() << "Component" << componentId << "not found in m_fetchingComponents, cannot update LCSC data";
626 }
627 } // 锁在这里释放
628
629 // 锁外发送信号和保存缓存（避免信号槽死锁和锁顺序问题）
630 if (hasValidUpdate) {
631 // 发送 LCSC 数据更新信号，以便 ComponentListViewModel 可以更新缓存的 ComponentData
632 emit lcscDataUpdated(componentId, manufacturerPart, datasheetUrl, imageUrls);
633
634 // 保存到磁盘缓存（异步，不阻塞UI）
635 ComponentCacheService::instance()->saveComponentMetadataAsync(componentId, updatedData, gen);
636
637 // 更新内存缓存，确保 startPreload 时能获取到最新数据
638 updateComponentCache(componentId, updatedData);
639 }

642void ComponentService::handleDatasheetReady(const QString& componentId, const QByteArray& datasheetData) {
643 qDebug() << "Datasheet downloaded for component:" << componentId << "size:" << datasheetData.size() << "bytes";
644
645 // 准备数据用于锁外处理
646 QString format;
647 bool hasValidUpdate = false;
648 bool shouldMarkCompleted = false;
649 ComponentData completedData;
650
651 // 加锁保护共享数据的访问
652 {
653 QMutexLocker locker(&m_fetchingComponentsMutex);
654
655 // 更新 m_fetchingComponents 中的数据
656 if (m_fetchingComponents.contains(componentId)) {
657 FetchingComponent& fetchingComponent = m_fetchingComponents[componentId];
658 fetchingComponent.data.setDatasheetData(datasheetData);
659
660 // 检测数据手册格式（基于内容）
661 format = fetchingComponent.data.datasheetFormat();
662 if (format == "pdf" && !isPDF(datasheetData)) {
663 format = "html";
664 fetchingComponent.data.setDatasheetFormat(format);
665 }
666
667 qDebug() << "Datasheet data saved to ComponentData, size:" << datasheetData.size() << "bytes"
668 << "format:" << format;
669
670 // 递减待处理的异步下载计数
671 if (fetchingComponent.pendingAsyncDownloads > 0) {
672 fetchingComponent.pendingAsyncDownloads--;
673 qDebug() << "Datasheet download complete, pending async downloads:"
674 << fetchingComponent.pendingAsyncDownloads;
675 if (fetchingComponent.pendingAsyncDownloads == 0) {
676 shouldMarkCompleted = true;
677 completedData = fetchingComponent.data;
678 }
679 }
680
681 hasValidUpdate = true;
682 } else {
683 qWarning() << "Component" << componentId
684 << "not found in m_fetchingComponents, datasheet data will be saved directly to cache";
685 }
686 } // 锁在这里释放
687
688 // 如果所有异步下载都完成了，标记组件完成
689 if (shouldMarkCompleted) {
690 ParallelFetchContext* parallelContext = nullptr;
691 {
692 QMutexLocker locker(&m_parallelContextMutex);
693 parallelContext = m_parallelContext;
694 }
695 if (parallelContext != nullptr) {
696 parallelContext->markCompleted(componentId, completedData);
697 }
698
699 if (m_queueManager != nullptr) {
700 m_queueManager->requestCompleted(componentId);
701 }
702 }
703
704 // 更新缓存中的数据手册数据
705 if (hasValidUpdate) {
706 // 避免嵌套锁：先从 fetchingComponents 获取数据副本，再更新缓存
707 ComponentData dataCopy;
708 bool hasDataCopy = false;
709 {
710 QMutexLocker fetchLocker(&m_fetchingComponentsMutex);
711 if (m_fetchingComponents.contains(componentId)) {
712 dataCopy = m_fetchingComponents[componentId].data;
713 hasDataCopy = true;
714 }
715 }
716
717 if (hasDataCopy) {
718 QMutexLocker cacheLocker(&m_componentCacheMutex);
719 if (m_componentCache.contains(componentId)) {
720 m_componentCache[componentId] = dataCopy;
721 qDebug() << "ComponentService: Updated cache with datasheet data for" << componentId;
722 }
723 }
724 } else {
725 // 组件不在 m_fetchingComponents 中（可能在批处理模式下已移除）
726 // 直接更新内存缓存
727 QMutexLocker cacheLocker(&m_componentCacheMutex);
728 if (m_componentCache.contains(componentId)) {
729 m_componentCache[componentId].setDatasheetData(datasheetData);
730 // 检测数据手册格式
731 format = m_componentCache[componentId].datasheetFormat();
732 if (format.isEmpty()) {
733 format = "pdf";
734 }
735 if (format == "pdf" && !isPDF(datasheetData)) {
736 format = "html";
737 }
738 m_componentCache[componentId].setDatasheetFormat(format);
739 qDebug() << "ComponentService: Updated cache directly for" << componentId;
740 hasValidUpdate = true;
741 }
742 }
743
744 // 锁外发送信号（避免信号槽死锁）
745 if (hasValidUpdate) {
746 // 发送数据手册就绪信号
747 emit datasheetReady(componentId, datasheetData);
748 }

763void ComponentService::handleAllImagesReady(const QString& componentId, const QStringList& imagePaths) {
764 qDebug() << "All images ready for component:" << componentId << "paths:" << imagePaths.size();
765
766 // 注意：预览图已由 LcscImageService::handleDownloadResponse 在下载时保存到磁盘
767 // 这里不需要再次保存，避免重复 I/O
768
769 // 使用 QtConcurrent 在后台线程执行文件读取和 Base64 编码，避免阻塞 UI
770 // 然后在主线程发送信号
771 QFuture<QStringList> future = QtConcurrent::run([this, componentId, imagePaths]() {
772 QStringList encodedImages(3);
773 QList<QByteArray> imageDataList;
774 imageDataList.resize(3);
775
776 for (const QString& path : imagePaths) {
777 const int imageIndex = previewImageIndexFromPath(path);
778 if (imageIndex < 0 || imageIndex >= encodedImages.size()) {
779 continue;
780 }
781
782 QFile file(path);
783 if (file.open(QIODevice::ReadOnly)) {
784 QByteArray data = file.readAll();
785 encodedImages[imageIndex] = QString::fromLatin1(data.toBase64().data());
786 imageDataList[imageIndex] = data;
787 file.close();
788 }
789 }
790
791 // 在后台线程更新 m_fetchingComponents（使用锁保护）
792 {
793 QMutexLocker locker(&m_fetchingComponentsMutex);
794 if (m_fetchingComponents.contains(componentId)) {
795 FetchingComponent& fetchingComponent = m_fetchingComponents[componentId];
796 fetchingComponent.data.setPreviewImageData(imageDataList);
797 qDebug() << "All image data updated in ComponentData for component:" << componentId
798 << "count:" << imageDataList.size();
799 }
800 }
801
802 return encodedImages;
803 });
804
805 // 使用 QFutureWatcher 在主线程接收结果并发送信号
806 auto* watcher = new QFutureWatcher<QStringList>(this);
807 connect(watcher, &QFutureWatcher<QStringList>::finished, this, [this, watcher, componentId, imagePaths]() {
808 QStringList encodedImages = watcher->result();
809 watcher->deleteLater();
810 emit previewImagesReady(componentId, encodedImages);
811 emit allImagesReady(componentId, imagePaths);
812 });
813 watcher->setFuture(future);
814}

842void ComponentService::handleCadDataFetched(const QString& componentId, const QJsonObject& data) {
843 // 从 FetchingComponent 读取请求创建时的 generation，而非当前值
844 uint64_t gen = 0;
845 {
846 QMutexLocker locker(&m_fetchingComponentsMutex);
847 auto it = m_fetchingComponents.find(componentId.toUpper());
848 if (it != m_fetchingComponents.end()) {
849 gen = it->cacheGeneration;
850 }
851 }
852 if (gen == 0) {
853 // 找不到 FetchingComponent，说明是已过期的孤儿回调，直接丢弃
854 qDebug() << "handleCadDataFetched: No FetchingComponent for" << componentId << ", discarding stale callback";
855 return;
856 }
857 auto future =
858 QtConcurrent::run([componentId, data]() { return CadDataLoader::parseCadPayload(componentId, data); });
859 auto* watcher = new QFutureWatcher<CadParseResult>(this);
860 connect(watcher, &QFutureWatcher<CadParseResult>::finished, this, [this, watcher, gen]() {
861 const CadParseResult parsed = watcher->result();
862 watcher->deleteLater();
863
864 if (!parsed.success) {
865 emitFetchErrorAndClearState(parsed.componentId, parsed.errorMessage);
866 return;
867 }
868
869 {
870 QMutexLocker locker(&m_fetchingComponentsMutex);
871 auto it = m_fetchingComponents.find(parsed.componentId);
872 if (it == m_fetchingComponents.end()) {
873 return;
874 }
875 it->data = parsed.componentData;
876 it->hasCadData = true;
877 it->requestActive = false;
878 }
879
880 updateComponentCache(parsed.componentId, parsed.componentData);
881 emit cadDataReady(parsed.componentId, parsed.componentData);
882 // 使用异步保存，不阻塞UI
883 ComponentCacheService::instance()->saveComponentMetadataAsync(parsed.componentId, parsed.componentData, gen);
884 ComponentCacheService::instance()->saveCadDataJson(
885 parsed.componentId, QJsonDocument(parsed.resultData).toJson(QJsonDocument::Compact), gen);
886
887 if (m_parallelContext != nullptr) {
888 handleParallelDataCollected(parsed.componentId, parsed.componentData);
889 }
890 });
891 watcher->setFuture(future);
892}
