164void ComponentService::handleCadDataFetched(const QJsonObject& data) {
165 // 从数据中提取 LCSC ID
166 QString lcscId;
167 if (data.contains("lcscId")) {
168 lcscId = data["lcscId"].toString();
169 } else {
170 // 如果没有 lcscId 字段，尝试从 lcsc.szlcsc.number 中提取
171 if (data.contains("lcsc")) {
172 QJsonObject lcsc = data["lcsc"].toObject();
173 if (lcsc.contains("number")) {
174 lcscId = lcsc["number"].toString();
175 }
176 }
177 }
178
179 if (lcscId.isEmpty()) {
180 qWarning() << "Cannot extract LCSC ID from CAD data";
181 return;
182 }
183
184 qDebug() << "CAD data fetched for:" << lcscId;
185
186 // 临时保存当前的组件ID
187 QString savedComponentId = m_currentComponentId;
188 m_currentComponentId = lcscId;
189
190 // 获取 fetch3DModel 配置
191 bool need3DModel = true; // 默认值
192 if (m_fetchingComponents.contains(lcscId)) {
193 need3DModel = m_fetchingComponents[lcscId].fetch3DModel;
194 } else {
195 // 如果找不到配置（可能是旧的串行逻辑遗留，或者 m_fetchingComponents 被清除），
196 // 尝试回退到 m_fetch3DModel，但这在并行下不可靠。
197 // 不过由于我们在 fetchComponentDataInternal 中强制设置了 map，这里应该能找到。
198 qWarning() << "Fetching component info not found in map for:" << lcscId << "Using default fetch3DModel=true";
199 }
200
201 // 提取 result 数据
202 QJsonObject resultData;
203 if (data.contains("result")) {
204 resultData = data["result"].toObject();
205 } else {
206 // 直接使用 data
207 resultData = data;
208 }
209
210 if (resultData.isEmpty()) {
211 emit fetchError(lcscId, "Empty CAD data");
212 m_currentComponentId = savedComponentId;
213 return;
214 }
215
216 // 调试：打印resultData的结构
217 qDebug() << "=== CAD Data Structure ===";
218 qDebug() << "Top-level keys:" << resultData.keys();
219 if (resultData.contains("dataStr")) {
220 QJsonObject dataStr = resultData["dataStr"].toObject();
221 qDebug() << "dataStr keys:" << dataStr.keys();
222 if (dataStr.contains("shape")) {
223 QJsonArray shapes = dataStr["shape"].toArray();
224 qDebug() << "dataStr.shape size:" << shapes.size();
225 if (!shapes.isEmpty()) {
226 qDebug() << "First shape:" << shapes[0].toString().left(100);
227 }
228 } else {
229 qDebug() << "WARNING: dataStr does NOT contain 'shape' field!";
230 }
231 } else {
232 qDebug() << "WARNING: resultData does NOT contain 'dataStr' field!";
233 }
234 qDebug() << "===========================";
235
236 // 创建 ComponentData 对象
237 ComponentData componentData;
238 componentData.setLcscId(lcscId);
239
240 // 提取基本信息
241 if (resultData.contains("title")) {
242 componentData.setName(resultData["title"].toString());
243 }
244 if (resultData.contains("package")) {
245 componentData.setPackage(resultData["package"].toString());
246 }
247 if (resultData.contains("manufacturer")) {
248 componentData.setManufacturer(resultData["manufacturer"].toString());
249 }
250 if (resultData.contains("datasheet")) {
251 componentData.setDatasheet(resultData["datasheet"].toString());
252 }
253
254 // 导入符号数据
255 QSharedPointer<SymbolData> symbolData = m_importer->importSymbolData(resultData);
256 if (symbolData) {
257 componentData.setSymbolData(symbolData);
258 qDebug() << "Symbol imported successfully - Name:" << symbolData->info().name;
259 } else {
260 qWarning() << "Failed to import symbol data for:" << m_currentComponentId;
261 emit fetchError(lcscId, "Failed to parse Symbol data from EasyEDA JSON");
262 m_currentComponentId = savedComponentId;
263 return;
264 }
265
266 // 导入封装数据
267 QSharedPointer<FootprintData> footprintData = m_importer->importFootprintData(resultData);
268 if (footprintData) {
269 componentData.setFootprintData(footprintData);
270 qDebug() << "Footprint imported successfully - Name:" << footprintData->info().name;
271 } else {
272 qWarning() << "Failed to import footprint data for:" << m_currentComponentId;
273 emit fetchError(lcscId, "Failed to parse Footprint data from EasyEDA JSON (Library might be corrupted)");
274 m_currentComponentId = savedComponentId;
275 return;
276 }
277
278 // 检查是否需要获取3D 模型
279 // 使用本地变量 need3DModel 而不是成员变量 m_fetch3DModel
280 if (need3DModel && footprintData) {
281 // 检查封装数据中是否包含 3D 模型 UUID
282 QString modelUuid = footprintData->model3D().uuid();
283
284 // 如果封装数据中没有UUID，尝试从 head.uuid_3d 字段中提取
285 if (modelUuid.isEmpty() && resultData.contains("head")) {
286 QJsonObject head = resultData["head"].toObject();
287 if (head.contains("uuid_3d")) {
288 modelUuid = head["uuid_3d"].toString();
289 }
290 }
291
292 if (!modelUuid.isEmpty()) {
293 qDebug() << "Fetching 3D model with UUID:" << modelUuid;
294
295 // 创建 Model3DData 对象
296 QSharedPointer<Model3DData> model3DData(new Model3DData());
297 model3DData->setUuid(modelUuid);
298
299 // 如果封装数据中有 3D 模型信息，复制平移和旋转
300 if (!footprintData->model3D().uuid().isEmpty()) {
301 model3DData->setName(footprintData->model3D().name());
302 model3DData->setTranslation(footprintData->model3D().translation());
303 model3DData->setRotation(footprintData->model3D().rotation());
304 }
305
306 componentData.setModel3DData(model3DData);
307
308 // 在并行模式下，使用 m_fetchingComponents 存储待处理的组件数据
309 if (m_parallelFetching) {
310 // 更新已有的 entry
311 FetchingComponent& fetchingComponent = m_fetchingComponents[m_currentComponentId];
312 fetchingComponent.data = componentData;
313 fetchingComponent.hasComponentInfo = true;
314 fetchingComponent.hasCadData = true;
315 fetchingComponent.hasObjData = false;
316 fetchingComponent.hasStepData = false;
317 // fetch3DModel 已经在 internal 调用时设置了
318 } else {
319 // 串行模式下的处理
320 m_pendingComponentData = componentData;
321 m_pendingModelUuid = modelUuid;
322 m_hasDownloadedWrl = false;
323 }
324
325 // 获取 WRL 格式的3D 模型
326 m_api->fetch3DModelObj(modelUuid);
327 return; // 等待 3D 模型数据
328 } else {
329 qDebug() << "No 3D model UUID found for:" << m_currentComponentId;
330 }
331 }
332
333 // 不需要3D 模型或没有找到UUID，直接发送完成信号
334 emit cadDataReady(m_currentComponentId, componentData);
335
336 // 如果在并行模式下，处理并行数据收集
337 if (m_parallelFetching) {
338 handleParallelDataCollected(m_currentComponentId, componentData);
339 }
340
341 // 恢复组件 ID
342 m_currentComponentId = savedComponentId;
343}

345void ComponentService::handleModel3DFetched(const QString& uuid, const QByteArray& data) {
346 qDebug() << "3D model data fetched for UUID:" << uuid << "Size:" << data.size();
347
348 // 在并行模式下，查找对应的组件
349 if (m_parallelFetching) {
350 // 在并行模式下，查找对应UUID的组件
351 for (auto it = m_fetchingComponents.begin(); it != m_fetchingComponents.end(); ++it) {
352 if (it.value().data.model3DData() && it.value().data.model3DData()->uuid() == uuid) {
353 QString componentId = it.key();
354 FetchingComponent& fetchingComponent = it.value();
355
356 if (!fetchingComponent.hasObjData) {
357 // 这是 WRL 格式的3D 模型
358 fetchingComponent.data.model3DData()->setRawObj(QString::fromUtf8(data));
359 fetchingComponent.hasObjData = true;
360 qDebug() << "WRL data saved for:" << uuid << "Size:" << data.size();
361
362 // 继续下载 STEP 格式的3D 模型
363 qDebug() << "Fetching STEP model with UUID:" << uuid;
364 m_api->fetch3DModelStep(uuid);
365 } else {
366 // 这是 STEP 格式的3D 模型
367 fetchingComponent.data.model3DData()->setStep(data);
368 fetchingComponent.hasStepData = true;
369 qDebug() << "STEP data saved for:" << uuid << "Size:" << data.size();
370
371 // 发送完成信号
372 emit cadDataReady(componentId, fetchingComponent.data);
373
374 // 处理并行数据收集
375 handleParallelDataCollected(componentId, fetchingComponent.data);
376
377 // 从待处理列表中移除
378 m_fetchingComponents.remove(componentId);
379 }
380 return;
381 }
382 }
383 qWarning() << "Received 3D model data for unexpected UUID:" << uuid;
384 } else {
385 // 串行模式下的处理
386 if (m_pendingComponentData.model3DData() && m_pendingComponentData.model3DData()->uuid() == uuid) {
387 if (!m_hasDownloadedWrl) {
388 // 这是 WRL 格式的3D 模型
389 m_pendingComponentData.model3DData()->setRawObj(QString::fromUtf8(data));
390 qDebug() << "WRL data saved for:" << uuid << "Size:" << data.size();
391
392 // 标记已经下载了WRL 格式
393 m_hasDownloadedWrl = true;
394
395 // 继续下载 STEP 格式的3D 模型
396 qDebug() << "Fetching STEP model with UUID:" << uuid;
397 m_api->fetch3DModelStep(uuid);
398 } else {
399 // 这是 STEP 格式的3D 模型
400 m_pendingComponentData.model3DData()->setStep(data);
401 qDebug() << "STEP data saved for:" << uuid << "Size:" << data.size();
402
403 // 发送完成信号
404 emit cadDataReady(m_currentComponentId, m_pendingComponentData);
405
406 // 清空待处理数据
407 m_pendingComponentData = ComponentData();
408 m_pendingModelUuid.clear();
409 m_hasDownloadedWrl = false;
410 }
411 } else {
412 qWarning() << "Received 3D model data for unexpected UUID:" << uuid;
413 }
414 }

429void ComponentService::handleFetchErrorWithId(const QString& idOrUuid, const QString& error) {
430 QString componentId = idOrUuid;
431
432 // 如果在并行模式下，尝试解析 UUID 为组件 ID
433 if (m_parallelFetching && !m_parallelFetchingStatus.contains(idOrUuid)) {
434 for (auto it = m_fetchingComponents.begin(); it != m_fetchingComponents.end(); ++it) {
435 if (it.value().data.model3DData() && it.value().data.model3DData()->uuid() == idOrUuid) {
436 componentId = it.key();
437 qDebug() << "Resolved UUID" << idOrUuid << "to component ID" << componentId;
438 break;
439 }
440 }
441 }
442
443 qDebug() << "Fetch error for component:" << componentId << "Error:" << error;
444
445 // 如果在并行模式下，处理并行错误
446 if (m_parallelFetching) {
447 handleParallelFetchError(componentId, error);
448 }

525QStringList ComponentService::extractComponentIdFromText(const QString& text) const {
526 // 依然保留简单的提取逻辑，或者可以进一步整合进 BomParser
527 QStringList extractedIds;
528 QRegularExpression re("[Cc]\\d{4,}");
529 QRegularExpressionMatchIterator it = re.globalMatch(text);
530
531 while (it.hasNext()) {
532 QRegularExpressionMatch match = it.next();
533 QString id = match.captured().toUpper();
534 if (!BomParser::getExcludedIds().contains(id) && !extractedIds.contains(id)) {
535 extractedIds.append(id);
536 }
537 }
538 return extractedIds;
539}

555void ComponentService::completeComponentData(const QString& componentId) {
556 if (m_fetchingComponents.contains(componentId)) {
557 FetchingComponent& fc = m_fetchingComponents[componentId];
558 // 判定标准：CAD 数据必须有，如果需要 3D 则 OBJ 和 STEP 也必须有
559 bool isComplete = fc.hasCadData && (!fc.fetch3DModel || (fc.hasObjData && fc.hasStepData));
560
561 if (isComplete) {
562 emit cadDataReady(componentId, fc.data);
563 if (m_parallelFetching) {
564 handleParallelDataCollected(componentId, fc.data);
565 }
566 m_fetchingComponents.remove(componentId);
567 }
568 }
