180

183 FetchingComponent& fetchingComponent = m_fetchingComponents[componentId];

296void ComponentService::handleCadDataFetched(const QJsonObject& data) {
297 // 从数据中提取 LCSC ID
298 QString lcscId;
299 if (data.contains("lcscId")) {
300 lcscId = data["lcscId"].toString();
301 } else {
302 // 如果没有 lcscId 字段，尝试从 lcsc.szlcsc.number 中提取
303 if (data.contains("lcsc")) {
304 QJsonObject lcsc = data["lcsc"].toObject();
305 if (lcsc.contains("number")) {
306 lcscId = lcsc["number"].toString();
307 }
308 }
309 }
310
311 if (lcscId.isEmpty()) {
312 qWarning() << "Cannot extract LCSC ID from CAD data";
313 return;
314 }
315
316 qDebug() << "CAD data fetched for:" << lcscId;
317
318 // 临时保存当前的组件ID
319 QString savedComponentId = m_currentComponentId;
320 m_currentComponentId = lcscId;
321
322 // 获取 fetch3DModel 配置
323 bool need3DModel = true; // 默认值
324 if (m_fetchingComponents.contains(lcscId)) {
325 need3DModel = m_fetchingComponents[lcscId].fetch3DModel;
326 } else {
327 // 如果找不到配置（可能是旧的串行逻辑遗留，或者 m_fetchingComponents 被清除），
328 // 尝试回退到 m_fetch3DModel，但这在并行下不可靠。
329 // 不过由于我们在 fetchComponentDataInternal 中强制设置了 map，这里应该能找到。
330 qWarning() << "Fetching component info not found in map for:" << lcscId << "Using default fetch3DModel=true";
331 }
332
333 // 提取 result 数据
334 QJsonObject resultData;
335 if (data.contains("result")) {
336 resultData = data["result"].toObject();
337 } else {
338 // 直接使用 data
339 resultData = data;
340 }
341
342 if (resultData.isEmpty()) {
343 emit fetchError(lcscId, "Empty CAD data");
344 m_currentComponentId = savedComponentId;
345 return;
346 }
347
348 // 调试：打印resultData的结构
349 qDebug() << "=== CAD Data Structure ===";
350 qDebug() << "Top-level keys:" << resultData.keys();
351 if (resultData.contains("dataStr")) {
352 QJsonObject dataStr = resultData["dataStr"].toObject();
353 qDebug() << "dataStr keys:" << dataStr.keys();
354 if (dataStr.contains("shape")) {
355 QJsonArray shapes = dataStr["shape"].toArray();
356 qDebug() << "dataStr.shape size:" << shapes.size();
357 if (!shapes.isEmpty()) {
358 qDebug() << "First shape:" << shapes[0].toString().left(100);
359 }
360 } else {
361 qDebug() << "WARNING: dataStr does NOT contain 'shape' field!";
362 }
363 } else {
364 qDebug() << "WARNING: resultData does NOT contain 'dataStr' field!";
365 }
366 qDebug() << "===========================";
367
368 // 创建 ComponentData 对象
369 ComponentData componentData;
370 componentData.setLcscId(lcscId);
371
372 // 提取基本信息
373 if (resultData.contains("title")) {
374 componentData.setName(resultData["title"].toString());
375 }
376 if (resultData.contains("package")) {
377 componentData.setPackage(resultData["package"].toString());
378 }
379 if (resultData.contains("manufacturer")) {
380 componentData.setManufacturer(resultData["manufacturer"].toString());
381 }
382 if (resultData.contains("datasheet")) {
383 componentData.setDatasheet(resultData["datasheet"].toString());
384 }
385
386 // 导入符号数据
387 QSharedPointer<SymbolData> symbolData = m_importer->importSymbolData(resultData);
388 if (symbolData) {
389 componentData.setSymbolData(symbolData);
390 qDebug() << "Symbol imported successfully - Name:" << symbolData->info().name;
391
392 // 从符号数据中提取预览图（作为备用，稍后会被 LCSC API 的数据覆盖）
393 if (!symbolData->info().thumb.isEmpty()) {
394 QString thumbUrl = symbolData->info().thumb;
395 // 检查是否是相对路径，如果是则拼接完整的 URL
396 if (thumbUrl.startsWith("/")) {
397 thumbUrl = "https://image.lceda.cn" + thumbUrl;
398 qDebug() << "Preview image is relative path, constructed full URL:" << thumbUrl;
399 }
400 componentData.setPreviewImages(QStringList() << thumbUrl);
401 qDebug() << "Preview image extracted from symbolData (EasyEDA, will be overridden by LCSC if available):"
402 << thumbUrl;
403 } else {
404 qDebug() << "No preview image found in symbolData";
405 }
406
407 // 从符号数据中提取手册（作为备用，稍后会被 LCSC API 的数据覆盖）
408 if (!symbolData->info().datasheet.isEmpty() && componentData.datasheet().isEmpty()) {
409 QString datasheetUrl = symbolData->info().datasheet;
410 // 检查是否是相对路径，如果是则拼接完整的 URL
411 if (datasheetUrl.startsWith("/")) {
412 datasheetUrl = "https://image.lceda.cn" + datasheetUrl;
413 qDebug() << "Datasheet is relative path, constructed full URL:" << datasheetUrl;
414 }
415 componentData.setDatasheet(datasheetUrl);
416 qDebug() << "Datasheet extracted from symbolData (EasyEDA, will be overridden by LCSC if available):"
417 << datasheetUrl;
418 }
419 } else {
420 qWarning() << "Failed to import symbol data for:" << m_currentComponentId;
421 emit fetchError(lcscId, "Failed to parse Symbol data from EasyEDA JSON");
422 m_currentComponentId = savedComponentId;
423 return;
424 }
425
426 // 导入封装数据
427 QSharedPointer<FootprintData> footprintData = m_importer->importFootprintData(resultData);
428 if (footprintData) {
429 componentData.setFootprintData(footprintData);
430 qDebug() << "Footprint imported successfully - Name:" << footprintData->info().name;
431 } else {
432 qWarning() << "Failed to import footprint data for:" << m_currentComponentId;
433 emit fetchError(lcscId, "Failed to parse Footprint data from EasyEDA JSON (Library might be corrupted)");
434 m_currentComponentId = savedComponentId;
435 return;
436 }
437
438 // 检查是否需要获取3D 模型
439 // 使用本地变量 need3DModel 而不是成员变量 m_fetch3DModel
440 if (need3DModel && footprintData) {
441 // 检查封装数据中是否包含 3D 模型 UUID
442 QString modelUuid = footprintData->model3D().uuid();
443
444 // 如果封装数据中没有UUID，尝试从 head.uuid_3d 字段中提取
445 if (modelUuid.isEmpty() && resultData.contains("head")) {
446 QJsonObject head = resultData["head"].toObject();
447 if (head.contains("uuid_3d")) {
448 modelUuid = head["uuid_3d"].toString();
449 }
450 }
451
452 if (!modelUuid.isEmpty()) {
453 qDebug() << "Fetching 3D model with UUID:" << modelUuid;
454
455 // 创建 Model3DData 对象
456 QSharedPointer<Model3DData> model3DData(new Model3DData());
457 model3DData->setUuid(modelUuid);
458
459 // 如果封装数据中有 3D 模型信息，复制平移和旋转
460 if (!footprintData->model3D().uuid().isEmpty()) {
461 model3DData->setName(footprintData->model3D().name());
462 model3DData->setTranslation(footprintData->model3D().translation());
463 model3DData->setRotation(footprintData->model3D().rotation());
464 }
465
466 componentData.setModel3DData(model3DData);
467
468 // 在并行模式下，使用 m_fetchingComponents 存储待处理的组件数据
469 if (m_parallelFetching) {
470 // 更新已有的 entry
471 FetchingComponent& fetchingComponent = m_fetchingComponents[m_currentComponentId];
472 fetchingComponent.data = componentData;
473 fetchingComponent.hasComponentInfo = true;
474 fetchingComponent.hasCadData = true;
475 fetchingComponent.hasObjData = false;
476 fetchingComponent.hasStepData = false;
477 // fetch3DModel 已经在 internal 调用时设置了
478 } else {
479 // 串行模式下的处理
480 m_pendingComponentData = componentData;
481 m_pendingModelUuid = modelUuid;
482 m_hasDownloadedWrl = false;
483 }
484
485 // 获取 WRL 格式的3D 模型
486 m_api->fetch3DModelObj(modelUuid);
487 return; // 等待 3D 模型数据
488 } else {
489 qDebug() << "No 3D model UUID found for:" << m_currentComponentId;
490 }
491 }
492
493 // 调用 LCSC API 获取数据手册和预览图 URL（异步）
494 // 注意：这个调用是异步的，handleLcscDataReady 会更新 ComponentData
495 // 我们需要将数据保存到 m_fetchingComponents 中，以便 handleLcscDataReady 可以更新
496 m_fetchingComponents[m_currentComponentId].data = componentData;
497
498 // 先发送完成信号，确保 ComponentListViewModel 获取到完整的 ComponentData
499 // 图片下载是异步的，会在后台继续进行
500 emit cadDataReady(m_currentComponentId, componentData);
501
502 // 然后调用 LCSC API 获取数据手册和预览图
503 m_imageService->fetchPreviewImages(m_currentComponentId);
504 qDebug() << "Called fetchPreviewImages to get LCSC datasheet and preview images for" << m_currentComponentId;
505
506 // 如果在并行模式下，处理并行数据收集
507 if (m_parallelFetching) {
508 handleParallelDataCollected(m_currentComponentId, componentData);
509 }
510
511 // 恢复组件 ID
512 m_currentComponentId = savedComponentId;
513}

515void ComponentService::handleModel3DFetched(const QString& uuid, const QByteArray& data) {
516 qDebug() << "3D model data fetched for UUID:" << uuid << "Size:" << data.size();
517
518 // 在并行模式下，查找对应的组件
519 if (m_parallelFetching) {
520 // 在并行模式下，查找对应UUID的组件
521 for (auto it = m_fetchingComponents.begin(); it != m_fetchingComponents.end(); ++it) {
522 if (it.value().data.model3DData() && it.value().data.model3DData()->uuid() == uuid) {
523 QString componentId = it.key();
524 FetchingComponent& fetchingComponent = it.value();
525
526 if (!fetchingComponent.hasObjData) {
527 // 这是 WRL 格式的3D 模型
528 fetchingComponent.data.model3DData()->setRawObj(QString::fromUtf8(data));
529 fetchingComponent.hasObjData = true;
530 qDebug() << "WRL data saved for:" << uuid << "Size:" << data.size();
531
532 // 继续下载 STEP 格式的3D 模型
533 qDebug() << "Fetching STEP model with UUID:" << uuid;
534 m_api->fetch3DModelStep(uuid);
535 } else {
536 // 这是 STEP 格式的3D 模型
537 fetchingComponent.data.model3DData()->setStep(data);
538 fetchingComponent.hasStepData = true;
539 qDebug() << "STEP data saved for:" << uuid << "Size:" << data.size();
540
541 // 调用 LCSC API 获取数据手册和预览图 URL（异步）
542 // 注意：这个调用是异步的，handleLcscDataReady 会更新 ComponentData
543 m_fetchingComponents[componentId].data = fetchingComponent.data;
544
545 // 先发送完成信号，确保 ComponentListViewModel 获取到完整的 ComponentData
546 // 图片下载是异步的，会在后台继续进行
547 emit cadDataReady(componentId, fetchingComponent.data);
548
549 // 处理并行数据收集
550 handleParallelDataCollected(componentId, fetchingComponent.data);
551
552 // 然后调用 LCSC API 获取数据手册和预览图
553 m_imageService->fetchPreviewImages(componentId);
554 qDebug() << "Called fetchPreviewImages after 3D model loaded for" << componentId;
555
556 // 从待处理列表中移除
557 m_fetchingComponents.remove(componentId);
558 }
559 return;
560 }
561 }
562 qWarning() << "Received 3D model data for unexpected UUID:" << uuid;
563 } else {
564 // 串行模式下的处理
565 if (m_pendingComponentData.model3DData() && m_pendingComponentData.model3DData()->uuid() == uuid) {
566 if (!m_hasDownloadedWrl) {
567 // 这是 WRL 格式的3D 模型
568 m_pendingComponentData.model3DData()->setRawObj(QString::fromUtf8(data));
569 qDebug() << "WRL data saved for:" << uuid << "Size:" << data.size();
570
571 // 标记已经下载了WRL 格式
572 m_hasDownloadedWrl = true;
573
574 // 继续下载 STEP 格式的3D 模型
575 qDebug() << "Fetching STEP model with UUID:" << uuid;
576 m_api->fetch3DModelStep(uuid);
577 } else {
578 // 这是 STEP 格式的3D 模型
579 m_pendingComponentData.model3DData()->setStep(data);
580 qDebug() << "STEP data saved for:" << uuid << "Size:" << data.size();
581
582 // 调用 LCSC API 获取数据手册和预览图 URL（异步）
583 m_fetchingComponents[m_currentComponentId].data = m_pendingComponentData;
584 m_imageService->fetchPreviewImages(m_currentComponentId);
585 qDebug() << "Called fetchPreviewImages after 3D model loaded (serial mode) for" << m_currentComponentId;
586
587 // 发送完成信号
588 emit cadDataReady(m_currentComponentId, m_pendingComponentData);
589
590 // 清空待处理数据
591 m_pendingComponentData = ComponentData();
592 m_pendingModelUuid.clear();
593 m_hasDownloadedWrl = false;
594 }
595 } else {
596 qWarning() << "Received 3D model data for unexpected UUID:" << uuid;
597 }
598 }

614 QString componentId = idOrUuid;
615
616 // 如果在并行模式下，尝试解析 UUID 为组件 ID
617 if (m_parallelFetching && !m_parallelFetchingStatus.contains(idOrUuid)) {
618 for (auto it = m_fetchingComponents.begin(); it != m_fetchingComponents.end(); ++it) {
619 if (it.value().data.model3DData() && it.value().data.model3DData()->uuid() == idOrUuid) {
620 componentId = it.key();
621 qDebug() << "Resolved UUID" << idOrUuid << "to component ID" << componentId;
622 break;
623 }
624 }
625 }
626
627 qDebug() << "Fetch error for component:" << componentId << "Error:" << error;
628
629 // 如果在并行模式下，处理并行错误
630 if (m_parallelFetching) {
631 handleParallelFetchError(componentId, error);
632 }

709QStringList ComponentService::extractComponentIdFromText(const QString& text) const {
710 // 依然保留简单的提取逻辑，或者可以进一步整合进 BomParser
711 QStringList extractedIds;
712 QRegularExpression re("[Cc]\\d{4,}");
713 QRegularExpressionMatchIterator it = re.globalMatch(text);
714
715 while (it.hasNext()) {
716 QRegularExpressionMatch match = it.next();
717 QString id = match.captured().toUpper();
718 if (!BomParser::getExcludedIds().contains(id) && !extractedIds.contains(id)) {
719 extractedIds.append(id);
720 }
721 }
722 return extractedIds;
723}
