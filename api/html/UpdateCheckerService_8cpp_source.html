109QString UpdateCheckerService::normalizeVersion(const QString& version) {
110 QString normalized = version.trimmed();
111 normalized.remove(QRegularExpression(QStringLiteral("^[^0-9]*")));
112 const QRegularExpression semverExpr(QStringLiteral("(\\d+)(?:\\.(\\d+))?(?:\\.(\\d+))?"));
113 const QRegularExpressionMatch match = semverExpr.match(normalized);
114 if (!match.hasMatch()) {
115 return QString();
116 }
117
118 const QString major = match.captured(1);
119 const QString minor = match.captured(2).isEmpty() ? QStringLiteral("0") : match.captured(2);
120 const QString patch = match.captured(3).isEmpty() ? QStringLiteral("0") : match.captured(3);
121 return QStringLiteral("%1.%2.%3").arg(major, minor, patch);
122}

124bool UpdateCheckerService::isRemoteVersionNewer(const QString& currentVersion, const QString& latestVersion) {
125 const QStringList currentParts = splitVersion(currentVersion);
126 const QStringList latestParts = splitVersion(latestVersion);
127 const int maxParts = qMax(currentParts.size(), latestParts.size());
128
129 for (int i = 0; i < maxParts; ++i) {
130 const int currentPart = i < currentParts.size() ? currentParts.at(i).toInt() : 0;
131 const int latestPart = i < latestParts.size() ? latestParts.at(i).toInt() : 0;
132 if (latestPart > currentPart) {
133 return true;
134 }
135 if (latestPart < currentPart) {
136 return false;
137 }
138 }
139
140 return false;
141}
