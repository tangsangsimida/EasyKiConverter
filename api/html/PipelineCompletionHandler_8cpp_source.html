72bool PipelineCompletionHandler::exportPreviewImagesFromMemory(const QList<QByteArray>& imageDataList,
73 const QString& outputPath,
74 const QString& componentName) {
75 QDir outputDir(outputPath);
76 QString imagesDir = outputDir.filePath("images");
77 if (!outputDir.exists(imagesDir)) {
78 if (!outputDir.mkpath(imagesDir)) {
79 qWarning() << "Failed to create images directory:" << imagesDir;
80 return false;
81 }
82 }
83
84 QString safeName = componentName;
85 safeName.replace(INVALID_FILENAME_CHARS, "_");
86
87 bool allSuccess = true;
88 int exportedCount = 0;
89
90 for (int i = 0; i < imageDataList.size(); ++i) {
91 if (imageDataList[i].isEmpty()) {
92 continue;
93 }
94
95 QString filename = QString("%1_%2.jpg").arg(safeName).arg(i);
96 QString filePath = QDir(imagesDir).filePath(filename);
97
98 QFile file(filePath);
99 if (file.open(QIODevice::WriteOnly)) {
100 file.write(imageDataList[i]);
101 file.close();
102 exportedCount++;
103 } else {
104 allSuccess = false;
105 }
106 }
107
108 return allSuccess;
109}

144void PipelineCompletionHandler::exportPreviewImagesFromViewModel(ComponentListViewModel* componentListViewModel,
145 ExportProgressViewModel* progressViewModel,
146 const ExportOptions& options) {
147 if (!options.exportPreviewImages || !componentListViewModel) {
148 return;
149 }
150
151 // 从 ViewModel 获取最新的数据
152 QMap<QString, QSharedPointer<ComponentData>> allData = componentListViewModel->getAllPreloadedData();
153 if (allData.isEmpty()) {
154 qDebug() << "No preloaded data available from ViewModel for preview images export";
155 emit previewImagesExported(0);
156 return;
157 }
158
159 int successCount = 0;
160 for (auto it = allData.constBegin(); it != allData.constEnd(); ++it) {
161 const QSharedPointer<ComponentData>& componentData = it.value();
162 QString componentId = componentData->lcscId();
163
164 QList<QByteArray> previewImageDataList = componentData->previewImageData();
165
166 // 1. 如果内存中有数据，直接使用
167 // 2. 如果内存中没有但有 URL，尝试从磁盘缓存加载
168 if (previewImageDataList.isEmpty() && !componentData->previewImages().isEmpty()) {
169 const QStringList& imageUrls = componentData->previewImages();
170 for (int i = 0; i < imageUrls.size() && i < 3; ++i) {
171 QByteArray imageData = ComponentCacheService::instance()->loadPreviewImage(componentId, i);
172 if (!imageData.isEmpty()) {
173 previewImageDataList.append(imageData);
174 qDebug() << "Preview image loaded from disk cache for component:" << componentId << "index:" << i;
175 } else {
176 // 3. 磁盘缓存也没有，从 URL 下载并缓存
177 qDebug() << "Preview image not in cache, downloading for component:" << componentId
178 << "index:" << i;
179 imageData = ComponentCacheService::instance()->downloadPreviewImage(
180 componentId, imageUrls[i], i, nullptr, nullptr, options.weakNetworkSupport);
181 if (!imageData.isEmpty()) {
182 previewImageDataList.append(imageData);
183 }
184 }
185 }
186 }
187
188 bool exportSuccess = false;
189 if (!previewImageDataList.isEmpty()) {
190 QString componentName = componentData->name().isEmpty() ? componentId : componentData->name();
191
192 if (exportPreviewImagesFromMemory(previewImageDataList, options.outputPath, componentName)) {
193 successCount++;
194 exportSuccess = true;
195 } else {
196 qWarning() << "Failed to export preview images for component:" << componentId;
197 }
198 } else {
199 // 没有预览图数据但不需要导出时，也标记为成功（没有预览图不代表失败）
200 if (!options.exportPreviewImages || componentData->previewImages().isEmpty()) {
201 exportSuccess = true;
202 }
203 }
204
205 // 更新预览图导出状态
206 componentListViewModel->updateExportStatus(componentId, exportSuccess ? 1 : 0, -1);
207 if (progressViewModel) {
208 progressViewModel->updateComponentExportStatus(componentId, exportSuccess ? 1 : 0, -1);
209 }
210 }
211 qDebug() << "Preview images export from ViewModel completed:" << successCount << "components";
212 emit previewImagesExported(successCount);
213}
