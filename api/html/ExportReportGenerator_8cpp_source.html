23 // 只在调试模式下生成详细报告
24 if (!options.debugMode) {
25 return;

27
28 if (options.outputPath.trimmed().isEmpty()) {
29 return;

31
32 QDir outputDir(options.outputPath);
33 if (!outputDir.exists() && !outputDir.mkpath(QStringLiteral("."))) {
34 qWarning() << "ParallelExportService: Failed to create diagnostics report directory:" << outputDir.path();

37
38 const QString reportPath = outputDir.filePath(QStringLiteral("easykiconverter_export_detailed_report.md"));
39 QSaveFile file(reportPath);

41 qWarning() << "ParallelExportService: Failed to open export detailed report for writing:" << reportPath;

44
45 QTextStream out(&file);
46 out.setEncoding(QStringConverter::Utf8);
47 out << "# EasyKiConverter Export Detailed Report\n\n";
48 out << "- Generated: " << QDateTime::currentDateTime().toString(Qt::ISODate) << "\n";

56

60 out << "- Export footprint: " << (options.exportFootprint ? "yes" : "no") << "\n";

68 out << "## Preload\n\n";
69 out << "- Completed: " << progress.preloadProgress.completedCount << "/" << progress.preloadProgress.totalCount
70 << "\n";
71 out << "- Success: " << progress.preloadProgress.successCount << "\n";
72 out << "- Failed: " << progress.preloadProgress.failedCount << "\n";
73 out << "- In progress: " << progress.preloadProgress.inProgressCount << "\n";
74 if (!progress.preloadProgress.failedComponents.isEmpty()) {
75 out << "\n### Preload Failures\n\n";
76 for (auto it = progress.preloadProgress.failedComponents.cbegin();
77 it != progress.preloadProgress.failedComponents.cend();
78 ++it) {
79 out << "- `" << it.key() << "`: " << it.value() << "\n";
80 }
81 }
82
83 out << "\n## Export Type Progress\n\n";
84 for (auto it = progress.exportTypeProgress.cbegin(); it != progress.exportTypeProgress.cend(); ++it) {
85 const ExportTypeProgress& typeProgress = it.value();
86 out << "### " << it.key() << "\n\n";
87 out << "- Completed: " << typeProgress.completedCount << "/" << typeProgress.totalCount << "\n";
88 out << "- Success: " << typeProgress.successCount << "\n";
89 out << "- Failed: " << typeProgress.failedCount << "\n";
90 out << "- Skipped: " << typeProgress.skippedCount << "\n";
91 out << "- In progress: " << typeProgress.inProgressCount << "\n";
92 }
93
94 out << "\n## Weak Network Diagnostics\n\n";
95 out << "### Network Runtime Stats\n\n```\n";
96 out << NetworkClient::instance().formatRuntimeStats() << "\n";
97 out << "```\n";
98
99 if (!file.commit()) {
100 qWarning() << "ParallelExportService: Failed to commit export detailed report:" << reportPath;
101 return;
102 }
103
104 qInfo() << "ParallelExportService: Export detailed report written to" << reportPath;
105}
