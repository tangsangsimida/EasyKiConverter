35void Model3DExportWorker::run() {
36 if (m_cancelled.load()) {
37 emit completed(m_componentId, false, QStringLiteral("Cancelled"));
38 return;
39 }
40 const uint64_t gen = ComponentCacheService::instance()->currentGeneration();
41
42 if (!m_data) {
43 emit completed(m_componentId, false, QStringLiteral("No data available"));
44 return;
45 }
46
47 qDebug() << "Model3DExportWorker: Exporting" << m_componentId;
48
49 QString uuid;
50 if (m_data->model3DData()) {
51 uuid = m_data->model3DData()->uuid();
52 }
53 if (uuid.isEmpty()) {
54 QSharedPointer<ComponentData> cachedComponent =
55 ComponentCacheService::instance()->loadComponentData(m_componentId);
56 if (cachedComponent && cachedComponent->model3DData()) {
57 uuid = cachedComponent->model3DData()->uuid();
58 }
59 }
60 if (uuid.isEmpty()) {
61 qDebug() << "Model3DExportWorker: 3D model UUID not available, skipping" << m_componentId;
62 emit completed(m_componentId, true, QStringLiteral("3D model UUID not available, skipped"));
63 return;
64 }
65
66 // 根据选项决定是否导出 WRL 和 STEP
67 const bool needWrl = m_options.needsModel3DWrl();
68 const bool needStep = m_options.needsModel3DStep();
69
70 qInfo() << "Model3DExportWorker::run() for" << m_componentId
71 << "exportModel3DFormat:" << m_options.exportModel3DFormat << "needWrl:" << needWrl
72 << "needStep:" << needStep;
73
74 // 如果两种格式都不需要导出，直接完成
75 if (!needWrl && !needStep) {
76 qDebug() << "Model3DExportWorker: No format selected, skipping" << m_componentId;
77 emit completed(m_componentId, true, QString());
78 return;
79 }
80
81 Exporter3DModel exporter;
82
83 // 获取模型名称用于命名文件（仅在需要时）
84 QString modelName;
85 if (m_data && m_data->model3DData()) {
86 modelName = m_data->model3DData()->name();
87 }
88 if (modelName.isEmpty()) {
89 modelName = m_componentId;
90 } else {
91 // 使用 PathSecurity 清理文件名中的非法字符
92 modelName = PathSecurity::sanitizeFilename(modelName);
93 }
94
95 // 构建输出路径
96 QString outputDir = m_options.outputPath;
97 if (outputDir.isEmpty()) {
98 outputDir =
99 QDir(QDir(QDir::currentPath()).filePath(QStringLiteral("export"))).filePath(QStringLiteral("3dmodels"));
100 }
101
102 QDir dir;
103 if (!dir.mkpath(outputDir)) {
104 emit completed(m_componentId, false, QStringLiteral("Failed to create output directory"));
105 return;
106 }
107
108 const QDir outputDirectory(outputDir);
109 const QString wrlFinalPath = needWrl ? outputDirectory.filePath(modelName + QStringLiteral(".wrl")) : QString();
110 const QString stepFinalPath = needStep ? outputDirectory.filePath(modelName + QStringLiteral(".step")) : QString();
111 const QString wrlWritePath = m_outputPaths.wrlTempPath.isEmpty() ? wrlFinalPath : m_outputPaths.wrlTempPath;
112 const QString stepWritePath = m_outputPaths.stepTempPath.isEmpty() ? stepFinalPath : m_outputPaths.stepTempPath;
113
114 const auto ensureParentDir = [](const QString& filePath) {
115 if (filePath.isEmpty())
116 return true;
117 QFileInfo fileInfo(filePath);
118 return fileInfo.absoluteDir().exists() || QDir().mkpath(fileInfo.absolutePath());
119 };
120
121 if ((needWrl && !ensureParentDir(wrlWritePath)) || (needStep && !ensureParentDir(stepWritePath))) {
122 emit completed(m_componentId, false, QStringLiteral("Failed to create 3D model output directory"));
123 return;
124 }
125
126 if (!m_options.overwriteExistingFiles && m_outputPaths.wrlTempPath.isEmpty() &&
127 m_outputPaths.stepTempPath.isEmpty()) {
128 bool bothExist = true;
129 if (needWrl && !QFile::exists(wrlFinalPath))
130 bothExist = false;
131 if (needStep && !QFile::exists(stepFinalPath))
132 bothExist = false;
133 if (bothExist && (needWrl || needStep)) {
134 qDebug() << "Model3DExportWorker: Files already exist, skipping" << wrlFinalPath << stepFinalPath;
135 emit completed(m_componentId, true, QString());
136 return;
137 }
138 }
139
140 ComponentCacheService* cache = ComponentCacheService::instance();
141
142 // WRL 是由 OBJ 转换得到的派生文件；有 OBJ 时重新生成，避免旧 WRL 缓存携带过期归一化。
143 // 只有在没有 OBJ 且不能下载时才复用 WRL 缓存，封装阶段会按该 WRL 的实际几何计算 STEP 偏移。
144 // STEP 文件保持服务器原始坐标系，不使用派生缓存。
145
146 auto buildModelData = [this, &uuid]() {
147 Model3DData modelData;
148 modelData.setUuid(uuid);
149 if (m_data && m_data->model3DData()) {
150 modelData.setName(m_data->model3DData()->name());
151 modelData.setTranslation(m_data->model3DData()->translation());
152 modelData.setRotation(m_data->model3DData()->rotation());
153 }
154 return modelData;
155 };
156
157 QString error;
158 // 只在需要 WRL 时导出
159 if (needWrl) {
160 QByteArray objData;
161 if (m_data && !m_data->model3DObjRaw().isEmpty()) {
162 objData = m_data->model3DObjRaw();
163 }
164 if (objData.isEmpty()) {
165 objData = cache->loadModel3D(uuid, QStringLiteral("obj"));
166 }
167 if (objData.isEmpty() && cache->hasModel3DCached(uuid, QStringLiteral("wrl")) &&
168 cache->copyModel3DToFile(uuid, QStringLiteral("wrl"), wrlWritePath)) {
169 qDebug() << "Model3DExportWorker: WRL cache fallback for" << uuid;
170 } else if (objData.isEmpty() && !exporter.downloadObjDataSync(uuid, &objData, &error)) {
171 if (error.isEmpty()) {
172 error = QStringLiteral("Failed to download OBJ data for WRL export");
173 }
174 } else if (objData.isEmpty()) {
175 error = QStringLiteral("OBJ data is empty for WRL export");
176 } else {
177 cache->saveModel3D(uuid, objData, QStringLiteral("obj"), gen);
178 Model3DData modelData = buildModelData();
179 modelData.setRawObj(QString::fromUtf8(objData));
180 if (!exporter.exportToWrl(modelData, wrlWritePath)) {
181 error = QStringLiteral("Failed to convert OBJ to WRL");
182 } else {
183 QFile file(wrlWritePath);
184 if (file.open(QIODevice::ReadOnly)) {
185 cache->saveModel3D(uuid, file.readAll(), QStringLiteral("wrl"), gen);
186 qDebug() << "Model3DExportWorker: Saved WRL to cache for" << m_componentId << "uuid" << uuid;
187 }
188 }
189 }
190 }
191 // STEP 文件保持服务器原始坐标系，优先使用磁盘缓存
192 if (error.isEmpty() && needStep && !m_cancelled.load()) {
193 QByteArray stepData;
194
195 if (cache->copyModel3DToFile(uuid, QStringLiteral("step"), stepWritePath)) {
196 qDebug() << "Model3DExportWorker: STEP cache hit for" << m_componentId << "uuid" << uuid;
197 } else if (!m_cancelled.load()) {
198 if (!exporter.downloadStepDataSync(uuid, &stepData, &error)) {
199 if (error.isEmpty()) {
200 error = QStringLiteral("Failed to download STEP data");
201 }
202 }
203
204 if (!stepData.isEmpty()) {
205 Model3DData modelData = buildModelData();
206 modelData.setStep(stepData);
207 if (!exporter.exportToStep(modelData, stepWritePath)) {
208 error = QStringLiteral("Failed to write STEP file");
209 } else {
210 cache->saveModel3D(uuid, stepData, QStringLiteral("step"), gen);
211 qDebug() << "Model3DExportWorker: Saved STEP to cache for" << m_componentId << "uuid" << uuid;
212 }
213 }
214 }
215 }
216
217 if (m_cancelled.load() && error.isEmpty()) {
218 error = QStringLiteral("Cancelled");
219 }
220
221 if (error.isEmpty()) {
222 qDebug() << "Model3DExportWorker: Successfully exported" << (needWrl ? "WRL" : "") << (needStep ? "STEP" : "")
223 << "for" << m_componentId;
224
225 if (m_options.debugMode) {
226 (void)QtConcurrent::run([componentId = m_componentId, data = m_data, outputPath = m_options.outputPath]() {
227 DebugExportHelper::exportDebugData(componentId, data, outputPath);
228 });
229 }
230
231 emit completed(m_componentId, true, QString());
232 return;
233 }
234
235 qCritical() << "Model3DExportWorker: Export failed for" << m_componentId << "uuid" << uuid << "error:" << error;
236 emit completed(m_componentId, false, error);
237}
