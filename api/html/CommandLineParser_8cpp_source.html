103bool CommandLineParser::parse() {
104 // 重要：使用存储的 argv 而非 QCoreApplication::arguments()
105 // 因为此函数可能在 QCoreApplication 创建前被调用
106 QStringList args;
107 if (m_argv) {
108 args.reserve(m_argc);
109 for (int i = 0; i < m_argc; ++i) {
110 args.append(QString::fromLocal8Bit(m_argv[i]));
111 }
112 }
113
114 bool result = m_parser.parse(args);
115
116 if (result) {
117 // 检测 CLI 子命令
118 // 使用 args 而非 QCoreApplication::arguments() 保证可靠性
119
120 // 检查是否包含 "convert" 命令
121 for (int i = 1; i < args.size(); ++i) {
122 QString arg = args[i].toLower();
123 if (arg == "convert") {
124 m_hasConvertCommand = true;
125 // 检查下一个参数
126 if (i + 1 < args.size()) {
127 QString subcommand = args[i + 1].toLower();
128 if (subcommand == "bom") {
129 m_hasBomSubcommand = true;
130 m_cliMode = CliMode::ConvertBom;
131 } else if (subcommand == "component") {
132 m_hasComponentSubcommand = true;
133 m_cliMode = CliMode::ConvertComponent;
134 } else if (subcommand == "batch") {
135 m_hasBatchSubcommand = true;
136 m_cliMode = CliMode::ConvertBatch;
137 }
138 }
139 break;
140 }
141 }
142 }
143
144 return result;
145}

219bool CommandLineParser::validate() const {
220 // 验证日志级别
221 if (m_parser.isSet(m_logLevelOption)) {
222 QString level = logLevel();
223 QStringList validLevels = {"trace", "debug", "info", "warn", "error", "fatal"};
224 if (!validLevels.contains(level)) {
225 return false;
226 }
227 }
228
229 // 验证语言设置
230 if (m_parser.isSet(m_languageOption)) {
231 QString lang = language();
232 QStringList validLangs = {"zh_CN", "en"};
233 if (!validLangs.contains(lang)) {
234 return false;
235 }
236 }
237
238 // 验证主题设置
239 if (m_parser.isSet(m_themeOption)) {
240 QString theme = this->theme();
241 QStringList validThemes = {"dark", "light"};
242 if (!validThemes.contains(theme)) {
243 return false;
244 }
245 }
246
247 // CLI 模式验证
248 if (m_parser.isSet(m_cacheDirOption) && cacheDir().trimmed().isEmpty()) {
249 return false;
250 }
251
252 if (m_parser.isSet(m_diskCacheLimitOption)) {
253 bool ok = false;
254 const int sizeMB = m_parser.value(m_diskCacheLimitOption).toInt(&ok);
255 if (!ok || sizeMB <= 0) {
256 return false;
257 }
258 }
259
260 // CLI 模式验证
261 if (isCliMode()) {
262 if (m_parser.isSet(m_3dModelFormatOption)) {
263 if (!VALID_3D_MODEL_FORMATS.contains(model3DFormat())) {
264 return false;
265 }
266 }
267
268 if (m_parser.isSet(m_3dPathModeOption)) {
269 if (!VALID_3D_PATH_MODES.contains(model3DPathMode())) {
270 return false;
271 }
272 }
273
274 // 验证输出目录
275 if (!m_parser.isSet(m_outputOption)) {
276 return false;
277 }
278
279 // BOM 转换需要输入文件
280 if (m_cliMode == CliMode::ConvertBom && !m_parser.isSet(m_inputOption)) {
281 return false;
282 }
283
284 // 单个元器件转换需要组件编号
285 if (m_cliMode == CliMode::ConvertComponent && !m_parser.isSet(m_componentOption)) {
286 return false;
287 }
288
289 // 批量转换需要输入文件
290 if (m_cliMode == CliMode::ConvertBatch && !m_parser.isSet(m_inputOption)) {
291 return false;
292 }
293 }
294
295 return true;
296}

298QString CommandLineParser::validationError() const {
299 QStringList errors;
300
301 // 检查日志级别
302 if (m_parser.isSet(m_logLevelOption)) {
303 QString level = logLevel();
304 QStringList validLevels = {"trace", "debug", "info", "warn", "error", "fatal"};
305 if (!validLevels.contains(level)) {
306 errors.append(QCoreApplication::translate("CommandLineParser", "无效的日志级别: %1（有效值: %2）")
307 .arg(level)
308 .arg(validLevels.join(", ")));
309 }
310 }
311
312 // 检查语言设置
313 if (m_parser.isSet(m_languageOption)) {
314 QString lang = language();
315 QStringList validLangs = {"zh_CN", "en"};
316 if (!validLangs.contains(lang)) {
317 errors.append(QCoreApplication::translate("CommandLineParser", "无效的语言设置: %1（有效值: %2）")
318 .arg(lang)
319 .arg(validLangs.join(", ")));
320 }
321 }
322
323 // 检查主题设置
324 if (m_parser.isSet(m_themeOption)) {
325 QString theme = this->theme();
326 QStringList validThemes = {"dark", "light"};
327 if (!validThemes.contains(theme)) {
328 errors.append(QCoreApplication::translate("CommandLineParser", "无效的主题设置: %1（有效值: %2）")
329 .arg(theme)
330 .arg(validThemes.join(", ")));
331 }
332 }
333
334 // CLI 模式验证错误
335 if (m_parser.isSet(m_cacheDirOption) && cacheDir().trimmed().isEmpty()) {
336 errors.append(QCoreApplication::translate("CommandLineParser", "缓存目录不能为空"));
337 }
338
339 if (m_parser.isSet(m_diskCacheLimitOption)) {
340 bool ok = false;
341 const int sizeMB = m_parser.value(m_diskCacheLimitOption).toInt(&ok);
342 if (!ok || sizeMB <= 0) {
343 errors.append(
344 QCoreApplication::translate("CommandLineParser", "磁盘缓存大小必须是大于 0 的整数（单位: MB）"));
345 }
346 }
347
348 // CLI 模式验证错误
349 if (isCliMode()) {
350 if (m_parser.isSet(m_3dModelFormatOption)) {
351 const QString format = model3DFormat();
352 if (!VALID_3D_MODEL_FORMATS.contains(format)) {
353 errors.append(QCoreApplication::translate("CommandLineParser", "无效的 3D 模型格式: %1（有效值: %2）")
354 .arg(format)
355 .arg(VALID_3D_MODEL_FORMATS.join(", ")));
356 }
357 }
358
359 if (m_parser.isSet(m_3dPathModeOption)) {
360 const QString pathMode = model3DPathMode();
361 if (!VALID_3D_PATH_MODES.contains(pathMode)) {
362 errors.append(
363 QCoreApplication::translate("CommandLineParser", "无效的 3D 模型路径模式: %1（有效值: %2）")
364 .arg(pathMode, VALID_3D_PATH_MODES.join(", ")));
365 }
366 }
367
368 if (!m_parser.isSet(m_outputOption)) {
369 errors.append(QCoreApplication::translate("CommandLineParser", "CLI 模式必须指定输出目录 (-o/--output)"));
370 }
371
372 if (m_cliMode == CliMode::ConvertBom && !m_parser.isSet(m_inputOption)) {
373 errors.append(QCoreApplication::translate("CommandLineParser", "BOM 表转换必须指定输入文件 (-i/--input)"));
374 }
375
376 if (m_cliMode == CliMode::ConvertComponent && !m_parser.isSet(m_componentOption)) {
377 errors.append(
378 QCoreApplication::translate("CommandLineParser", "单个元器件转换必须指定 LCSC 编号 (-c/--component)"));
379 }
380
381 if (m_cliMode == CliMode::ConvertBatch && !m_parser.isSet(m_inputOption)) {
382 errors.append(QCoreApplication::translate("CommandLineParser", "批量转换必须指定输入文件 (-i/--input)"));
383 }
384 }
385
386 return errors.join("\n");
387}
