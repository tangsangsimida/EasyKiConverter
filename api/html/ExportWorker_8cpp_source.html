39void ExportWorker::run() {
40 qDebug() << "ExportWorker started for:" << m_componentId;
41
42 try {
43 // 创建输出目录
44 if (!createOutputDirectories()) {
45 emit exportFinished(m_componentId, false, "Failed to create output directories");
46 return;
47 }
48
49 int progress = 0;
50 int totalSteps = (m_exportSymbol ? 1 : 0) + (m_exportFootprint ? 1 : 0) + (m_exportModel3D ? 1 : 0);
51 int currentStep = 0;
52
53 // 导出符号
54 if (m_exportSymbol) {
55 qDebug() << "Exporting symbol for:" << m_componentId;
56 if (!exportSymbolLibrary()) {
57 emit exportFinished(m_componentId, false, "Failed to export symbol library");
58 return;
59 }
60 currentStep++;
61 progress = static_cast<int>((static_cast<double>(currentStep) / totalSteps) * 100);
62 emit exportProgress(m_componentId, progress);
63 }
64
65 // 准备3D模型路径
66 QString model3DPath;
67 if (m_exportModel3D && m_footprintData && !m_footprintData->model3D().uuid().isEmpty()) {
68 Model3DData model3D = m_footprintData->model3D();
69
70 // 清理模型名称，移除文件系统不支持的字符
71 QString sanitizedName = model3D.name();
72 sanitizedName.replace(QRegularExpression("[<>:\"/\\\\|?*]"), "_");
73
74 // 计算3D模型导出路径
75 model3DPath = QString("%1/%2.3dshapes/%3.wrl").arg(m_outputPath, m_libName, sanitizedName);
76 }
77
78 // 导出封装
79 if (m_exportFootprint) {
80 qDebug() << "Exporting footprint for:" << m_componentId;
81 if (!exportFootprintLibrary(model3DPath)) {
82 emit exportFinished(m_componentId, false, "Failed to export footprint library");
83 return;
84 }
85 currentStep++;
86 progress = static_cast<int>((static_cast<double>(currentStep) / totalSteps) * 100);
87 emit exportProgress(m_componentId, progress);
88 }
89
90 // 导出3D模型
91 if (m_exportModel3D && m_footprintData && !m_footprintData->model3D().uuid().isEmpty()) {
92 qDebug() << "Exporting 3D model for:" << m_componentId;
93 if (!export3DModel(m_footprintData->model3D(), model3DPath)) {
94 emit exportFinished(m_componentId, false, "Failed to export 3D model");
95 return;
96 }
97 currentStep++;
98 progress = static_cast<int>((static_cast<double>(currentStep) / totalSteps) * 100);
99 emit exportProgress(m_componentId, progress);
100 }
101
102 // 导出成功
103 emit exportFinished(m_componentId, true, "Export successful");
104 } catch (const std::exception& e) {
105 qWarning() << "Exception in ExportWorker for" << m_componentId << ":" << e.what();
106 emit exportFinished(m_componentId, false, QString("Exception: %1").arg(e.what()));
107 } catch (...) {
108 qWarning() << "Unknown exception in ExportWorker for:" << m_componentId;
109 emit exportFinished(m_componentId, false, "Unknown exception occurred");
110 }
