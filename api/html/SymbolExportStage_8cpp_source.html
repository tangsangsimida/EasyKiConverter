27void SymbolExportStage::start(const QStringList& componentIds,
28 const QMap<QString, QSharedPointer<ComponentData>>& cachedData) {
29 if (m_isExporting.load()) {
30 qWarning() << "SymbolExportStage: Export already in progress";
31 return;
32 }
33
34 if (componentIds.isEmpty()) {
35 qWarning() << "SymbolExportStage: No components to export";
36 emit completed(0, 0, 0);
37 return;
38 }
39
40 m_componentIds = componentIds;
41 m_cachedData = cachedData;
42 m_cancelled.store(false);
43 m_isRunning.store(true);
44
45 {
46 QMutexLocker locker(&m_progressMutex);
47 m_progress = ExportTypeProgress();
48 m_progress.typeName = QStringLiteral("Symbol");
49 m_progress.totalCount = componentIds.size();
50 m_progress.completedCount = 0;
51 m_progress.successCount = 0;
52 m_progress.failedCount = 0;
53 m_progress.skippedCount = 0;
54 m_progress.inProgressCount = componentIds.size();
55 for (const QString& componentId : componentIds) {
56 ExportItemStatus itemStatus;
57 itemStatus.status = ExportItemStatus::Status::InProgress;
58 itemStatus.startTime = QDateTime::currentDateTime();
59 m_progress.itemStatus[componentId] = itemStatus;
60 }
61 }
62
63 m_tempManager.setOutputPath(m_options.outputPath);
64
65 m_isExporting.store(true);
66 m_workerThread = QThread::create([this, componentIds, cachedData]() { doLibraryExport(componentIds, cachedData); });
67 connect(m_workerThread, &QThread::finished, this, [this]() { m_workerThread = nullptr; });
68 m_workerThread->start();
69}

94void SymbolExportStage::doLibraryExport(const QStringList& componentIds,
95 const QMap<QString, QSharedPointer<ComponentData>>& cachedData) {
96 qDebug() << "SymbolExportStage: Starting library export in worker thread for" << componentIds.size()
97 << "components";
98
99 QList<SymbolData> symbolList;
100 QStringList collectedIds;
101 QStringList failedIds;
102 int successCount = 0;
103 int skippedCount = 0;
104
105 for (const QString& componentId : componentIds) {
106 if (m_cancelled.load()) {
107 qDebug() << "SymbolExportStage: Export cancelled during data collection";
108 break;
109 }
110
111 auto it = cachedData.find(componentId);
112 if (it == cachedData.end() || !it.value()) {
113 qWarning() << "SymbolExportStage: No data for component:" << componentId;
114 failedIds.append(componentId);
115 ExportItemStatus status;
116 status.status = ExportItemStatus::Status::Failed;
117 status.errorMessage = "No component data";
118 emit itemStatusChanged(componentId, status);
119 continue;
120 }
121
122 QSharedPointer<ComponentData> data = it.value();
123
124 if (!data->symbolData()) {
125 qWarning() << "SymbolExportStage: No symbol data for component:" << componentId;
126 failedIds.append(componentId);
127 ExportItemStatus status;
128 status.status = ExportItemStatus::Status::Failed;
129 status.errorMessage = "No symbol data";
130 emit itemStatusChanged(componentId, status);
131 continue;
132 }
133
134 symbolList.append(*data->symbolData());
135 collectedIds.append(componentId);
136 successCount++;
137
138 ExportItemStatus status;
139 status.status = ExportItemStatus::Status::Success;
140 emit itemStatusChanged(componentId, status);
141
142 if (m_options.debugMode) {
143 QThreadPool::globalInstance()->start([componentId, data, outputPath = m_options.outputPath]() {
144 DebugExportHelper::exportDebugData(componentId, data, outputPath);
145 });
146 }
147
148 qDebug() << "SymbolExportStage: Collected symbol for" << componentId;
149 }
150
151 if (m_cancelled.load()) {
152 m_isExporting.store(false);
153 m_isRunning.store(false);
154 emit completed(0, componentIds.size(), 0);
155 return;
156 }
157
158 if (symbolList.isEmpty()) {
159 qWarning() << "SymbolExportStage: No valid symbols to export";
160 m_isExporting.store(false);
161 m_isRunning.store(false);
162 emit completed(0, componentIds.size(), 0);
163 return;
164 }
165
166 const auto failCollectedSymbols = [this, &collectedIds, &failedIds, &successCount](const QString& errorMessage) {
167 for (const QString& componentId : collectedIds) {
168 failedIds.append(componentId);
169 ExportItemStatus status;
170 status.status = ExportItemStatus::Status::Failed;
171 status.errorMessage = errorMessage;
172 status.endTime = QDateTime::currentDateTime();
173 emit itemStatusChanged(componentId, status);
174 }
175 successCount = 0;
176 };
177
178 const auto abortExport = [&](const QString& errorMessage) {
179 qCritical() << "SymbolExportStage:" << errorMessage;
180 failCollectedSymbols(errorMessage);
181 m_tempManager.rollbackAll();
182 m_isExporting.store(false);
183 m_isRunning.store(false);
184 emit completed(0, failedIds.size(), 0);
185 };
186
187 QString libName = m_options.libName.isEmpty() ? QStringLiteral("EasyKiConverter") : m_options.libName;
188 QString fileName = libName + QStringLiteral(".kicad_sym");
189 QString outputDir = m_options.outputPath;
190 if (outputDir.isEmpty()) {
191 outputDir = QDir::currentPath() + QStringLiteral("/export");
192 }
193 QString finalPath = outputDir + QDir::separator() + fileName;
194
195 QDir dir;
196 if (!dir.mkpath(outputDir)) {
197 abortExport(QStringLiteral("Failed to create output directory: %1").arg(outputDir));
198 return;
199 }
200
201 const bool finalFileExists = QFile::exists(finalPath);
202
203 QString tempPath = m_tempManager.createSymbolTempPath(libName, QStringLiteral(".kicad_sym"));
204 if (tempPath.isEmpty()) {
205 abortExport(QStringLiteral("Failed to create temp file path"));
206 return;
207 }
208
209 // 追加/更新到现有符号库时，先把最终文件复制到临时文件，再在临时文件上执行 merge。
210 if (finalFileExists && (!m_options.overwriteExistingFiles || m_options.updateMode || m_options.retryMode)) {
211 const QString tempDirPath = QFileInfo(tempPath).absolutePath();
212 if (!QDir().mkpath(tempDirPath)) {
213 abortExport(QStringLiteral("Failed to create temp symbol library directory: %1").arg(tempDirPath));
214 return;
215 }
216
217 QFile tempFile(tempPath);
218 if (tempFile.exists() && !tempFile.remove()) {
219 abortExport(QStringLiteral("Failed to remove stale temp symbol library: %1").arg(tempPath));
220 return;
221 }
222
223 QFile sourceFile(finalPath);
224 if (!sourceFile.open(QIODevice::ReadOnly)) {
225 if (!QFileInfo::exists(finalPath)) {
226 qWarning()
227 << "SymbolExportStage: Existing symbol library disappeared before copy, creating new library:"
228 << finalPath;
229 } else {
230 abortExport(QStringLiteral("Failed to open existing symbol library for copy: %1").arg(finalPath));
231 return;
232 }
233 } else if (!tempFile.open(QIODevice::WriteOnly | QIODevice::Truncate)) {
234 abortExport(QStringLiteral("Failed to create temp symbol library for copy: %1").arg(tempPath));
235 return;
236 } else {
237 constexpr qint64 kChunkSize = 65536;
238 char buf[kChunkSize];
239 bool copyError = false;
240 while (!sourceFile.atEnd()) {
241 qint64 bytesRead = sourceFile.read(buf, kChunkSize);
242 if (bytesRead < 0) {
243 abortExport(QStringLiteral("Failed to read existing symbol library: %1").arg(finalPath));
244 copyError = true;
245 break;
246 }
247 if (tempFile.write(buf, bytesRead) != bytesRead) {
248 abortExport(QStringLiteral("Failed to write temp symbol library copy: %1").arg(tempPath));
249 copyError = true;
250 break;
251 }
252 }
253 if (copyError) {
254 return;
255 }
256 tempFile.close();
257 }
258 }
259
260 qDebug() << "SymbolExportStage: Exporting" << symbolList.size() << "symbols to temp:" << tempPath;
261
262 bool exportSuccess = false;
263 QString libraryDescription = m_options.symbolLibraryDescription;
264 {
265 ExporterSymbol exporter;
266 bool appendMode = !m_options.overwriteExistingFiles;
267 exportSuccess = exporter.exportSymbolLibrary(
268 symbolList, libName, tempPath, appendMode, m_options.updateMode, libraryDescription);
269 }
270
271 if (m_cancelled.load()) {
272 m_tempManager.rollbackAll();
273 m_isExporting.store(false);
274 m_isRunning.store(false);
275 emit completed(0, symbolList.size(), 0);
276 return;
277 }
278
279 if (!exportSuccess) {
280 abortExport(QStringLiteral("Failed to export symbol library"));
281 return;
282 }
283
284 if (!m_tempManager.commitWithBackup(tempPath, finalPath)) {
285 abortExport(QStringLiteral("Failed to commit temp file"));
286 return;
287 }
288 qDebug() << "SymbolExportStage: Successfully exported to:" << finalPath;
289
290 if (!libraryDescription.isEmpty()) {
291 ExporterSymbol symTableExporter;
292 symTableExporter.generateSymLibTable(libName, finalPath, outputDir, libraryDescription);
293 KiCadLibraryTableManager::registerSymbolLibrary(outputDir, libName, finalPath, libraryDescription);
294 }
295
296 m_tempManager.cleanupTempDirectory();
297
298 qDebug() << "SymbolExportStage: Completed. Success:" << successCount << "Failed:" << failedIds.size();
299
300 m_isExporting.store(false);
301 m_isRunning.store(false);
302 emit completed(successCount, failedIds.size(), skippedCount);
303}
