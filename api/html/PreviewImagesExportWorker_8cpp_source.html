33 if (m_cancelled.load()) {
34 emit completed(m_componentId, false, QStringLiteral("Cancelled"));
35 return;
36 }
37
38 if (!m_data) {
39 emit completed(m_componentId, false, QStringLiteral("No data available"));
40 return;
41 }
42
43 qDebug() << "PreviewImagesExportWorker: Exporting" << m_componentId;
44
45 // 检查预览图数据是否可用
46 QList<QByteArray> previewDataList = m_data->previewImageData();
47
48 // 如果内存中没有预览图数据，尝试从磁盘缓存加载
49 if (previewDataList.isEmpty()) {
50 ComponentCacheService* cache = ComponentCacheService::instance();
51 for (int i = 0; i < 3; ++i) { // 最多尝试加载3张预览图
52 QByteArray imageData = cache->loadPreviewImage(m_componentId, i);
53 if (imageData.isEmpty()) {
54 break;
55 }
56 previewDataList.append(imageData);
57 }
58 qDebug() << "PreviewImagesExportWorker: Loaded" << previewDataList.size() << "preview images from cache for"
59 << m_componentId;
60 }
61
62 // 如果缓存被清空，但内存里仍保留了预览图 URL，则在导出阶段直接回补下载。
63 if (previewDataList.isEmpty()) {
64 const QStringList previewUrls = m_data->previewImages();
65 if (!previewUrls.isEmpty()) {
66 ComponentCacheService* cache = ComponentCacheService::instance();
67 for (int i = 0; i < previewUrls.size(); ++i) {
68 const QByteArray imageData = cache->downloadPreviewImage(
69 m_componentId, previewUrls[i], i, nullptr, nullptr, m_options.weakNetworkSupport);
70 if (!imageData.isEmpty()) {
71 previewDataList.append(imageData);
72 }
73 }
74 qDebug() << "PreviewImagesExportWorker: Re-downloaded" << previewDataList.size()
75 << "preview images from URLs for" << m_componentId;
76 }
77 }
78
79 if (previewDataList.isEmpty()) {
80 // 无预览图数据时按跳过处理，不阻塞整体导出
81 emit completed(m_componentId, true, QStringLiteral("Preview image data not available, skipped"));
82 return;
83 }
84
85 // 构建输出路径
86 // 注意：如果没有使用临时路径（previewCount <= 0 的情况），
87 // m_options.outputPath 仍然是根目录，需要手动添加 .preview 子目录
88 QString outputDir;
89 if (m_tempPaths.isEmpty()) {
90 // 没有使用临时路径，使用 .preview 子目录
91 QString libName = m_options.libName.isEmpty() ? QStringLiteral("EasyKiConverter") : m_options.libName;
92 QString baseOutputDir = m_options.outputPath;
93 if (baseOutputDir.isEmpty()) {
94 baseOutputDir = QDir::currentPath() + QStringLiteral("/export");
95 }
96 outputDir = baseOutputDir + QDir::separator() + libName + QStringLiteral(".preview");
97 } else {
98 // 使用 setTempPaths 设置的路径
99 outputDir = m_options.outputPath;
100 }
101
102 QDir dir;
103 if (!dir.mkpath(outputDir)) {
104 emit completed(m_componentId, false, QStringLiteral("Failed to create output directory"));
105 return;
106 }
107
108 // 遍历导出所有预览图
109 int successCount = 0;
110 int totalCount = previewDataList.size();
111
112 for (int i = 0; i < totalCount; ++i) {
113 if (m_cancelled.load()) {
114 emit completed(m_componentId, false, QStringLiteral("Cancelled"));
115 return;
116 }
117
118 const auto& previewData = previewDataList[i];
119 QString fileName = QStringLiteral("%1_preview_%2.png").arg(m_componentId).arg(i + 1);
120 QString finalPath = outputDir + QStringLiteral("/") + fileName;
121
122 // 使用temp路径如果可用
123 QString filePath = m_tempPaths.contains(fileName) ? m_tempPaths[fileName] : finalPath;
124
125 // 确保目标目录存在（temp目录按组件拆分，可能尚未创建）
126 const QString targetDir = QFileInfo(filePath).absolutePath();
127 if (!QDir().mkpath(targetDir)) {
128 qWarning() << "PreviewImagesExportWorker: Failed to create target dir" << targetDir;
129 continue;
130 }
131
132 // 检查文件是否已存在（仅对最终路径且非temp模式）
133 if (filePath == finalPath && QFile::exists(finalPath) && !m_options.overwriteExistingFiles) {
134 qDebug() << "PreviewImagesExportWorker: File already exists, skipping" << finalPath;
135 successCount++;
136 continue;
137 }
138
139 try {
140 // 解码并保存预览图
141 QImage image = QImage::fromData(previewData);
142 if (image.isNull()) {
143 qWarning() << "PreviewImagesExportWorker: Failed to decode preview image" << i;
144 continue;
145 }
146
147 QSaveFile file(filePath);
148 if (!file.open(QIODevice::WriteOnly)) {
149 qWarning() << "PreviewImagesExportWorker: Failed to open file" << filePath << file.errorString();
150 continue;
151 }
152
153 if (!image.save(&file, "PNG")) {
154 qWarning() << "PreviewImagesExportWorker: Failed to save image" << filePath;
155 continue;
156 }
157
158 if (!file.commit()) {
159 qWarning() << "PreviewImagesExportWorker: Failed to commit file" << filePath;
160 continue;
161 }
162
163 successCount++;
164
165 } catch (const std::exception& e) {
166 qWarning() << "PreviewImagesExportWorker: Exception during export:" << e.what();
167 }
168 }
169
170 if (successCount == totalCount) {
171 qDebug() << "PreviewImagesExportWorker: Successfully exported all previews for" << m_componentId;
172
173 if (m_options.debugMode) {
174 (void)QtConcurrent::run([componentId = m_componentId, data = m_data, outputPath = m_options.outputPath]() {
175 DebugExportHelper::exportDebugData(componentId, data, outputPath);
176 });
177 }
178
179 emit completed(m_componentId, true, QString());
180 } else if (successCount > 0) {
181 // 部分成功也视为失败，避免提交不完整的临时文件
182 qWarning() << "PreviewImagesExportWorker: Partially exported" << successCount << "/" << totalCount
183 << "- treating as failure";
184 emit completed(
185 m_componentId,
186 false,
187 QStringLiteral("Partial export failure: %1 of %2 images exported").arg(successCount).arg(totalCount));
188 } else {
189 emit completed(m_componentId, false, QStringLiteral("Failed to export any preview images"));
190 }
191}
