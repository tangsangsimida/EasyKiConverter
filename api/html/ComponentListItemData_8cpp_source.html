106void ComponentListItemData::insertPreviewImage(const QImage& image, int index) {
107 if (image.isNull()) {
108 return;
109 }
110
111 // 确保列表大小足够
112 while (m_previewImages.size() <= index) {
113 m_previewImages.append(QImage());
114 }
115
116 // 插入或替换指定位置的图片
117 if (m_previewImages[index].isNull()) {
118 m_previewImages[index] = image;
119 qDebug() << "Inserted preview image at index:" << index;
120 } else {
121 // 如果该位置已有图片，替换它
122 m_previewImages[index] = image;
123 qDebug() << "Replaced preview image at index:" << index;
124 }
125
126 // 第一张预览图始终覆盖缩略图显示
127 if (index == 0) {
128 setThumbnail(image);
129 qDebug() << "First preview image (index 0) set as thumbnail";
130 } else if (!hasThumbnail()) {
131 // 如果没有缩略图且不是第一张，使用当前图片作为缩略图
132 setThumbnail(image);
133 }
134
135 emit previewImagesChanged();
