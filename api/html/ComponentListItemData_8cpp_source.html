132void ComponentListItemData::updatePreviewImagesCache() const {
133 m_previewImagesCache.clear();
134 for (const QImage& image : m_previewImages) {
135 if (image.isNull()) {
136 m_previewImagesCache.append(QVariant());

140 QBuffer buffer(&byteArray);
141 buffer.open(QIODevice::WriteOnly);
142 image.save(&buffer, "PNG");
143 m_previewImagesCache.append(QString::fromLatin1(byteArray.toBase64().data()));
144 }

154int ComponentListItemData::previewImageCount() const {
155 int cacheCount = 0;
156 for (const QVariant& cached : m_previewImagesCache) {
157 if (cached.isValid() && !cached.toString().isEmpty()) {
158 ++cacheCount;
159 }
160 }
161
162 int rawCount = 0;
163 for (const QImage& image : m_previewImages) {
164 if (!image.isNull()) {
165 ++rawCount;
166 }
167 }
168
169 return qMax(cacheCount, rawCount);
170}
