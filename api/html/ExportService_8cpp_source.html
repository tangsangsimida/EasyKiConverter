114void ExportService::executeExportPipelineWithData(const QList<ComponentData>& componentDataList,
115 const ExportOptions& options) {
116 qDebug() << "Executing export pipeline with data for" << componentDataList.size() << "components";
117
118 QMutexLocker locker(m_mutex);
119
120 if (m_isExporting) {
121 qWarning() << "Export already in progress";
122 return;
123 }
124
125 m_isExporting = true;
126 m_options = options;
127 m_currentProgress = 0;
128 m_totalProgress = componentDataList.size();
129 m_successCount = 0;
130 m_failureCount = 0;
131
132 // 清空之前的数
133 m_exportDataList.clear();
134
135 // 收集所有符号和封装数据
136 QList<SymbolData> allSymbols;
137 QList<FootprintData> allFootprints;
138 QList<Model3DData> allModels;
139
140 locker.unlock();
141
142 // 为每个元件收集数
143 for (const ComponentData& componentData : componentDataList) {
144 ExportData exportData;
145 exportData.componentId = componentData.lcscId();
146
147 // 检查并复制符号数据
148 if (componentData.symbolData() && !componentData.symbolData()->info().name.isEmpty()) {
149 exportData.symbolData = *componentData.symbolData();
150 allSymbols.append(exportData.symbolData);
151 }
152
153 // 检查并复制封装数据
154 if (componentData.footprintData() && !componentData.footprintData()->info().name.isEmpty()) {
155 exportData.footprintData = *componentData.footprintData();
156 allFootprints.append(exportData.footprintData);
157 }
158
159 // 检查并复制3D模型数据
160 if (componentData.model3DData() && !componentData.model3DData()->uuid().isEmpty()) {
161 exportData.model3DData = *componentData.model3DData();
162 allModels.append(exportData.model3DData);
163 }
164
165 exportData.success = false;
166 m_exportDataList.append(exportData);
167
168 qDebug() << "Added export data for:" << exportData.componentId
169 << "Symbol:" << !exportData.symbolData.info().name.isEmpty()
170 << "Footprint:" << !exportData.footprintData.info().name.isEmpty()
171 << "3D Model:" << !exportData.model3DData.uuid().isEmpty();
172 }
173
174 // 创建输出目录
175 if (!createOutputDirectory(m_options.outputPath)) {
176 qWarning() << "Failed to create output directory:" << m_options.outputPath;
177 m_isExporting = false;
178 emit exportFailed("Failed to create output directory");
179 return;
180 }
181
182 // 导出符号库
183 if (m_options.exportSymbol && !allSymbols.isEmpty()) {
184 QString symbolLibPath = QString("%1/%2.kicad_sym").arg(m_options.outputPath, m_options.libName);
185 if (exportSymbolLibrary(allSymbols, m_options.libName, symbolLibPath)) {
186 qDebug() << "Symbol library exported successfully:" << symbolLibPath;
187 // 在追加模式下，如果所有符号都已存在，exportSymbolLibrary 会返true
188 // 但实际上没有导出任何新符号，这种情况下我们认为这些符号已经存
189 // 不应该计入成功或失败计数
190 // 只有在实际导出新符号或覆盖现有符号时才增加计
191 // 由于我们无法从返回值判断是否实际导出了新符
192 // 这里简单处理：如果文件存在，认为所有符号都已成功导出（包括之前已存在的
193 m_successCount += allSymbols.size();
194 } else {
195 qWarning() << "Failed to export symbol library";
196 m_failureCount += allSymbols.size();
197 }
198 }
199
200 // 导出3D模型（需要在导出封装库之前完成）
201 QString modelsDirPath = QString("%1/3dmodels").arg(m_options.outputPath);
202 if (m_options.exportModel3D && !allModels.isEmpty()) {
203 // 创建 3D 模型目录（添加库名称前缀
204 QString modelsDirPath = QString("%1/%2.3dmodels").arg(m_options.outputPath, m_options.libName);
205 if (!createOutputDirectory(modelsDirPath)) {
206 qWarning() << "Failed to create 3D models directory:" << modelsDirPath;
207 } else {
208 // 为每3D 模型设置文件名（使用封装名称
209 QMap<QString, QString> modelPathMap; // UUID -> 文件路径映射
210
211 for (auto& model : allModels) {
212 // 查找对应的封装名
213 QString footprintName;
214 for (const auto& exportData : m_exportDataList) {
215 if (exportData.model3DData.uuid() == model.uuid() &&
216 !exportData.footprintData.info().name.isEmpty()) {
217 footprintName = exportData.footprintData.info().name;
218 break;
219 }
220 }
221
222 // 使用封装名称作为文件
223 QString modelName = footprintName.isEmpty() ? model.uuid() : footprintName;
224 QString wrlPath = QString("%1/%2.wrl").arg(modelsDirPath, modelName);
225 QString stepPath = QString("%1/%2.step").arg(modelsDirPath, modelName);
226
227 // 保存 WRL 模型
228 if (!model.rawObj().isEmpty()) {
229 if (!m_options.overwriteExistingFiles && QFile::exists(wrlPath)) {
230 qWarning() << "WRL model file already exists:" << wrlPath;
231 } else {
232 if (m_modelExporter->exportToWrl(model, wrlPath)) {
233 qDebug() << "WRL model exported successfully:" << wrlPath;
234 } else {
235 qWarning() << "Failed to export WRL model:" << modelName;
236 }
237 }
238 }
239
240 // 保存 STEP 模型
241 if (!model.step().isEmpty() && model.step().size() > 100) {
242 if (!m_options.overwriteExistingFiles && QFile::exists(stepPath)) {
243 qWarning() << "STEP model file already exists:" << stepPath;
244 } else {
245 if (m_modelExporter->exportToStep(model, stepPath)) {
246 qDebug() << "STEP model exported successfully:" << stepPath;
247 } else {
248 qWarning() << "Failed to export STEP model:" << modelName;
249 }
250 }
251 }
252
253 // 保存模型路径映射（使用相对路径，相对于封装库目录
254 QString relativePath = QString("${KIPRJMOD}/%1.3dmodels/%2").arg(m_options.libName, modelName);
255 modelPathMap[model.uuid()] = relativePath;
256 }
257
258 // 更新封装数据中的 3D 模型路径
259 for (auto& footprint : allFootprints) {
260 if (!footprint.model3D().uuid().isEmpty() && modelPathMap.contains(footprint.model3D().uuid())) {
261 Model3DData updatedModel = footprint.model3D();
262 updatedModel.setName(modelPathMap[footprint.model3D().uuid()]);
263 footprint.setModel3D(updatedModel);
264 }
265 }
266 }
267 }
268
269 // 创建封装库目录
270 QString footprintDirPath = QString("%1/%2.pretty").arg(m_options.outputPath, m_options.libName);
271 if (m_options.exportFootprint && !allFootprints.isEmpty()) {
272 if (exportFootprintLibrary(allFootprints, m_options.libName, footprintDirPath)) {
273 qDebug() << "Footprint library exported successfully with 3D model paths:" << footprintDirPath;
274 } else {
275 m_failureCount += allFootprints.size();
276 }
277 }
278
279 // 更新进度和统
280 m_successCount = m_totalProgress - m_failureCount;
281 m_currentProgress = m_totalProgress;
282
283 // 发送导出完成信
284 emit exportProgress(m_currentProgress, m_totalProgress);
285
286 // 为每个元件发送导出成功信
287 for (const ExportData& exportData : m_exportDataList) {
288 emit componentExported(exportData.componentId, true, "Export successful");
289 }
290
291 // 完成导出
292 m_isExporting = false;
293 qDebug() << "Export pipeline completed:" << m_successCount << "success," << m_failureCount << "failed";
294 emit exportCompleted(m_totalProgress, m_successCount);
295}

431void ExportService::processNextExport() {
432 QMutexLocker locker(m_mutex);
433
434 if (!m_isExporting) {
435 return;
436 }
437
438 // 检查是否所有元件都已处
439 if (m_currentProgress >= m_totalProgress) {
440 m_isExporting = false;
441 locker.unlock();
442
443 qDebug() << "Export pipeline completed:" << m_successCount << "success," << m_failureCount << "failed";
444 emit exportCompleted(m_totalProgress, m_successCount);
445 return;
446 }
447
448 // 获取下一个待处理的元
449 int index = m_currentProgress;
450 if (index >= m_exportDataList.size()) {
451 return;
452 }
453
454 ExportData& exportData = m_exportDataList[index];
455 QString componentId = exportData.componentId;
456 locker.unlock();
457
458 // 检查是否有数据可以导出
459 bool hasSymbolData = !exportData.symbolData.info().name.isEmpty();
460 bool hasFootprintData = !exportData.footprintData.info().name.isEmpty();
461 bool hasModel3DData = !exportData.model3DData.uuid().isEmpty();
462
463 if (!hasSymbolData && !hasFootprintData && !hasModel3DData) {
464 qWarning() << "No data available for export:" << componentId;
465 handleExportTaskFinished(componentId, false, "No data available for export");
466 return;
467 }
468
469 // 创建输出目录
470 if (!createOutputDirectory(m_options.outputPath)) {
471 qWarning() << "Failed to create output directory:" << m_options.outputPath;
472 handleExportTaskFinished(componentId, false, "Failed to create output directory");
473 return;
474 }
475
476 // 导出符号
477 if (m_options.exportSymbol && hasSymbolData) {
478 QString symbolPath = QString("%1/%2.kicad_sym").arg(m_options.outputPath, componentId);
479 if (!exportSymbol(exportData.symbolData, symbolPath)) {
480 qWarning() << "Failed to export symbol for:" << componentId;
481 handleExportTaskFinished(componentId, false, "Failed to export symbol");
482 return;
483 }
484 }
485
486 // 导出封装
487 if (m_options.exportFootprint && hasFootprintData) {
488 QString footprintPath = QString("%1/%2.kicad_mod").arg(m_options.outputPath, componentId);
489 if (!exportFootprint(exportData.footprintData, footprintPath)) {
490 qWarning() << "Failed to export footprint for:" << componentId;
491 handleExportTaskFinished(componentId, false, "Failed to export footprint");
492 return;
493 }
494 }
495
496 // 导出3D模型
497 if (m_options.exportModel3D && hasModel3DData) {
498 QString modelPath = QString("%1/%2.wrl").arg(m_options.outputPath, exportData.model3DData.uuid());
499 if (!export3DModel(exportData.model3DData, modelPath)) {
500 qWarning() << "Failed to export 3D model for:" << componentId;
501 handleExportTaskFinished(componentId, false, "Failed to export 3D model");
502 return;
503 }
504 }
505
506 // 导出成功
507 handleExportTaskFinished(componentId, true, "Export successful");
508}

642 updatedModel.setName(modelPathMap[footprint.model3D().uuid()]);
643 // 保留 STEP 数据
644 for (const auto& model : allModels) {
645 if (model.uuid() == footprint.model3D().uuid()) {
646 updatedModel.setStep(model.step());
647 break;
648 }
649 }
650 footprint.setModel3D(updatedModel);
651 }
652 }
653 }
654 }
655
656 // 创建封装库目录
657 QString footprintDirPath = QString("%1/%2.pretty").arg(m_options.outputPath, m_options.libName);
658 if (m_options.exportFootprint && !allFootprints.isEmpty()) {
659 if (exportFootprintLibrary(allFootprints, m_options.libName, footprintDirPath)) {
660 qDebug() << "Footprint library exported successfully with 3D model paths:" << footprintDirPath;
661 m_successCount += allFootprints.size();
662 } else {
663 m_failureCount += allFootprints.size();
664 }
665 }
666
667 // 更新进度和统
668 m_currentProgress = m_totalProgress;
669
670 // 发送导出完成信
671 m_isExporting = false;
672 m_parallelExporting = false;
673 emit exportProgress(m_currentProgress, m_totalProgress);
674 emit exportCompleted(m_totalProgress, m_successCount);
675}

677void ExportService::handleParallelExportTaskFinished(const QString& componentId, bool success, const QString& message) {
678 qDebug() << "Parallel export task finished for:" << componentId << "Success:" << success;
679
680 QMutexLocker locker(m_mutex);
681
682 // 更新统计信息
683 if (success) {
684 m_successCount++;
685 } else {
686 m_failureCount++;
687 }
688
689 // 更新状
690 m_parallelExportStatus[componentId] = false;
691 m_parallelCompletedCount++;
692
693 // 更新进度
694 m_currentProgress++;
695 locker.unlock();
696
697 emit exportProgress(m_currentProgress, m_totalProgress);
698 emit componentExported(componentId, success, message);
699
700 // 检查是否所有任务都已完
701 if (m_parallelCompletedCount >= m_parallelTotalCount) {
702 qDebug() << "All parallel export tasks completed:" << m_successCount << "success," << m_failureCount
703 << "failed";
704
705 m_isExporting = false;
706 m_parallelExporting = false;
707 emit exportCompleted(m_totalProgress, m_successCount);
708 }
