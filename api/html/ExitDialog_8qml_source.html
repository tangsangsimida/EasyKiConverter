221 function exitFillDiameterFromOrigin(originX, originY) {
222 var w = root.dialogBox ? root.dialogBox.width : root.width;
223 var h = root.dialogBox ? root.dialogBox.height : root.height;
224 var distances = [Math.sqrt(originX * originX + originY * originY), Math.sqrt((w - originX) * (w - originX) + originY * originY), Math.sqrt(originX * originX + (h - originY) * (h - originY)), Math.sqrt((w - originX) * (w - originX) + (h - originY) * (h - originY))];
225 return Math.max.apply(Math, distances) * 2 + AppStyle.spacing.xl;
226 }

274 Keys.onPressed: event => {
275 if (event.key === Qt.Key_Up || event.key === Qt.Key_Left) {
276 if (focusArea === "button") {
277 // 向上/左导航
278 var currentIndex = 0;
279 var targetIndex = 0;
280 var btnCount = 0;
281 for (var i = 0; i < buttonSpecs.length; i++) {
282 if (!buttonSpecs[i].isSeparator) {
283 if (buttonColLayout.children[i].actualButton === selectedButton) {
284 currentIndex = i;
285 }
286 btnCount++;
287 }
288 }
289 // 找到上一个按钮
290 var prevBtnIndex = -1;
291 for (var j = currentIndex - 1; j >= 0; j--) {
292 if (!buttonSpecs[j].isSeparator) {
293 prevBtnIndex = j;
294 break;
295 }
296 }
297 if (prevBtnIndex >= 0) {
298 selectedButton = buttonColLayout.children[prevBtnIndex].actualButton;
299 updateSlider(selectedButton, buttonSpecs[prevBtnIndex].color);
300 }
301 }
302 event.accepted = true;
303 } else if (event.key === Qt.Key_Down || event.key === Qt.Key_Right) {
304 if (focusArea === "button") {
305 // 向下/右导航
306 var currentIdx = 0;
307 var nextIdx = -1;
308 for (var k = 0; k < buttonSpecs.length; k++) {
309 if (!buttonSpecs[k].isSeparator) {
310 if (buttonColLayout.children[k].actualButton === selectedButton) {
311 currentIdx = k;
312 }
313 }
314 }
315 for (var l = currentIdx + 1; l < buttonSpecs.length; l++) {
316 if (!buttonSpecs[l].isSeparator) {
317 nextIdx = l;
318 break;
319 }
320 }
321 if (nextIdx >= 0) {
322 selectedButton = buttonColLayout.children[nextIdx].actualButton;
323 updateSlider(selectedButton, buttonSpecs[nextIdx].color);
324 }
325 }
326 event.accepted = true;
327 } else if (event.key === Qt.Key_Tab) {
328 // Tab 键：在按钮和复选框之间切换
329 if (focusArea === "button") {
330 focusArea = "checkbox";
331 hideSlider();
332 } else {
333 focusArea = "button";
334 if (selectedButton) {
335 for (var m = 0; m < buttonSpecs.length; m++) {
336 if (buttonColLayout.children[m].actualButton === selectedButton) {
337 updateSlider(selectedButton, buttonSpecs[m].color);
338 break;
339 }
340 }
341 }
342 }
343 event.accepted = true;
344 } else if (event.key === Qt.Key_Space) {
345 if (focusArea === "checkbox") {
346 root.rememberChoice = !root.rememberChoice;
347 event.accepted = true;
348 }
349 } else if (event.key === Qt.Key_Return || event.key === Qt.Key_Enter) {
350 if (focusArea === "checkbox") {
351 root.rememberChoice = !root.rememberChoice;
352 event.accepted = true;
353 } else if (selectedButton) {
354 // 执行选中按钮的动作
355 for (var n = 0; n < buttonSpecs.length; n++) {
356 if (buttonColLayout.children[n].actualButton === selectedButton) {
357 if (buttonSpecs[n].action) {
358 buttonSpecs[n].action();
359 }
360 break;
361 }
362 }
363 }
364 event.accepted = true;
365 } else if (event.key === Qt.Key_Escape) {
366 // ESC 键不处理，让 Main.qml 的 Shortcut 处理
367 event.accepted = false;
368 } else {
369 event.accepted = true;
370 }
371 }

383 var childIndex = 0;
384 for (var i = 0; i < buttonSpecs.length; i++) {
385 if (buttonSpecs[i].isSeparator) {
386 continue;
387 }
388 // 找到第一个非分隔线的按钮
389 var loader = buttonColLayout.children[childIndex];
390 if (loader && loader.actualButton) {
391 selectedButton = loader.actualButton;
392 updateSlider(selectedButton, buttonSpecs[i].color);

395 childIndex++;
396 }
