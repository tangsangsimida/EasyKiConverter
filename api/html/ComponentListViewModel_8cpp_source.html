101void ComponentListViewModel::addComponent(const QString& componentId) {
102 QString trimmedId = componentId.trimmed();
103
104 if (trimmedId.isEmpty()) {
105 qWarning() << "Component ID is empty";
106 return;
107 }
108
109 // 统一转换为大写，支持用户输入小写 c
110 trimmedId = trimmedId.toUpper();
111
112 // 验证元件ID格式
113 if (!validateComponentId(trimmedId)) {
114 qWarning() << "Invalid component ID format:" << trimmedId;
115
116 // 尝试从文本中智能提取元件编号
117 QStringList extractedIds = extractComponentIdFromText(trimmedId);
118 if (extractedIds.isEmpty()) {
119 qWarning() << "Failed to extract component ID from text:" << trimmedId;
120 emit componentAdded(trimmedId, false, "Invalid LCSC component ID format");
121 return;
122 }
123
124 addComponentsBatch(extractedIds);
125 return;
126 }
127
128 // 检查元件是否已存在
129 if (componentExists(trimmedId)) {
130 qWarning() << "Component already exists:" << trimmedId;
131 emit componentAdded(trimmedId, false, "Component already exists");
132 return;
133 }
134
135 beginInsertRows(QModelIndex(), m_componentList.count(), m_componentList.count());
136 auto item = new ComponentListItemData(trimmedId, this);
137 item->setFetching(true);
138 item->setValid(false); // 验证开始时先设置为 false
139 m_componentList.append(item);
140 m_componentIdIndex.insert(trimmedId); // 维护索引
141 endInsertRows();
142
143 // 触发获取数据
144 m_service->fetchComponentData(trimmedId, false);
145
146 // 两阶段控制：单个添加也使用两阶段
147 m_pendingValidationCount = 1;
148 m_validatedComponentIds.clear();
149 m_previewFetchEnabled = false;
150
151 emit componentCountChanged();
152 emit filteredCountChanged();
153 emit componentAdded(trimmedId, true, "Component added");
154}

156void ComponentListViewModel::removeComponent(int index) {
157 if (index >= 0 && index < m_componentList.count()) {
158 beginRemoveRows(QModelIndex(), index, index);
159 auto item = m_componentList.takeAt(index);
160 QString removedId = item->componentId();
161 m_componentIdIndex.remove(removedId); // 维护索引
162
163 // 两阶段控制：更新状态
164 m_validatedComponentIds.removeAll(removedId);
165 // 只有当元器件还在验证中（isFetching=true）时才减少计数
166 if (item->isFetching()) {
167 m_pendingValidationCount--;
168 }
169
170 delete item;
171 endRemoveRows();
172
173 // 更新 hasInvalidComponents 状态
174 updateHasInvalidComponents();
175
176 // 如果删除后验证队列为空，尝试开始预览图获取
177 if (m_pendingValidationCount <= 0 && !m_validatedComponentIds.isEmpty()) {
178 m_previewFetchEnabled = true;
179 fetchAllPreviewImages();
180 }
181
182 emit componentCountChanged();
183 emit filteredCountChanged();
184 emit componentRemoved(removedId);
185 }

224void ComponentListViewModel::addComponentsBatch(const QStringList& componentIds) {
225 // 收集需要添加的新 ID（去重 + 验证）
226 QStringList newIds;
227 for (const QString& rawId : componentIds) {
228 QString id = rawId.trimmed().toUpper();
229 if (id.isEmpty())
230 continue;
231
232 // 验证格式
233 if (!validateComponentId(id)) {
234 // 尝试提取
235 QStringList extracted = extractComponentIdFromText(id);
236 for (const QString& extractedId : extracted) {
237 if (!componentExists(extractedId) && !newIds.contains(extractedId)) {
238 newIds.append(extractedId);
239 }
240 }
241 continue;
242 }
243
244 if (!componentExists(id) && !newIds.contains(id)) {
245 newIds.append(id);
246 }
247 }
248
249 if (newIds.isEmpty())
250 return;
251
252 // 重置两阶段控制变量
253 m_pendingValidationCount = newIds.count();
254 m_validatedComponentIds.clear();
255 m_previewFetchEnabled = false;
256
257 // 一次性插入所有新元器件，只触发 1 次 UI 重建
258 int first = m_componentList.count();
259 int last = first + newIds.count() - 1;
260 beginInsertRows(QModelIndex(), first, last);
261
262 for (const QString& id : newIds) {
263 auto item = new ComponentListItemData(id, this);
264 item->setFetching(true);
265 item->setValid(false); // 验证开始时先设置为 false
266 m_componentList.append(item);
267 m_componentIdIndex.insert(id);
268 }
269
270 endInsertRows();
271 emit componentCountChanged();
272 emit filteredCountChanged();
273
274 // 延迟分散发送网络请求，避免同时回调集中到达卡 UI
275 // 增加延迟间隔到 150ms，减少同时触发的网络请求数量
276 for (int i = 0; i < newIds.count(); ++i) {
277 const QString& id = newIds[i];
278 // 每个请求间隔 150ms，避免回调集中在同一帧
279 QTimer::singleShot(i * 150, this, [this, id]() { m_service->fetchComponentData(id, false); });
280 emit componentAdded(id, true, "Component added");
281 }

284void ComponentListViewModel::pasteFromClipboard() {
285 QClipboard* clipboard = QGuiApplication::clipboard();
286 QString text = clipboard->text();
287
288 if (text.isEmpty()) {
289 qWarning() << "Clipboard is empty";
290 return;
291 }
292
293 QStringList extractedIds = extractComponentIdFromText(text);
294 if (extractedIds.isEmpty()) {
295 qWarning() << "No valid component IDs found in clipboard";
296 emit pasteCompleted(0, 0);
297 return;
298 }
299
300 // 统计新增和跳过数量
301 int skipped = 0;
302 QStringList newIds;
303 for (const QString& id : extractedIds) {
304 if (componentExists(id)) {
305 skipped++;
306 } else {
307 newIds.append(id);
308 }
309 }
310
311 // 使用批量添加
312 if (!newIds.isEmpty()) {
313 addComponentsBatch(newIds);
314 }
315
316 emit pasteCompleted(newIds.count(), skipped);
317}

343void ComponentListViewModel::selectBomFile(const QString& filePath) {
344 qDebug() << "BOM file selected:" << filePath;
345
346 if (m_bomFilePath != filePath) {
347 m_bomFilePath = filePath;
348 emit bomFilePathChanged();
349 }
350
351 // 将 URL 转换为本地文件路径
352 QString localPath = filePath;
353 if (filePath.startsWith("file:///")) {
354 localPath = QUrl(filePath).toLocalFile();
355 qDebug() << "Converted URL to local path:" << localPath;
356 }
357
358 // 使用 ComponentService 解析 BOM 文件
359 QStringList componentIds = m_service->parseBomFile(localPath);
360
361 if (componentIds.isEmpty()) {
362 m_bomResult = "No valid component IDs found in BOM file";
363 qWarning() << "No component IDs found in BOM file:" << localPath;
364 emit bomResultChanged();
365 return;
366 }
367
368 // 使用批量添加方法
369 QStringList newIds;
370 int skipped = 0;
371
372 for (const QString& id : componentIds) {
373 if (!componentExists(id)) {
374 newIds.append(id);
375 } else {
376 skipped++;
377 }
378 }
379
380 if (!newIds.isEmpty()) {
381 addComponentsBatch(newIds);
382 }
383
384 QString resultMsg = QString("BOM file imported: %1 components added, %2 skipped").arg(newIds.count()).arg(skipped);
385 m_bomResult = resultMsg;
386 qDebug() << resultMsg;
387 emit bomResultChanged();
388}

487void ComponentListViewModel::handleFetchError(const QString& componentId, const QString& error) {
488 qWarning() << "Fetch error for:" << componentId << "-" << error;
489
490 auto item = findItemData(componentId);
491
492 if (item) {
493 item->setFetching(false);
494
495 // 标记为无效
496 item->setValid(false);
497
498 // 根据错误类型设置不同的错误消息
499 if (error.contains("Request timeout", Qt::CaseInsensitive) || error.contains("timeout", Qt::CaseInsensitive)) {
500 item->setErrorMessage("验证超时（网络不稳定）");
501 } else if (error.contains("No result", Qt::CaseInsensitive) || error.contains("404") ||
502 error.contains("not found", Qt::CaseInsensitive)) {
503 item->setErrorMessage("元件不存在");
504 } else {
505 item->setErrorMessage(error);
506 }
507
508 // 更新 hasInvalidComponents 状态
509 updateHasInvalidComponents();
510 emit filteredCountChanged();
511
512 // 两阶段控制：验证失败也要计数
513 m_pendingValidationCount--;
514
515 // 所有验证（成功+失败）完成后，统一获取预览图
516 if (m_pendingValidationCount <= 0) {
517 m_previewFetchEnabled = true;
518 fetchAllPreviewImages();
519 }
520
521 // 添加到防抖集合
522
523 m_pendingUpdateIndices.insert(componentId);
524
525 m_debounceTimer->start();
526 }

682void ComponentListViewModel::retryAllInvalidComponents() {
683 int retryCount = 0;
684 for (int i = 0; i < m_componentList.count(); ++i) {

687 item->setFetching(true);
688 item->setValid(false); // 验证开始时先设置为 false
689 item->setErrorMessage("");
690 m_service->fetchComponentData(item->componentId(), false);
691 retryCount++;
692 }
693 }
694
695 // 两阶段控制
696 if (retryCount > 0) {
697 m_pendingValidationCount = retryCount;
698 m_validatedComponentIds.clear();
699 m_previewFetchEnabled = false;
700 }
