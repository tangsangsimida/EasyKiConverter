62void ComponentListViewModel::addComponent(const QString& componentId) {
63 QString trimmedId = componentId.trimmed();
64
65 if (trimmedId.isEmpty()) {
66 qWarning() << "Component ID is empty";
67 return;
68 }
69
70 // 统一转换为大写，支持用户输入小写 c
71 trimmedId = trimmedId.toUpper();
72
73 // 验证元件ID格式
74 if (!validateComponentId(trimmedId)) {
75 qWarning() << "Invalid component ID format:" << trimmedId;
76
77 // 尝试从文本中智能提取元件编号
78 QStringList extractedIds = extractComponentIdFromText(trimmedId);
79 if (extractedIds.isEmpty()) {
80 qWarning() << "Failed to extract component ID from text:" << trimmedId;
81 emit componentAdded(trimmedId, false, "Invalid LCSC component ID format");
82 return;
83 }
84
85 // 添加提取的元件编号 - 使用批量添加
86 addComponentsBatch(extractedIds);
87 return;
88 }
89
90 // 检查元件是否已存在
91 if (componentExists(trimmedId)) {
92 qWarning() << "Component already exists:" << trimmedId;
93 emit componentAdded(trimmedId, false, "Component already exists");
94 return;
95 }
96
97 beginInsertRows(QModelIndex(), m_componentList.count(), m_componentList.count());
98 auto item = new ComponentListItemData(trimmedId, this);
99 item->setFetching(true);
100 m_componentList.append(item);
101 m_componentIdIndex.insert(trimmedId); // 维护索引
102 endInsertRows();
103
104 // 触发获取数据
105 m_service->fetchComponentData(trimmedId, false);
106
107 emit componentCountChanged();
108 emit componentAdded(trimmedId, true, "Component added");
109}

147void ComponentListViewModel::addComponentsBatch(const QStringList& componentIds) {
148 // 收集需要添加的新 ID（去重 + 验证）
149 QStringList newIds;
150 for (const QString& rawId : componentIds) {
151 QString id = rawId.trimmed().toUpper();
152 if (id.isEmpty())
153 continue;
154
155 // 验证格式
156 if (!validateComponentId(id)) {
157 // 尝试提取
158 QStringList extracted = extractComponentIdFromText(id);
159 for (const QString& extractedId : extracted) {
160 if (!componentExists(extractedId) && !newIds.contains(extractedId)) {
161 newIds.append(extractedId);
162 }
163 }
164 continue;
165 }
166
167 if (!componentExists(id) && !newIds.contains(id)) {
168 newIds.append(id);
169 }
170 }
171
172 if (newIds.isEmpty())
173 return;
174
175 // 一次性插入所有新元器件，只触发 1 次 UI 重建
176 int first = m_componentList.count();
177 int last = first + newIds.count() - 1;
178 beginInsertRows(QModelIndex(), first, last);
179
180 for (const QString& id : newIds) {
181 auto item = new ComponentListItemData(id, this);
182 item->setFetching(true);
183 m_componentList.append(item);
184 m_componentIdIndex.insert(id);
185 }
186
187 endInsertRows();
188 emit componentCountChanged();
189
190 // 延迟分散发送网络请求，避免同时回调集中到达卡 UI
191 for (int i = 0; i < newIds.count(); ++i) {
192 const QString& id = newIds[i];
193 // 每个请求间隔 50ms，避免回调集中在同一帧
194 QTimer::singleShot(i * 50, this, [this, id]() { m_service->fetchComponentData(id, false); });
195 emit componentAdded(id, true, "Component added");
196 }
197}

199void ComponentListViewModel::pasteFromClipboard() {
200 QClipboard* clipboard = QGuiApplication::clipboard();
201 QString text = clipboard->text();
202
203 if (text.isEmpty()) {
204 qWarning() << "Clipboard is empty";
205 return;
206 }
207
208 QStringList extractedIds = extractComponentIdFromText(text);
209 if (extractedIds.isEmpty()) {
210 qWarning() << "No valid component IDs found in clipboard";
211 emit pasteCompleted(0, 0);
212 return;
213 }
214
215 // 统计新增和跳过数量
216 int skipped = 0;
217 QStringList newIds;
218 for (const QString& id : extractedIds) {
219 if (componentExists(id)) {
220 skipped++;
221 } else {
222 newIds.append(id);
223 }
224 }
225
226 // 使用批量添加
227 if (!newIds.isEmpty()) {
228 addComponentsBatch(newIds);
229 }
230
231 emit pasteCompleted(newIds.count(), skipped);
232}

258void ComponentListViewModel::selectBomFile(const QString& filePath) {
259 qDebug() << "BOM file selected:" << filePath;
260
261 if (m_bomFilePath != filePath) {
262 m_bomFilePath = filePath;
263 emit bomFilePathChanged();
264 }
265
266 // 将 URL 转换为本地文件路径
267 QString localPath = filePath;
268 if (filePath.startsWith("file:///")) {
269 localPath = QUrl(filePath).toLocalFile();
270 qDebug() << "Converted URL to local path:" << localPath;
271 }
272
273 // 使用 ComponentService 解析 BOM 文件
274 QStringList componentIds = m_service->parseBomFile(localPath);
275
276 if (componentIds.isEmpty()) {
277 m_bomResult = "No valid component IDs found in BOM file";
278 qWarning() << "No component IDs found in BOM file:" << localPath;
279 emit bomResultChanged();
280 return;
281 }
282
283 // 使用批量添加方法
284 QStringList newIds;
285 int skipped = 0;
286
287 for (const QString& id : componentIds) {
288 if (!componentExists(id)) {
289 newIds.append(id);
290 } else {
291 skipped++;
292 }
293 }
294
295 if (!newIds.isEmpty()) {
296 addComponentsBatch(newIds);
297 }
298
299 QString resultMsg = QString("BOM file imported: %1 components added, %2 skipped").arg(newIds.count()).arg(skipped);
300 m_bomResult = resultMsg;
301 qDebug() << resultMsg;
302 emit bomResultChanged();
303}

349void ComponentListViewModel::handleCadDataReady(const QString& componentId, const ComponentData& data) {
350 qDebug() << "CAD data ready for:" << componentId;
351 auto item = findItemData(componentId);
352 if (item) {
353 QSharedPointer<ComponentData> dataPtr = QSharedPointer<ComponentData>::create(data);
354 item->setComponentData(dataPtr);
355 item->setFetching(false);
356 item->setValid(true);
357 item->setErrorMessage("");
358
359 // 异步生成缩略图 - 使用 QPointer 防止悬垂指针
360 QPointer<ComponentListItemData> safeItem = item;
361 QThreadPool::globalInstance()->start(QRunnable::create([safeItem, dataPtr]() {
362 QImage thumbnail = ThumbnailGenerator::generateThumbnail(dataPtr);
363 if (safeItem) { // 检查对象是否仍然有效
364 QMetaObject::invokeMethod(safeItem, [safeItem, thumbnail]() {
365 if (safeItem) { // 再次检查，确保在 invokeMethod 执行时仍然有效
366 safeItem->setThumbnail(thumbnail);
367 }
368 });
369 }
370 }));
371
372 // 尝试获取 LCSC 预览图覆盖生成的缩略图
373 m_service->fetchLcscPreviewImage(componentId);
374 }
