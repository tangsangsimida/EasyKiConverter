98void ComponentListViewModel::addComponent(const QString& componentId) {
99 QString trimmedId = componentId.trimmed();
100
101 if (trimmedId.isEmpty()) {
102 qWarning() << "Component ID is empty";
103 return;
104 }
105
106 // 统一转换为大写，支持用户输入小写 c
107 trimmedId = trimmedId.toUpper();
108
109 // 验证元件ID格式
110 if (!validateComponentId(trimmedId)) {
111 qWarning() << "Invalid component ID format:" << trimmedId;
112
113 // 尝试从文本中智能提取元件编号
114 QStringList extractedIds = extractComponentIdFromText(trimmedId);
115 if (extractedIds.isEmpty()) {
116 qWarning() << "Failed to extract component ID from text:" << trimmedId;
117 emit componentAdded(trimmedId, false, "Invalid LCSC component ID format");
118 return;
119 }
120
121 // 添加提取的元件编号 - 使用批量添加
122 addComponentsBatch(extractedIds);
123 return;
124 }
125
126 // 检查元件是否已存在
127 if (componentExists(trimmedId)) {
128 qWarning() << "Component already exists:" << trimmedId;
129 emit componentAdded(trimmedId, false, "Component already exists");
130 return;
131 }
132
133 beginInsertRows(QModelIndex(), m_componentList.count(), m_componentList.count());
134 auto item = new ComponentListItemData(trimmedId, this);
135 item->setFetching(true);
136 m_componentList.append(item);
137 m_componentIdIndex.insert(trimmedId); // 维护索引
138 endInsertRows();
139
140 // 触发获取数据
141 m_service->fetchComponentData(trimmedId, false);
142
143 emit componentCountChanged();
144 emit componentAdded(trimmedId, true, "Component added");
145}

188void ComponentListViewModel::addComponentsBatch(const QStringList& componentIds) {
189 // 收集需要添加的新 ID（去重 + 验证）
190 QStringList newIds;
191 for (const QString& rawId : componentIds) {
192 QString id = rawId.trimmed().toUpper();
193 if (id.isEmpty())
194 continue;
195
196 // 验证格式
197 if (!validateComponentId(id)) {
198 // 尝试提取
199 QStringList extracted = extractComponentIdFromText(id);
200 for (const QString& extractedId : extracted) {
201 if (!componentExists(extractedId) && !newIds.contains(extractedId)) {
202 newIds.append(extractedId);
203 }
204 }
205 continue;
206 }
207
208 if (!componentExists(id) && !newIds.contains(id)) {
209 newIds.append(id);
210 }
211 }
212
213 if (newIds.isEmpty())
214 return;
215
216 // 一次性插入所有新元器件，只触发 1 次 UI 重建
217 int first = m_componentList.count();
218 int last = first + newIds.count() - 1;
219 beginInsertRows(QModelIndex(), first, last);
220
221 for (const QString& id : newIds) {
222 auto item = new ComponentListItemData(id, this);
223 item->setFetching(true);
224 m_componentList.append(item);
225 m_componentIdIndex.insert(id);
226 }
227
228 endInsertRows();
229 emit componentCountChanged();
230
231 // 延迟分散发送网络请求，避免同时回调集中到达卡 UI
232 // 增加延迟间隔到 150ms，减少同时触发的网络请求数量
233 for (int i = 0; i < newIds.count(); ++i) {
234 const QString& id = newIds[i];
235 // 每个请求间隔 150ms，避免回调集中在同一帧
236 QTimer::singleShot(i * 150, this, [this, id]() { m_service->fetchComponentData(id, false); });
237 emit componentAdded(id, true, "Component added");
238 }

241void ComponentListViewModel::pasteFromClipboard() {
242 QClipboard* clipboard = QGuiApplication::clipboard();
243 QString text = clipboard->text();
244
245 if (text.isEmpty()) {
246 qWarning() << "Clipboard is empty";
247 return;
248 }
249
250 QStringList extractedIds = extractComponentIdFromText(text);
251 if (extractedIds.isEmpty()) {
252 qWarning() << "No valid component IDs found in clipboard";
253 emit pasteCompleted(0, 0);
254 return;
255 }
256
257 // 统计新增和跳过数量
258 int skipped = 0;
259 QStringList newIds;
260 for (const QString& id : extractedIds) {
261 if (componentExists(id)) {
262 skipped++;
263 } else {
264 newIds.append(id);
265 }
266 }
267
268 // 使用批量添加
269 if (!newIds.isEmpty()) {
270 addComponentsBatch(newIds);
271 }
272
273 emit pasteCompleted(newIds.count(), skipped);
274}

300void ComponentListViewModel::selectBomFile(const QString& filePath) {
301 qDebug() << "BOM file selected:" << filePath;
302
303 if (m_bomFilePath != filePath) {
304 m_bomFilePath = filePath;
305 emit bomFilePathChanged();
306 }
307
308 // 将 URL 转换为本地文件路径
309 QString localPath = filePath;
310 if (filePath.startsWith("file:///")) {
311 localPath = QUrl(filePath).toLocalFile();
312 qDebug() << "Converted URL to local path:" << localPath;
313 }
314
315 // 使用 ComponentService 解析 BOM 文件
316 QStringList componentIds = m_service->parseBomFile(localPath);
317
318 if (componentIds.isEmpty()) {
319 m_bomResult = "No valid component IDs found in BOM file";
320 qWarning() << "No component IDs found in BOM file:" << localPath;
321 emit bomResultChanged();
322 return;
323 }
324
325 // 使用批量添加方法
326 QStringList newIds;
327 int skipped = 0;
328
329 for (const QString& id : componentIds) {
330 if (!componentExists(id)) {
331 newIds.append(id);
332 } else {
333 skipped++;
334 }
335 }
336
337 if (!newIds.isEmpty()) {
338 addComponentsBatch(newIds);
339 }
340
341 QString resultMsg = QString("BOM file imported: %1 components added, %2 skipped").arg(newIds.count()).arg(skipped);
342 m_bomResult = resultMsg;
343 qDebug() << resultMsg;
344 emit bomResultChanged();
345}
