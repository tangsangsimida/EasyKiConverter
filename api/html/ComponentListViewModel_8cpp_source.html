303void ComponentListViewModel::removeComponent(int index) {
304 clearAttentionHints();
305
306 // 检查是否是最后一个元器件（需要锁保护）
307 int listCount;
308 {
309 QMutexLocker locker(&m_listMutex);
310 listCount = m_componentList.count();
311 if (index < 0 || index >= listCount) {
312 return;
313 }
314 }
315
316 // 当删除最后一个元器件时，触发与清空列表相同的效果
317 if (listCount == 1) {
318 clearComponentList();
319 return;
320 }
321
322 // 获取要删除的 item 和 ID（需要锁保护）
323 ComponentListItemData* item = nullptr;
324 QString removedId;
325
326 {
327 QMutexLocker locker(&m_listMutex);
328 item = m_componentList.takeAt(index);
329 removedId = item->componentId();
330 m_componentIdIndex.remove(removedId);
331 m_validatedComponentIds.removeAll(removedId);
332 }
333
334 // 取消该元器件的网络请求
335 if (m_service) {
336 m_service->cancelRequestForComponent(removedId);
337 }
338
339 // 检查元器件是否已经在飞行中（已 dispatch）
340 bool wasInFlight = m_inFlightComponentIds.contains(removedId);
341
342 // 从验证队列和飞行中列表中移除
343 m_validationQueue.removeAll(removedId);
344 m_inFlightComponentIds.remove(removedId);
345
346 // 检查是否正在获取中（锁外操作，因为 m_pendingValidationCount 有自己的逻辑）
347 if (item->isFetching()) {
348 m_pendingValidationCount--;
349 // 只有当元器件已经在飞行中（已 dispatch）时才减少 m_validationPendingCount
350 // 因为 m_validationPendingCount 只在 dispatch 时增加
351 if (wasInFlight) {
352 m_validationPendingCount--;
353 }
354 // 通知 ValidationStateManager 取消验证
355 m_validationStateManager->cancelValidation(1);
356 }
357
358 beginRemoveRows(QModelIndex(), index, index);
359 delete item;
360 endRemoveRows();
361
362 rebuildComponentIdIndex();
363 updateHasInvalidComponents();
364
365 scheduleListUpdate();
366 emit componentRemoved(removedId);
367}

437 clearAttentionHints();

439 for (const QString& rawId : componentIds) {
440 QString id = rawId.trimmed().toUpper();
441 if (id.isEmpty())

477void ComponentListViewModel::processNextBatchAdd() {
478 if (m_pendingComponentIds.isEmpty()) {
479 m_batchAddTimer->stop();
480 return;
481 }
482
483 QStringList batch;
484 int count = qMin(BATCH_ADD_SIZE, m_pendingComponentIds.size());
485 for (int i = 0; i < count; ++i) {
486 batch.append(m_pendingComponentIds.takeFirst());
487 }
488
489 int pendingCount = batch.count();
490
491 // 获取起始索引（需要锁保护）
492 int startIndex;
493 bool isFirstBatch;
494 bool isLastBatch;
495 {
496 QMutexLocker locker(&m_listMutex);
497 startIndex = m_componentList.count();
498 isFirstBatch = (startIndex == 0);
499 isLastBatch = m_pendingComponentIds.isEmpty();
500 }
501
502 // 创建新的 items（在锁外创建，因为不涉及共享数据）
503 QList<ComponentListItemData*> newItems;
504 for (const QString& id : batch) {
505 auto item = new ComponentListItemData(id, this);
506 item->setFetching(true);
507 item->setValid(false);
508 item->setValidationPhase("validating");
509 newItems.append(item);
510 }
511
512 // 通知视图即将插入
513 beginInsertRows(QModelIndex(), startIndex, startIndex + pendingCount - 1);
514
515 // 修改列表（需要锁保护）
516 {
517 QMutexLocker locker(&m_listMutex);
518 for (auto item : newItems) {
519 m_componentList.append(item);
520 m_componentIdIndex.insert(item->componentId(), m_componentList.count() - 1);
521 }
522 }
523
524 // 通知视图插入完成
525 endInsertRows();
526
527 // 只有在最后一批时才启动验证和刷新统计，避免 BOM 导入时 UI 与验证并发抖动
528 if (isLastBatch) {
529 const int validationCount = m_pendingBatchValidationCount;
530 m_pendingBatchValidationCount = 0;
531
532 // BOM 导入模式结束，恢复正常 UI 更新
533 if (m_bomImportMode) {
534 // 保存状态以便后续处理
535 bool hadPendingUpdates = (m_bomImportPendingUpdates > 0);
536 m_bomImportMode = false;
537 // 如果有累积的验证完成计数，触发一次 UI 更新
538 if (hadPendingUpdates) {
539 m_bomImportPendingUpdates = 0;
540 scheduleListUpdate();
541 }
542 }
543
544 if (validationCount > 0) {
545 if (m_validationStateManager->pendingCount() > 0 || !m_validationQueue.isEmpty() ||
546 !m_inFlightComponentIds.isEmpty()) {
547 m_validationStateManager->addValidation(validationCount);
548 } else {
549 m_validationStateManager->startValidation(validationCount);
550 }
551 }
552
553 // BOM 导入完成，标记并启动验证队列
554 m_bomImportComplete = true;
555 startValidationQueue();
556 }
557
558 scheduleListUpdate();
559}

561void ComponentListViewModel::startValidationQueue() {
562 const int CONCURRENT_WORKERS = ConfigService::instance()->getValidationConcurrentCount();
563
564 // 检查需要添加到队列的新组件
565 for (auto item : m_componentList) {
566 // 只添加正在验证的组件，排除已验证完成或已验证失败的组件
567 // 注意：isFetching() 在 handleCadDataReady 时立即设为 false，
568 // 但 onValidationComplete 通过 singleShot(0) 延迟调用，
569 // 所以需要额外检查 isValid() 来排除已完成的组件
570 if (item->isFetching() && !item->isValid()) {
571 QString componentId = item->componentId();
572 // 检查是否已经在队列中或正在处理中（飞行中），避免重复添加
573 if (!m_validationQueue.contains(componentId) && !m_inFlightComponentIds.contains(componentId)) {
574 m_validationQueue.append(componentId);
575 }
576 }
577 }
578
579 // 如果没有待验证的组件，直接返回
580 if (m_validationQueue.isEmpty()) {
581 return;
582 }
583
584 // 如果 m_validationTotalCount 为 0，说明是首次启动，设置总数
585 // 否则说明是延续之前的处理，不需要额外设置（已在 pending 中）
586 if (m_validationTotalCount == 0) {
587 m_validationTotalCount = m_validationQueue.count();
588 }
589
590 // 启动并发验证 worker（如果当前没有活跃的 worker）
591 int initialCount = qMin(CONCURRENT_WORKERS, m_validationQueue.count());
592 for (int i = 0; i < initialCount; ++i) {
593 QString componentId = m_validationQueue.takeFirst();
594 m_inFlightComponentIds.insert(componentId); // 标记为飞行中
595 m_service->fetchComponentData(componentId, false);
596 m_validationPendingCount++;
597 }
598}

611void ComponentListViewModel::onValidationComplete(const QString& componentId) {
612 m_validationCompletedCount++;
613 m_inFlightComponentIds.remove(componentId); // 从飞行中移除
614
615 if (m_validationPendingCount > 0) {
616 m_validationPendingCount--;
617 }
618
619 // BOM 导入完成后，跳过 scheduleListUpdate 和 startValidationQueue
620 // 避免频繁调用导致 O(n²) 复杂度和 UI 阻塞
621 // 验证流程已在 processNextBatchAdd -> startValidationQueue 中启动
622 if (m_bomImportComplete) {
623 // BOM 导入已完成，只处理队列中的下一个验证
624 if (!m_validationQueue.isEmpty()) {
625 processNextValidation();
626 } else if (m_validationPendingCount == 0) {
627 m_bomImportComplete = false;
628 }
629 return;
630 }
631
632 // BOM 导入模式下，跳过频繁的 scheduleListUpdate，避免 UI 阻塞
633 // 只在最后所有验证完成时再更新 UI
634 if (!m_bomImportMode) {
635 scheduleListUpdate();
636 } else {
637 m_bomImportPendingUpdates++;
638 }
639
640 if (!m_validationQueue.isEmpty()) {
641 processNextValidation();
642 } else {
643 // 队列为空，尝试补充新组件（可能在验证期间新添加的）
644 startValidationQueue();

646 // 不再使用计数器判断，因为计数器存在溢出问题
647 }

848void ComponentListViewModel::handleFetchError(const QString& componentId, const QString& error) {
849 qWarning() << "Fetch error for:" << componentId << "-" << error;
850
851 auto item = findItemData(componentId);
852
853 if (item) {
854 // 重要：预览图获取失败不应该改变元器件的验证状态
855 // 元器件的验证状态只由 CAD 数据获取结果决定
856 // 如果组件已经验证通过（isValid 为 true），保持验证状态不变
857 bool wasAlreadyValid = item->isValid();
858
859 item->setFetching(false);
860
861 if (!wasAlreadyValid) {
862 // 组件尚未验证通过，说明 CAD 数据获取失败或还未完成
863 // 判断是否是 CAD 数据获取失败
864 // 扩大判断范围：包括网络错误和 HTTP 错误，因为这些也可能导致 CAD 数据获取失败
865 bool isCadDataFailure =
866 error.contains("CAD data") || error.contains("Symbol data") || error.contains("Footprint data") ||
867 error.contains("Empty CAD") || error.contains("parse.*EasyEDA") ||
868 error.contains("No result", Qt::CaseInsensitive) ||
869 // API 业务层错误
870 error.contains("API returned success=false", Qt::CaseInsensitive) ||
871 error.contains("API response missing result field", Qt::CaseInsensitive) ||
872 // 网络相关错误（CAD 数据获取也可能触发这些）
873 error.contains("403") || error.contains("404") || error.contains("timeout", Qt::CaseInsensitive) ||
874 error.contains("access denied", Qt::CaseInsensitive) ||
875 error.contains("forbidden", Qt::CaseInsensitive) || error.contains("not found", Qt::CaseInsensitive) ||
876 error.contains("connection closed", Qt::CaseInsensitive) ||
877 error.contains("operation canceled", Qt::CaseInsensitive) ||
878 error.contains("Request cancelled", Qt::CaseInsensitive) ||
879 error.contains("network error", Qt::CaseInsensitive) ||
880 error.contains("fetch error", Qt::CaseInsensitive);
881
882 if (isCadDataFailure) {
883 // CAD 数据获取失败，标记为验证失败
884 item->setValid(false);
885 item->setValidationPhase("failed");
886 const bool nonRetryable = ComponentListViewModel::isNonRetryableValidationError(error);
887 item->setRetryable(!nonRetryable);
888 if (error.contains("No result", Qt::CaseInsensitive) ||
889 error.contains("not found", Qt::CaseInsensitive) || error.contains("404", Qt::CaseInsensitive)) {
890 item->setErrorMessage(tr("元器件不存在（404）"));
891 } else if (nonRetryable) {
892 item->setErrorMessage(error);
893 } else {
894 item->setErrorMessage(error);
895 }
896 m_validationStateManager->onComponentFailed(componentId);
897 scheduleListUpdate();
898 QTimer::singleShot(0, this, [this, componentId]() { onValidationComplete(componentId); });
899 } else {
900 // 预览图获取失败，保持验证状态（可能还未验证完成）
901 // 只更新错误消息，不改变验证状态
902 if (error.contains("Request timeout", Qt::CaseInsensitive) ||
903 error.contains("timeout", Qt::CaseInsensitive)) {
904 item->setErrorMessage(tr("预览图获取超时（网络不稳定）"));
905 } else if (error.contains("No result", Qt::CaseInsensitive) || error.contains("404") ||
906 error.contains("not found", Qt::CaseInsensitive)) {
907 item->setErrorMessage(tr("预览图不存在"));
908 } else if (error.contains("403")) {
909 item->setErrorMessage(tr("预览图获取被拒绝"));
910 } else {
911 item->setErrorMessage(tr("预览图获取失败"));
912 }
913 }
914 } else {
915 // 组件已经验证通过，只是预览图获取失败，保持验证状态不变
916 }
917
918 scheduleListUpdate();
919 }
