24CadParseResult CadDataLoader::parseCadPayload(const QString& componentId, const QJsonObject& rawData) {
25 CadParseResult parsed;
26 parsed.componentId = componentId;
27
28 QJsonObject resultData = rawData;
29 if (rawData.contains("result")) {
30 resultData = rawData.value("result").toObject();
31 }
32
33 if (resultData.isEmpty()) {
34 parsed.errorMessage = QStringLiteral("Empty CAD data");
35 return parsed;
36 }
37
38 ComponentData componentData;
39 componentData.setLcscId(componentId);
40 componentData.setName(resultData.value("title").toString());
41 componentData.setPackage(resultData.value("package").toString());
42 componentData.setManufacturer(resultData.value("manufacturer").toString());
43 componentData.setDatasheet(resultData.value("datasheet").toString());
44
45 EasyedaSymbolImporter symbolImporter;
46 QSharedPointer<SymbolData> symbolData = symbolImporter.importSymbolData(resultData);
47 if (!symbolData) {
48 parsed.errorMessage = QStringLiteral("Failed to parse Symbol data from EasyEDA JSON");
49 return parsed;
50 }
51 componentData.setSymbolData(symbolData);
52
53 if (!symbolData->info().thumb.isEmpty()) {
54 const QString thumbUrl = UrlUtils::normalizePreviewImageUrl(symbolData->info().thumb);
55 componentData.setPreviewImages(QStringList() << thumbUrl);
56 }
57
58 if (!symbolData->info().datasheet.isEmpty() && componentData.datasheet().isEmpty()) {
59 const QString datasheetUrl = UrlUtils::normalizePreviewImageUrl(symbolData->info().datasheet);
60 componentData.setDatasheet(datasheetUrl);
61 }
62
63 EasyedaFootprintImporter footprintImporter;
64 QSharedPointer<FootprintData> footprintData = footprintImporter.importFootprintData(resultData);
65 if (!footprintData) {
66 parsed.errorMessage =
67 QStringLiteral("Failed to parse Footprint data from EasyEDA JSON (Library might be corrupted)");
68 return parsed;
69 }
70 componentData.setFootprintData(footprintData);
71
72 // 优先使用 head.uuid_3d（EasyEDA API 3D 模型端点使用的规范 UUID），
73 // 其次使用 SVGNODE outline3D 的 uuid（可能是不同 UUID）。
74 QString modelUuid;
75 if (resultData.contains("dataStr")) {
76 const QJsonObject dataStr = resultData.value("dataStr").toObject();
77 modelUuid = dataStr.value("head").toObject().value("uuid_3d").toString();
78 }
79 if (modelUuid.isEmpty()) {
80 modelUuid = footprintData->model3D().uuid();
81 }
82 if (modelUuid.isEmpty() && resultData.contains("dataStr")) {
83 const QJsonObject dataStr = resultData.value("dataStr").toObject();
84 const QJsonArray shapes = dataStr.value("shape").toArray();
85 for (const QJsonValue& shapeVal : shapes) {
86 const QString shapeStr = shapeVal.toString();
87 if (!shapeStr.startsWith(QStringLiteral("SVGNODE~"))) {
88 continue;
89 }
90
91 const int tildeIndex = shapeStr.indexOf('~');
92 if (tildeIndex < 0) {
93 continue;
94 }
95
96 const QJsonDocument nodeDoc = QJsonDocument::fromJson(shapeStr.mid(tildeIndex + 1).toUtf8());
97 const QJsonObject attrs = nodeDoc.object().value("attrs").toObject();
98 if (attrs.value("c_etype").toString() == QLatin1String("outline3D")) {
99 modelUuid = attrs.value("uuid").toString();
100 if (!modelUuid.isEmpty()) {
101 break;
102 }
103 }
104 }
105 }
106
107 if (!modelUuid.isEmpty()) {
108 auto model3DData = QSharedPointer<Model3DData>::create();
109 model3DData->setUuid(modelUuid);
110 model3DData->setName(footprintData->model3D().name());
111 model3DData->setTranslation(footprintData->model3D().translation());
112 model3DData->setRotation(footprintData->model3D().rotation());
113 componentData.setModel3DData(model3DData);
114 }
115
116 parsed.resultData = resultData;
117 parsed.componentData = componentData;
118 parsed.success = true;
119 return parsed;
120}

122CadFetchTaskResult CadDataLoader::fetchAndParseCadData(const QString& componentId, INetworkClient* networkClient) {
123 CadFetchTaskResult taskResult;
124 taskResult.componentId = componentId;
125
126 const RequestProfile profile = RequestProfiles::cadData();
127 const bool weakNetworkSupport = networkClient ? false : ConfigService::instance()->getWeakNetworkSupport();
128 const RetryPolicy policy = RetryPolicy::fromProfile(profile, weakNetworkSupport);
129 const QUrl url(EASYEDA_COMPONENT_API_ENDPOINT.arg(componentId));
130 const NetworkResult networkResult = networkClient
131 ? networkClient->get(url, ResourceType::CadData, policy)
132 : NetworkClient::instance().get(url, ResourceType::CadData, policy);
133 if (!networkResult.success) {
134 taskResult.errorMessage =
135 networkResult.error.isEmpty() ? QStringLiteral("CAD request failed") : networkResult.error;
136 return taskResult;
137 }
138
139 QJsonParseError parseError;
140 const QJsonDocument jsonDoc = QJsonDocument::fromJson(networkResult.data, &parseError);
141 if (parseError.error != QJsonParseError::NoError || !jsonDoc.isObject()) {
142 taskResult.errorMessage = QStringLiteral("Failed to parse CAD response JSON");
143 return taskResult;
144 }
145
146 const QJsonObject root = jsonDoc.object();
147
148 // 检查业务层错误：success == false
149 if (root.contains("success") && !root.value("success").toBool()) {
150 const int code = root.value("code").toInt();
151 const QString message = root.value("message").toString();
152 if (code == 404) {
153 taskResult.errorMessage = QStringLiteral("元器件不存在（404）");
154 } else {
155 taskResult.errorMessage = message.isEmpty() ? QStringLiteral("API returned success=false") : message;
156 }
157 return taskResult;
158 }
159
160 // 检查 result 字段是否缺失或为 null
161 if (!root.contains("result") || root.value("result").isNull()) {
162 taskResult.errorMessage = QStringLiteral("API response missing result field");
163 return taskResult;
164 }
165
166 // 检查 result 是否为空对象
167 if (root.value("result").isObject() && root.value("result").toObject().isEmpty()) {
168 taskResult.errorMessage = QStringLiteral("Empty CAD data");
169 return taskResult;
170 }
171
172 taskResult.parsed = parseCadPayload(componentId, root);
173 if (!taskResult.parsed.success) {
174 taskResult.errorMessage = taskResult.parsed.errorMessage;
175 return taskResult;
176 }
177
178 taskResult.success = true;
179 return taskResult;
180}
