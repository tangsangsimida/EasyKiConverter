105void EasyedaApi::handleAsyncRequestFinished(AsyncNetworkRequest* request,
106 const QString& id,
107 ResourceType resourceType,
108 bool isBinary) {
109 const NetworkResult result = request->result();
110
111 {
112 QMutexLocker locker(&m_requestsMutex);
113 m_activeRequests.removeOne(QPointer<AsyncNetworkRequest>(request));
114 }
115
116 if (result.wasCancelled) {
117 if (resourceType == ResourceType::ComponentInfo) {
118 m_isFetching = false;
119 }
120 emit fetchError(id, "Request cancelled");
121 request->deleteLater();
122 return;
123 }
124
125 if (!result.success) {
126 if (resourceType == ResourceType::ComponentInfo) {
127 m_isFetching = false;
128 }
129 emit fetchError(id, result.error);
130 request->deleteLater();
131 return;
132 }
133
134 if (isBinary) {
135 emit model3DFetched(id, result.data);
136 request->deleteLater();
137 return;
138 }
139
140 QJsonParseError parseError;
141 QJsonDocument jsonDoc = QJsonDocument::fromJson(result.data, &parseError);
142 if (parseError.error != QJsonParseError::NoError) {
143 emit fetchError(id, QString("Failed to parse JSON response: %1").arg(parseError.errorString()));
144 request->deleteLater();
145 return;
146 }
147
148 if (!jsonDoc.isObject()) {
149 emit fetchError(id, "JSON response is not an object");
150 request->deleteLater();
151 return;
152 }
153
154 const QJsonObject jsonObject = jsonDoc.object();
155 switch (resourceType) {
156 case ResourceType::ComponentInfo:
157 m_isFetching = false;
158 handleComponentInfoResponse(id, jsonObject);
159 break;
160 case ResourceType::CadData:
161 handleCadDataResponse(id, jsonObject);
162 break;
163 default:
164 emit fetchError(id, "Unsupported JSON resource type");
165 break;
166 }
167
168 request->deleteLater();
169}
