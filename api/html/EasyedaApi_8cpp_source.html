236void EasyedaApi::cancelRequest() {
237 // Use QPointer to safely check if m_networkUtils is still valid
238 // (it might be during destruction when parent-child relationships are being torn down)
239 QPointer<INetworkAdapter> networkUtilsPtr(m_networkUtils);
240 if (networkUtilsPtr && !networkUtilsPtr.isNull()) {
241 networkUtilsPtr->cancelRequest();
242 }
243 m_isFetching = false;
244
245 QMutexLocker locker(&m_requestsMutex);
246 for (auto& req : m_activeRequests) {
247 if (req && !req.isNull()) {
248 req->cancelRequest();
249 }
250 }
251 m_activeRequests.clear();
252}
