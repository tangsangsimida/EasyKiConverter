255void ExportProgressViewModel::cancelExport() {
256 if (!m_exportService)
257 return;
258
259 qDebug() << "ExportProgressViewModel::cancelExport() called";
260
261 // 1. 瞬时响应：立即通知后端中断（原子操作，<1ms）
262 m_exportService->cancelExport();
263
264 // 2. 立即更新按钮状态（同步执行，<1ms）
265 m_isStopping = true;
266 emit isStoppingChanged();
267 m_status = "Stopping export...";
268 emit statusChanged();
269
270 qDebug() << "Cancel operation initiated in <1ms";
271
272 // 3. 异步处理 UI 列表更新（不阻塞取消操作）
273 QTimer::singleShot(0, this, [this]() {
274 bool updated = false;
275 for (int i = 0; i < m_resultsList.size(); ++i) {
276 QVariantMap item = m_resultsList[i].toMap();
277 QString status = item["status"].toString();
278
279 // 更新所有未完成的状态 (v3.0.5+ 改进)
280 if (status == "pending" || status == "fetching" || status == "processing" || status == "writing") {
281 item["status"] = "failed";
282 item["message"] = "Export cancelled";
283 // CRITICAL FIX: 在取消时，将所有分项成功标志也重置为 false
284 item["symbolSuccess"] = false;
285 item["footprintSuccess"] = false;
286 item["model3DSuccess"] = false;
287 m_resultsList[i] = item;
288 updated = true;
289 }
290 }
291
292 if (updated) {
293 emit resultsListChanged();
294 updateStatistics();
295 }
296 qDebug() << "UI update completed after cancellation";
297 });
298}

380QString ExportProgressViewModel::getStatusString(int stage,
381 bool success,
382 bool symbolSuccess,
383 bool footprintSuccess,
384 bool model3DSuccess) const {
385 if (!success) {
386 return "failed";
387 }
388
389 switch (stage) {
390 case 0:
391 return "fetch_completed";
392 case 1:
393 return "processing";
394 case 2: {
395 // 严格成功判定：检查用户请求的所有项是否都已成功写入
396 bool allRequestedItemsDone = true;
397 if (m_exportOptions.exportSymbol && !symbolSuccess)
398 allRequestedItemsDone = false;
399 if (m_exportOptions.exportFootprint && !footprintSuccess)
400 allRequestedItemsDone = false;
401 if (m_exportOptions.exportModel3D && !model3DSuccess)
402 allRequestedItemsDone = false;
403
404 return allRequestedItemsDone ? "success" : "failed";
405 }
406 default:
407 return "success";
408 }
409}

452void ExportProgressViewModel::updateStatistics() {

455 int globalSymbol = 0;

468

473 globalFootprint++;
474 if (m_exportOptions.exportModel3D && item["model3DSuccess"].toBool())
475 globalModel3D++;
476 }
477
478 if (m_successCount != globalSuccess) {
479 m_successCount = globalSuccess;
480 emit successCountChanged();
481 }
482
483 if (m_failureCount != globalFailed) {
484 m_failureCount = globalFailed;
485 emit failureCountChanged();
486 }
487
488 // 更新分项计数
489 if (m_successSymbolCount != globalSymbol || m_successFootprintCount != globalFootprint ||
490 m_successModel3DCount != globalModel3D) {
491 m_successSymbolCount = globalSymbol;
492 m_successFootprintCount = globalFootprint;
493 m_successModel3DCount = globalModel3D;
494 emit successCountChanged(); // 触发 UI 刷新

632void ExportProgressViewModel::startExportInternal(const QStringList& componentIds, bool isRetry) {
633 if (!m_exportService || !m_componentService)
634 return;
635
636 m_isExporting = true;
637 m_progress = 0;
638 m_fetchedCount = 0;
639 m_fetchProgress = 0;
640 m_processProgress = 0;
641 m_writeProgress = 0;
642 m_componentIds = componentIds;
643 m_collectedData.clear();
644 m_status = isRetry ? "Retrying..." : "Starting export...";
645
646 if (!isRetry) {
647 m_successCount = 0;
648 m_failureCount = 0;
649 prepopulateResultsList(componentIds);
650 } else {
651 // 重试模式：仅更新被重试项的状态
652 for (const QString& id : componentIds) {
653 int index = m_idToIndexMap.value(id, -1);
654 if (index >= 0) {
655 QVariantMap item = m_resultsList[index].toMap();
656 item["status"] = "pending";
657 item["message"] = "Waiting to retry...";
658 m_resultsList[index] = item;
659 }
660 }
661 }
662
663 // 无论是新开始还是重试，都先执行一次全量统计同步
664 updateStatistics();
665 emit resultsListChanged();
666 emit isExportingChanged();
667 emit progressChanged();
668 emit statusChanged();
669
670 if (isRetry) {
671 m_exportService->retryExport(componentIds, m_exportOptions);
672 } else {
673 if (m_usePipelineMode && qobject_cast<ExportServicePipeline*>(m_exportService)) {
674 auto* pipelineService = qobject_cast<ExportServicePipeline*>(m_exportService);
675
676 // Gather preloaded data
677 if (m_componentListViewModel) {
678 QMap<QString, QSharedPointer<ComponentData>> preloadedData;
679 for (const QString& id : componentIds) {
680 auto data = m_componentListViewModel->getPreloadedData(id);
681 if (data) {
682 qDebug() << "Preloaded component:" << id
683 << "PreviewImages (URLs):" << data->previewImages().size()
684 << "PreviewImageData (bytes):" << data->previewImageData().size() << "Datasheet (URL):"
685 << (data->datasheet().isEmpty() ? "empty" : data->datasheet().left(50))
686 << "DatasheetData (bytes):" << data->datasheetData().size();
687 preloadedData.insert(id, data);
688 }
689 }
690 if (!preloadedData.isEmpty()) {
691 qDebug() << "Passing" << preloadedData.size() << "preloaded components to pipeline";
692 pipelineService->setPreloadedData(preloadedData);
693 pipelineService->setComponentListViewModel(m_componentListViewModel);
694 } else {
695 qDebug() << "No preloaded data available";
696 }
697 }
698
699 pipelineService->executeExportPipelineWithStages(componentIds, m_exportOptions);
700 } else {
701 m_componentService->setOutputPath(m_exportOptions.outputPath);
702 m_componentService->fetchMultipleComponentsData(componentIds, m_exportOptions.exportModel3D);
703 }
704 }

707void ExportProgressViewModel::showExportCompleteNotification() {
708 if (!m_systemTrayIcon) {
709 qDebug() << "System tray icon not initialized";
710 return;
711 }
712
713 // 检查是否支持消息
714 if (!QSystemTrayIcon::supportsMessages()) {
715 qDebug() << "System tray does not support messages on this platform";
716 return;
717 }
718
719 // 准备通知内容
720 QString title;
721 QString message;
722 QSystemTrayIcon::MessageIcon iconType = QSystemTrayIcon::Information;
723
724 if (m_failureCount > 0) {
725 // 有失败的情况
726 title = QObject::tr("导出完成");
727
728 // 计算成功率
729 double successRate = m_resultsList.size() > 0 ? (m_successCount * 100.0 / m_resultsList.size()) : 0.0;
730
731 if (m_successCount == 0) {
732 // 全部失败
733 iconType = QSystemTrayIcon::Critical;
734 message = QObject::tr("导出失败：%1 个元器件全部失败").arg(m_resultsList.size());

766 .arg(m_successSymbolCount)
767 .arg(m_successFootprintCount)
768 .arg(m_successModel3DCount);

770 // 添加耗时信息

774 if (durationSec < 60.0) {

778 double seconds = durationSec - (minutes * 60.0);

782 }

784 message += "\n" + stats.join("\n");
785 }

790 m_systemTrayIcon->showMessage(title, message, iconType, 12000);

792 qDebug() << "Notification shown successfully";
