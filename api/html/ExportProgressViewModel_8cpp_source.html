163void ExportProgressViewModel::cancelExport() {
164 if (!m_exportService)
165 return;
166
167 qDebug() << "ExportProgressViewModel::cancelExport() called";
168
169 // 1. 瞬时响应：立即通知后端中断（原子操作，<1ms）
170 m_exportService->cancelExport();

172 // 2. 立即更新按钮状态（同步执行，<1ms）
173 m_isStopping = true;
174 emit isStoppingChanged();
175 m_status = "Stopping export...";
176 emit statusChanged();
177
178 qDebug() << "Cancel operation initiated in <1ms";
179
180 // 3. 异步处理 UI 列表更新（不阻塞取消操作）
181 QTimer::singleShot(0, this, [this]() {
182 bool updated = false;
183 for (int i = 0; i < m_resultsList.size(); ++i) {
184 QVariantMap item = m_resultsList[i].toMap();
185 QString status = item["status"].toString();
186
187 // 更新所有未完成的状态 (v3.0.5+ 改进)
188 if (status == "pending" || status == "fetching" || status == "processing" || status == "writing") {
189 item["status"] = "failed";
190 item["message"] = "Export cancelled";
191 // CRITICAL FIX: 在取消时，将所有分项成功标志也重置为 false
192 item["symbolSuccess"] = false;
193 item["footprintSuccess"] = false;
194 item["model3DSuccess"] = false;
195 m_resultsList[i] = item;
196 updated = true;
197 }
198 }
199

201 emit resultsListChanged();

203 }

288QString ExportProgressViewModel::getStatusString(int stage,
289 bool success,
290 bool symbolSuccess,
291 bool footprintSuccess,
292 bool model3DSuccess) const {
293 if (!success) {
294 return "failed";
295 }
296
297 switch (stage) {
298 case 0:
299 return "fetch_completed";
300 case 1:
301 return "processing";
302 case 2: {
303 // 严格成功判定：检查用户请求的所有项是否都已成功写入
304 bool allRequestedItemsDone = true;
305 if (m_exportOptions.exportSymbol && !symbolSuccess)
306 allRequestedItemsDone = false;
307 if (m_exportOptions.exportFootprint && !footprintSuccess)
308 allRequestedItemsDone = false;
309 if (m_exportOptions.exportModel3D && !model3DSuccess)
310 allRequestedItemsDone = false;
311
312 return allRequestedItemsDone ? "success" : "failed";
313 }
314 default:
315 return "success";
316 }

321 const QString& message,
322 int stage,
323 bool symbolSuccess,
324 bool footprintSuccess,

329
330 if (index >= 0 && index < m_resultsList.size()) {
331 QVariantMap result = m_resultsList[index].toMap();

336 result["footprintSuccess"] = footprintSuccess;
337 result["model3DSuccess"] = model3DSuccess;
338 m_resultsList[index] = result;

342 result["status"] = statusStr;

345 result["footprintSuccess"] = footprintSuccess;
346 result["model3DSuccess"] = model3DSuccess;

349 }

360void ExportProgressViewModel::updateStatistics() {
361 int globalSuccess = 0;
362 int globalFailed = 0;
363 int globalSymbol = 0;
364 int globalFootprint = 0;
365 int globalModel3D = 0;
366
367 for (const auto& var : m_resultsList) {
368 QVariantMap item = var.toMap();
369 QString status = item["status"].toString();
370
371 if (status == "success") {
372 globalSuccess++;
373 } else if (status == "failed") {
374 globalFailed++;
375 }
376
377 // 累加分项成功数（只有当该类型被启用时才计数）
378 if (m_exportOptions.exportSymbol && item["symbolSuccess"].toBool())
379 globalSymbol++;
380 if (m_exportOptions.exportFootprint && item["footprintSuccess"].toBool())
381 globalFootprint++;
382 if (m_exportOptions.exportModel3D && item["model3DSuccess"].toBool())
383 globalModel3D++;
384 }
385
386 if (m_successCount != globalSuccess) {
387 m_successCount = globalSuccess;
388 emit successCountChanged();
389 }
390
391 if (m_failureCount != globalFailed) {
392 m_failureCount = globalFailed;
393 emit failureCountChanged();
394 }
395
396 // 更新分项计数
397 if (m_successSymbolCount != globalSymbol || m_successFootprintCount != globalFootprint ||
398 m_successModel3DCount != globalModel3D) {
399 m_successSymbolCount = globalSymbol;
400 m_successFootprintCount = globalFootprint;
401 m_successModel3DCount = globalModel3D;
402 emit successCountChanged(); // 触发 UI 刷新

540void ExportProgressViewModel::startExportInternal(const QStringList& componentIds, bool isRetry) {
541 if (!m_exportService || !m_componentService)
542 return;
543
544 m_isExporting = true;
545 m_progress = 0;
546 m_fetchedCount = 0;
547 m_fetchProgress = 0;
548 m_processProgress = 0;
549 m_writeProgress = 0;
550 m_componentIds = componentIds;
551 m_collectedData.clear();
552 m_status = isRetry ? "Retrying..." : "Starting export...";
553
554 if (!isRetry) {
555 m_successCount = 0;
556 m_failureCount = 0;
557 prepopulateResultsList(componentIds);
558 } else {
559 // 重试模式：仅更新被重试项的状态
560 for (const QString& id : componentIds) {
561 int index = m_idToIndexMap.value(id, -1);
562 if (index >= 0) {
563 QVariantMap item = m_resultsList[index].toMap();
564 item["status"] = "pending";
565 item["message"] = "Waiting to retry...";
566 m_resultsList[index] = item;
567 }
568 }
569 }
570
571 // 无论是新开始还是重试，都先执行一次全量统计同步
572 updateStatistics();
573 emit resultsListChanged();
574 emit isExportingChanged();
575 emit progressChanged();
576 emit statusChanged();
577
578 if (isRetry) {
579 m_exportService->retryExport(componentIds, m_exportOptions);
580 } else {
581 if (m_usePipelineMode && qobject_cast<ExportServicePipeline*>(m_exportService)) {
582 auto* pipelineService = qobject_cast<ExportServicePipeline*>(m_exportService);
583
584 // Gather preloaded data
585 if (m_componentListViewModel) {
586 QMap<QString, QSharedPointer<ComponentData>> preloadedData;
587 for (const QString& id : componentIds) {
588 auto data = m_componentListViewModel->getPreloadedData(id);
589 if (data) {
590 preloadedData.insert(id, data);
591 }
592 }
593 if (!preloadedData.isEmpty()) {
594 qDebug() << "Passing" << preloadedData.size() << "preloaded components to pipeline";
595 pipelineService->setPreloadedData(preloadedData);
596 }
597 }
598
599 pipelineService->executeExportPipelineWithStages(componentIds, m_exportOptions);
600 } else {
601 m_componentService->setOutputPath(m_exportOptions.outputPath);
602 m_componentService->fetchMultipleComponentsData(componentIds, m_exportOptions.exportModel3D);
603 }
604 }

607void ExportProgressViewModel::showExportCompleteNotification() {
608 if (!m_systemTrayIcon) {
609 qDebug() << "System tray icon not initialized";
610 return;
611 }
612
613 // 检查是否支持消息
614 if (!QSystemTrayIcon::supportsMessages()) {
615 qDebug() << "System tray does not support messages on this platform";
616 return;
617 }
618
619 // 准备通知内容
620 QString title;
621 QString message;
622 QSystemTrayIcon::MessageIcon iconType = QSystemTrayIcon::Information;
623
624 if (m_failureCount > 0) {
625 // 有失败的情况
626 title = QObject::tr("导出完成");
627
628 // 计算成功率
629 double successRate = m_resultsList.size() > 0 ? (m_successCount * 100.0 / m_resultsList.size()) : 0.0;
630
631 if (m_successCount == 0) {
632 // 全部失败
633 iconType = QSystemTrayIcon::Critical;
634 message = QObject::tr("导出失败：%1 个元器件全部失败").arg(m_resultsList.size());
635 } else if (successRate < 50.0) {
636 // 失败较多
637 iconType = QSystemTrayIcon::Warning;
638 message = QObject::tr("成功 %1 个，失败 %2 个").arg(m_successCount).arg(m_failureCount);
639 } else {
640 // 部分失败
641 message = QObject::tr("成功 %1 个，失败 %2 个").arg(m_successCount).arg(m_failureCount);
642 }
643
644 // 添加详细的输出统计
645 if (m_successCount > 0) {
646 message += QObject::tr("\n输出：符号 %1 · 封装 %2 · 3D %3")
647 .arg(m_successSymbolCount)
648 .arg(m_successFootprintCount)
649 .arg(m_successModel3DCount);
650 }
651 } else {
652 // 全部成功
653 iconType = QSystemTrayIcon::Information;
654 title = QObject::tr("导出完成");
655
656 // 根据成功数量显示不同的成功消息
657 if (m_successCount == 1) {
658 message = QObject::tr("成功导出 1 个元器件");
659 } else {
660 message = QObject::tr("成功导出 %1 个元器件").arg(m_successCount);
661 }
662
663 // 添加输出统计和耗时信息
664 QStringList stats;
665 stats << QObject::tr("输出：符号 %1 · 封装 %2 · 3D %3")
666 .arg(m_successSymbolCount)
667 .arg(m_successFootprintCount)
668 .arg(m_successModel3DCount);
669
670 // 添加耗时信息
671 if (m_statistics.totalDurationMs > 0) {
672 double durationSec = m_statistics.totalDurationMs / 1000.0;
673 QString timeStr;
674 if (durationSec < 60.0) {
675 timeStr = QObject::tr("%1 秒").arg(QString::number(durationSec, 'f', 1));
676 } else {
677 int minutes = static_cast<int>(durationSec / 60.0);
678 double seconds = durationSec - (minutes * 60.0);
679 timeStr = QObject::tr("%1 分 %2 秒").arg(minutes).arg(QString::number(seconds, 'f', 0));
680 }
681 stats << QObject::tr("耗时：%1").arg(timeStr);
682 }
683
684 message += "\n" + stats.join("\n");
685 }
686
687 qDebug() << "Attempting to show notification:" << title << message;
688
689 // 显示通知（延长显示时间到 12 秒，让用户有足够时间阅读）
690 m_systemTrayIcon->showMessage(title, message, iconType, 12000);
691
692 qDebug() << "Notification shown successfully";
693}
