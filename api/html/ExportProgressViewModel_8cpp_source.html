222 beginExportRun(componentIds, QStringLiteral("Preloading component data..."));

224 emit failureCountChanged();

227 m_exportService->startPreload(componentIds);

309void ExportProgressViewModel::handleProgressChanged(const ExportOverallProgress& progress) {
310 // Update stage progress values for QML binding
311 switch (progress.currentStage) {
312 case ExportOverallProgress::Stage::Preloading:
313 setStatus("Preloading components...");
314 m_fetchProgress = progress.preloadProgress.percentage();
315 m_processProgress = 0;
316 m_writeProgress = 0;
317 setProgress(weightedOverallProgress());
318 break;
319 case ExportOverallProgress::Stage::Exporting: {
320 setStatus("Exporting components...");
321 // 阶段定义统一为：
322 // 1. 抓取：需要额外获取的数据是否准备完成（预览图/手册/3D源数据）
323 // 2. 处理：符号/封装/3D转换任务是否完成
324 // 3. 写入：最终文件是否已经落盘
325 m_fetchProgress = stageTypeProgress(progress, {TYPE_PREVIEW, TYPE_DATASHEET, TYPE_MODEL3D});
326
327 const int rawProcessProgress = stageTypeProgress(progress, {TYPE_SYMBOL, TYPE_FOOTPRINT, TYPE_MODEL3D});
328 const int rawWriteProgress =
329 stageTypeProgress(progress, {TYPE_SYMBOL, TYPE_FOOTPRINT, TYPE_MODEL3D, TYPE_PREVIEW, TYPE_DATASHEET});
330
331 // UI 上的三段进度条必须遵守抓取 -> 处理 -> 写入的阶段依赖关系。
332 // 即使底层某些类型先完成，后续阶段也不能先于前置阶段显示完成。
333 m_processProgress = qMin(rawProcessProgress, m_fetchProgress);
334 m_writeProgress = qMin(rawWriteProgress, m_processProgress);
335 setProgress(weightedOverallProgress());
336 break;
337 }
338 case ExportOverallProgress::Stage::Completed:
339 m_fetchProgress = 100;
340 m_processProgress = 100;
341 m_writeProgress = 100;
342 setProgress(100);
343 setStatus("Export completed");
344 break;
345 case ExportOverallProgress::Stage::Cancelled:
346 m_fetchProgress = 0;
347 m_processProgress = 0;
348 m_writeProgress = 0;
349 setProgress(0);
350 setStatus("Export cancelled");
351 break;
352 case ExportOverallProgress::Stage::Failed:
353 m_fetchProgress = 0;
354 m_processProgress = 0;
355 m_writeProgress = 0;
356 setStatus("Export failed");
357 break;
358 default:
359 break;
360 }
361
362 emit stageProgressChanged();
363}

365void ExportProgressViewModel::handleItemStatusChanged(const QString& componentId,
366 const QString& typeName,
367 const ExportItemStatus& status) {
368 if (m_idToIndexMap.contains(componentId)) {
369 int index = m_idToIndexMap[componentId];
370 QVariantMap result = m_resultsList[index].toMap();
371
372 // Update individual type success flags
373 if (typeName == "Symbol") {
374 result["symbolSuccess"] = (status.status == ExportItemStatus::Status::Success);
375 } else if (typeName == "Footprint") {
376 result["footprintSuccess"] = (status.status == ExportItemStatus::Status::Success);
377 } else if (typeName == "Model3D") {
378 result["model3DSuccess"] = (status.status == ExportItemStatus::Status::Success);
379 } else if (typeName == "PreviewImages") {
380 result["previewSuccess"] = (status.status == ExportItemStatus::Status::Success);
381 } else if (typeName == "Datasheet") {
382 result["datasheetSuccess"] = (status.status == ExportItemStatus::Status::Success);
383 }
384
385 if (status.status == ExportItemStatus::Status::Failed) {
386 result["error"] = status.errorMessage;
387 }
388
389 QString statusText = "pending";
390 switch (status.status) {
391 case ExportItemStatus::Status::Pending:
392 statusText = "pending";
393 break;
394 case ExportItemStatus::Status::InProgress:
395 statusText = "in_progress";
396 break;
397 case ExportItemStatus::Status::Success:
398 statusText = "success";
399 break;
400 case ExportItemStatus::Status::Failed:
401 statusText = "failed";
402 break;
403 case ExportItemStatus::Status::Skipped:
404 statusText = "skipped";
405 break;
406 }
407 result[typeStatusKey(typeName)] = statusText;
408
409 updateOverallItemStatus(result);
410
411 m_resultsList[index] = result;
412
413 markResultsDirty();
414 }

445void ExportProgressViewModel::handleCancelled() {
446 qInfo() << "Export cancelled";
447
448 // 将所有未完成的项标记为失败，并同步各类型状态，避免小球状态残留
449 for (int i = 0; i < m_resultsList.size(); ++i) {
450 QVariantMap result = m_resultsList[i].toMap();
451 QString currentStatus = result["status"].toString();
452 if (currentStatus == "pending" || currentStatus == "in_progress") {
453 result["error"] = "Export cancelled";
454 const QStringList statusKeys = {
455 QStringLiteral("symbolStatus"),
456 QStringLiteral("footprintStatus"),
457 QStringLiteral("model3DStatus"),
458 QStringLiteral("previewStatus"),
459 QStringLiteral("datasheetStatus"),
460 };
461 for (const QString& key : statusKeys) {
462 const QString currentTypeStatus = result.value(key).toString();
463 if (currentTypeStatus == "pending" || currentTypeStatus == "in_progress") {
464 result[key] = "failed";
465 }
466 }
467 updateOverallItemStatus(result);
468 m_resultsList[i] = result;
469 }
470 }
471
472 if (m_isStopping) {
473 m_isStopping = false;
474 emit isStoppingChanged();
475 }
476 setIsExporting(false);
477 setHasCompletedExport(false);
478 setProgress(0);
479 setStatus("Export cancelled");
480 m_fetchProgress = 0;
481 m_processProgress = 0;
482 m_writeProgress = 0;
483 updateResultsList(); // 更新失败数量统计
484 emit stageProgressChanged();
485 flushPendingUpdates();
486}

678void ExportProgressViewModel::retryComponent(const QString& componentId) {
679 qDebug() << "Retry requested for component:" << componentId;
680
681 if (!m_idToIndexMap.contains(componentId)) {
682 qWarning() << "Component not found in results:" << componentId;
683 return;
684 }
685
686 int index = m_idToIndexMap[componentId];
687 QVariantMap result = m_resultsList[index].toMap();
688
689 resetItemForRetry(result);
690 m_resultsList[index] = result;
691
692 // Update counts and notify UI
693 markResultsDirty();
694 flushPendingUpdates();
695
696 // Re-trigger export if not currently exporting
697 if (!m_isExporting && m_exportService) {
698 QStringList idsToRetry = {componentId};
699 ExportOptions opts = m_exportService->options();
700 opts.retryMode = true;
701 m_exportService->setOptions(opts);
702 beginExportRun(idsToRetry, QStringLiteral("Preloading component data..."));
703 m_exportService->startPreload(idsToRetry);
704 }

707void ExportProgressViewModel::retryFailedComponents() {
708 qDebug() << "Retry requested for all failed components";
709
710 // Collect all failed component IDs
711 QStringList failedIds;
712 for (const auto& item : m_resultsList) {
713 QVariantMap map = item.toMap();
714 if (map.value("status") == "failed") {
715 QString id = map.value("componentId").toString();
716 failedIds.append(id);
717
718 // Reset status to pending
719 if (m_idToIndexMap.contains(id)) {
720 int index = m_idToIndexMap[id];
721 QVariantMap result = m_resultsList[index].toMap();
722 resetItemForRetry(result);
723 m_resultsList[index] = result;
724 }
725 }
726 }
727
728 if (failedIds.isEmpty()) {
729 qDebug() << "No failed components to retry";
730 return;
731 }
732
733 qDebug() << "Retrying" << failedIds.size() << "failed components";
734
735 // Update counts and notify UI
736 markResultsDirty();
737 flushPendingUpdates();
738
739 // Re-trigger export if not currently exporting
740 if (!m_isExporting && m_exportService) {
741 ExportOptions opts = m_exportService->options();
742 opts.retryMode = true;
743 m_exportService->setOptions(opts);
744 beginExportRun(failedIds, QStringLiteral("Preloading component data..."));
745 m_exportService->startPreload(failedIds);

821void ExportProgressViewModel::updateOverallItemStatus(QVariantMap& result) const {
822 struct TypeRule {
823 bool enabled;
824 QString statusKey;
825 };
826
827 const QList<TypeRule> rules = {
828 {m_exportSymbolEnabled, "symbolStatus"},
829 {m_exportFootprintEnabled, "footprintStatus"},
830 {m_exportModel3DEnabled, "model3DStatus"},
831 {m_exportPreviewEnabled, "previewStatus"},
832 {m_exportDatasheetEnabled, "datasheetStatus"},
833 };
834
835 bool hasEnabledType = false;
836 bool anyFailed = false;
837 bool anyInProgress = false;
838 bool allDone = true;
839
840 for (const TypeRule& rule : rules) {
841 if (!rule.enabled) {
842 continue;
843 }
844
845 hasEnabledType = true;
846 const QString status = result.value(rule.statusKey, "pending").toString();
847 if (status == "failed") {
848 anyFailed = true;
849 }
850 if (status == "in_progress") {
851 anyInProgress = true;
852 }
853 if (status != "success" && status != "failed" && status != "skipped") {
854 allDone = false;
855 }
856 }
857
858 if (!hasEnabledType) {
859 result["status"] = "success";
860 } else if (allDone) {
861 result["status"] = anyFailed ? "failed" : "success";
862 } else if (anyInProgress) {
863 result["status"] = "in_progress";
864 } else {
865 result["status"] = "pending";
866 }
867}

884int ExportProgressViewModel::averageTypeProgress(const ExportOverallProgress& progress,
885 const QStringList& typeNames) const {
886 int sum = 0;
887 int count = 0;
888 for (const QString& typeName : typeNames) {
889 auto it = progress.exportTypeProgress.constFind(typeName);
890 if (it == progress.exportTypeProgress.constEnd()) {
891 continue;
892 }
893 sum += it.value().percentage();
894 count++;
895 }
896
897 if (count == 0) {
898 return 0;
899 }
900 return qBound(0, sum / count, 100);
901}

903int ExportProgressViewModel::stageTypeProgress(const ExportOverallProgress& progress,

905 bool hasEnabledType = false;
906 for (const QString& typeName : typeNames) {
907 const bool enabled = (typeName == "Symbol" && m_exportSymbolEnabled) ||
908 (typeName == "Footprint" && m_exportFootprintEnabled) ||
909 (typeName == "Model3D" && m_exportModel3DEnabled) ||
910 (typeName == "PreviewImages" && m_exportPreviewEnabled) ||
911 (typeName == "Datasheet" && m_exportDatasheetEnabled);
912 if (enabled) {
913 hasEnabledType = true;
914 break;
915 }
916 }
917
918 if (!hasEnabledType) {
919 return 100;
920 }
921
922 return averageTypeProgress(progress, typeNames);
