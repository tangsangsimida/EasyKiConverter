51 var optionTexts = [qsTranslate("MainWindow", "符号库"), qsTranslate("MainWindow", "封装库"), qsTranslate("MainWindow", "3D模型"), qsTranslate("MainWindow", "预览图"), qsTranslate("MainWindow", "手册"), qsTranslate("MainWindow", "追加"), qsTranslate("MainWindow", "更新")];

53 var descriptionTexts = [qsTranslate("MainWindow", "保留已存在的元器件"), qsTranslate("MainWindow", "覆盖已存在的元器件")];
54 // 合并所有文本用于测量
55 var texts = optionTexts.concat(descriptionTexts);
56 // 测量文本宽度
57 var maxOptionTextWidth = 0;
58 for (var i = 0; i < texts.length; i++) {
59 var textWidth = textMetrics.advanceWidth(texts[i]);

73 var radioButtonSpacing = 8; // 单选按钮与文本之间的间距

75 // 第一行：单选按钮 + 选项文本
76 var firstLineWidth = radioButtonWidth + radioButtonSpacing + maxOptionTextWidth;

79 // 取两行中较大的，并根据语言调整缓冲区

101 console.log("计算的最小窗口宽度:", minWidth);
