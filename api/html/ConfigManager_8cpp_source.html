31bool ConfigManager::save() {
32 try {
33 // 保存导出路径
34 QJsonValue exportPathValue = m_config.value("export_path");
35 m_settings->setValue("export/export_path", exportPathValue.isUndefined() ? "" : exportPathValue.toString());
36
37 QJsonValue libNameValue = m_config.value("lib_name");
38 m_settings->setValue("export/lib_name",
39 libNameValue.isUndefined() ? "easyeda_convertlib" : libNameValue.toString());
40
41 // 保存导出选项
42 QJsonValue exportOptionsValue = m_config.value("export_options");
43 QJsonObject exportOptions = exportOptionsValue.isUndefined() ? QJsonObject() : exportOptionsValue.toObject();
44
45 QJsonValue symbolValue = exportOptions.value("symbol");
46 m_settings->setValue("export/export_symbol", symbolValue.isUndefined() ? true : symbolValue.toBool());
47
48 QJsonValue footprintValue = exportOptions.value("footprint");
49 m_settings->setValue("export/export_footprint", footprintValue.isUndefined() ? true : footprintValue.toBool());
50
51 QJsonValue model3dValue = exportOptions.value("model3d");
52 m_settings->setValue("export/export_3d_model", model3dValue.isUndefined() ? true : model3dValue.toBool());
53
54 // 同步到磁
55 m_settings->sync();
56
57 qDebug() << "Configuration saved successfully";
58 return true;
59 } catch (const std::exception& e) {
60 qWarning() << "Failed to save configuration:" << e.what();
61 return false;
62 }

65bool ConfigManager::load() {
