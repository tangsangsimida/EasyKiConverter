368struct RetryPolicy {
369 int maxRetries = 3;
370 int baseTimeoutMs = 30000;
371 double jitterFactor = 0.2;
372 QVector<int> delays = {1000, 2000, 4000}; // Exponential backoff
373
374 // Status codes that should trigger retry
375 QVector<int> retryableStatusCodes = {429, 500, 502, 503, 504};
376
380 static RetryPolicy fromProfile(const RequestProfile& profile, bool isWeakNetwork = false) {
381 RetryPolicy policy;
382 const int baseTimeout = qMax(profile.connectTimeoutMs, profile.readTimeoutMs);
383 const int timeout = isWeakNetwork ? baseTimeout * profile.weakNetworkTimeoutMultiplier : baseTimeout;
384 policy.baseTimeoutMs = timeout;
385 policy.maxRetries = isWeakNetwork ? profile.weakNetworkMaxRetries : profile.maxRetries;
386 policy.jitterFactor = profile.jitterFactor;
387
388 // Generate backoff delays
389 policy.delays.clear();
390 double delay = profile.backoffBaseMs;
391 for (int i = 0; i < policy.maxRetries; ++i) {
392 policy.delays.append(static_cast<int>(delay));
393 delay *= profile.backoffMultiplier;
394 }
395
396 return policy;
397 }
