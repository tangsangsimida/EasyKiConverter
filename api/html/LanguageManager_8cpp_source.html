40 << ", initializing:" << m_isInitializing << ", loadedFromConfig:" << m_loadedFromConfig << ")";
41

122void LanguageManager::installTranslator(const QString& languageCode) {
123 QCoreApplication* app = QCoreApplication::instance();
124 if (!app) {
125 qWarning() << "[LanguageManager] QCoreApplication instance not available";
126 return;
127 }
128
129 // 如果已有翻译器，先移除并删除
130 if (m_translator) {
131 app->removeTranslator(m_translator);
132 m_translator->deleteLater();
133 m_translator = nullptr;
134 qInfo() << "[LanguageManager] Removed previous translator";
135 }
136
137 // Qt 6 qml_module 资源路径 - 使用 compiled .qm files
138 QString translationPath =
139 QString(":/qt/qml/EasyKiconverter_Cpp_Version/resources/translations/translations_easykiconverter_%1.qm")
140 .arg(languageCode);
141
142 m_translator = new QTranslator(this);
143
144 // 尝试加载翻译文件（.qm 格式）
145 if (m_translator->load(translationPath)) {
146 app->installTranslator(m_translator);
147 qInfo() << "[LanguageManager] ✓ Successfully installed translator from resources:" << translationPath;
148 return;
149 }
150
151 // 如果资源文件不存在，尝试从文件系统加载
152 QString localPath = QString("resources/translations/translations_easykiconverter_%1.qm").arg(languageCode);
153 if (m_translator->load(localPath)) {
154 app->installTranslator(m_translator);
155 qInfo() << "[LanguageManager] ✓ Successfully installed translator from file system:" << localPath;
156 return;
157 }
158
159 qWarning() << "[LanguageManager] ✗ Failed to load translation file for language:" << languageCode;
160 qWarning() << "[LanguageManager] Tried paths:";
161 qWarning() << " 1. Resource path:" << translationPath;
162 qWarning() << " 2. File system path:" << localPath;
163
164 // 加载失败时，清理创建的翻译器对象
165 m_translator->deleteLater();
166 m_translator = nullptr;
167}
