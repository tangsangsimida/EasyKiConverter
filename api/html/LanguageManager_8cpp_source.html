38void LanguageManager::setLanguage(const QString& languageCode, bool force) {
39 qInfo() << "[LanguageManager] setLanguage() called with:" << languageCode << " (current:" << m_currentLanguage
40 << ", initializing:" << m_isInitializing << ", loadedFromConfig:" << m_loadedFromConfig
41 << ", force:" << force << ")";
42
43 if (m_currentLanguage == languageCode) {
44 qInfo() << "[LanguageManager] Language unchanged, returning";
45
46 return;
47 }
48
49 // 如果正在初始化且已经从配置加载了语言，则拒绝覆盖（除非 force=true）
50 // 这样可以防止 QML 组件在初始化时覆盖配置文件的语言设置
51 if (m_isInitializing && m_loadedFromConfig && !force) {
52 qWarning() << "[LanguageManager] Refusing to override language during initialization after loading from "
53 "config. Requested:"
54 << languageCode << ", Keeping:" << m_currentLanguage;
55
56 return;
57 }
58
59 m_currentLanguage = languageCode;
60
61 qInfo() << "[LanguageManager] Language changed to" << m_currentLanguage;
62
63 // 只在非初始化期间保存设置
64
65 if (!m_isInitializing) {
66 saveLanguageSettings();
67
68 } else {
69 qInfo() << "[LanguageManager] Skipping save during initialization";
70 }
71
72 // 安装新的翻译器
73
74 installTranslator(languageCode);
75
76 emit languageChanged(languageCode);
77
78 emit refreshRequired();
79
80 qInfo() << "[LanguageManager] Language changed signal emitted:" << languageCode;
81}

97 qInfo() << "[LanguageManager] Loaded language settings from ConfigService:" << m_currentLanguage;

102

121void LanguageManager::installTranslator(const QString& languageCode) {
122 QCoreApplication* app = QCoreApplication::instance();
123 if (!app) {
124 qWarning() << "[LanguageManager] QCoreApplication instance not available";
125 return;
126 }
127
128 // 移除旧翻译器
129 if (m_translator) {
130 app->removeTranslator(m_translator);
131 m_translator->deleteLater();
132 m_translator = nullptr;
133 }
134
135 // Qt 资源系统中的翻译文件路径
136 // 由 qt_add_translations 自动生成并注册到资源系统
137 // qt_add_translations 默认使用 /i18n 前缀
138 QString resourcePath = QString(":/i18n/translations_easykiconverter_%1.qm").arg(languageCode);
139
140 m_translator = new QTranslator(this);
141
142 // 尝试从 Qt 资源系统加载
143 if (m_translator->load(resourcePath)) {
144 app->installTranslator(m_translator);
145 qInfo() << "[LanguageManager] [OK] Loaded translator from resource:" << resourcePath;
146 return;
147 }
148
149 // 如果资源文件不存在，尝试从文件系统加载（开发时可能有用）
150 QString localPath = QString("resources/translations/translations_easykiconverter_%1.qm").arg(languageCode);
151 if (m_translator->load(localPath)) {
152 app->installTranslator(m_translator);
153 qInfo() << "[LanguageManager] [OK] Loaded translator from filesystem:" << localPath;
154 return;
155 }
156
157 qWarning() << "[LanguageManager] [FAIL] Cannot load translation:" << languageCode;
158 qWarning() << "[LanguageManager] Tried paths:";
159 qWarning() << " 1. Resource path:" << resourcePath;
160 qWarning() << " 2. File system path:" << localPath;
161 m_translator->deleteLater();
162 m_translator = nullptr;
163}
