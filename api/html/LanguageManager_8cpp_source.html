95void LanguageManager::installTranslator(const QString& languageCode) {
96 QCoreApplication* app = QCoreApplication::instance();
97 if (!app) {
98 qWarning() << "QCoreApplication instance not available";
99 return;
100 }
101
102 // 如果已有翻译器，先移除并删除
103 if (m_translator) {
104 app->removeTranslator(m_translator);
105 m_translator->deleteLater();
106 m_translator = nullptr;
107 }
108
109 // Qt 6 qml_module 资源路径 - 使用 compiled .qm files
110 QString translationPath =
111 QString(":/qt/qml/EasyKiconverter_Cpp_Version/resources/translations/translations_easykiconverter_%1.qm")
112 .arg(languageCode);
113
114 m_translator = new QTranslator(this);
115
116 // 尝试加载翻译文件（.qm 格式）
117 if (m_translator->load(translationPath)) {
118 app->installTranslator(m_translator);
119 qDebug() << "Installed translator:" << translationPath;
120 return;
121 }
122
123 // 如果资源文件不存在，尝试从文件系统加载
124 QString localPath = QString("resources/translations/translations_easykiconverter_%1.qm").arg(languageCode);
125 if (m_translator->load(localPath)) {
126 app->installTranslator(m_translator);
127 qDebug() << "Installed translator from file system:" << localPath;
128 return;
129 }
130
131 qWarning() << "Failed to load translation file for language:" << languageCode;
132 qWarning() << "Tried paths:" << translationPath << localPath;
133
134 // 加载失败时，清理创建的翻译器对象
135 m_translator->deleteLater();
136 m_translator = nullptr;
137}
