34bool calculateStepGeometryCenter(const QByteArray& stepData, Model3DBase* center) {
35 if (stepData.isEmpty() || center == nullptr) {
36 return false;
37 }
38
39 const QString content = QString::fromLatin1(stepData);
40 // Thread-safe: const QRegularExpression; globalMatch() is reentrant.
41 static const QRegularExpression pointRegex(
42 QStringLiteral("#(\\d+)\\s*=\\s*CARTESIAN_POINT\\s*\\(\\s*('[^']*'|\\$)\\s*,\\s*\\(([^()]*)\\)\\s*\\)"),
43 QRegularExpression::CaseInsensitiveOption);
44 static const QRegularExpression vertexPointRegex(QStringLiteral("VERTEX_POINT\\s*\\([^,]*,\\s*#(\\d+)\\s*\\)"),
45 QRegularExpression::CaseInsensitiveOption);
46
47 // 单次遍历收集所有 CARTESIAN_POINT，避免对整个文件做两次正则扫描
48 struct Vec3 {
49 double x = 0, y = 0, z = 0;
50 };
51
52 QHash<int, Vec3> allPoints;
53 QRegularExpressionMatchIterator pointMatches = pointRegex.globalMatch(content);
54 while (pointMatches.hasNext()) {
55 const QRegularExpressionMatch match = pointMatches.next();
56 bool idOk = false;
57 int pointId = match.captured(1).toInt(&idOk);
58 if (!idOk) {
59 continue;
60 }
61
62 const QStringList coordinateParts = match.captured(3).split(',', Qt::SkipEmptyParts);
63 if (coordinateParts.size() < 3) {
64 continue;
65 }
66
67 bool okX = false;
68 bool okY = false;
69 bool okZ = false;
70 const double x = coordinateParts.at(0).trimmed().toDouble(&okX);
71 const double y = coordinateParts.at(1).trimmed().toDouble(&okY);
72 const double z = coordinateParts.at(2).trimmed().toDouble(&okZ);
73 if (!okX || !okY || !okZ) {
74 continue;
75 }
76
77 allPoints.insert(pointId, {x, y, z});
78 }
79
80 // 第二次遍历仅扫描 VERTEX_POINT（数量远少于 CARTESIAN_POINT）
81 double minX = std::numeric_limits<double>::max();
82 double maxX = std::numeric_limits<double>::lowest();
83 double minY = std::numeric_limits<double>::max();
84 double maxY = std::numeric_limits<double>::lowest();
85 double minZ = std::numeric_limits<double>::max();
86 bool hasGeometryPoint = false;
87
88 QRegularExpressionMatchIterator vertexMatches = vertexPointRegex.globalMatch(content);
89 while (vertexMatches.hasNext()) {
90 bool ok = false;
91 int id = vertexMatches.next().captured(1).toInt(&ok);
92 if (!ok) {
93 continue;
94 }
95
96 auto it = allPoints.constFind(id);
97 if (it == allPoints.constEnd()) {
98 continue;
99 }
100
101 const Vec3& p = it.value();
102 minX = qMin(minX, p.x);
103 maxX = qMax(maxX, p.x);
104 minY = qMin(minY, p.y);
105 maxY = qMax(maxY, p.y);
106 minZ = qMin(minZ, p.z);
107 hasGeometryPoint = true;
108 }
109
110 if (!hasGeometryPoint) {
111 return false;
112 }
113
114 center->x = (minX + maxX) / 2.0;
115 center->y = (minY + maxY) / 2.0;
116 center->z = minZ; // 有意取最小值而非中心——用于 Z 轴对齐
117 return true;
118}

120double calculateStepZOffset(double wrlDisplayMinZ, double stepMinZ) {
121 if (wrlDisplayMinZ == std::numeric_limits<double>::max()) {
122 return stepMinZ > MODEL_Z_OFFSET_EPSILON_MM ? -stepMinZ : 0.0;
123 }
124
125 const double offset = wrlDisplayMinZ - stepMinZ;
126 return qAbs(offset) < MODEL_Z_OFFSET_EPSILON_MM ? 0.0 : offset;
127}

129bool readModelSourceZMm(const QByteArray& cadJsonRaw, const QString& modelUuid, double* zMm) {
130 if (cadJsonRaw.isEmpty() || modelUuid.isEmpty() || zMm == nullptr) {
131 return false;
132 }
133
134 QJsonParseError parseError;
135 const QJsonDocument cadDoc = QJsonDocument::fromJson(cadJsonRaw, &parseError);
136 if (parseError.error != QJsonParseError::NoError || !cadDoc.isObject()) {
137 return false;
138 }
139
140 const QJsonObject packageData =
141 cadDoc.object().value(QStringLiteral("packageDetail")).toObject().value(QStringLiteral("dataStr")).toObject();
142 const QJsonArray shapes = packageData.value(QStringLiteral("shape")).toArray();
143
144 // 第一遍：精确匹配 UUID（attrs.uuid == modelUuid）
145 // 第二遍：匹配第一个 outline3D SVGNODE（UUID 可能是 head.uuid_3d 而非 attrs.uuid）
146 for (int pass = 0; pass < 2; ++pass) {
147 for (const QJsonValue& shapeValue : shapes) {
148 const QString shape = shapeValue.toString();
149 if (!shape.startsWith(QStringLiteral("SVGNODE~"))) {
150 continue;
151 }
152
153 const QByteArray svgNodeJson = shape.mid(QStringLiteral("SVGNODE~").size()).toUtf8();
154 QJsonParseError svgParseError;
155 const QJsonDocument svgDoc = QJsonDocument::fromJson(svgNodeJson, &svgParseError);
156 if (svgParseError.error != QJsonParseError::NoError || !svgDoc.isObject()) {
157 continue;
158 }
159
160 const QJsonObject attrs = svgDoc.object().value(QStringLiteral("attrs")).toObject();
161 if (pass == 0 && attrs.value(QStringLiteral("uuid")).toString() != modelUuid) {
162 continue;
163 }
164 if (pass == 1 && attrs.value(QStringLiteral("c_etype")).toString() != QLatin1String("outline3D")) {
165 continue;
166 }
167
168 bool ok = false;
169 double z = 0.0;
170 const QJsonValue zValue = attrs.value(QStringLiteral("z"));
171 if (zValue.isString()) {
172 z = zValue.toString().toDouble(&ok);
173 } else if (zValue.isDouble()) {
174 z = zValue.toDouble();
175 ok = true;
176 }
177 if (!ok) {
178 return false;
179 }
180
181 *zMm = GeometryUtils::convertToMm(z);
182 return true;
183 }
184 }
185
186 return false;
187}

189double calculateWrlBaseZOffset(const QString& fpType, double wrlDisplayMinZ, double sourceZMm) {
190 if (fpType != QLatin1String(FP_TYPE_SMD) || wrlDisplayMinZ == std::numeric_limits<double>::max()) {
191 return 0.0;
192 }
193
194 const double offset = -wrlDisplayMinZ + sourceZMm;
195 return offset > MODEL_Z_OFFSET_EPSILON_MM ? offset : 0.0;
196}

198double zOffsetMmToEasyEdaUnits(double zOffsetMm) {
199 const double roundedOffset = std::round(zOffsetMm * 100.0) / 100.0;
200 return -(roundedOffset - EASYEDA_Z_OFFSET_BIAS) / EASYEDA_UNIT_TO_MM;
201}
