18AsyncNetworkRequest* AsyncNetworkRequest::createFinished(const NetworkResult& result, QObject* parent) {
19 auto* request =
20 new AsyncNetworkRequest(QUrl(), nullptr, result.diagnostic.resourceType, RetryPolicy(), QByteArray(), parent);
21 if (result.wasCancelled) {
22 request->m_cancelled.storeRelease(1);
23 }
24 // 设置已完成状态（同步），供调用方立即检查 isFinished() / result()
25 {
26 QMutexLocker locker(&request->m_resultMutex);
27 request->m_result = result;
28 }
29 request->m_finished.storeRelease(1);
30
31 // 延迟发射 finished 信号，确保调用方有机会先 connect
32 // 使用 QTimer::singleShot(0, ...) 投递到当前线程的事件循环
33 // 调用方必须在有事件循环的线程中使用（主线程、QThread 子类）
34 QTimer::singleShot(0, request, [request, result]() { emit request->finished(result); });
35 return request;
36}

65AsyncNetworkRequest::~AsyncNetworkRequest() {
66 // Cancel any pending operations

69 // Clean up any pending replies
70 QMutexLocker locker(&m_repliesMutex);
71 for (QNetworkReply* reply : m_pendingReplies) {

73 reply->abort();

75 if (reply) {
76 reply->deleteLater();
77 }
78 }
79 m_pendingReplies.clear();

152void AsyncNetworkRequest::startAttempt(int attemptNumber) {
153 if (isCancelled() || isFinished()) {
154 return;
155 }
156
157 m_currentAttemptTimedOut.storeRelease(0);
158
159 if (!m_networkManager) {
160 NetworkResult result = m_result;
161 result.error = "Network manager is null";
162 result.success = false;
163 result.wasCancelled = false;
164 result.diagnostic.errorType = NetworkErrorType::Other;
165 result.diagnostic.errorMessage = result.error;
166 completeWithResult(result);
167 return;
168 }
169
170 QNetworkRequest request(m_url);
171 request.setHeader(QNetworkRequest::UserAgentHeader, "EasyKiConverter/1.0");
172 request.setRawHeader("X-Requested-With", "XMLHttpRequest");
173 request.setAttribute(QNetworkRequest::CacheLoadControlAttribute, QNetworkRequest::AlwaysNetwork);
174 // 禁止不安全的重定向（HTTPS -> HTTP），防止降级攻击
175 request.setAttribute(QNetworkRequest::RedirectPolicyAttribute, QNetworkRequest::NoLessSafeRedirectPolicy);
176 // Note: We handle gzip manually in processResponse() since Qt doesn't auto-decompress
177
178 // Create the reply - use POST if body is provided
179 QNetworkReply* reply;
180 if (m_isPostRequest && !m_postBody.isEmpty()) {
181 // Auto-detect Content-Type based on body format
182 // If body contains = and &, it's likely URL-encoded form data
183 if (m_postBody.contains('=') && m_postBody.contains('&')) {
184 request.setHeader(QNetworkRequest::ContentTypeHeader, "application/x-www-form-urlencoded");
185 } else {
186 request.setHeader(QNetworkRequest::ContentTypeHeader, "application/json");
187 }
188 reply = m_networkManager->post(request, m_postBody);
189 } else {
190 reply = m_networkManager->get(request);
191 }
192
193 {
194 QMutexLocker locker(&m_repliesMutex);
195 m_currentReply = reply;
196 m_pendingReplies.append(reply);
197 }
198
199 // Set up timeout (no parent - created/destroyed in same thread pool thread)
200 m_timeoutTimer = new QTimer();
201 m_timeoutTimer->setSingleShot(true);
202 connect(m_timeoutTimer, &QTimer::timeout, this, &AsyncNetworkRequest::handleTimeout, Qt::DirectConnection);
203 if (m_timeoutMs > 0) {
204 m_timeoutTimer->start(m_timeoutMs);
205 }
206
207 // Connect signals - use DirectConnection since we want immediate handling
208 // The slot will run in this object's thread (main thread where AsyncNetworkRequest lives)
209 connect(reply, &QNetworkReply::finished, this, &AsyncNetworkRequest::handleAttemptFinished, Qt::DirectConnection);
210 connect(reply,
211 &QNetworkReply::downloadProgress,
212 this,
213 &AsyncNetworkRequest::handleDownloadProgress,
214 Qt::DirectConnection);
215
216 qDebug() << "AsyncNetworkRequest: Attempt" << attemptNumber << "started for" << m_url;
217}

293void AsyncNetworkRequest::processResponse(QNetworkReply* reply) {
294 NetworkResult result;
295 {
296 QMutexLocker locker(&m_resultMutex);
297 result = m_result;
298 }
299
300 result.statusCode = reply->attribute(QNetworkRequest::HttpStatusCodeAttribute).toInt();
301 result.diagnostic.statusCode = result.statusCode;
302 result.diagnostic.retryCount = m_currentRetryCount;
303
304 // Check for error
305 if (reply->error() != QNetworkReply::NoError) {
306 QString errorString = reply->errorString();
307
308 // Check if cancelled
309 if (isCancelled()) {
310 completeWithResult(cancelledResult());
311 return;
312 }
313
314 if (m_currentAttemptTimedOut.loadAcquire() != 0) {
315 if (shouldRetryInternal(result.statusCode, QNetworkReply::TimeoutError, m_currentRetryCount)) {
316 scheduleRetry(m_currentRetryCount);
317 } else {
318 completeWithResult(timeoutResult());
319 }
320 return;
321 }
322
323 // Check if should retry
324 if (shouldRetryInternal(result.statusCode, reply->error(), m_currentRetryCount)) {
325 scheduleRetry(m_currentRetryCount);
326 return;
327 }
328
329 // Final error
330 result.error = errorString;
331 result.success = false;
332 result.wasCancelled = false;
333 result.diagnostic.errorMessage = result.error;
334
335 switch (reply->error()) {
336 case QNetworkReply::TimeoutError:
337 result.diagnostic.errorType = NetworkErrorType::Timeout;
338 break;
339 case QNetworkReply::ConnectionRefusedError:
340 result.diagnostic.errorType = NetworkErrorType::ConnectionRefused;
341 break;
342 case QNetworkReply::HostNotFoundError:
343 result.diagnostic.errorType = NetworkErrorType::HostNotFound;
344 break;
345 case QNetworkReply::SslHandshakeFailedError:
346 result.diagnostic.errorType = NetworkErrorType::SSLError;
347 break;
348 case QNetworkReply::OperationCanceledError:
349 result.diagnostic.errorType = NetworkErrorType::Canceled;
350 result.diagnostic.wasCanceled = true;
351 break;
352 default:
353 if (result.statusCode == 429) {
354 result.diagnostic.errorType = NetworkErrorType::RateLimited;
355 result.diagnostic.wasRateLimited = true;
356 } else if (result.statusCode >= 500) {
357 result.diagnostic.errorType = NetworkErrorType::ServerError;
358 } else if (result.statusCode == 404) {
359 result.diagnostic.errorType = NetworkErrorType::NotFound;
360 } else if (result.statusCode == 403) {
361 result.diagnostic.errorType = NetworkErrorType::Forbidden;
362 } else {
363 result.diagnostic.errorType = NetworkErrorType::Other;
364 }
365 break;
366 }
367 completeWithResult(result);
368 return;
369 }
370
371 // Success - read data
372 QByteArray data = reply->readAll();
373
374 // 安全兜底：检查实际读取的数据大小（Content-Length 可能缺失或不准确）
375 {
376 const RequestProfile profile = RequestProfiles::fromType(m_resourceType);
377 if (profile.maxResponseBytes > 0 && data.size() > profile.maxResponseBytes) {
378 result.error = QStringLiteral("Response too large: %1 bytes (limit %2)")
379 .arg(data.size())
380 .arg(profile.maxResponseBytes);
381 result.success = false;
382 result.wasCancelled = false;
383 result.diagnostic.errorType = NetworkErrorType::Other;
384 result.diagnostic.errorMessage = result.error;
385 completeWithResult(result);
386 return;
387 }
388 }
389
390 // 校验重定向后的最终 URL 是否仍在白名单内
391 {
392 const QUrl finalUrl = reply->url();
393 if (finalUrl != m_url) {
394 QString errorMsg;
395 if (!UrlUtils::isAllowedUrl(finalUrl, m_resourceType, &errorMsg)) {
396 result.error = QStringLiteral("Redirect to disallowed URL: %1").arg(errorMsg);
397 result.success = false;
398 result.wasCancelled = false;
399 result.diagnostic.errorType = NetworkErrorType::Other;
400 result.diagnostic.errorMessage = result.error;
401 completeWithResult(result);
402 return;
403 }
404 }
405 }
406
407 // Check for gzip encoding and decompress if needed
408 QVariant encodingVariant = reply->rawHeader("Content-Encoding");
409 QString encoding = encodingVariant.toString().toLower();
410 m_gzipUsed = encoding.contains("gzip");
411
412 if (m_gzipUsed || GzipUtils::isGzipped(data)) {
413 GzipUtils::DecompressResult decompResult = GzipUtils::decompress(data);
414 if (decompResult.success) {
415 data = decompResult.data;
416
417 // 解压后重新检查大小限制（压缩可能放大数据）
418 const RequestProfile profile = RequestProfiles::fromType(m_resourceType);
419 if (profile.maxResponseBytes > 0 && data.size() > profile.maxResponseBytes) {
420 result.error = QStringLiteral("Decompressed response too large: %1 bytes (limit %2)")
421 .arg(data.size())
422 .arg(profile.maxResponseBytes);
423 result.success = false;
424 result.wasCancelled = false;
425 result.diagnostic.errorType = NetworkErrorType::Other;
426 result.diagnostic.errorMessage = result.error;
427 completeWithResult(result);
428 return;
429 }
430 } else {
431 result.error = "Gzip decompression failed";
432 result.success = false;
433 result.wasCancelled = false;
434 result.diagnostic.errorType = NetworkErrorType::DecompressionFailed;
435 result.diagnostic.errorMessage = result.error;
436 completeWithResult(result);
437 return;
438 }
439 }
440
441 result.data = data;
442 result.success = true;
443 result.retryCount = m_currentRetryCount;
444 result.wasCancelled = false;
445 result.diagnostic.errorType = NetworkErrorType::None;
446 result.diagnostic.errorMessage.clear();
447
448 completeWithResult(result);
449}

451void AsyncNetworkRequest::scheduleRetry(int retryCount) {
452 if (isCancelled() || isFinished()) {
453 return;
454 }
455
456 int delay = calculateRetryDelay(retryCount);
457 m_currentRetryCount = retryCount + 1;
458
459 qDebug() << "AsyncNetworkRequest: Scheduling retry" << m_currentRetryCount << "for" << m_url << "after" << delay
460 << "ms";
461
462 emit retrying(m_currentRetryCount, delay);
463
464 // Use a timer to delay retry - this allows cancellation to be checked
465 QTimer::singleShot(delay, this, [this]() {
466 if (isCancelled() || isFinished()) {
467 return;
468 }
469
470 // Clean up old reply
471 cleanupAttempt();
472
473 // Start new attempt in main thread where QNetworkAccessManager lives
474 QMetaObject::invokeMethod(this, [this]() { startAttempt(m_currentRetryCount); }, Qt::QueuedConnection);
475 });
476}

478int AsyncNetworkRequest::calculateRetryDelay(int retryCount) const {
479 if (retryCount >= static_cast<int>(m_policy.delays.size())) {
480 return m_policy.delays.back();
481 }
482 int baseDelay = m_policy.delays[retryCount];
483
484 // Add jitter
485 int jitterRange = static_cast<int>(baseDelay * m_policy.jitterFactor);
486 int jitter = QRandomGenerator::global()->bounded(-jitterRange, jitterRange + 1);
487 return baseDelay + jitter;
488}

490bool AsyncNetworkRequest::shouldRetryInternal(int statusCode, QNetworkReply::NetworkError error, int retryCount) const {
491 if (retryCount >= m_policy.maxRetries) {
492 return false;
493 }
494
495 if (m_policy.retryableStatusCodes.contains(statusCode)) {
496 return true;
497 }
498
499 switch (error) {
500 case QNetworkReply::TimeoutError:
501 case QNetworkReply::TemporaryNetworkFailureError:
502 case QNetworkReply::NetworkSessionFailedError:
503 case QNetworkReply::UnknownNetworkError:
504 case QNetworkReply::RemoteHostClosedError:
505 case QNetworkReply::HostNotFoundError:
506 case QNetworkReply::ConnectionRefusedError:
507 case QNetworkReply::ProxyTimeoutError:
508 return true;
509 default:
510 return false;
511 }
