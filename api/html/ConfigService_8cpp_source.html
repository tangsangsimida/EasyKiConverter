41bool ConfigService::loadConfig(const QString& path) {
42 QString configPath = path.isEmpty() ? getDefaultConfigPath() : path;
43
44 QFile file(configPath);
45 if (!file.open(QIODevice::ReadOnly)) {
46 qWarning() << "Config file not found, using defaults:" << configPath;
47 saveConfig();
48 return false;
49 }
50
51 QByteArray fileData = file.readAll();
52 file.close();
53
54 QJsonParseError parseError;
55 QJsonDocument doc = QJsonDocument::fromJson(fileData, &parseError);
56
57 if (parseError.error != QJsonParseError::NoError) {
58 qWarning() << "Failed to parse config file:" << parseError.errorString();
59 return false;
60 }
61
62 QMutexLocker locker(&m_configMutex);
63
64 // 合并配置：只更新文件中存在的字段，保留文件中缺失字段的默认值
65 QJsonObject loadedConfig = doc.object();
66 for (auto it = loadedConfig.begin(); it != loadedConfig.end(); ++it) {
67 m_config[it.key()] = it.value();
68 }
69
70 // 移除 debugMode 字段，因为它不应该从配置文件加载
71 // 调试模式完全由命令行参数和环境变量控制
72 m_config.remove("debugMode");
73
74 m_configPath = configPath;
75
76 qDebug() << "Config loaded from:" << configPath;
77 emit configChanged();
78
79 // 释放锁后保存，确保补全缺失的字段到文件中（但不包含 debugMode）
80 locker.unlock();
81 saveConfig();
82
83 return true;
84}

86bool ConfigService::saveConfig(const QString& path) {
87 QString configPath = path.isEmpty() ? (m_configPath.isEmpty() ? getDefaultConfigPath() : m_configPath) : path;
88
89 QMutexLocker locker(&m_configMutex);
90
91 // 创建配置的副本，移除 debugMode 字段
92 QJsonObject configToSave = m_config;
93 configToSave.remove("debugMode");
94
95 QJsonDocument doc(configToSave);
96
97 QFile file(configPath);
98 if (!file.open(QIODevice::WriteOnly)) {
99 qWarning() << "Failed to open config file for writing:" << configPath;
100 return false;
101 }
102
103 file.write(doc.toJson());
104 file.close();
105
106 qDebug() << "Config saved to:" << configPath;
107 return true;
108}

127void ConfigService::endBatchUpdate() {
128 QMutexLocker locker(&m_configMutex);
129 if (m_batchUpdateDepth > 0) {
130 m_batchUpdateDepth--;
131 if (m_batchUpdateDepth == 0 && m_batchDirty) {
132 m_batchDirty = false;

416void ConfigService::initializeDefaultConfig() {
417 m_config["outputPath"] = QStandardPaths::writableLocation(QStandardPaths::DesktopLocation);
418 m_config["libName"] = "easyeda_convertlib";
419 m_config["exportSymbol"] = true;
420 m_config["exportFootprint"] = true;
421 m_config["exportModel3D"] = true;
422 m_config["exportModel3DFormat"] = 3; // 默认 3=Both (WRL+STEP)
423 m_config["exportModel3DPathMode"] = 0; // 默认 0=相对路径
424 m_config["exportPreviewImages"] = false;
425 m_config["exportDatasheet"] = false;
426 m_config["overwriteExistingFiles"] = false;
427 m_config["exportMode"] = 0; // 默认 0=追加模式
428 m_config["weakNetworkSupport"] = false;
429 m_config["darkMode"] = false;
430 // 注意：debugMode 不在这里设置，因为它不应该保存在配置文件中
431 // 它完全由命令行参数和环境变量控制
432
433 // 窗口配置默认值（-1 或 -9999 表示使用默认值）
434 m_config["windowWidth"] = DEFAULT_WINDOW_WIDTH;
435 m_config["windowHeight"] = DEFAULT_WINDOW_HEIGHT;
436 m_config["windowX"] = DEFAULT_WINDOW_X;
437 m_config["windowY"] = DEFAULT_WINDOW_Y;
438 m_config["windowMaximized"] = false;
439 // 退出偏好默认值（空字符串表示未记住）
440 m_config["exitPreference"] = "";
441 // 语言设置默认值（英文）
442 m_config["language"] = "en";
443 m_config["cacheDir"] = defaultCacheDir();
444 m_config["diskCacheLimitMB"] = DEFAULT_DISK_CACHE_LIMIT_MB;
445}
