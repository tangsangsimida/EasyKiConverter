35bool ConfigService::loadConfig(const QString& path) {
36 QString configPath = path.isEmpty() ? getDefaultConfigPath() : path;
37
38 QFile file(configPath);
39 if (!file.open(QIODevice::ReadOnly)) {
40 qWarning() << "Config file not found, using defaults:" << configPath;
41 saveConfig();
42 return false;
43 }
44
45 QByteArray fileData = file.readAll();
46 file.close();
47
48 QJsonParseError parseError;
49 QJsonDocument doc = QJsonDocument::fromJson(fileData, &parseError);
50
51 if (parseError.error != QJsonParseError::NoError) {
52 qWarning() << "Failed to parse config file:" << parseError.errorString();
53 return false;
54 }
55
56 QMutexLocker locker(&m_configMutex);
57
58 // 合并配置：只更新文件中存在的字段，保留文件中缺失字段的默认值
59 QJsonObject loadedConfig = doc.object();
60 for (auto it = loadedConfig.begin(); it != loadedConfig.end(); ++it) {
61 m_config[it.key()] = it.value();
62 }
63
64 // 移除 debugMode 字段，因为它不应该从配置文件加载
65 // 调试模式完全由命令行参数和环境变量控制
66 m_config.remove("debugMode");
67
68 m_configPath = configPath;
69
70 qDebug() << "Config loaded from:" << configPath;
71 emit configChanged();
72
73 // 释放锁后保存，确保补全缺失的字段到文件中（但不包含 debugMode）
74 locker.unlock();
75 saveConfig();
76
77 return true;
78}

80bool ConfigService::saveConfig(const QString& path) {
81 QString configPath = path.isEmpty() ? (m_configPath.isEmpty() ? getDefaultConfigPath() : m_configPath) : path;
82
83 QMutexLocker locker(&m_configMutex);
84
85 // 创建配置的副本，移除 debugMode 字段
86 QJsonObject configToSave = m_config;
87 configToSave.remove("debugMode");
88
89 QJsonDocument doc(configToSave);
90
91 QFile file(configPath);
92 if (!file.open(QIODevice::WriteOnly)) {
93 qWarning() << "Failed to open config file for writing:" << configPath;
94 return false;
95 }
96
97 file.write(doc.toJson());
98 file.close();
99
100 qDebug() << "Config saved to:" << configPath;
101 return true;
102}

344 m_config["libName"] = "easyeda_convertlib";
345 m_config["exportSymbol"] = true;
