35bool ConfigService::loadConfig(const QString& path) {
36 QString configPath = path.isEmpty() ? getDefaultConfigPath() : path;
37
38 QFile file(configPath);
39 if (!file.open(QIODevice::ReadOnly)) {
40 qWarning() << "Failed to open config file:" << configPath;
41 return false;
42 }
43
44 QByteArray fileData = file.readAll();
45 file.close();
46
47 QJsonParseError parseError;
48 QJsonDocument doc = QJsonDocument::fromJson(fileData, &parseError);
49
50 if (parseError.error != QJsonParseError::NoError) {
51 qWarning() << "Failed to parse config file:" << parseError.errorString();
52 return false;
53 }
54
55 QMutexLocker locker(&m_configMutex);
56
57 // 合并配置：只更新文件中存在的字段，保留文件中缺失字段的默认值
58 QJsonObject loadedConfig = doc.object();
59 for (auto it = loadedConfig.begin(); it != loadedConfig.end(); ++it) {
60 m_config[it.key()] = it.value();
61 }
62
63 m_configPath = configPath;
64
65 qDebug() << "Config loaded from:" << configPath;
66 emit configChanged();
67
68 // 释放锁后保存，确保补全缺失的字段（如 debugMode）到文件中
69 locker.unlock();
70 saveConfig();
71
72 return true;
73}

75bool ConfigService::saveConfig(const QString& path) {
76 QString configPath = path.isEmpty() ? (m_configPath.isEmpty() ? getDefaultConfigPath() : m_configPath) : path;
77
78 QMutexLocker locker(&m_configMutex);
79
80 QJsonDocument doc(m_config);
81
82 QFile file(configPath);
83 if (!file.open(QIODevice::WriteOnly)) {
84 qWarning() << "Failed to open config file for writing:" << configPath;
85 return false;
86 }
87
88 file.write(doc.toJson());
89 file.close();
90
91 qDebug() << "Config saved to:" << configPath;
92 return true;
93}
