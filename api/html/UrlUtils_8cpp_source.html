87// 检查 host 是否匹配允许列表（支持子域名匹配）
88bool isHostAllowed(const QString& host, const QSet<QString>& allowedHosts) {
89 if (host.isEmpty()) {
90 return false;
91 }
92
93 // 精确匹配
94 if (allowedHosts.contains(host)) {
95 return true;
96 }
97
98 // 子域名匹配：检查 host 是否以 ".allowedHost" 结尾
99 for (const QString& allowed : allowedHosts) {
100 if (host.endsWith(QLatin1Char('.') + allowed)) {
101 return true;
102 }
103 }
104
105 return false;
106}

110QString normalizePreviewImageUrl(const QString& imageUrl) {
111 QString normalizedUrl = imageUrl.trimmed();
112 if (normalizedUrl.isEmpty()) {
113 return QString();
114 }
115
116 if (normalizedUrl.startsWith(QStringLiteral("//"))) {
117 normalizedUrl.prepend(QStringLiteral("https:"));
118 } else if (normalizedUrl.startsWith(QStringLiteral("/image.lceda.cn/")) ||
119 normalizedUrl.startsWith(QStringLiteral("/file.elecfans.com/")) ||
120 normalizedUrl.startsWith(QStringLiteral("/www.lcsc.com/"))) {
121 normalizedUrl.remove(0, 1);
122 normalizedUrl.prepend(QStringLiteral("https://"));
123 } else if (normalizedUrl.startsWith(QStringLiteral("/web1/")) ||
124 normalizedUrl.startsWith(QStringLiteral("/M00/"))) {
125 normalizedUrl.remove(0, 1);
126 normalizedUrl.prepend(QStringLiteral("https://file.elecfans.com/"));
127 } else if (normalizedUrl.startsWith('/')) {
128 normalizedUrl.remove(0, 1);
129 normalizedUrl.prepend(QStringLiteral("https://image.lceda.cn/"));
130 } else if (normalizedUrl.contains("alimg.szlcsc.com")) {
131 if (!normalizedUrl.startsWith("http")) {
132 normalizedUrl = "https://" + normalizedUrl;
133 }
134 } else if (!normalizedUrl.contains("://")) {
135 normalizedUrl = "https://image.lceda.cn/" + normalizedUrl;
136 }
137
138 normalizedUrl.replace(QStringLiteral("https://image.lceda.cn//image.lceda.cn/"),
139 QStringLiteral("https://image.lceda.cn/"));
140 normalizedUrl.replace(QStringLiteral("http://image.lceda.cn//image.lceda.cn/"),
141 QStringLiteral("https://image.lceda.cn/"));
142 normalizedUrl.replace(QStringLiteral("https://image.lceda.cn/image.lceda.cn/"),
143 QStringLiteral("https://image.lceda.cn/"));
144
145 return normalizedUrl;
146}

159bool isAllowedUrl(const QUrl& url, ResourceType resourceType, QString* errorMessage) {
160 if (!url.isValid()) {
161 if (errorMessage) {
162 *errorMessage = QStringLiteral("Invalid URL: %1").arg(url.toString());
163 }
164 return false;
165 }
166
167 // 仅允许 https scheme
168 const QString scheme = url.scheme().toLower();
169 if (scheme != QStringLiteral("https")) {
170 if (errorMessage) {
171 *errorMessage = QStringLiteral("URL scheme must be https, got: %1").arg(scheme);
172 }
173 return false;
174 }
175
176 const QString host = url.host().toLower();
177
178 // Unknown 类型：仅允许已知域名（拒绝未知 host）
179 if (resourceType == ResourceType::Unknown) {
180 static const QSet<QString> allKnown =
181 allowedHostsForType(ResourceType::ComponentInfo) + allowedHostsForType(ResourceType::ProductSearch) +
182 allowedHostsForType(ResourceType::PreviewImage) + allowedHostsForType(ResourceType::Datasheet) +
183 allowedHostsForType(ResourceType::Model3DObj) + allowedHostsForType(ResourceType::UpdateCheck) +
184 allowedHostsForType(ResourceType::WorkerRequest);
185 if (!isHostAllowed(host, allKnown)) {
186 if (errorMessage) {
187 *errorMessage = QStringLiteral("URL host '%1' is not in any known allowlist").arg(host);
188 }
189 return false;
190 }
191 return true;
192 }
193 const QSet<QString> allowed = allowedHostsForType(resourceType);
194
195 if (allowed.isEmpty()) {
196 // 空白名单 = 允许所有（向后兼容）
197 return true;
198 }
199
200 if (!isHostAllowed(host, allowed)) {
201 if (errorMessage) {
202 *errorMessage = QStringLiteral("URL host '%1' is not allowed for resource type %2")
203 .arg(host)
204 .arg(static_cast<int>(resourceType));
205 }
206 return false;
207 }
208
209 return true;
210}
