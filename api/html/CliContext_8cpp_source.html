9static int model3DFormatFromString(const QString& format) {

36ExportOptions CliContext::createExportOptions() const {
37 ExportOptions options;
38
39 // 设置输出路径
40 options.outputPath = m_parser.outputDir();
41 // 清洗 libName，防止路径穿越（如 ../foo）
42 options.libName = PathSecurity::sanitizeFilename(m_parser.libName());
43
44 // 设置导出类型（CLI 默认：符号库、封装库；3D 模型通过参数启用）
45 options.exportSymbol = m_parser.exportSymbol();
46 options.exportFootprint = m_parser.exportFootprint();
47 options.exportModel3D = m_parser.export3DModel();
48 options.exportModel3DFormat = model3DFormatFromString(m_parser.model3DFormat());
49 options.exportPreviewImages = m_parser.exportPreview();
50 options.exportDatasheet = m_parser.exportDatasheet();
51
52 // 设置默认值
53 options.debugMode = m_parser.isDebugMode();
54
55 // 弱网模式
56 options.weakNetworkSupport = m_parser.weakNetworkSupport();
57
58 // 更新模式
59 options.updateMode = m_parser.updateMode();
60
61 // 3D 模型路径模式
62 const QString pathMode = m_parser.model3DPathMode();
63 options.exportModel3DPathMode = (pathMode == QStringLiteral("absolute")) ? ExportOptions::MODEL_3D_PATH_ABSOLUTE
64 : ExportOptions::MODEL_3D_PATH_RELATIVE;
65
66 // 覆盖模式（默认 true，--no-overwrite 禁用）
67 options.overwriteExistingFiles = m_parser.overwriteExistingFiles();
68
69 // 库描述
70 options.symbolLibraryDescription = m_parser.symbolDescription();
71 options.footprintLibraryDescription = m_parser.footprintDescription();
72
73 return options;
74}
