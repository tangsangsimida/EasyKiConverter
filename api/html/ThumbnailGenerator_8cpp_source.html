24QImage ThumbnailGenerator::generateSymbolThumbnail(const QSharedPointer<SymbolData>& symbolData, int size) {
25 if (!symbolData)
26 return QImage();
27
28 QImage image(size, size, QImage::Format_ARGB32);
29 image.fill(Qt::white);
30
31 QPainter painter(&image);
32 painter.setRenderHint(QPainter::Antialiasing);
33
34 // 计算边框
35 QRectF bbox = calculateSymbolBoundingBox(symbolData);
36
37 // 如果没有有效内容，返回空图片
38 if (bbox.isNull() || bbox.width() <= 0 || bbox.height() <= 0) {
39 return image;
40 }
41
42 // 计算缩放比例，保留 10% 边距
43 double margin = size * 0.1;
44 double scaleX = (size - 2 * margin) / bbox.width();
45 double scaleY = (size - 2 * margin) / bbox.height();
46 double scale = qMin(scaleX, scaleY);
47
48 // 居中偏移
49 double offsetX = (size - bbox.width() * scale) / 2 - bbox.x() * scale;
50 double offsetY = (size - bbox.height() * scale) / 2 - bbox.y() * scale;
51
52 painter.translate(offsetX, offsetY);
53 painter.scale(scale, scale);
54
55 drawSymbol(painter, symbolData, QRect(0, 0, size, size)); // rect 参数其实在这里没用到，因为已经缩放了
56
57 return image;
58}

73void ThumbnailGenerator::drawSymbol(QPainter& painter,
74 const QSharedPointer<SymbolData>& symbolData,
75 const QRect& targetRect) {
76 Q_UNUSED(targetRect);
77
78 // 绘制样式
79 QPen pinPen(Qt::darkBlue, 1); // 考虑到缩放，线宽可能需要调整，但在 transform 下 1 代表逻辑单位
80 QPen bodyPen(Qt::darkRed, 2);
81 QBrush bodyBrush(Qt::NoBrush);
82
83 // 绘制矩形
84 painter.setPen(bodyPen);
85 painter.setBrush(bodyBrush);
86 for (const auto& rect : symbolData->rectangles()) {
87 painter.drawRect(QRectF(rect.posX, rect.posY, rect.width, rect.height));
88 }
89
90 // 绘制多边形
91 for (const auto& poly : symbolData->polygons()) {
92 QStringList pointsStr = poly.points.split(' ');
93 QPolygonF polygon;
94 for (int i = 0; i < pointsStr.size(); i += 2) {
95 if (i + 1 < pointsStr.size()) {
96 polygon.append(QPointF(pointsStr[i].toDouble(), pointsStr[i + 1].toDouble()));
97 }
98 }
99 painter.drawPolygon(polygon);
100 }
101
102 // 绘制多段线
103 for (const auto& poly : symbolData->polylines()) {
104 QStringList pointsStr = poly.points.split(' ');
105 QPolygonF polygon;
106 for (int i = 0; i < pointsStr.size(); i += 2) {
107 if (i + 1 < pointsStr.size()) {
108 polygon.append(QPointF(pointsStr[i].toDouble(), pointsStr[i + 1].toDouble()));
109 }
110 }
111 painter.drawPolyline(polygon);
112 }
113
114 // 绘制引脚
115 painter.setPen(pinPen);
116 for (const auto& pin : symbolData->pins()) {
117 if (!pin.settings.isDisplayed)
118 continue;
119
120 QPointF pos(pin.settings.posX, pin.settings.posY);
121
122 // 简单绘制一个小圆圈表示引脚位置
123 painter.drawEllipse(pos, 2, 2);
124
125 // 如果有引脚路径，解析并绘制
126 // 这里简化处理，通常引脚是一条线
127 // 根据旋转角度绘制一条短线
128 double length = 10.0; // 假设长度
129 double angle = pin.settings.rotation * M_PI / 180.0;
130 QPointF endPos = pos + QPointF(length * cos(angle), length * sin(angle));
131 painter.drawLine(pos, endPos);
132 }
133}

141QRectF ThumbnailGenerator::calculateSymbolBoundingBox(const QSharedPointer<SymbolData>& symbolData) {
142 QRectF bbox;
143 bool first = true;
144
145 auto updateBBox = [&](const QRectF& rect) {
146 if (first) {
147 bbox = rect;
148 first = false;
149 } else {
150 bbox = bbox.united(rect);
151 }
152 };
153
154 for (const auto& rect : symbolData->rectangles()) {
155 updateBBox(QRectF(rect.posX, rect.posY, rect.width, rect.height));
156 }
157
158 for (const auto& pin : symbolData->pins()) {
159 updateBBox(QRectF(pin.settings.posX - 5, pin.settings.posY - 5, 10, 10)); // 估算引脚范围
160 }
161
162 for (const auto& poly : symbolData->polygons()) {
163 QStringList pointsStr = poly.points.split(' ');
164 for (int i = 0; i < pointsStr.size(); i += 2) {
165 if (i + 1 < pointsStr.size()) {
166 updateBBox(QRectF(pointsStr[i].toDouble(), pointsStr[i + 1].toDouble(), 1, 1));
167 }
168 }
169 }
170
171 // 如果 bbox 仍然为空（例如只有引脚），给一个默认值
172 if (bbox.isNull()) {
173 return QRectF(0, 0, 100, 100);
174 }
175
176 return bbox;
177}
