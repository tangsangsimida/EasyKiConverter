55void ConsoleAppender::append(const LogRecord& record, const QString& formatted) {
56 QString output = formatted;
57
58 if (m_useColors) {
59 output = colorize(formatted, record.level);
60 }
61
62 output += "\n";
63 QByteArray utf8 = output.toUtf8();
64 bool isError = (record.level >= LogLevel::Error);
65
66 if (m_async) {
67 // 异步写入：加入队列（带大小限制）
68 QMutexLocker locker(&m_queueMutex);
69
70 // 检查队列大小
71 if (m_maxQueueSize > 0 && m_writeQueue.size() >= m_maxQueueSize) {
72 // 队列已满，丢弃最旧的消息
73 m_queueOverflow.storeRelaxed(1);
74 m_writeQueue.dequeue();
75 }
76
77 // 使用特殊前缀标记错误输出
78 QByteArray data = isError ? QByteArray("\x01") + utf8 : QByteArray("\x00") + utf8;
79 m_writeQueue.enqueue(data);
80 m_queueCondition.wakeOne();
81 } else {
82 // 同步写入
83 QMutexLocker locker(&m_mutex);
84 writeDirect(utf8, isError);
85 }

88void ConsoleAppender::flush() {
89 if (m_async) {
90 // 请求刷新
91 m_flushRequested.storeRelaxed(1);
92 m_queueCondition.wakeAll();
93
94 // 等待队列清空（带超时保护）
95 int timeout = 3000; // 3秒超时
96 int waited = 0;
97 while (waited < timeout) {
98 QMutexLocker locker(&m_queueMutex);
99 if (m_writeQueue.isEmpty()) {
100 break;
101 }
102 locker.unlock();
103 QThread::msleep(10);
104 waited += 10;
105 }
106 }
107
108 QMutexLocker locker(&m_mutex);
109 fflush(stdout);
110 fflush(stderr);
111}

139bool ConsoleAppender::supportsColors() {
140#ifdef Q_OS_WIN
141 // Windows 10+ 支持 ANSI 颜色
142 static bool checked = false;
143 static bool supported = false;
144
145 if (!checked) {
146 checked = true;
147
148 // 启用 Windows 控制台的 ANSI 转义序列处理
149 HANDLE hOut = GetStdHandle(STD_OUTPUT_HANDLE);
150 if (hOut != INVALID_HANDLE_VALUE) {
151 DWORD dwMode = 0;
152 if (GetConsoleMode(hOut, &dwMode)) {
153 dwMode |= ENABLE_VIRTUAL_TERMINAL_PROCESSING;
154 if (SetConsoleMode(hOut, dwMode)) {
155 supported = true;
156 }
157 }
158 }
159 }
160
161 return supported;
162#else
163 // Unix/Linux/macOS 默认支持颜色（检测 TERM 环境变量）
164 QByteArray term = qgetenv("TERM");
165 return !term.isEmpty() && term != "dumb";
166#endif
167}

188void ConsoleAppender::writerThreadFunc() {
189 while (m_running.loadRelaxed()) {
190 QByteArray data;
191 bool overflowed = false;
192
193 {
194 QMutexLocker locker(&m_queueMutex);
195
196 // 等待新消息或刷新请求
197 while (m_writeQueue.isEmpty() && m_running.loadRelaxed() && !m_flushRequested.loadRelaxed()) {
198 m_queueCondition.wait(&m_queueMutex, 100);
199 }
200
201 if (!m_writeQueue.isEmpty()) {
202 data = m_writeQueue.dequeue();
203 }
204
205 // 检查是否有溢出发生过
206 if (m_queueOverflow.loadRelaxed()) {
207 overflowed = true;
208 m_queueOverflow.storeRelaxed(0);
209 }
210 }
211
212 // 如果发生过溢出，添加警告消息
213 if (overflowed) {
214 writeDirect("[WARNING] Log queue overflow, some messages were dropped\n", false);
215 }
216
217 if (!data.isEmpty()) {
218 // 检查第一个字节判断输出流
219 bool isError = (data.at(0) == '\x01');
220 writeDirect(data.mid(1), isError);
221 }
222
223 // 处理刷新请求
224 if (m_flushRequested.loadRelaxed()) {
225 QMutexLocker locker(&m_mutex);
226 fflush(stdout);
227 fflush(stderr);
228 m_flushRequested.storeRelaxed(0);
229 }
230 }
231
232 // 写入剩余消息
233 while (true) {
234 QByteArray data;
235 {
236 QMutexLocker locker(&m_queueMutex);
237 if (m_writeQueue.isEmpty()) {
238 break;
239 }
240 data = m_writeQueue.dequeue();
241 }
242
243 if (!data.isEmpty()) {
244 bool isError = (data.at(0) == '\x01');
245 writeDirect(data.mid(1), isError);
246 }
247 }
248}
