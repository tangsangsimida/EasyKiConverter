104bool ProcessWorker::parseComponentInfo(ComponentExportStatus& status) {
105 if (status.componentInfoRaw.isEmpty()) {
106 status.componentData = QSharedPointer<ComponentData>::create();
107 status.componentData->setLcscId(status.componentId);
108 return true;
109 }
110
111 QJsonParseError parseError;
112 QJsonDocument doc = QJsonDocument::fromJson(status.componentInfoRaw, &parseError);
113 if (parseError.error != QJsonParseError::NoError) {
114 status.addDebugLog(QString("ERROR: Failed to parse component info JSON: %1").arg(parseError.errorString()));
115 return false;
116 }
117
118 QJsonObject obj = doc.object();
119 status.componentData = QSharedPointer<ComponentData>::create();
120 status.componentData->setLcscId(status.componentId);
121
122 if (obj.contains("name")) {
123 status.componentData->setName(obj["name"].toString());
124 }
125
126 return true;

129bool ProcessWorker::parseCadData(ComponentExportStatus& status) {
130 if (status.cadDataRaw.isEmpty()) {
131 return true;
132 }
133
134 QJsonParseError parseError;
135 QJsonDocument doc = QJsonDocument::fromJson(status.cadDataRaw, &parseError);
136 if (parseError.error != QJsonParseError::NoError) {
137 status.addDebugLog(QString("ERROR: Failed to parse CAD data JSON: %1").arg(parseError.errorString()));
138 return false;
139 }
140
141 QJsonObject rootObj = doc.object();
142
143 // API 返回的数据在 result 字段中，需要提取出来
144 QJsonObject obj;
145 if (rootObj.contains("result") && rootObj["result"].isObject()) {
146 obj = rootObj["result"].toObject();
147 status.addDebugLog("Extracted CAD data from 'result' field");
148 } else {
149 // 如果没有 result 字段，直接使用根对象
150 obj = rootObj;
151 }
152
153 // 调试：打印obj的结构
154 status.addDebugLog("=== ProcessWorker CAD Data Structure ===");
155 status.addDebugLog(QString("Top-level keys: %1").arg(obj.keys().join(", ")));
156 if (obj.contains("dataStr")) {
157 QJsonObject dataStr = obj["dataStr"].toObject();
158 status.addDebugLog(QString("dataStr keys: %1").arg(dataStr.keys().join(", ")));
159 if (dataStr.contains("shape")) {
160 QJsonArray shapes = dataStr["shape"].toArray();
161 status.addDebugLog(QString("dataStr.shape size: %1").arg(shapes.size()));
162 if (!shapes.isEmpty()) {
163 status.addDebugLog(QString("First shape: %1").arg(shapes[0].toString().left(100)));
164 }
165 } else {
166 status.addDebugLog("WARNING: dataStr does NOT contain 'shape' field!");
167 }
168 } else {
169 status.addDebugLog("WARNING: obj does NOT contain 'dataStr' field!");
170 }
171 status.addDebugLog("==========================================");
172
173 // 使用EasyedaImporter导入符号和封装数据
174 EasyedaImporter importer;
175
176 // 导入符号数据
177 status.symbolData = importer.importSymbolData(obj);
178
179 // 严格检查: 如果启用了符号导出但导入失败
180 if (!status.symbolData) {
181 status.addDebugLog("ERROR: Symbol data import failed or missing in CAD data.");
182 // 如果 symbolData 为空，说明解析失败。
183 return false;
184 }
185
186 // 调试：打印符号数据的统计信息
187 status.addDebugLog(QString("Symbol data imported successfully. Pins: %1").arg(status.symbolData->pins().size()));
188
189 // 导入封装数据
190 status.footprintData = importer.importFootprintData(obj);
191
192 // 严格检查: 如果启用了封装导出但导入失败
193 if (!status.footprintData || status.footprintData->info().name.isEmpty()) {
194 status.addDebugLog("ERROR: Footprint data import failed or footprint name is empty.");
195 return false;
196 }
197
198 status.addDebugLog(
199 QString("Footprint data imported successfully - Name: %1").arg(status.footprintData->info().name));
200
201 return true;
202}

204bool ProcessWorker::parse3DModelData(ComponentExportStatus& status) {
205 if (status.model3DObjRaw.isEmpty()) {
206 return true;
207 }
208
209 // 保存已有 model3DData 的属性（如果有的话）
210 QString existingUuid;
211 QString existingName;
212 Model3DBase existingTranslation;
213 Model3DBase existingRotation;
214 if (status.model3DData) {
215 existingUuid = status.model3DData->uuid();
216 existingName = status.model3DData->name();
217 existingTranslation = status.model3DData->translation();
218 existingRotation = status.model3DData->rotation();
219 }
220
221 // 创建3D模型数据
222 status.model3DData = QSharedPointer<Model3DData>::create();
223 status.model3DData->setRawObj(QString::fromUtf8(status.model3DObjRaw));
224
225 // 恢复 UUID：优先使用已存在的 UUID（来自预加载数据），其次从 footprintData 获取
226 if (!existingUuid.isEmpty()) {
227 status.model3DData->setUuid(existingUuid);
228 status.addDebugLog(QString("Preserved 3D model UUID from preloaded data: %1").arg(existingUuid));
229 } else if (status.footprintData && !status.footprintData->model3D().uuid().isEmpty()) {
230 status.model3DData->setUuid(status.footprintData->model3D().uuid());
231 status.addDebugLog(QString("Using 3D model UUID from footprintData: %1").arg(status.model3DData->uuid()));
232 }
233
234 // 恢复 name、translation、rotation：优先使用预加载数据，其次从 footprintData 获取
235 if (!existingName.isEmpty()) {
236 status.model3DData->setName(existingName);
237 } else if (status.footprintData && !status.footprintData->model3D().name().isEmpty()) {
238 status.model3DData->setName(status.footprintData->model3D().name());
239 }
240
241 if (existingTranslation.x != 0.0 || existingTranslation.y != 0.0 || existingTranslation.z != 0.0) {
242 status.model3DData->setTranslation(existingTranslation);
243 } else if (status.footprintData) {
244 status.model3DData->setTranslation(status.footprintData->model3D().translation());
245 }
246
247 if (existingRotation.x != 0.0 || existingRotation.y != 0.0 || existingRotation.z != 0.0) {
248 status.model3DData->setRotation(existingRotation);
249 } else if (status.footprintData) {
250 status.model3DData->setRotation(status.footprintData->model3D().rotation());
251 }
252
253 return true;
254}
