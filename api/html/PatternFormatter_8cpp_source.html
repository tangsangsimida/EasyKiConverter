12QString PatternFormatter::format(const LogRecord& record) {
13 QString result = m_pattern;
14
15 QRegularExpressionMatchIterator it = s_placeholderRegex.globalMatch(m_pattern);
16 QStringList processed;
17
18 while (it.hasNext()) {
19 QRegularExpressionMatch match = it.next();
20 QString placeholder = match.captured(1).toLower();
21
22 // 避免重复处理相同的占位符
23 QString fullMatch = match.captured(0);
24 if (processed.contains(fullMatch)) {
25 continue;
26 }
27 processed.append(fullMatch);

29 QString replacement = replacePlaceholder(placeholder, record);
30 result.replace(fullMatch, replacement);
31 }

47QString PatternFormatter::replacePlaceholder(const QString& placeholder, const LogRecord& record) const {
48 if (placeholder == "timestamp") {
49 return record.formattedTimestamp();
50 }
51 if (placeholder == "time") {
52 return QDateTime::fromMSecsSinceEpoch(record.timestamp).toString("HH:mm:ss.zzz");
53 }
54 if (placeholder == "date") {
55 return QDateTime::fromMSecsSinceEpoch(record.timestamp).toString("yyyy-MM-dd");
56 }
57 if (placeholder == "level") {
58 return logLevelToString(record.level);
59 }
60 if (placeholder == "module") {
61 return logModuleToString(record.module);
62 }
63 if (placeholder == "thread") {
64 return QString::number(reinterpret_cast<quintptr>(record.threadId), 16).toUpper().rightJustified(5, '0');
65 }
66 if (placeholder == "threadname") {
67 return record.threadName.isEmpty()
68 ? QString::number(reinterpret_cast<quintptr>(record.threadId), 16).toUpper().rightJustified(5, '0')
69 : record.threadName;
70 }
71 if (placeholder == "file") {
72 return record.fileName;
73 }
74 if (placeholder == "shortfile") {
75 return record.shortFileName();
76 }
77 if (placeholder == "line") {
78 return QString::number(record.line);
79 }
80 if (placeholder == "function") {
81 return record.functionName;
82 }
83 if (placeholder == "message") {
84 return record.message;
85 }
86 if (placeholder == "newline") {
87 return QStringLiteral("\n");
88 }

90 // 未知占位符，保持原样
91 return QStringLiteral("%{") + placeholder + QStringLiteral("}");
