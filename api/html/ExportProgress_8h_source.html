234 int totalSuccessCount() const {
235 int count = 0;
236 for (const auto& p : exportTypeProgress) {
237 count += p.successCount;
238 }
239 return count;
240 }
241
246 int totalFailedCount() const {
247 int count = 0;
248 for (const auto& p : exportTypeProgress) {
249 count += p.failedCount;
250 }
251 return count;
252 }
253
258 bool isComplete() const {
259 if (currentStage != Stage::Exporting && currentStage != Stage::Preloading) {
260 return currentStage == Stage::Completed || currentStage == Stage::Cancelled ||
261 currentStage == Stage::Failed;
262 }
263 if (currentStage == Stage::Preloading) {
264 return preloadProgress.isComplete();
265 }
266 for (const auto& p : exportTypeProgress) {
267 if (!p.isComplete())
268 return false;
269 }
270 return true;
271 }
272
284 int overallPercentage() const {
285 if (currentStage == Stage::Idle)
286 return 0;
287 if (currentStage == Stage::Preloading)
288 return preloadProgress.percentage();
289 if (currentStage == Stage::Exporting) {
290 int total = 0, completed = 0;
291 for (const auto& p : exportTypeProgress) {
292 total += p.totalCount;
293 completed += p.completedCount;
294 }
295 return total > 0 ? (completed * 100 / total) : 0;
296 }
297 if (currentStage == Stage::Completed)
298 return 100;
299 return 0;
300 }
