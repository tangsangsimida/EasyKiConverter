51QString FileUtils::toAbsolutePath(const QString& path) {
52 if (path.isEmpty()) {
53 return "";
54 }
55
56 QDir dir(path);
57 if (dir.isAbsolute()) {
58 // 已经是绝对路径，规范化路径分隔符
59 return dir.cleanPath(path);
60 }
61
62 // 相对路径，转换为绝对路径
63 QString absolutePath = dir.absolutePath();
64 qDebug() << "FileUtils::toAbsolutePath: Converting relative path" << path << "to" << absolutePath;
65 return absolutePath;
66}

77bool FileUtils::openFolderOnWindows(const QString& absolutePath) {
78#ifdef Q_OS_WIN
79 // 使用 explorer 命令打开文件夹
80 QStringList arguments;
81 arguments << "/select,"; // /select 参数可以同时选择文件夹
82 arguments << QDir::toNativeSeparators(absolutePath);
83
84 qDebug() << "FileUtils::openFolderOnWindows: Running explorer with arguments:" << arguments;
85
86 QProcess process;
87 bool success = process.startDetached("explorer", arguments);
88
89 if (!success) {
90 qWarning() << "FileUtils::openFolderOnWindows: Failed to start explorer, trying QDesktopServices fallback";
91 // 降级方案：使用 QDesktopServices
92 QUrl url = QUrl::fromLocalFile(absolutePath);
93 return QDesktopServices::openUrl(url);
94 }
95
96 return success;
97#else
98 Q_UNUSED(absolutePath);
99 return false;
100#endif
101}
