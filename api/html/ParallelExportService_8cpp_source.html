73void ParallelExportService::discardExportStage(const QString& typeName, ExportTypeStage* stage) {
74 if (m_exportStages.value(typeName) == stage) {
75 m_exportStages.remove(typeName);
76 }
77 if (stage) {
78 if (stage->isRunning() || stage->hasActiveWorkers()) {
79 retireExportStage(stage);
80 } else {
81 for (int i = m_retiredExportStages.size() - 1; i >= 0; --i) {
82 if (m_retiredExportStages.at(i).data() == stage || m_retiredExportStages.at(i).isNull()) {
83 m_retiredExportStages.removeAt(i);
84 }
85 }
86 stage->deleteLater();
87 }
88 }
89 cleanupRetiredExportStages();

92void ParallelExportService::retireExportStage(ExportTypeStage* stage) {
93 if (!stage) {
94 return;
95 }
96
97 stage->setParent(this);
98 for (const QPointer<ExportTypeStage>& retiredStage : std::as_const(m_retiredExportStages)) {

102 }
103 m_retiredExportStages.append(QPointer<ExportTypeStage>(stage));

623void ParallelExportService::onPreloadItemCompleted(const QString& componentId, bool success, const QString& error) {
624 QMutexLocker locker(&m_progressMutex);
625
626 m_progress.preloadProgress.inProgressCount--;
627
628 if (success) {
629 m_progress.preloadProgress.successCount++;
630 } else {
631 m_progress.preloadProgress.failedCount++;
632 m_progress.preloadProgress.failedComponents[componentId] = error;
633 }
634
635 m_progress.preloadProgress.completedCount++;
636
637 locker.unlock();
638 emit preloadProgressChanged(m_progress.preloadProgress);
639
640 if (m_progress.preloadProgress.completedCount >= m_progress.preloadProgress.totalCount) {
641 m_preloadCompleted = true;
642 m_progress.currentStage = ExportOverallProgress::Stage::Idle;
643 emit preloadCompleted(m_progress.preloadProgress.successCount, m_progress.preloadProgress.failedCount);
644 }

660void ParallelExportService::onExportTypeCompleted(const QString& typeName,
661 int successCount,
662 int failedCount,
663 int skippedCount) {
664 {
665 QMutexLocker locker(&m_progressMutex);
666
667 ExportTypeProgress typeProgress = m_progress.exportTypeProgress.value(typeName);
668 typeProgress.typeName = typeName;
669 typeProgress.totalCount = m_componentIds.size();
670 typeProgress.successCount = successCount;
671 typeProgress.failedCount = failedCount;
672 typeProgress.skippedCount = skippedCount;
673 typeProgress.completedCount = successCount + failedCount + skippedCount;
674 typeProgress.inProgressCount = 0;
675
676 m_progress.exportTypeProgress[typeName] = typeProgress;
677
678 if (m_runningExportStages > 0) {
679 m_runningExportStages--;
680 }
681 }
682
683 emit typeCompleted(typeName, successCount, failedCount, skippedCount);
684
685 if (auto* finishedStage = m_exportStages.value(typeName)) {
686 discardExportStage(typeName, finishedStage);
687 }
688
689 checkAllExportCompleted();

718void ParallelExportService::processNextPreloadBatch() {
719 if (m_cancelRequested || m_progress.currentStage != ExportOverallProgress::Stage::Preloading) {
720 return;
721 }
722
723 constexpr int BATCH_SIZE = 8;
724
725 if (!m_componentService) {
726 qWarning() << "ParallelExportService: ComponentService not set, cannot load data";
727 {
728 QMutexLocker locker(&m_progressMutex);
729 m_progress.preloadProgress.completedCount = m_componentIds.size();
730 m_progress.preloadProgress.failedCount = m_componentIds.size();
731 m_progress.preloadProgress.inProgressCount = 0;
732 m_progress.preloadProgress.currentComponentId.clear();
733 m_progress.currentStage = ExportOverallProgress::Stage::Idle;
734 m_progress.endTime = QDateTime::currentDateTime();
735 }
736 emit preloadProgressChanged(m_progress.preloadProgress);
737 logNetworkRuntimeStats(QStringLiteral("preload-failed-no-component-service"));
738 writeExportDetailedReport(QStringLiteral("preload-failed-no-component-service"));
739 emit preloadCompleted(0, m_componentIds.size());
740 return;
741 }
742
743 int processedInBatch = 0;
744 while (processedInBatch < BATCH_SIZE && m_nextPreloadIndex < m_componentIds.size() && !m_cancelRequested) {
745 const QString componentId = m_componentIds.at(m_nextPreloadIndex++);
746 {
747 QMutexLocker locker(&m_progressMutex);
748 m_progress.preloadProgress.currentComponentId = componentId;
749 m_progress.preloadProgress.inProgressCount = 1;
750 }
751
752 ComponentData data = m_componentService->getComponentData(componentId);
753 const QSharedPointer<ComponentData> diskCachedData =
754 ExportWorkerHelpers::loadDiskCachedComponentData(componentId);
755 ExportWorkerHelpers::mergeComponentData(data, diskCachedData);
756
757 {
758 QMutexLocker locker(&m_progressMutex);
759 // 严格校验：必须有 lcscId + symbolData + footprintData + isValid 才算有效
760 bool hasValidData = !data.lcscId().isEmpty() && data.isValid() && data.symbolData() != nullptr &&
761 data.footprintData() != nullptr;
762 if (hasValidData) {
763 auto sharedData = QSharedPointer<ComponentData>::create(data);
764 m_cachedData[componentId] = sharedData;
765 m_progress.preloadProgress.successCount++;
766 qDebug() << "ParallelExportService: Loaded data for" << componentId;
767 } else {
768 m_progress.preloadProgress.failedCount++;
769 m_progress.preloadProgress.failedComponents[componentId] = QStringLiteral("Incomplete component data");
770 qWarning() << "ParallelExportService: No data found for component:" << componentId;
771 }
772
773 m_progress.preloadProgress.completedCount++;
774 m_progress.preloadProgress.inProgressCount = 0;
775 }
776
777 processedInBatch++;
778 }
779
780 const bool finished = m_nextPreloadIndex >= m_componentIds.size();
781 if (finished) {
782 {
783 QMutexLocker locker(&m_progressMutex);
784 m_progress.preloadProgress.currentComponentId.clear();
785 m_progress.currentStage = ExportOverallProgress::Stage::Idle;
786 m_progress.endTime = QDateTime::currentDateTime();
787 }
788 qDebug() << "ParallelExportService: Preload completed. Success:" << m_progress.preloadProgress.successCount
789 << "Failed:" << m_progress.preloadProgress.failedCount;
790 m_preloadCompleted = true;
791 }
792
793 emit preloadProgressChanged(m_progress.preloadProgress);
794 updateOverallProgress();
795
796 if (finished) {
797 emit preloadCompleted(m_progress.preloadProgress.successCount, m_progress.preloadProgress.failedCount);
798 return;
799 }
800
801 QTimer::singleShot(0, this, &ParallelExportService::processNextPreloadBatch);
802}

804void ParallelExportService::onAllComponentDataCollected(const QList<ComponentData>& componentDataList) {
805 // 断开连接，避免重复处理
806 if (m_componentService) {
807 disconnect(m_componentService, &ComponentService::allComponentsDataCollected, this, nullptr);
808 }
809
810 // 检查是否已请求取消 - 如果是则直接返回，避免覆盖取消状态
811 {
812 QMutexLocker locker(&m_progressMutex);
813 if (m_cancelRequested) {
814 qDebug() << "ParallelExportService: Ignoring late callback after cancel requested";
815 return;
816 }
817 }
818
819 qDebug() << "ParallelExportService: Received" << componentDataList.size() << "component data from parallel fetch";
820
821 // 统计成功和失败数量
822 int successCount = 0;
823 int failedCount = 0;
824
825 // 处理每个组件数据
826 for (const ComponentData& data : componentDataList) {
827 const QString componentId = data.lcscId();
828
829 if (componentId.isEmpty()) {
830 failedCount++;
831 continue;
832 }
833
834 // 严格校验：必须同时有 symbol 和 footprint 数据且通过验证
835 bool hasValidData = data.isValid() && data.symbolData() != nullptr && data.footprintData() != nullptr;
836
837 if (hasValidData) {
838 auto sharedData = QSharedPointer<ComponentData>::create(data);
839 m_cachedData[componentId] = sharedData;
840 successCount++;
841 qDebug() << "ParallelExportService: Cached data for" << componentId;
842 } else {
843 failedCount++;
844 m_progress.preloadProgress.failedComponents[componentId] = QStringLiteral("No valid data found");
845 qWarning() << "ParallelExportService: No valid data for component:" << componentId;
846 }
847 }
848
849 // 更新进度
850 {
851 QMutexLocker locker(&m_progressMutex);
852 m_progress.preloadProgress.successCount = successCount;
853 m_progress.preloadProgress.failedCount = failedCount;
854 m_progress.preloadProgress.completedCount = componentDataList.size();
855 m_progress.preloadProgress.inProgressCount = 0;
856 m_progress.preloadProgress.currentComponentId.clear();
857 m_progress.currentStage = ExportOverallProgress::Stage::Idle;
858 m_progress.endTime = QDateTime::currentDateTime();
859 }
860
861 qDebug() << "ParallelExportService: Parallel preload completed. Success:" << successCount
862 << "Failed:" << failedCount;
863 logNetworkRuntimeStats(QStringLiteral("preload-completed"));
864 writeExportDetailedReport(QStringLiteral("preload-completed"));
865
866 m_preloadCompleted = true;
867 emit preloadProgressChanged(m_progress.preloadProgress);
868 emit preloadCompleted(successCount, failedCount);
869}

881void ParallelExportService::checkAllExportCompleted() {
882 if (m_runningExportStages > 0) {
883 return;
884 }
885
886 if (m_cancelRequested || m_progress.currentStage == ExportOverallProgress::Stage::Cancelled) {
887 qDebug() << "ParallelExportService: Ignoring completion because export was cancelled";
888 cleanupExportStages();
889 return;
890 }
891
892 qDebug() << "ParallelExportService: All export stages completed";
893
894 m_progress.currentStage = ExportOverallProgress::Stage::Completed;
895 m_progress.endTime = QDateTime::currentDateTime();
896
897 int totalSuccess = 0;
898 int totalFailed = 0;
899 for (const QString& componentId : m_componentIds) {
900 bool anyFailed = false;
901 bool allDone = true;
902
903 for (auto it = m_progress.exportTypeProgress.cbegin(); it != m_progress.exportTypeProgress.cend(); ++it) {
904 const ExportItemStatus itemStatus =
905 it.value().itemStatus.value(componentId, ExportItemStatus{ExportItemStatus::Status::Pending});
906 if (itemStatus.status == ExportItemStatus::Status::Failed) {
907 anyFailed = true;
908 }
909 if (itemStatus.status != ExportItemStatus::Status::Success &&
910 itemStatus.status != ExportItemStatus::Status::Failed &&
911 itemStatus.status != ExportItemStatus::Status::Skipped) {
912 allDone = false;
913 }
914 }
915
916 if (!allDone) {
917 continue;
918 }
919 if (anyFailed) {
920 totalFailed++;
921 } else {
922 totalSuccess++;
923 }
924 }
925
926 logNetworkRuntimeStats(QStringLiteral("export-completed"));
927 writeExportDetailedReport(QStringLiteral("export-completed"));
928 emit completed(totalSuccess, totalFailed);
929 cleanupExportStages();
930}
