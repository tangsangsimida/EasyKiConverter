423 holesArray.append(toJson(hole));
424 }
425 json["holes"] = holesArray;

428 QJsonArray circlesArray;
429 for (const FootprintCircle& circle : data.circles()) {
430 circlesArray.append(toJson(circle));

435 QJsonArray rectanglesArray;
436 for (const FootprintRectangle& rect : data.rectangles()) {

441 // 圆弧
442 QJsonArray arcsArray;
443 for (const FootprintArc& arc : data.arcs()) {
444 arcsArray.append(toJson(arc));

449 QJsonArray textsArray;
450 for (const FootprintText& text : data.texts()) {
451 textsArray.append(toJson(text));
