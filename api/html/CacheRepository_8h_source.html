23 void fetchPreviewImageAsync(const QString& lcscId,
24 const QString& imageUrl,
25 int imageIndex,
26 QAtomicInt* cancelled,
27 bool weakNetwork,
28 const MediaFetchCallback& onComplete) {
29 ComponentCacheService* cache = ComponentCacheService::instance();
30 if (!cache) {
31 ComponentExportStatus::NetworkDiagnostics diag;
32 diag.url = imageUrl;
33 diag.errorString = QStringLiteral("Cache service unavailable");
34 onComplete(QByteArray(), diag);
35 return;
36 }
37
38 if (cancelled && cancelled->loadRelaxed()) {
39 ComponentExportStatus::NetworkDiagnostics diag;
40 diag.url = imageUrl;
41 diag.errorString = QStringLiteral("Cancelled before start");
42 onComplete(QByteArray(), diag);
43 return;
44 }
45
46 const QString previewFilePath = cache->previewImagePath(lcscId, imageIndex);
47 if (QFileInfo::exists(previewFilePath)) {
48 QFile file(previewFilePath);
49 if (file.open(QIODevice::ReadOnly)) {
50 ComponentExportStatus::NetworkDiagnostics diag;
51 diag.url = imageUrl;
52 diag.statusCode = 200;
53 diag.retryCount = 0;
54 diag.wasRateLimited = false;
55 onComplete(file.readAll(), diag);
56 return;
57 }
58 }
59
60 const RetryPolicy policy = RetryPolicy::fromProfile(RequestProfiles::previewImage(), weakNetwork);
61 const uint64_t gen = cache->currentGeneration();
62 AsyncNetworkRequest* request =
63 NetworkClient::instance().getAsync(QUrl(imageUrl), ResourceType::PreviewImage, policy);
64
65 QObject::connect(request, &AsyncNetworkRequest::finished, request, [=](const NetworkResult& result) {
66 ComponentExportStatus::NetworkDiagnostics diag;
67 diag.url = imageUrl;
68 diag.statusCode = result.statusCode;
69 diag.retryCount = result.retryCount;
70 diag.wasRateLimited = result.diagnostic.wasRateLimited;
71 diag.latencyMs = result.elapsedMs;
72
73 if ((cancelled && cancelled->loadRelaxed()) || result.wasCancelled) {
74 diag.errorString = QStringLiteral("Cancelled");
75 onComplete(QByteArray(), diag);
76 } else if (result.success) {
77 cache->savePreviewImage(lcscId, result.data, imageIndex, gen);
78 onComplete(result.data, diag);
79 } else {
80 diag.errorString = result.error;
81 onComplete(QByteArray(), diag);
82 }
83
84 request->deleteLater();
85 });
86 }
