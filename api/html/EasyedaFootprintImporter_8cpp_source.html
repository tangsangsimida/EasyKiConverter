14QSharedPointer<FootprintData> EasyedaFootprintImporter::importFootprintData(const QJsonObject& cadData) {
15 auto footprintData = QSharedPointer<FootprintData>::create();
16
17 // 导入封装信息（从 packageDetail 中获取）
18 if (cadData.contains("packageDetail")) {
19 QJsonObject packageDetail = cadData["packageDetail"].toObject();
20
21 // packageDetail 顶层字段
22 FootprintInfo info;
23 info.uuid = packageDetail["uuid"].toString();
24 info.docType = QString::number(packageDetail["docType"].toInt());
25 info.datastrid = packageDetail["datastrid"].toString();
26 info.writable = packageDetail["writable"].toBool(false);
27 info.updateTime = packageDetail["updateTime"].toVariant().toLongLong();
28
29 if (packageDetail.contains("dataStr")) {
30 QJsonObject dataStr = packageDetail["dataStr"].toObject();
31
32 if (dataStr.contains("head")) {
33 QJsonObject head = dataStr["head"].toObject();
34
35 info.editorVersion = head["editorVersion"].toString();
36
37 info.puuid = head["puuid"].toString();
38 info.utime = head["utime"].toVariant().toLongLong();
39 info.importFlag = head["importFlag"].toBool(false);
40 info.hasIdFlag = head["hasIdFlag"].toBool(false);
41 info.newgId = head["newgId"].toBool(false);
42
43 if (head.contains("c_para")) {
44 QJsonObject c_para = head["c_para"].toObject();
45 info.name = c_para["package"].toString();
46
47 bool isSmd = cadData.contains("SMT") && cadData["SMT"].toBool();
48 info.type = isSmd ? "smd" : "tht";
49
50 if (c_para.contains("3DModel")) {
51 info.model3DName = c_para["3DModel"].toString();
52 }
53
54 // 附加参数
55 info.link = c_para["link"].toString();
56 info.contributor = c_para["Contributor"].toString();
57 info.uuid3d = head["uuid_3d"].toString();
58 }
59
60 // 保存画布信息
61 if (dataStr.contains("canvas")) {
62 info.canvas = dataStr["canvas"].toString();
63 }
64
65 if (dataStr.contains("layers")) {
66 QJsonArray layersArray = dataStr["layers"].toArray();
67 QStringList layers;
68 for (const QJsonValue& layerValue : layersArray) {
69 layers.append(layerValue.toString());
70 }
71 info.layers = layers.join("\n");
72 }
73
74 if (dataStr.contains("objects")) {
75 QJsonArray objectsArray = dataStr["objects"].toArray();
76 QStringList objects;
77 for (const QJsonValue& objectValue : objectsArray) {
78 objects.append(objectValue.toString());
79 }
80 info.objects = objects.join("\n");
81 }
82 }
83
84 // 导入边界框（包围盒）
85 if (dataStr.contains("head")) {
86 QJsonObject head = dataStr["head"].toObject();
87 FootprintBBox footprintBbox;
88 if (dataStr.contains("BBox")) {
89 QJsonObject bbox = dataStr["BBox"].toObject();
90 footprintBbox.x = bbox["x"].toDouble();
91 footprintBbox.y = bbox["y"].toDouble();
92 footprintBbox.width = bbox["width"].toDouble();
93 footprintBbox.height = bbox["height"].toDouble();
94 } else {
95 footprintBbox.x = head["x"].toDouble();
96 footprintBbox.y = head["y"].toDouble();
97 footprintBbox.width = 0;
98 footprintBbox.height = 0;
99 }
100 footprintData->setBbox(footprintBbox);
101 }
102
103 // 导入几何数据（焊盘、走线、孔等）
104 if (dataStr.contains("shape")) {
105 QJsonArray shapes = dataStr["shape"].toArray();
106
107 for (const QJsonValue& shapeValue : shapes) {
108 QString shapeString = shapeValue.toString();
109 QStringList parts = shapeString.split("~");
110
111 if (parts.isEmpty())
112 continue;
113
114 QString designator = parts[0];
115
116 if (designator == "PAD") {
117 FootprintPad pad = importPadData(shapeString);
118 footprintData->addPad(pad);
119 } else if (designator == "TRACK") {
120 FootprintTrack track = importTrackData(shapeString);
121 footprintData->addTrack(track);
122 } else if (designator == "HOLE") {
123 FootprintHole hole = importHoleData(shapeString);
124 footprintData->addHole(hole);
125 } else if (designator == "CIRCLE") {
126 FootprintCircle circle = importFootprintCircleData(shapeString);
127 footprintData->addCircle(circle);
128 } else if (designator == "ARC") {
129 FootprintArc arc = importFootprintArcData(shapeString);
130 footprintData->addArc(arc);
131 } else if (designator == "RECT") {
132 FootprintRectangle rectangle = importFootprintRectangleData(shapeString);
133 footprintData->addRectangle(rectangle);
134 } else if (designator == "TEXT") {
135 FootprintText text = importFootprintTextData(shapeString);
136 footprintData->addText(text);
137 } else if (designator == "POLYLINE" || designator == "PL") {
138 FootprintTrack track = importTrackData(shapeString);
139 footprintData->addTrack(track);
140 } else if (designator == "POLYGON" || designator == "PG") {
141 FootprintTrack track = importTrackData(shapeString);
142 footprintData->addTrack(track);
143 } else if (designator == "PATH" || designator == "PT") {
144 FootprintTrack track = importTrackData(shapeString);
145 footprintData->addTrack(track);
146 } else if (designator == "SVGNODE") {
147 importSvgNodeData(shapeString, footprintData);
148 } else if (designator == "SOLIDREGION") {
149 FootprintSolidRegion solidRegion = importSolidRegionData(shapeString);
150 footprintData->addSolidRegion(solidRegion);
151 }
152 }
153
154 if (dataStr.contains("layers")) {
155 QJsonArray layers = dataStr["layers"].toArray();
156 for (const QJsonValue& layerValue : layers) {
157 LayerDefinition layer = parseLayerDefinition(layerValue.toString());
158 footprintData->addLayer(layer);
159 }
160 }
161
162 if (dataStr.contains("objects")) {
163 QJsonArray objects = dataStr["objects"].toArray();
164 for (const QJsonValue& objectValue : objects) {
165 ObjectVisibility visibility = parseObjectVisibility(objectValue.toString());
166 footprintData->addObjectVisibility(visibility);
167 }
168 }
169
170 bool hasHole = false;
171 for (const FootprintPad& pad : footprintData->pads()) {
172 if (pad.holeRadius > 0 || pad.holeLength > 0) {
173 hasHole = true;
174 break;
175 }
176 }
177 if (hasHole) {
178 info.type = "tht";
179 }
180 }
181
182 footprintData->setInfo(info);
183 }
184 }
185
186 return footprintData;

338void EasyedaFootprintImporter::importSvgNodeData(const QString& svgNodeData,
339 QSharedPointer<FootprintData> footprintData) {
340 if (!footprintData) {
341 qWarning() << "importSvgNodeData called with null footprintData";
342 return;
343 }
344
345 QStringList parts = svgNodeData.split("~");
346 if (parts.size() < 2) {
347 qWarning() << "Invalid SVGNODE data format";
348 return;
349 }
350
351 QJsonDocument doc = QJsonDocument::fromJson(parts[1].toUtf8());
352 if (!doc.isObject()) {
353 qWarning() << "Failed to parse SVGNODE JSON";
354 return;
355 }
356
357 QJsonObject root = doc.object();
358 if (!root.contains("attrs")) {
359 qWarning() << "SVGNODE missing attrs";
360 return;
361 }
362
363 QJsonObject attrs = root["attrs"].toObject();
364
365 if (attrs.contains("c_etype") && attrs["c_etype"].toString() == "outline3D") {
366 Model3DData model3D;
367 if (attrs.contains("uuid")) {
368 QString modelUuid = attrs["uuid"].toString();
369 model3D.setUuid(modelUuid);
370 model3D.setName(attrs.contains("title") ? attrs["title"].toString() : "");
371
372 if (attrs.contains("c_origin")) {
373 QString c_origin = attrs["c_origin"].toString();
374 QStringList originParts = c_origin.split(",");
375 if (originParts.size() >= 2) {
376 Model3DBase translation;
377 double originX = originParts[0].toDouble();
378 double originY = originParts[1].toDouble();
379
380 if (qAbs(originX) < 1000 && qAbs(originY) < 1000) {
381 double bboxCenterX = footprintData->bbox().x + footprintData->bbox().width / 2.0;
382 double bboxCenterY = footprintData->bbox().y + footprintData->bbox().height / 2.0;
383
384 translation.x = bboxCenterX;
385 translation.y = bboxCenterY;
386 } else {
387 translation.x = originX;
388 translation.y = originY;
389 }
390
391 translation.z = attrs.contains("z") ? attrs["z"].toDouble() : 0.0;
392 model3D.setTranslation(translation);
393 }
394 }
395
396 if (attrs.contains("c_rotation")) {
397 QString c_rotation = attrs["c_rotation"].toString();
398 QStringList rotationParts = c_rotation.split(",");
399 if (rotationParts.size() >= 3) {
400 Model3DBase rotation;
401 rotation.x = rotationParts[0].toDouble();
402 rotation.y = rotationParts[1].toDouble();
403 rotation.z = rotationParts[2].toDouble();
404 model3D.setRotation(rotation);
405 }
406 }
407
408 footprintData->setModel3D(model3D);
409 }
410
411 FootprintOutline outline;
412 outline.id = attrs["id"].toString();
413 outline.strokeWidth = attrs.contains("c_width") ? attrs["c_width"].toString().toDouble() : 0.0;
414 outline.isLocked = false;
415
416 if (attrs.contains("layerid")) {
417 outline.layerId = attrs["layerid"].toString().toInt();
418 } else {
419 outline.layerId = 19; // 默认3DModel 层
420 }
421
422 if (root.contains("childNodes")) {
423 QJsonArray childNodes = root["childNodes"].toArray();
424 for (const QJsonValue& child : childNodes) {
425 QJsonObject childObj = child.toObject();
426 if (childObj.contains("attrs")) {
427 QJsonObject childAttrs = childObj["attrs"].toObject();
428 if (childAttrs.contains("points")) {
429 outline.path = childAttrs["points"].toString();
430 break;
431 }
432 }
433 }
434 }
435
436 footprintData->addOutline(outline);
437 } else {
438 Model3DData model3D;
439 model3D.setName(attrs.contains("title") ? attrs["title"].toString() : "");
440 model3D.setUuid(attrs.contains("uuid") ? attrs["uuid"].toString() : "");
441
442 if (attrs.contains("c_origin")) {
443 QString c_origin = attrs["c_origin"].toString();
444 QStringList originParts = c_origin.split(",");
445 if (originParts.size() >= 2) {
446 Model3DBase translation;
447 translation.x = originParts[0].toDouble();
448 translation.y = originParts[1].toDouble();
449 translation.z = attrs.contains("z") ? attrs["z"].toDouble() : 0.0;
450 model3D.setTranslation(translation);
451 }
452 }
453
454 if (attrs.contains("c_rotation")) {
455 QString c_rotation = attrs["c_rotation"].toString();
456 QStringList rotationParts = c_rotation.split(",");
457 if (rotationParts.size() >= 3) {
458 Model3DBase rotation;
459 rotation.x = rotationParts[0].toDouble();
460 rotation.y = rotationParts[1].toDouble();
461 rotation.z = rotationParts[2].toDouble();
462 model3D.setRotation(rotation);
463 }
464 }
465
466 footprintData->setModel3D(model3D);
467 }
