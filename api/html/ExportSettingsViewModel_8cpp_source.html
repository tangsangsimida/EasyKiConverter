257void ExportSettingsViewModel::startExport(const QStringList& componentIds) {
258 qDebug() << "Starting export for" << componentIds.size() << "components";
259
260 if (m_isExporting) {
261 qWarning() << "Export already in progress";
262 return;
263 }
264
265 if (!m_exportService) {
266 qWarning() << "ParallelExportService is not available";
267 setStatus("Export service not available");
268 return;
269 }
270
271 if (componentIds.isEmpty()) {
272 qWarning() << "No components to export";
273 setStatus("No components to export");
274 return;
275 }
276
277 // Save component IDs for after preload
278 m_pendingComponentIds = componentIds;
279
280 // Build and set export options
281 buildExportOptions();
282
283 setIsExporting(true);
284 setStatus("Preloading component data...");
285
286 // Start preload first
287 m_exportService->startPreload(componentIds);
288}

290void ExportSettingsViewModel::buildExportOptions() {
291 ExportOptions options;
292
293 // Handle output path
294 QString absoluteOutputPath = m_outputPath;
295 if (absoluteOutputPath.isEmpty()) {
296 QString documentsPath = QStandardPaths::writableLocation(QStandardPaths::DocumentsLocation);
297 QDir exportDir(documentsPath);
298
299 if (!exportDir.exists("EasyKiConverter")) {
300 exportDir.mkdir("EasyKiConverter");
301 }
302 exportDir.cd("EasyKiConverter");
303 absoluteOutputPath = exportDir.absoluteFilePath(m_libName);
304 } else {
305 QDir dir(absoluteOutputPath);
306 if (dir.isAbsolute()) {
307 absoluteOutputPath = dir.cleanPath(absoluteOutputPath);
308 } else {
309 QString documentsPath = QStandardPaths::writableLocation(QStandardPaths::DocumentsLocation);
310 QDir exportDir(documentsPath);
311 if (!exportDir.exists("EasyKiConverter")) {
312 exportDir.mkdir("EasyKiConverter");
313 }
314 exportDir.cd("EasyKiConverter");
315 absoluteOutputPath = exportDir.absoluteFilePath(absoluteOutputPath);
316 }
317 }
318
319 if (absoluteOutputPath.isEmpty()) {
320 absoluteOutputPath = QStandardPaths::writableLocation(QStandardPaths::DesktopLocation);
321 absoluteOutputPath = QDir(absoluteOutputPath).absoluteFilePath(m_libName);
322 }
323
324 options.outputPath = absoluteOutputPath;
325 options.libName = m_libName;
326 options.exportSymbol = m_exportSymbol;
327 options.exportFootprint = m_exportFootprint;
328 options.exportModel3D = m_exportModel3D;
329 options.exportModel3DFormat = m_exportModel3DFormat;
330 options.exportModel3DPathMode = m_exportModel3DPathMode;
331 options.exportPreviewImages = m_exportPreviewImages;
332 options.exportDatasheet = m_exportDatasheet;
333 options.overwriteExistingFiles = m_overwriteExistingFiles;
334 options.weakNetworkSupport = m_weakNetworkSupport;
335 options.updateMode = (m_exportMode == 1);
336 options.debugMode = m_debugMode;
337 options.exportSymbolDescription = m_exportSymbolDescription;
338 options.exportFootprintDescription = m_exportFootprintDescription;
339 options.symbolLibraryDescription = m_symbolLibraryDescription;
340 options.footprintLibraryDescription = m_footprintLibraryDescription;
341 options.footprintLibraryKeywords = m_footprintLibraryKeywords;
342
343 qInfo() << "Export options:" << "OutputPath:" << options.outputPath << "LibName:" << options.libName
344 << "Symbol:" << options.exportSymbol << "Footprint:" << options.exportFootprint
345 << "3D Model:" << options.exportModel3D << "3D Model Format:" << options.exportModel3DFormat
346 << "(1=WRL, 2=STEP, 3=Both)" << "3D Model Path Mode:" << options.exportModel3DPathMode
347 << "(0=Relative, 1=Absolute)" << "Preview Images:" << options.exportPreviewImages
348 << "Datasheet:" << options.exportDatasheet
349 << "Client Weak Network Adaptation:" << options.weakNetworkSupport << "Update Mode:" << options.updateMode
350 << "Debug Mode:" << options.debugMode << "Symbol Description:" << options.exportSymbolDescription
351 << "Footprint Description:" << options.exportFootprintDescription
352 << "Symbol Library Description:" << options.symbolLibraryDescription
353 << "Footprint Library Description:" << options.footprintLibraryDescription;
354
355 m_exportService->setOptions(options);
356 m_exportService->setOutputPath(absoluteOutputPath);
357}

453void ExportSettingsViewModel::loadFromConfig() {
454 // 保存旧值以便比较，仅在值变化时发射信号（避免不必要的 QML 绑定刷新）
455 const QString oldOutputPath = m_outputPath;
456 const QString oldLibName = m_libName;
457 const bool oldExportSymbol = m_exportSymbol;
458 const bool oldExportFootprint = m_exportFootprint;
459 const bool oldExportModel3D = m_exportModel3D;
460 const int oldExportModel3DFormat = m_exportModel3DFormat;
461 const int oldExportModel3DPathMode = m_exportModel3DPathMode;
462 const bool oldExportPreviewImages = m_exportPreviewImages;
463 const bool oldExportDatasheet = m_exportDatasheet;
464 const bool oldOverwriteExistingFiles = m_overwriteExistingFiles;
465 const bool oldWeakNetworkSupport = m_weakNetworkSupport;
466 const int oldExportMode = m_exportMode;
467 const bool oldDebugMode = m_debugMode;
468 const QString oldCacheDir = m_cacheDir;
469 const int oldDiskCacheLimitMB = m_diskCacheLimitMB;
470
471 m_outputPath = m_configService->getOutputPath();
472 m_libName = PathSecurity::sanitizeFilename(m_configService->getLibName());
473 m_exportSymbol = m_configService->getExportSymbol();
474 m_exportFootprint = m_configService->getExportFootprint();
475 m_exportModel3D = m_configService->getExportModel3D();
476 m_exportModel3DFormat = m_configService->getExportModel3DFormat();
477 m_exportModel3DPathMode = m_configService->getExportModel3DPathMode();
478 m_exportPreviewImages = m_configService->getExportPreviewImages();
479 m_exportDatasheet = m_configService->getExportDatasheet();
480 m_overwriteExistingFiles = m_configService->getOverwriteExistingFiles();
481 m_weakNetworkSupport = m_configService->getWeakNetworkSupport();
482 m_exportMode = m_configService->getExportMode();
483 m_cacheDir = m_configService->getCacheDir();
484 m_diskCacheLimitMB = m_configService->getDiskCacheLimitMB();
485
486 bool envDebugMode = qEnvironmentVariableIsSet("EASYKICONVERTER_DEBUG_MODE");
487 if (envDebugMode) {
488 QString debugValue = qEnvironmentVariable("EASYKICONVERTER_DEBUG_MODE", "false").toLower();
489 m_debugMode = (debugValue == "true" || debugValue == "1" || debugValue == "yes");
490 } else {
491 m_debugMode = m_configService->getDebugMode();
492 }
493
494 // 以下字段 ConfigService 暂未支持持久化，使用默认值
495 m_exportSymbolDescription = true;
496 m_exportFootprintDescription = true;
497 m_symbolLibraryDescription.clear();
498 m_footprintLibraryDescription.clear();
499 m_footprintLibraryKeywords.clear();
500
501 // 仅在值变化时发射信号
502 if (m_outputPath != oldOutputPath)
503 emit outputPathChanged();
504 if (m_libName != oldLibName)
505 emit libNameChanged();
506 if (m_exportSymbol != oldExportSymbol)
507 emit exportSymbolChanged();
508 if (m_exportFootprint != oldExportFootprint)
509 emit exportFootprintChanged();
510 if (m_exportModel3D != oldExportModel3D)
511 emit exportModel3DChanged();
512 if (m_exportModel3DFormat != oldExportModel3DFormat)
513 emit exportModel3DFormatChanged();
514 if (m_exportModel3DPathMode != oldExportModel3DPathMode)
515 emit exportModel3DPathModeChanged();
516 if (m_exportPreviewImages != oldExportPreviewImages)
517 emit exportPreviewImagesChanged();
518 if (m_exportDatasheet != oldExportDatasheet)
519 emit exportDatasheetChanged();
520 if (m_overwriteExistingFiles != oldOverwriteExistingFiles)
521 emit overwriteExistingFilesChanged();
522 if (m_weakNetworkSupport != oldWeakNetworkSupport)
523 emit weakNetworkSupportChanged();
524 if (m_exportMode != oldExportMode)
525 emit exportModeChanged();
526 if (m_debugMode != oldDebugMode)
527 emit debugModeChanged();
528 // 非持久化字段始终发射（首次加载时需要通知 QML）
529 emit exportSymbolDescriptionChanged();
530 emit exportFootprintDescriptionChanged();
531 emit symbolLibraryDescriptionChanged();
532 emit footprintLibraryDescriptionChanged();
533 emit footprintLibraryKeywordsChanged();
534 if (m_cacheDir != oldCacheDir)
535 emit cacheDirChanged();
536 if (m_diskCacheLimitMB != oldDiskCacheLimitMB)
537 emit diskCacheLimitMBChanged();
538}
