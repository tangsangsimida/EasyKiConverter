87void ExportSettingsViewModel::setExportMode(int mode) {
88 if (m_exportMode != mode) {
89 m_exportMode = mode;
90 emit exportModeChanged();
91 qDebug() << "Export mode changed to:" << mode << "(0=append, 1=update)";
92
93 // 同步更新 overwriteExistingFiles
94 // 0 = 追加模式（保留已存在的元器件，跳过重复的）-> overwriteExistingFiles = false
95 // 1 = 更新模式（替换相同的元器件，保留不同的元器件，添加新的元器件）-> overwriteExistingFiles = false
96 // 注意：更新模式不删除整个文件，而是智能合并
97 bool newOverwrite = false; // 两种模式都不删除整个文件
98 if (m_overwriteExistingFiles != newOverwrite) {
99 m_overwriteExistingFiles = newOverwrite;
100 m_configService->setOverwriteExistingFiles(newOverwrite);
101 emit overwriteExistingFilesChanged();
102 qDebug() << "overwriteExistingFiles changed to:" << newOverwrite;
103 }
104 }
105}

128void ExportSettingsViewModel::startExport(const QStringList& componentIds) {
129 qDebug() << "Starting export for" << componentIds.size() << "components";
130
131 if (m_isExporting) {
132 qWarning() << "Export already in progress";
133 return;
134 }
135
136 if (!m_exportService) {
137 qWarning() << "ExportService is not available";
138 setStatus("Export service not available");
139 return;
140 }
141
142 if (componentIds.isEmpty()) {
143 qWarning() << "No components to export";
144 setStatus("No components to export");
145 return;
146 }
147
148 // 设置导出选项
149 ExportOptions options;
150 options.outputPath = m_outputPath;
151 options.libName = m_libName;
152 options.exportSymbol = m_exportSymbol;
153 options.exportFootprint = m_exportFootprint;
154 options.exportModel3D = m_exportModel3D;
155 options.updateMode = (m_exportMode == 1); // 1 = 更新模式
156 options.debugMode = m_debugMode;
157
158 qDebug() << "Export options:"
159 << "OutputPath:" << options.outputPath << "LibName:" << options.libName
160 << "Symbol:" << options.exportSymbol << "Footprint:" << options.exportFootprint
161 << "3D Model:" << options.exportModel3D << "Update Mode:" << options.updateMode
162 << "Debug Mode:" << options.debugMode;
163
164 // 设置导出状态
165 setIsExporting(true);
166 setStatus("Export started");
167 setProgress(0);
168
169 // 调用 ExportService 执行导出
170 m_exportService->executeExportPipeline(componentIds, options);
171}

240void ExportSettingsViewModel::loadFromConfig() {
241 // 从配置服务加载设置
242 m_outputPath = m_configService->getOutputPath();
243 m_libName = m_configService->getLibName();
244 m_exportSymbol = m_configService->getExportSymbol();
245 m_exportFootprint = m_configService->getExportFootprint();
246 m_exportModel3D = m_configService->getExportModel3D();
247 m_overwriteExistingFiles = m_configService->getOverwriteExistingFiles();
248
249 // 从环境变量读取调试模式（优先级高于配置文件）
250 bool envDebugMode = qEnvironmentVariableIsSet("EASYKICONVERTER_DEBUG_MODE");
251 if (envDebugMode) {
252 QString debugValue = qEnvironmentVariable("EASYKICONVERTER_DEBUG_MODE", "false").toLower();
253 m_debugMode = (debugValue == "true" || debugValue == "1" || debugValue == "yes");
254 qDebug() << "Debug mode enabled via environment variable EASYKICONVERTER_DEBUG_MODE:" << m_debugMode;
255 } else {
256 m_debugMode = m_configService->getDebugMode();
257 }
258
259 // 发送变更信号以更新 UI
260 emit outputPathChanged();
261 emit libNameChanged();
262 emit exportSymbolChanged();
263 emit exportFootprintChanged();
264 emit exportModel3DChanged();
265 emit overwriteExistingFilesChanged();
266 emit debugModeChanged();
267
268 qDebug() << "Loading configuration from ConfigService";
269}
