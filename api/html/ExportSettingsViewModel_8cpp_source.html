116void ExportSettingsViewModel::setExportMode(int mode) {
117 if (m_exportMode != mode) {
118 m_exportMode = mode;
119 emit exportModeChanged();
120 qDebug() << "Export mode changed to:" << mode << "(0=append, 1=update)";
121
122 // 同步更新 overwriteExistingFiles
123 // 0 = 追加模式（保留已存在的元器件，跳过重复的）-> overwriteExistingFiles = false
124 // 1 = 更新模式（替换相同的元器件，保留不同的元器件，添加新的元器件）-> overwriteExistingFiles = false
125 // 注意：更新模式不删除整个文件，而是智能合并
126 bool newOverwrite = false; // 两种模式都不删除整个文件
127 if (m_overwriteExistingFiles != newOverwrite) {
128 m_overwriteExistingFiles = newOverwrite;
129 m_configService->setOverwriteExistingFiles(newOverwrite);
130 emit overwriteExistingFilesChanged();
131 qDebug() << "overwriteExistingFiles changed to:" << newOverwrite;
132 }
133 }

157void ExportSettingsViewModel::startExport(const QStringList& componentIds) {
158 qDebug() << "Starting export for" << componentIds.size() << "components";
159
160 if (m_isExporting) {
161 qWarning() << "Export already in progress";
162 return;
163 }
164
165 if (!m_exportService) {
166 qWarning() << "ExportService is not available";
167 setStatus("Export service not available");
168 return;
169 }
170
171 if (componentIds.isEmpty()) {
172 qWarning() << "No components to export";
173 setStatus("No components to export");
174 return;
175 }
176
177 // 设置导出选项
178 ExportOptions options;
179
180 // 调试：打印原始输入值
181 qDebug() << "Original input values - outputPath:" << m_outputPath << "libName:" << m_libName;
182
183 // 处理输出路径
184 QString absoluteOutputPath = m_outputPath;
185 if (absoluteOutputPath.isEmpty()) {
186 // 如果导出路径为空，使用默认路径：~/Documents/EasyKiConverter/库名称
187 QString documentsPath = QStandardPaths::writableLocation(QStandardPaths::DocumentsLocation);
188 QDir exportDir(documentsPath);
189
190 // 创建 EasyKiConverter 目录
191 if (!exportDir.exists("EasyKiConverter")) {
192 exportDir.mkdir("EasyKiConverter");
193 }
194 exportDir.cd("EasyKiConverter");
195
196 // 使用库名称作为导出路径
197 absoluteOutputPath = exportDir.absoluteFilePath(m_libName);
198 qDebug() << "Using default export path (empty input):" << absoluteOutputPath;
199 } else {
200 QDir dir(absoluteOutputPath);
201
202 if (dir.isAbsolute()) {
203 // 如果是绝对路径，规范化路径（处理 .. 和 . 等）
204 absoluteOutputPath = dir.cleanPath(absoluteOutputPath);
205 qDebug() << "Normalized absolute path:" << m_outputPath << "->" << absoluteOutputPath;
206 } else {
207 // 如果是相对路径，相对于 ~/Documents/EasyKiConverter/ 转换为绝对路径
208 QString documentsPath = QStandardPaths::writableLocation(QStandardPaths::DocumentsLocation);
209 QDir exportDir(documentsPath);
210
211 // 创建 EasyKiConverter 目录
212 if (!exportDir.exists("EasyKiConverter")) {
213 exportDir.mkdir("EasyKiConverter");
214 }
215 exportDir.cd("EasyKiConverter");
216
217 // 直接拼接相对路径，不添加库名称
218 absoluteOutputPath = exportDir.absoluteFilePath(absoluteOutputPath);
219 qDebug() << "Converted relative path to absolute path (Documents/EasyKiConverter/userPath):" << m_outputPath
220 << "->" << absoluteOutputPath;
221 }
222 }
223
224 // 最终验证路径
225 if (absoluteOutputPath.isEmpty()) {
226 qWarning() << "Final output path is still empty! Using fallback to Desktop.";
227 absoluteOutputPath = QStandardPaths::writableLocation(QStandardPaths::DesktopLocation);
228 absoluteOutputPath = QDir(absoluteOutputPath).absoluteFilePath(m_libName);
229 }
230
231 options.outputPath = absoluteOutputPath;
232 options.libName = m_libName;
233 options.exportSymbol = m_exportSymbol;
234 options.exportFootprint = m_exportFootprint;
235 options.exportModel3D = m_exportModel3D;
236 options.exportPreviewImages = m_exportPreviewImages;
237 options.exportDatasheet = m_exportDatasheet;
238 options.overwriteExistingFiles = m_overwriteExistingFiles;
239 options.updateMode = (m_exportMode == 1); // 1 = 更新模式
240 options.debugMode = m_debugMode;
241
242 qDebug() << "Export options:" << "OutputPath:" << options.outputPath << "LibName:" << options.libName
243 << "Symbol:" << options.exportSymbol << "Footprint:" << options.exportFootprint
244 << "3D Model:" << options.exportModel3D << "Preview Images:" << options.exportPreviewImages
245 << "Datasheet:" << options.exportDatasheet << "Update Mode:" << options.updateMode
246 << "Debug Mode:" << options.debugMode;
247
248 // 设置导出状态
249 setIsExporting(true);
250 setStatus("Export started");
251 setProgress(0);
252
253 // 调用 ExportService 执行导出
254 m_exportService->executeExportPipeline(componentIds, options);
255}

334void ExportSettingsViewModel::loadFromConfig() {
335 // 从配置服务加载设置
336 m_outputPath = m_configService->getOutputPath();
337 m_libName = m_configService->getLibName();
338 m_exportSymbol = m_configService->getExportSymbol();
339 m_exportFootprint = m_configService->getExportFootprint();
340 m_exportModel3D = m_configService->getExportModel3D();
341 m_exportPreviewImages = m_configService->getExportPreviewImages();
342 m_exportDatasheet = m_configService->getExportDatasheet();
343 m_overwriteExistingFiles = m_configService->getOverwriteExistingFiles();
344
345 // 从环境变量读取调试模式（优先级高于配置文件）
346 bool envDebugMode = qEnvironmentVariableIsSet("EASYKICONVERTER_DEBUG_MODE");
347 if (envDebugMode) {
348 QString debugValue = qEnvironmentVariable("EASYKICONVERTER_DEBUG_MODE", "false").toLower();
349 m_debugMode = (debugValue == "true" || debugValue == "1" || debugValue == "yes");
350 qDebug() << "Debug mode enabled via environment variable EASYKICONVERTER_DEBUG_MODE:" << m_debugMode;
351 } else {
352 m_debugMode = m_configService->getDebugMode();
353 }
354
355 // 发送变更信号以更新 UI
356 emit outputPathChanged();
357 emit libNameChanged();
358 emit exportSymbolChanged();
359 emit exportFootprintChanged();
360 emit exportModel3DChanged();
361 emit exportPreviewImagesChanged();
362 emit exportDatasheetChanged();
363 emit overwriteExistingFilesChanged();
364 emit debugModeChanged();
365
366 qDebug() << "Loading configuration from ConfigService";
367}
